### 1.2 Token成本计算

```python
"""
Token成本计算
"""

# 主流模型价格（每1K tokens）
MODEL_PRICES = {
    "gpt-4": {"input": 0.03, "output": 0.06},
    "gpt-4-turbo": {"input": 0.01, "output": 0.03},
    "gpt-3.5-turbo": {"input": 0.0005, "output": 0.0015},
    "claude-3-opus": {"input": 0.015, "output": 0.075},
    "claude-3-sonnet": {"input": 0.003, "output": 0.015},
    "claude-3-haiku": {"input": 0.00025, "output": 0.00125}
}

def calculate_cost(
    model: str,
    input_tokens: int,
    output_tokens: int
) -> float:
    """计算成本"""
    if model not in MODEL_PRICES:
        return 0

    prices = MODEL_PRICES[model]
    cost = (
        input_tokens * prices["input"] / 1000 +
        output_tokens * prices["output"] / 1000
    )

    return cost

# 示例
cost = calculate_cost("gpt-4", 1000, 500)
print(f"成本: ${cost:.4f}")
```

## 二、Token优化

### 2.1 Token追踪器

```python
"""
Token使用追踪
"""

from dataclasses import dataclass, field
from typing import Dict, List
from datetime import datetime

@dataclass
class TokenUsage:
    """Token使用记录"""
    model: str
    input_tokens: int
    output_tokens: int
    cost: float
    timestamp: datetime
    metadata: dict = field(default_factory=dict)

class TokenTracker:
    """Token追踪器"""

    def __init__(self):
        self.usages: List[TokenUsage] = []
        self.total_cost = 0

    def record(
        self,
        model: str,
        input_tokens: int,
        output_tokens: int,
        metadata: dict = None
    ):
        """记录使用"""
        cost = calculate_cost(model, input_tokens, output_tokens)

        usage = TokenUsage(
            model=model,
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            cost=cost,
            timestamp=datetime.now(),
            metadata=metadata or {}
        )

        self.usages.append(usage)
        self.total_cost += cost

        return cost

    def get_summary(self) -> Dict:
        """获取摘要"""
        if not self.usages:
            return {"total_cost": 0}

        total_input = sum(u.input_tokens for u in self.usages)
        total_output = sum(u.output_tokens for u in self.usages)

        by_model = {}
        for u in self.usages:
            if u.model not in by_model:
                by_model[u.model] = {
                    "calls": 0,
                    "input_tokens": 0,
                    "output_tokens": 0,
                    "cost": 0
                }
            by_model[u.model]["calls"] += 1
            by_model[u.model]["input_tokens"] += u.input_tokens
            by_model[u.model]["output_tokens"] += u.output_tokens
            by_model[u.model]["cost"] += u.cost

        return {
            "total_cost": self.total_cost,
            "total_input_tokens": total_input,
            "total_output_tokens": total_output,
            "total_calls": len(self.usages),
            "by_model": by_model
        }

    def get_daily_usage(self) -> Dict:
        """获取每日使用"""
        daily = {}

        for u in self.usages:
            date = u.timestamp.date().isoformat()
            if date not in daily:
                daily[date] = {"cost": 0, "calls": 0}
            daily[date]["cost"] += u.cost
            daily[date]["calls"] += 1

        return daily
```

### 2.2 Token预算控制

```python
"""
Token预算控制
"""

from datetime import datetime, timedelta

class TokenBudget:
    """Token预算管理"""

    def __init__(
        self,
        daily_budget: float = 10.0,
        monthly_budget: float = 300.0
    ):
        self.daily_budget = daily_budget
        self.monthly_budget = monthly_budget
        self.tracker = TokenTracker()

    def check_budget(self) -> tuple:
        """检查预算"""
        # 获取今日使用
        daily = self.tracker.get_daily_usage()
        today = datetime.now().date().isoformat()
        today_cost = daily.get(today, {"cost": 0})["cost"]

        # 获取本月使用
        monthly_cost = sum(
            data["cost"] for date, data in daily.items()
            if datetime.fromisoformat(date).month == datetime.now().month
        )

        # 检查
        daily_ok = today_cost < self.daily_budget
        monthly_ok = monthly_cost < self.monthly_budget

        return daily_ok and monthly_ok, {
            "daily_used": today_cost,
            "daily_remaining": self.daily_budget - today_cost,
            "monthly_used": monthly_cost,
            "monthly_remaining": self.monthly_budget - monthly_cost
        }

    def estimate_cost(
        self,
        model: str,
        input_tokens: int,
        estimated_output: int
    ) -> float:
        """估算成本"""
        return calculate_cost(model, input_tokens, estimated_output)

    def can_afford(
        self,
        estimated_cost: float
    ) -> tuple:
        """是否负担得起"""
        ok, status = self.check_budget()

        if not ok:
            return False, "预算已用完"

        remaining = status["daily_remaining"]

        if estimated_cost > remaining:
            return False, f"估算成本${estimated_cost:.4f}超过剩余预算${remaining:.4f}"

        return True, "OK"
```

### 2.3 Token优化策略

```python
"""
Token优化策略实现
"""

class TokenOptimizer:
    """Token优化器"""

    def __init__(self, llm):
        self.llm = llm

    def optimize_prompt(self, prompt: str) -> str:
        """优化提示词"""
        # 移除多余空白
        prompt = " ".join(prompt.split())

        # 移除不必要的描述
        # 简化措辞

        return prompt

    def compress_history(
        self,
        messages: List[dict],
        max_tokens: int = 1000
    ) -> List[dict]:
        """压缩历史"""
        # 使用摘要替代完整历史
        if self._count_tokens(messages) <= max_tokens:
            return messages

        # 生成摘要
        history_text = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages[:-5]  # 除了最近5条
        ])

        summary = self.llm.invoke(
            f"简要总结：{history_text}"
        ).content

        return [
            {"role": "system", "content": f"历史摘要：{summary}"}
        ] + messages[-5:]

    def select_important_messages(
        self,
        messages: List[dict],
        max_tokens: int
    ) -> List[dict]:
        """选择重要消息"""
        # 保留系统消息
        system_msgs = [m for m in messages if m.get("role") == "system"]

        # 保留最近消息
        other_msgs = [m for m in messages if m.get("role") != "system"]

        selected = system_msgs[:]
        current_tokens = self._count_tokens(selected)

        for msg in reversed(other_msgs):
            msg_tokens = self._count_tokens([msg])
            if current_tokens + msg_tokens <= max_tokens:
                selected.append(msg)
                current_tokens += msg_tokens

        return selected

    def _count_tokens(self, messages: List[dict]) -> int:
        """计算Token数"""
        total = 0
        for msg in messages:
            total += len(msg.get("content", "").split()) * 1.3
        return int(total)
```

### 3.1 动态模型选择

```python
"""
动态模型选择
"""

class ModelSelector:
    """模型选择器"""

    def __init__(self):
        self.model_capabilities = {
            "gpt-4": {
                "reasoning": 10,
                "creativity": 9,
                "coding": 10,
                "cost": 1  # 成本评分（越高越贵）
            },
            "gpt-3.5-turbo": {
                "reasoning": 7,
                "creativity": 7,
                "coding": 8,
                "cost": 10
            },
            "claude-3-haiku": {
                "reasoning": 6,
                "creativity": 6,
                "coding": 7,
                "cost": 10
            }
        }

    def select_model(
        self,
        task_type: str,
        complexity: str = "medium",
        budget: float = None
    ) -> str:
        """选择模型"""
        # 任务类型映射
        task_capability = {
            "reasoning": "reasoning",
            "creative": "creativity",
            "coding": "coding",
            "simple": "reasoning"
        }

        capability = task_capability.get(task_type, "reasoning")

        # 根据复杂度确定最低要求
        min_score = {
            "low": 5,
            "medium": 7,
            "high": 9
        }.get(complexity, 7)

        # 筛选候选模型
        candidates = []
        for model, caps in self.model_capabilities.items():
            if caps[capability] >= min_score:
                candidates.append(model)

        if not candidates:
            return "gpt-4"  # 默认最强

        # 按成本排序
        candidates.sort(
            key=lambda m: self.model_capabilities[m]["cost"],
            reverse=True
        )

        return candidates[0]

    def select_by_input_size(self, input_tokens: int) -> str:
        """根据输入大小选择"""
        if input_tokens < 1000:
            return "gpt-3.5-turbo"
        elif input_tokens < 5000:
            return "gpt-3.5-turbo"
        else:
            return "gpt-4-turbo"  # 支持更大上下文
```

### 3.2 成本感知路由

```python
"""
成本感知路由
"""

class CostAwareRouter:
    """成本感知路由"""

    def __init__(self, budget_manager, model_selector):
        self.budget = budget_manager
        self.selector = model_selector

    def route_request(
        self,
        prompt: str,
        task_type: str = "general"
    ) -> dict:
        """路由请求"""
        # 估算输入Token
        input_tokens = len(prompt.split()) * 1.3

        # 估算输出Token
        estimated_output = 500 if task_type == "simple" else 1000

        # 选择模型
        model = self.selector.select_model(task_type)

        # 估算成本
        estimated_cost = calculate_cost(model, int(input_tokens), estimated_output)

        # 检查预算
        can_afford, reason = self.budget.can_afford(estimated_cost)

        if not can_afford:
            # 尝试选择更便宜的模型
            cheaper_model = self.selector.select_model(
                task_type,
                complexity="low"
            )
            if cheaper_model != model:
                model = cheaper_model
                estimated_cost = calculate_cost(
                    model, int(input_tokens), estimated_output
                )

        return {
            "model": model,
            "estimated_cost": estimated_cost,
            "input_tokens": int(input_tokens),
            "estimated_output": estimated_output
        }
```

```python
        age = time.time() - entry["timestamp"]
        return age < entry["ttl"]

    def _hash_key(self, content: str) -> str:
        """生成哈希键"""
        return hashlib.md5(content.encode()).hexdigest()
```

```python
"""
智能缓存决策
"""

class SmartCacheDecision:
    """智能缓存决策"""

    def __init__(self, llm):
        self.llm = llm
        self.access_patterns = {}

    def should_cache(self, prompt: str) -> tuple:
        """判断是否应该缓存"""
        # 检查是否包含动态内容
        if self._has_dynamic_content(prompt):
            return False, "包含动态内容"

        # 检查访问模式
        key = self._get_pattern_key(prompt)
        if key in self.access_patterns:
            frequency = self.access_patterns[key]
            if frequency > 3:  # 访问超过3次
                return True, "高频访问"

        return True, "默认缓存"

    def estimate_savings(
        self,
        prompt: str,
        cached: bool,
        estimated_calls: int
    ) -> float:
        """估算节省"""
        tokens = len(prompt.split()) * 1.3
        cost_per_call = calculate_cost("gpt-4", int(tokens), 500)

        if cached:
            # 缓存命中时只计算存储成本
            storage_cost = 0.0001 * len(prompt)  # 简化
            return (cost_per_call * estimated_calls) - storage_cost
        else:
            return 0

    def _has_dynamic_content(self, prompt: str) -> bool:
        """检查动态内容"""
        dynamic_indicators = [
            "现在", "今天", "当前",
            "latest", "current", "now"
        ]
        return any(ind in prompt.lower() for ind in dynamic_indicators)

    def _get_pattern_key(self, prompt: str) -> str:
        """获取模式键"""
        # 提取固定部分
        words = prompt.split()[:10]  # 取前10个词
        return " ".join(words)
```

### 5.1 成本监控器

```python
"""
成本监控系统
"""

from datetime import datetime, timedelta

class CostMonitor:
    """成本监控器"""

    def __init__(self, tracker: TokenTracker):
        self.tracker = tracker
        self.alerts = []

    def check_budget_status(
        self,
        daily_budget: float,
        monthly_budget: float
    ) -> dict:
        """检查预算状态"""
        daily = self.tracker.get_daily_usage()
        today = datetime.now().date().isoformat()

        today_cost = daily.get(today, {"cost": 0})["cost"]

        # 本月成本
        month_start = datetime.now().replace(day=1).date()
        monthly_cost = sum(
            data["cost"] for date, data in daily.items()
            if datetime.fromisoformat(date).date() >= month_start
        )

        # 计算百分比
        daily_pct = (today_cost / daily_budget) * 100
        monthly_pct = (monthly_cost / monthly_budget) * 100

        # 警告
        warnings = []
        if daily_pct > 80:
            warnings.append(f"今日预算已使用{daily_pct:.1f}%")
        if monthly_pct > 80:
            warnings.append(f"本月预算已使用{monthly_pct:.1f}%")

        return {
            "daily": {
                "used": today_cost,
                "budget": daily_budget,
                "percentage": daily_pct
            },
            "monthly": {
                "used": monthly_cost,
                "budget": monthly_budget,
                "percentage": monthly_pct
            },
            "warnings": warnings,
            "status": "warning" if warnings else "ok"
        }

    def get_cost_trend(self, days: int = 7) -> list:
        """获取成本趋势"""
        daily = self.tracker.get_daily_usage()
        trend = []

        for i in range(days):
            date = (datetime.now() - timedelta(days=i)).date().isoformat()
            cost = daily.get(date, {"cost": 0})["cost"]
            trend.append({"date": date, "cost": cost})

        return list(reversed(trend))

    def get_cost_breakdown(self) -> dict:
        """获取成本分解"""
        summary = self.tracker.get_summary()
        by_model = summary.get("by_model", {})

        breakdown = {}
        for model, data in by_model.items():
            breakdown[model] = {
                "calls": data["calls"],
                "cost": data["cost"],
                "percentage": (data["cost"] / summary["total_cost"] * 100)
                if summary["total_cost"] > 0 else 0
            }

        return breakdown
```

```python
"""
成本报告生成
"""

class CostReporter:
    """成本报告生成器"""

    def generate_report(
        self,
        monitor: CostMonitor,
        format: str = "markdown"
    ) -> str:
        """生成报告"""
        status = monitor.check_budget_status(10, 300)
        trend = monitor.get_cost_trend()
        breakdown = monitor.get_cost_breakdown()

        if format == "markdown":
            return self._generate_markdown(status, trend, breakdown)
        else:
            return self._generate_text(status, trend, breakdown)

    def _generate_markdown(self, status, trend, breakdown):
        """生成Markdown报告"""
        report = ["# 成本报告\n"]
        report.append(f"生成时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}\n")

        report.append("## 预算状态\n")
        report.append(f"- 今日: ${status['daily']['used']:.4f} / ${status['daily']['budget']:.2f} ({status['daily']['percentage']:.1f}%)")
        report.append(f"- 本月: ${status['monthly']['used']:.4f} / ${status['monthly']['budget']:.2f} ({status['monthly']['percentage']:.1f}%)")

        report.append("\n## 成本分解\n")
        for model, data in breakdown.items():
            report.append(f"- {model}: ${data['cost']:.4f} ({data['percentage']:.1f}%)")

        report.append("\n## 近期趋势\n")
        for t in trend:
            report.append(f"- {t['date']}: ${t['cost']:.4f}")

        return "\n".join(report)
```

### 核心要点

1. **成本分析**：了解成本构成和来源
2. **Token优化**：追踪、预算、优化
3. **模型选择**：动态选择、成本感知
4. **缓存策略**：多层缓存、智能决策
5. **成本监控**：实时监控、趋势分析

### 成本优化建议

- 选择合适的模型（简单任务用便宜模型）
- 使用缓存减少重复调用
- 优化提示词减少Token消耗
- 设置预算限制和告警
