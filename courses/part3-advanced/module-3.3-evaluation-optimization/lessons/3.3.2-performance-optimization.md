### 1.2 性能分析工具

```python
"""
性能分析工具
"""

import time
from functools import wraps
from typing import Callable

class PerformanceAnalyzer:
    """性能分析器"""

    def __init__(self):
        self.metrics = {}

    def profile(self, func_name: str = None):
        """性能分析装饰器"""
        def decorator(func: Callable):
            name = func_name or func.__name__

            @wraps(func)
            def wrapper(*args, **kwargs):
                start_time = time.time()

                result = func(*args, **kwargs)

                duration = time.time() - start_time

                if name not in self.metrics:
                    self.metrics[name] = []

                self.metrics[name].append(duration)

                return result

            return wrapper
        return decorator

    def get_stats(self, func_name: str) -> dict:
        """获取统计"""
        if func_name not in self.metrics:
            return {}

        durations = self.metrics[func_name]

        return {
            "count": len(durations),
            "total": sum(durations),
            "avg": sum(durations) / len(durations),
            "min": min(durations),
            "max": max(durations)
        }

    def get_report(self) -> str:
        """获取报告"""
        report = ["性能分析报告:\n"]

        for name in self.metrics:
            stats = self.get_stats(name)
            report.append(
                f"- {name}: "
                f"调用{stats['count']}次, "
                f"平均{stats['avg']:.3f}s, "
                f"总计{stats['total']:.3f}s"
            )

        return "\n".join(report)

# 使用
analyzer = PerformanceAnalyzer()

@analyzer.profile("llm_call")
def call_llm(prompt):
    return llm.invoke(prompt)

@analyzer.profile("tool_execution")
def execute_tool(tool_name, params):
    return tools[tool_name](**params)
```

### 2.1 批量处理

```python
"""
批量LLM调用
"""

import asyncio
from typing import List

class BatchLLMProcessor:
    """批量LLM处理器"""

    def __init__(self, llm, max_concurrent=10):
        self.llm = llm
        self.max_concurrent = max_concurrent

    async def process_batch_async(
        self,
        prompts: List[str]
    ) -> List[str]:
        """异步批量处理"""
        semaphore = asyncio.Semaphore(self.max_concurrent)

        async def process_one(prompt):
            async with semaphore:
                # 假设LLM有异步接口
                return await self.llm.ainvoke(prompt)

        tasks = [process_one(p) for p in prompts]
        results = await asyncio.gather(*tasks)

        return results

    def process_batch(
        self,
        prompts: List[str]
    ) -> List[str]:
        """同步批量处理"""
        # 使用并发执行
        from concurrent.futures import ThreadPoolExecutor

        with ThreadPoolExecutor(max_workers=self.max_concurrent) as executor:
            results = list(executor.map(
                lambda p: self.llm.invoke(p),
                prompts
            ))

        return results

# 使用
processor = BatchLLMProcessor(llm, max_concurrent=5)
results = processor.process_batch(prompts)
```

### 2.2 缓存机制

```python
"""
LLM响应缓存
"""

import hashlib
from typing import Any

class LLMCache:
    """LLM缓存"""

    def __init__(self, ttl=3600):
        self.cache = {}
        self.ttl = ttl

    def _get_key(self, prompt: str, **kwargs) -> str:
        """生成缓存键"""
        content = f"{prompt}{kwargs}"
        return hashlib.md5(content.encode()).hexdigest()

    def get(self, prompt: str, **kwargs) -> Any:
        """获取缓存"""
        key = self._get_key(prompt, **kwargs)

        if key in self.cache:
            entry = self.cache[key]

            # 检查是否过期
            if time.time() - entry["timestamp"] < self.ttl:
                return entry["response"]

            # 过期，删除
            del self.cache[key]

        return None

    def set(self, prompt: str, response: Any, **kwargs):
        """设置缓存"""
        key = self._get_key(prompt, **kwargs)
        self.cache[key] = {
            "response": response,
            "timestamp": time.time()
        }

    def cached_call(self, llm, prompt: str, **kwargs):
        """带缓存的调用"""
        # 尝试从缓存获取
        cached = self.get(prompt, **kwargs)
        if cached is not None:
            return cached

        # 调用LLM
        response = llm.invoke(prompt, **kwargs)

        # 缓存结果
        self.set(prompt, response, **kwargs)

        return response

# 使用
cache = LLMCache(ttl=3600)
response = cache.cached_call(llm, "你好")
```

```python
class StreamingResponse:
    """流式响应处理器"""

    def __init__(self, llm):
        self.llm = llm

    def stream_response(
        self,
        prompt: str,
        callback: Callable[[str], None] = None
    ):
        """流式生成响应"""
        full_response = ""

        for chunk in self.llm.stream(prompt):
            content = chunk.content if hasattr(chunk, 'content') else str(chunk)
            full_response += content

            if callback:
                callback(content)

        return full_response

    async def stream_async(
        self,
        prompt: str,
        callback: Callable[[str], None] = None
    ):
        """异步流式生成"""
        full_response = ""

        async for chunk in self.llm.astream(prompt):
            content = chunk.content if hasattr(chunk, 'content') else str(chunk)
            full_response += content

            if callback:
                callback(content)

        return full_response
```

## 三、工具调用优化

### 3.1 并行工具执行

```python
"""
并行工具执行
"""

import asyncio
from typing import List, Dict

class ParallelToolExecutor:
    """并行工具执行器"""

    def __init__(self, tools: Dict[str, Callable], max_workers=5):
        self.tools = tools
        self.max_workers = max_workers

    def execute_parallel(
        self,
        tool_calls: List[dict]
    ) -> List[dict]:
        """并行执行工具"""
        from concurrent.futures import ThreadPoolExecutor

        results = []

        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = {}

            for call in tool_calls:
                tool_name = call["name"]
                params = call.get("params", {})

                if tool_name in self.tools:
                    future = executor.submit(
                        self.tools[tool_name],
                        **params
                    )
                    futures[future] = call

            for future in futures:
                try:
                    result = future.result(timeout=30)
                    results.append({
                        "call": futures[future],
                        "result": result,
                        "success": True
                    })
                except Exception as e:
                    results.append({
                        "call": futures[future],
                        "error": str(e),
                        "success": False
                    })

        return results

    async def execute_parallel_async(
        self,
        tool_calls: List[dict]
    ) -> List[dict]:
        """异步并行执行"""
        semaphore = asyncio.Semaphore(self.max_workers)

        async def execute_one(call):
            async with semaphore:
                tool_name = call["name"]
                params = call.get("params", {})

                if tool_name in self.tools:
                    try:
                        # 假设工具有异步版本
                        result = await self.tools[tool_name](**params)
                        return {"call": call, "result": result, "success": True}
                    except Exception as e:
                        return {"call": call, "error": str(e), "success": False}

        tasks = [execute_one(c) for c in tool_calls]
        results = await asyncio.gather(*tasks)

        return results
```

### 3.2 工具结果缓存

```python
"""
工具结果缓存
"""

class ToolResultCache:
    """工具结果缓存"""

    def __init__(self, default_ttl=300):
        self.cache = {}
        self.default_ttl = default_ttl

    def get_or_execute(
        self,
        tool_name: str,
        params: dict,
        execute_func: Callable,
        ttl: int = None
    ):
        """获取或执行"""
        key = self._make_key(tool_name, params)

        if key in self.cache:
            entry = self.cache[key]
            if time.time() - entry["timestamp"] < (ttl or self.default_ttl):
                return entry["result"]

        # 执行
        result = execute_func(**params)

        # 缓存
        self.cache[key] = {
            "result": result,
            "timestamp": time.time()
        }

        return result

    def _make_key(self, tool_name: str, params: dict) -> str:
        """生成键"""
        import json
        content = f"{tool_name}:{json.dumps(params, sort_keys=True)}"
        return hashlib.md5(content.encode()).hexdigest()

    def invalidate(self, tool_name: str = None):
        """使缓存失效"""
        if tool_name:
            # 删除特定工具的缓存
            keys_to_remove = [
                k for k in self.cache
                if k.startswith(hashlib.md5(tool_name.encode()).hexdigest()[:8])
            ]
            for k in keys_to_remove:
                del self.cache[k]
        else:
            self.cache.clear()
```

## 四、上下文优化

### 4.1 智能上下文选择

```python
"""
智能上下文选择
"""

class SmartContextSelector:
    """智能上下文选择器"""

    def __init__(self, embeddings, max_tokens=4000):
        self.embeddings = embeddings
        self.max_tokens = max_tokens

    def select_relevant_context(
        self,
        query: str,
        context_items: List[str],
        max_items: int = 10
    ) -> List[str]:
        """选择相关上下文"""
        # 计算查询向量
        query_vec = self.embeddings.embed_query(query)

        # 计算每个项目的相关性
        scored_items = []
        for item in context_items:
            item_vec = self.embeddings.embed_query(item)
            similarity = self._cosine_similarity(query_vec, item_vec)
            scored_items.append((item, similarity))

        # 排序
        scored_items.sort(key=lambda x: x[1], reverse=True)

        # 选择直到达到Token限制
        selected = []
        current_tokens = 0

        for item, score in scored_items[:max_items]:
            tokens = len(item) // 4  # 粗略估计

            if current_tokens + tokens <= self.max_tokens:
                selected.append(item)
                current_tokens += tokens

        return selected

    def _cosine_similarity(self, vec1, vec2) -> float:
        """计算余弦相似度"""
        import numpy as np
        return np.dot(vec1, vec2) / (
            np.linalg.norm(vec1) * np.linalg.norm(vec2)
        )
```

```python
"""
上下文压缩优化
"""

class ContextCompressor:
    """上下文压缩器"""

    def __init__(self, llm):
        self.llm = llm

    def compress_for_query(
        self,
        context: str,
        query: str,
        target_tokens: int = 1000
    ) -> str:
        """为查询压缩上下文"""
        prompt = f"""
上下文：
{context}

问题：{query}

请提取与问题相关的关键信息，压缩到约{target_tokens}个token：
"""
        return self.llm.invoke(prompt).content

    def compress_preserve_structure(
        self,
        context: str,
        max_tokens: int = 2000
    ) -> str:
        """保留结构的压缩"""
        # 分段
        paragraphs = context.split("\n\n")

        # 压缩每段
        compressed = []
        for para in paragraphs:
            if len(para) > 200:
                # 压缩长段落
                prompt = f"压缩以下内容，保留关键信息：\n{para}"
                compressed_para = self.llm.invoke(prompt).content
                compressed.append(compressed_para)
            else:
                compressed.append(para)

        return "\n\n".join(compressed)
```

## 五、流程优化

### 5.1 减少迭代次数

```python
"""
减少Agent迭代次数
"""

class IterationOptimizer:
    """迭代优化器"""

    def __init__(self, max_iterations=5):
        self.max_iterations = max_iterations
        self.iteration_history = []

    def should_continue(
        self,
        current_iteration: int,
        current_result: str,
        previous_results: List[str]
    ) -> bool:
        """判断是否继续迭代"""
        # 检查最大迭代
        if current_iteration >= self.max_iterations:
            return False

        # 检查结果质量
        if self._is_good_result(current_result):
            return False

        # 检查是否陷入循环
        if self._is_looping(current_result, previous_results):
            return False

        return True

    def _is_good_result(self, result: str) -> bool:
        """判断结果质量"""
        # 简化判断
        if "错误" in result or "失败" in result:
            return False
        if len(result) < 50:
            return False
        return True

    def _is_looping(
        self,
        current: str,
        previous: List[str]
    ) -> bool:
        """检测循环"""
        if not previous:
            return False

        # 检查与最近几个结果是否相似
        for prev in previous[-3:]:
            if self._similarity(current, prev) > 0.9:
                return True

        return False

    def _similarity(self, text1: str, text2: str) -> float:
        """计算相似度"""
        words1 = set(text1.lower().split())
        words2 = set(text2.lower().split())

        if not words1 or not words2:
            return 0

        return len(words1 & words2) / max(len(words1), len(words2))
```

### 5.2 提前终止

```python
"""
提前终止机制
"""

class EarlyTermination:
    """提前终止"""

    def __init__(self, llm):
        self.llm = llm

    def check_completion(
        self,
        task: str,
        current_state: dict
    ) -> tuple:
        """检查是否完成"""
        prompt = f"""
任务：{task}
当前状态：{current_state}

任务是否已完成？
- 如果完成，回复 "COMPLETE: 最终答案"
- 如果未完成，回复 "CONTINUE: 下一步"

回复：
"""
        response = self.llm.invoke(prompt).content

        if response.startswith("COMPLETE"):
            answer = response.replace("COMPLETE:", "").strip()
            return True, answer
        else:
            return False, None

    def check_feasibility(
        self,
        task: str,
        attempts: int,
        errors: List[str]
    ) -> tuple:
        """检查可行性"""
        if attempts >= 3 and errors:
            prompt = f"""
任务：{task}
尝试次数：{attempts}
错误：{errors}

基于错误历史，任务是否可行？
回复 "FEASIBLE" 或 "INFEASIBLE: 原因"
"""
            response = self.llm.invoke(prompt).content

            if response.startswith("INFEASIBLE"):
                reason = response.replace("INFEASIBLE:", "").strip()
                return False, reason

        return True, None
```

### 6.1 性能优化器

```python
"""
综合性能优化系统
"""

class PerformanceOptimizer:
    """性能优化器"""

    def __init__(self, llm, tools):
        self.llm = llm
        self.tools = tools

        # 初始化各优化组件
        self.llm_cache = LLMCache()
        self.tool_cache = ToolResultCache()
        self.batch_processor = BatchLLMProcessor(llm)
        self.parallel_executor = ParallelToolExecutor(tools)
        self.analyzer = PerformanceAnalyzer()

    @PerformanceAnalyzer().profile("optimized_agent_run")
    def run_optimized(
        self,
        task: str,
        context: List[str] = None
    ):
        """优化运行"""
        # 1. 检查缓存
        cached = self.llm_cache.get(task)
        if cached:
            return cached

        # 2. 选择相关上下文
        if context:
            selector = SmartContextSelector(self.llm.embeddings)
            relevant = selector.select_relevant_context(task, context)
        else:
            relevant = []

        # 3. 构建提示
        prompt = self._build_prompt(task, relevant)

        # 4. 执行（带缓存）
        response = self.llm_cache.cached_call(self.llm, prompt)

        return response

    def _build_prompt(self, task: str, context: List[str]) -> str:
        """构建提示"""
        context_str = "\n\n".join(context) if context else ""
        return f"""
上下文：
{context_str}

任务：{task}

请完成上述任务。
"""
```
