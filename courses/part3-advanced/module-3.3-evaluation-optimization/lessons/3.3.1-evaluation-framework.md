```python
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class AgentMetrics:
    """Agent评估指标"""

    # 效果指标
    success_rate: float        # 任务成功率
    accuracy: float            # 准确性
    relevance: float           # 相关性
    completeness: float        # 完整性

    # 效率指标
    avg_latency: float         # 平均延迟（秒）
    avg_iterations: float      # 平均迭代次数
    timeout_rate: float        # 超时率

    # 成本指标
    total_tokens: int          # 总Token消耗
    avg_tokens_per_task: float # 每任务平均Token
    total_cost: float          # 总成本

    # 用户指标
    user_satisfaction: float   # 用户满意度
    feedback_score: float      # 反馈评分

    def to_dict(self) -> dict:
        return {
            "effectiveness": {
                "success_rate": self.success_rate,
                "accuracy": self.accuracy,
                "relevance": self.relevance,
                "completeness": self.completeness
            },
            "efficiency": {
                "avg_latency": self.avg_latency,
                "avg_iterations": self.avg_iterations,
                "timeout_rate": self.timeout_rate
            },
            "cost": {
                "total_tokens": self.total_tokens,
                "avg_tokens_per_task": self.avg_tokens_per_task,
                "total_cost": self.total_cost
            },
            "user": {
                "satisfaction": self.user_satisfaction,

        }
```

### 2.2 指标计算器

```python
"""
评估指标计算器
"""

class MetricsCalculator:
    """指标计算器"""

    def calculate_success_rate(
        self,
        results: List[dict]
    ) -> float:
        """计算成功率"""
        if not results:
            return 0.0

        successes = sum(1 for r in results if r.get("success", False))
        return successes / len(results)

    def calculate_accuracy(
        self,
        predictions: List[str],
        ground_truths: List[str],
        llm=None
    ) -> float:
        """计算准确性"""
        if not predictions or not ground_truths:
            return 0.0

        scores = []
        for pred, truth in zip(predictions, ground_truths):
            if llm:
                score = self._llm_evaluate(pred, truth, llm)
            else:
                score = self._exact_match(pred, truth)
            scores.append(score)

        return sum(scores) / len(scores)

    def calculate_latency(
        self,
        durations: List[float]
    ) -> dict:
        """计算延迟统计"""
        if not durations:
            return {"avg": 0, "min": 0, "max": 0, "p95": 0}

        import numpy as np
        durations = sorted(durations)

        return {
            "avg": np.mean(durations),
            "min": min(durations),
            "max": max(durations),
            "p95": durations[int(len(durations) * 0.95)] if len(durations) > 20 else max(durations)
        }

    def calculate_token_usage(
        self,
        usages: List[dict]
    ) -> dict:
        """计算Token使用统计"""
        if not usages:
            return {"total": 0, "avg_input": 0, "avg_output": 0}

        total_input = sum(u.get("input_tokens", 0) for u in usages)
        total_output = sum(u.get("output_tokens", 0) for u in usages)

        return {
            "total": total_input + total_output,
            "avg_input": total_input / len(usages),
            "avg_output": total_output / len(usages)
        }

    def _llm_evaluate(self, prediction: str, truth: str, llm) -> float:
        """使用LLM评估"""
        prompt = f"""
预测答案：{prediction}
标准答案：{truth}

评估预测答案的准确性（0-1分）：
"""
        response = llm.invoke(prompt)
        try:
            return float(response.content.strip())
        except:
            return 0.0

    def _exact_match(self, prediction: str, truth: str) -> float:
        """精确匹配"""
        return 1.0 if prediction.strip() == truth.strip() else 0.0
```

### 3.1 测试用例设计

```python
"""
测试用例设计
"""

from dataclasses import dataclass
from typing import List, Optional

@dataclass
class TestCase:
    """测试用例"""
    id: str
    input: str
    expected_output: Optional[str] = None
    expected_tools: Optional[List[str]] = None
    difficulty: str = "medium"  # easy, medium, hard
    category: str = "general"
    metadata: dict = None

class TestSetBuilder:
    """测试集构建器"""

    def __init__(self):
        self.test_cases = []

    def add_case(
        self,
        input: str,
        expected_output: str = None,
        expected_tools: List[str] = None,
        difficulty: str = "medium",
        category: str = "general"
    ):
        """添加测试用例"""
        case = TestCase(
            id=f"tc_{len(self.test_cases) + 1}",
            input=input,
            expected_output=expected_output,
            expected_tools=expected_tools,
            difficulty=difficulty,
            category=category
        )
        self.test_cases.append(case)
        return self

    def build(self) -> List[TestCase]:
        """构建测试集"""
        return self.test_cases

# 使用示例
builder = TestSetBuilder()

builder.add_case(
    input="北京今天的天气怎么样？",
    expected_tools=["get_weather"],
    difficulty="easy",
    category="tool_usage"
).add_case(
    input="帮我写一首关于春天的诗",
    expected_output="一首完整的诗",
    difficulty="medium",
    category="creative"
).add_case(
    input="分析Python和JavaScript的区别，并给出学习建议",
    expected_output="详细对比分析",
    difficulty="hard",
    category="analysis"
)

test_set = builder.build()
```

```python
"""
测试集管理
"""

import json
from pathlib import Path

class TestSetManager:
    """测试集管理器"""

    def __init__(self, storage_path: str = "./test_sets"):
        self.storage_path = Path(storage_path)
        self.storage_path.mkdir(exist_ok=True)

    def save_test_set(
        self,
        test_set: List[TestCase],
        name: str
    ):
        """保存测试集"""
        data = {
            "name": name,
            "created_at": datetime.now().isoformat(),
            "cases": [
                {
                    "id": c.id,
                    "input": c.input,
                    "expected_output": c.expected_output,
                    "expected_tools": c.expected_tools,
                    "difficulty": c.difficulty,
                    "category": c.category
                }
                for c in test_set
            ]
        }

        file_path = self.storage_path / f"{name}.json"
        with open(file_path, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def load_test_set(self, name: str) -> List[TestCase]:
        """加载测试集"""
        file_path = self.storage_path / f"{name}.json"

        if not file_path.exists():
            return []

        with open(file_path, "r", encoding="utf-8") as f:
            data = json.load(f)

        return [
            TestCase(**c) for c in data["cases"]
        ]

    def list_test_sets(self) -> List[str]:
        """列出所有测试集"""
        return [f.stem for f in self.storage_path.glob("*.json")]
```

## 四、自动化评估

### 4.1 评估Pipeline

```python
"""
自动化评估Pipeline
"""

from typing import Callable

class EvaluationPipeline:
    """评估Pipeline"""

    def __init__(self, agent, test_set: List[TestCase]):
        self.agent = agent
        self.test_set = test_set
        self.results = []
        self.metrics_calculator = MetricsCalculator()

    def run(self, verbose: bool = False) -> dict:
        """运行评估"""
        results = []

        for case in self.test_set:
            if verbose:
                print(f"测试用例: {case.id}")

            # 执行测试
            start_time = time.time()
            try:
                output = self.agent.run(case.input)
                success = True
                error = None
            except Exception as e:
                output = None
                success = False
                error = str(e)

            duration = time.time() - start_time

            # 记录结果
            result = {
                "case_id": case.id,
                "input": case.input,
                "output": output,
                "expected": case.expected_output,
                "success": success,
                "error": error,
                "duration": duration,
                "tools_used": getattr(self.agent, "last_tools_used", [])
            }
            results.append(result)

        # 计算指标
        metrics = self._calculate_metrics(results)

        return {
            "results": results,
            "metrics": metrics
        }

    def _calculate_metrics(self, results: List[dict]) -> dict:
        """计算指标"""
        return {
            "success_rate": self.metrics_calculator.calculate_success_rate(results),
            "latency": self.metrics_calculator.calculate_latency(
                [r["duration"] for r in results]
            ),
            "total_tests": len(results),
            "passed": sum(1 for r in results if r["success"]),
            "failed": sum(1 for r in results if not r["success"])
        }

    def generate_report(self, output_path: str = None) -> str:
        """生成报告"""
        report = []
        report.append("# Agent评估报告")
        report.append(f"\n评估时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append(f"\n## 总体指标\n")

        metrics = self._calculate_metrics(self.results)
        for key, value in metrics.items():
            report.append(f"- {key}: {value}")

        report.append(f"\n## 详细结果\n")
        for result in self.results:
            status = "✅" if result["success"] else "❌"
            report.append(f"- {status} {result['case_id']}: {result['duration']:.2f}s")

        report_text = "\n".join(report)

        if output_path:
            with open(output_path, "w", encoding="utf-8") as f:
                f.write(report_text)

        return report_text
```

### 4.2 LLM辅助评估

```python
"""
LLM辅助评估
"""

class LLMEvaluator:
    """LLM评估器"""

    def __init__(self, llm):
        self.llm = llm

    def evaluate_answer(
        self,
        question: str,
        answer: str,
        ground_truth: str = None
    ) -> dict:
        """评估答案质量"""
        prompt = f"""
问题：{question}
答案：{answer}
{'标准答案：' + ground_truth if ground_truth else ''}

请从以下维度评估答案（1-5分）：
1. 准确性：内容是否正确
2. 相关性：是否直接回答问题
3. 完整性：是否包含必要信息
4. 清晰度：是否易于理解

输出JSON格式：
{{"accuracy": X, "relevance": X, "completeness": X, "clarity": X, "comment": "评语"}}
"""
        response = self.llm.invoke(prompt)
        import json
        return json.loads(response.content)

    def compare_answers(
        self,
        question: str,
        answer_a: str,
        answer_b: str
    ) -> dict:
        """比较两个答案"""
        prompt = f"""
问题：{question}
答案A：{answer_a}
答案B：{answer_b}

比较两个答案的优劣：
- 哪个更准确？
- 哪个更完整？
- 哪个更清晰？

输出JSON：
{{"better": "A或B", "reason": "原因", "scores": {{"A": X, "B": Y}}}}
"""
        response = self.llm.invoke(prompt)
        import json
        return json.loads(response.content)

    def evaluate_agent_behavior(
        self,
        task: str,
        actions: List[dict],
        outcome: str
    ) -> dict:
        """评估Agent行为"""
        actions_text = "\n".join([
            f"- {a.get('action', 'unknown')}: {a.get('input', '')}"
            for a in actions
        ])

        prompt = f"""
任务：{task}
执行动作：
{actions_text}
结果：{outcome}

评估Agent的行为是否合理：
1. 动作选择是否恰当
2. 执行顺序是否合理
3. 是否有更优方案

输出JSON：
{{"score": X, "appropriate_actions": true/false, "improvements": ["建议"]}}
"""
        response = self.llm.invoke(prompt)
        import json
        return json.loads(response.content)
```

### 5.1 报告生成

```python
"""
评估报告生成
"""

class EvaluationReporter:
    """评估报告生成器"""

    def generate(
        self,
        results: dict,
        output_format: str = "markdown"
    ) -> str:
        """生成报告"""
        if output_format == "markdown":
            return self._generate_markdown(results)
        elif output_format == "html":
            return self._generate_html(results)
        elif output_format == "json":
            return self._generate_json(results)
        else:
            return self._generate_markdown(results)

    def _generate_markdown(self, results: dict) -> str:
        """生成Markdown报告"""
        report = []
        report.append("# Agent评估报告\n")

        # 概览
        report.append("## 概览\n")
        metrics = results.get("metrics", {})
        report.append(f"- 总测试数: {metrics.get('total_tests', 0)}")
        report.append(f"- 通过数: {metrics.get('passed', 0)}")
        report.append(f"- 失败数: {metrics.get('failed', 0)}")
        report.append(f"- 成功率: {metrics.get('success_rate', 0):.2%}\n")

        # 详细结果
        report.append("## 详细结果\n")
        for result in results.get("results", []):
            status = "✅" if result["success"] else "❌"
            report.append(f"### {status} {result['case_id']}\n")
            report.append(f"- 输入: {result['input']}")
            report.append(f"- 输出: {result.get('output', 'N/A')}")
            report.append(f"- 耗时: {result.get('duration', 0):.2f}s")
            if result.get("error"):
                report.append(f"- 错误: {result['error']}")
            report.append("")

        return "\n".join(report)

    def _generate_html(self, results: dict) -> str:
        """生成HTML报告"""
        html = """
<!DOCTYPE html>
<html>
<head>
    <title>Agent评估报告</title>
    <style>
        body { font-family: Arial; margin: 20px; }
        .success { color: green; }
        .failure { color: red; }
        table { border-collapse: collapse; width: 100%; }
        th, td { border: 1px solid #ddd; padding: 8px; }
    </style>
</head>
<body>
"""
        # 添加内容...
        html += "</body></html>"
        return html

    def _generate_json(self, results: dict) -> str:
        """生成JSON报告"""
        import json
        return json.dumps(results, ensure_ascii=False, indent=2)
```

### 6.1 评估监控

```python
"""
持续评估监控
"""

class EvaluationMonitor:
    """评估监控器"""

    def __init__(self):
        self.history = []

    def record_evaluation(
        self,
        metrics: dict,
        timestamp: datetime = None
    ):
        """记录评估结果"""
        self.history.append({
            "timestamp": timestamp or datetime.now(),
            "metrics": metrics
        })

    def get_trend(self, metric_name: str) -> List[dict]:
        """获取指标趋势"""
        return [
            {
                "timestamp": h["timestamp"],
                "value": h["metrics"].get(metric_name)
            }
            for h in self.history
        ]

    def detect_regression(
        self,
        metric_name: str,
        threshold: float = 0.1
    ) -> bool:
        """检测性能退化"""
        if len(self.history) < 2:
            return False

        current = self.history[-1]["metrics"].get(metric_name, 0)
        previous = self.history[-2]["metrics"].get(metric_name, 0)

        if previous == 0:
            return False

        change = (previous - current) / previous

        return change > threshold

    def get_summary(self) -> dict:
        """获取摘要"""
        if not self.history:
            return {}

        latest = self.history[-1]["metrics"]
        return {
            "current_metrics": latest,
            "evaluation_count": len(self.history),
            "last_evaluation": self.history[-1]["timestamp"]
        }
```

### 核心要点

1. **评估维度**：效果、效率、成本、用户
2. **评估指标**：成功率、准确性、延迟、成本

- 建立全面的评估体系
