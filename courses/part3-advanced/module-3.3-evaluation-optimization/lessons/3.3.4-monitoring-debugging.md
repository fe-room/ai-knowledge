## 二、执行追踪

### 2.1 执行日志

```python
"""
执行追踪系统
"""

from dataclasses import dataclass, field
from typing import List, Dict, Any
from datetime import datetime
import json

@dataclass
class ExecutionTrace:
    """执行追踪记录"""
    trace_id: str
    task: str
    start_time: datetime
    end_time: datetime = None
    steps: List[dict] = field(default_factory=list)
    result: Any = None
    error: str = None
    metadata: dict = field(default_factory=dict)

class ExecutionTracer:
    """执行追踪器"""

    def __init__(self):
        self.traces: Dict[str, ExecutionTrace] = {}
        self.current_trace: ExecutionTrace = None

    def start_trace(self, task: str) -> str:
        """开始追踪"""
        import uuid
        trace_id = str(uuid.uuid4())[:8]

        self.current_trace = ExecutionTrace(
            trace_id=trace_id,
            task=task,
            start_time=datetime.now()
        )

        self.traces[trace_id] = self.current_trace
        return trace_id

    def log_step(
        self,
        step_name: str,
        action: str,
        input_data: Any = None,
        output_data: Any = None,
        duration: float = None
    ):
        """记录步骤"""
        if not self.current_trace:
            return

        step = {
            "step": step_name,
            "action": action,
            "input": str(input_data)[:500] if input_data else None,
            "output": str(output_data)[:500] if output_data else None,
            "duration": duration,
            "timestamp": datetime.now().isoformat()
        }

        self.current_trace.steps.append(step)

    def end_trace(self, result: Any = None, error: str = None):
        """结束追踪"""
        if not self.current_trace:
            return

        self.current_trace.end_time = datetime.now()
        self.current_trace.result = result
        self.current_trace.error = error

        # 计算总时长
        duration = (
            self.current_trace.end_time - self.current_trace.start_time
        ).total_seconds()

        self.current_trace.metadata["total_duration"] = duration

        self.current_trace = None

    def get_trace(self, trace_id: str) -> ExecutionTrace:
        """获取追踪记录"""
        return self.traces.get(trace_id)

    def export_trace(self, trace_id: str) -> str:
        """导出追踪记录"""
        trace = self.get_trace(trace_id)
        if not trace:
            return ""

        return json.dumps({
            "trace_id": trace.trace_id,
            "task": trace.task,
            "start_time": trace.start_time.isoformat(),
            "end_time": trace.end_time.isoformat() if trace.end_time else None,
            "steps": trace.steps,
            "result": str(trace.result)[:1000] if trace.result else None,
            "error": trace.error,
            "metadata": trace.metadata
        }, ensure_ascii=False, indent=2)
```

### 2.2 追踪装饰器

```python
"""
追踪装饰器
"""

def traced(tracer: ExecutionTracer):
    """追踪装饰器工厂"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            step_name = func.__name__
            start_time = time.time()

            try:
                result = func(*args, **kwargs)
                duration = time.time() - start_time

                tracer.log_step(
                    step_name=step_name,
                    action="function_call",
                    input_data={"args": args[:3], "kwargs": kwargs},
                    output_data=result,
                    duration=duration
                )

                return result

            except Exception as e:
                duration = time.time() - start_time
                tracer.log_step(
                    step_name=step_name,
                    action="function_call",
                    input_data={"args": args[:3], "kwargs": kwargs},
                    output_data={"error": str(e)},
                    duration=duration
                )
                raise

        return wrapper
    return decorator

# 使用
tracer = ExecutionTracer()

@traced(tracer)
def call_llm(prompt):
    return llm.invoke(prompt)

@traced(tracer)
def execute_tool(name, params):
    return tools[name](**params)
```

```python
"""
结构化日志系统
"""

import logging
import json
from datetime import datetime

class StructuredLogger:
    """结构化日志器"""

    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.INFO)

        # 添加处理器
        handler = logging.StreamHandler()
        handler.setFormatter(logging.Formatter('%(message)s'))
        self.logger.addHandler(handler)

    def log(
        self,
        level: str,
        event: str,
        **kwargs
    ):
        """记录日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "level": level,
            "event": event,
            **kwargs
        }

        message = json.dumps(log_entry, ensure_ascii=False)

        if level == "ERROR":
            self.logger.error(message)
        elif level == "WARNING":
            self.logger.warning(message)
        elif level == "DEBUG":
            self.logger.debug(message)
        else:
            self.logger.info(message)

    def info(self, event: str, **kwargs):
        self.log("INFO", event, **kwargs)

    def error(self, event: str, **kwargs):
        self.log("ERROR", event, **kwargs)

    def warning(self, event: str, **kwargs):
        self.log("WARNING", event, **kwargs)

# 使用
logger = StructuredLogger("agent")

logger.info(
    "llm_call",
    model="gpt-4",
    input_tokens=100,
    output_tokens=50,
    duration=1.5
)

logger.error(
    "tool_error",
    tool_name="search",
    error="timeout"
)
```

### 3.2 日志分析

```python
"""
日志分析器
"""

from typing import List, Dict
from collections import Counter

class LogAnalyzer:
    """日志分析器"""

    def __init__(self):
        self.logs = []

    def ingest(self, log_entry: dict):
        """导入日志"""
        self.logs.append(log_entry)

    def analyze_errors(self, time_range: tuple = None) -> Dict:
        """分析错误"""
        errors = [
            log for log in self.logs
            if log.get("level") == "ERROR"
        ]

        # 按错误类型分组
        error_types = Counter(
            log.get("event") for log in errors
        )

        return {
            "total_errors": len(errors),
            "error_types": dict(error_types),
            "error_rate": len(errors) / len(self.logs) if self.logs else 0
        }

    def analyze_performance(self) -> Dict:
        """分析性能"""
        durations = [
            log.get("duration") for log in self.logs
            if "duration" in log
        ]

        if not durations:
            return {}

        import numpy as np

        return {
            "count": len(durations),
            "avg": np.mean(durations),
            "min": min(durations),
            "max": max(durations),
            "p95": np.percentile(durations, 95) if len(durations) > 20 else max(durations)
        }

    def get_event_timeline(self) -> List[Dict]:
        """获取事件时间线"""
        return sorted(
            self.logs,
            key=lambda x: x.get("timestamp", "")
        )
```

```python
"""

class AgentDebugger:
    """Agent调试器"""

    def __init__(self, agent):
        self.agent = agent
        self.breakpoints = set()
        self.watch_variables = {}
        self.trace = []
        self.paused = False

    def set_breakpoint(self, step_name: str):
        """设置断点"""
        self.breakpoints.add(step_name)

    def remove_breakpoint(self, step_name: str):
        """移除断点"""
        self.breakpoints.discard(step_name)

    def watch(self, variable_name: str, value: Any):
        """监视变量"""
        self.watch_variables[variable_name] = {
            "value": value,
            "timestamp": datetime.now()
        }

    def debug_step(
        self,
        step_name: str,
        context: dict
    ):
        """调试步骤"""
        # 记录
        self.trace.append({
            "step": step_name,
            "context": context,
            "watch": dict(self.watch_variables),
            "timestamp": datetime.now()
        })

        # 检查断点
        if step_name in self.breakpoints:
            self.paused = True
            return self._handle_breakpoint(step_name, context)

        return {"action": "continue"}

    def _handle_breakpoint(self, step_name: str, context: dict):
        """处理断点"""
        return {
            "action": "breakpoint",
            "step": step_name,
            "context": context,
            "watch": dict(self.watch_variables)
        }

    def get_trace(self) -> List[dict]:
        """获取追踪"""
        return self.trace

    def export_debug_session(self) -> str:
        """导出调试会话"""
        import json
        return json.dumps({
            "trace": self.trace,
            "watch_variables": {
                k: {"value": str(v["value"]), "timestamp": v["timestamp"].isoformat()}
                for k, v in self.watch_variables.items()
            }
        }, ensure_ascii=False, indent=2)
```

```python
"""
问题诊断器
"""

class ProblemDiagnoser:
    """问题诊断器"""

    def __init__(self):
        self.known_issues = {
            "timeout": {
                "symptoms": ["请求超时", "timeout"],
                "causes": ["网络问题", "服务过载", "请求太大"],
                "solutions": ["重试", "减少请求大小", "使用流式响应"]
            },
            "rate_limit": {
                "symptoms": ["429", "rate limit", "请求过多"],
                "causes": ["超出API限制", "并发过高"],
                "solutions": ["添加延迟", "减少并发", "升级套餐"]
            },
            "token_limit": {
                "symptoms": ["token", "context length", "超出限制"],
                "causes": ["输入太长", "历史太多"],
                "solutions": ["压缩历史", "减少输入", "使用更大上下文模型"]
            }
        }

    def diagnose(self, error_message: str) -> List[dict]:
        """诊断问题"""
        issues = []

        error_lower = error_message.lower()

        for issue_type, info in self.known_issues.items():
            # 检查症状
            for symptom in info["symptoms"]:
                if symptom.lower() in error_lower:
                    issues.append({
                        "type": issue_type,
                        "causes": info["causes"],
                        "solutions": info["solutions"]
                    })
                    break

        return issues

    def suggest_fix(self, error_message: str) -> str:
        """建议修复方案"""
        issues = self.diagnose(error_message)

        if not issues:
            return "未识别的问题类型，请检查错误日志"

        suggestions = []
        for issue in issues:
            suggestions.append(f"问题类型: {issue['type']}")
            suggestions.append(f"可能原因: {', '.join(issue['causes'])}")
            suggestions.append(f"建议方案: {', '.join(issue['solutions'])}")
            suggestions.append("")

        return "\n".join(suggestions)
```

```python
"""
告警规则系统
"""

from dataclasses import dataclass
from typing import Callable, List

@dataclass
class AlertRule:
    """告警规则"""
    name: str
    condition: Callable[[dict], bool]
    severity: str  # info, warning, error, critical
    message: str
    cooldown: int = 300  # 冷却时间（秒）

class AlertSystem:
    """告警系统"""

    def __init__(self):
        self.rules: List[AlertRule] = []
        self.alerts = []
        self.last_alert_time = {}

    def add_rule(self, rule: AlertRule):
        """添加规则"""
        self.rules.append(rule)

    def check(self, metrics: dict) -> List[dict]:
        """检查告警"""
        triggered = []

        for rule in self.rules:
            # 检查冷却
            if rule.name in self.last_alert_time:
                elapsed = time.time() - self.last_alert_time[rule.name]
                if elapsed < rule.cooldown:
                    continue

            # 检查条件
            if rule.condition(metrics):
                alert = {
                    "rule": rule.name,
                    "severity": rule.severity,
                    "message": rule.message,
                    "timestamp": datetime.now()
                }

                triggered.append(alert)
                self.alerts.append(alert)
                self.last_alert_time[rule.name] = time.time()

        return triggered

    def get_recent_alerts(self, hours: int = 24) -> List[dict]:
        """获取最近告警"""
        cutoff = datetime.now() - timedelta(hours=hours)

        return [
            a for a in self.alerts
            if a["timestamp"] > cutoff
        ]

# 预定义规则
def create_default_rules():
    """创建默认规则"""
    rules = [

            name="slow_response",
            condition=lambda m: m.get("avg_latency", 0) > 10,
            severity="warning",
            message="平均响应时间超过10秒"
        ),
        AlertRule(
            name="high_cost",
            condition=lambda m: m.get("daily_cost", 0) > 50,
            severity="warning",
            message="日成本超过$50"

    ]

    return rules
```

```python
"""
监控面板数据
"""

class DashboardDataProvider:
    """监控面板数据提供者"""

    def __init__(
        self,
        tracer: ExecutionTracer,
        logger: StructuredLogger,
        cost_monitor: CostMonitor
    ):
        self.tracer = tracer
        self.logger = logger
        self.cost_monitor = cost_monitor

    def get_summary(self) -> dict:
        """获取摘要"""
        return {
            "total_traces": len(self.tracer.traces),
            "total_logs": len(self.logger.logs),
            "cost_status": self.cost_monitor.check_budget_status(10, 300)
        }

    def get_performance_metrics(self) -> dict:
        """获取性能指标"""
        traces = list(self.tracer.traces.values())

        durations = [
            t.metadata.get("total_duration", 0)
            for t in traces
            if t.metadata.get("total_duration")
        ]

        import numpy as np

        return {
            "total_requests": len(traces),
            "avg_duration": np.mean(durations) if durations else 0,
            "p95_duration": np.percentile(durations, 95) if len(durations) > 20 else 0,
            "success_rate": sum(1 for t in traces if not t.error) / len(traces) if traces else 0
        }

    def get_recent_activity(self, limit: int = 10) -> list:
        """获取最近活动"""
        traces = sorted(
            self.tracer.traces.values(),
            key=lambda t: t.start_time,
            reverse=True
        )[:limit]

        return [
            {
                "trace_id": t.trace_id,
                "task": t.task[:50],
                "duration": t.metadata.get("total_duration"),
                "status": "error" if t.error else "success"
            }
            for t in traces
        ]
```
