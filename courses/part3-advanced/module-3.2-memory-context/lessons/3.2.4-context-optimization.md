# 3.2.4 上下文窗口优化

> 最大化利用有限的上下文窗口

## 一、上下文窗口挑战

### 1.1 Token限制

```
不同模型的上下文窗口：

| 模型 | 上下文窗口 | 说明 |
|-----|-----------|------|
| GPT-3.5-turbo | 16K | 标准 |
| GPT-4 | 8K/32K | 需要选择 |
| GPT-4-turbo | 128K | 超长上下文 |
| Claude 3 | 200K | 最长上下文 |

实际可用空间计算：
总Token = 系统提示词 + 用户输入 + 历史消息 + 响应预留
```

### 1.2 优化目标

```
上下文优化目标：

1. 保留关键信息
   - 不丢失重要内容

2. 提高相关性
   - 上下文与当前任务相关

3. 节省Token
   - 减少不必要的占用

4. 提高效率
   - 减少无效信息处理
```

## 二、内容截断策略

### 2.1 智能截断

```python
"""
智能内容截断
"""

class SmartTruncator:
    """智能截断器"""

    def __init__(self, max_tokens=4000):
        self.max_tokens = max_tokens
        self.tokenizer = tiktoken.encoding_for_model("gpt-4")

    def truncate_text(
        self,
        text: str,
        strategy: str = "middle"
    ) -> str:
        """截断文本"""
        tokens = self.tokenizer.encode(text)

        if len(tokens) <= self.max_tokens:
            return text

        if strategy == "start":
            # 保留开头
            truncated = tokens[:self.max_tokens]
        elif strategy == "end":
            # 保留结尾
            truncated = tokens[-self.max_tokens:]
        elif strategy == "middle":
            # 保留开头和结尾
            half = self.max_tokens // 2
            truncated = tokens[:half] + tokens[-half:]
        elif strategy == "important":
            # 基于重要性
            truncated = self._truncate_by_importance(tokens)
        else:
            truncated = tokens[:self.max_tokens]

        return self.tokenizer.decode(truncated)

    def _truncate_by_importance(self, tokens: list) -> list:
        """按重要性截断"""
        # 分段
        text = self.tokenizer.decode(tokens)
        paragraphs = text.split("\n\n")

        # 评估每段重要性
        scored = []
        for para in paragraphs:
            score = self._score_importance(para)
            scored.append((para, score))

        # 按重要性排序
        scored.sort(key=lambda x: x[1], reverse=True)

        # 选择直到达到限制
        selected = []
        current_tokens = 0

        for para, score in scored:
            para_tokens = len(self.tokenizer.encode(para))
            if current_tokens + para_tokens <= self.max_tokens:
                selected.append(para)
                current_tokens += para_tokens

        return selected

    def _score_importance(self, text: str) -> float:
        """评估重要性"""
        # 简化评分
        score = 0.5

        # 包含关键词加分
        keywords = ["重要", "关键", "必须", "注意"]
        for kw in keywords:
            if kw in text:
                score += 0.1

        # 数字和事实加分
        import re
        if re.search(r'\d+', text):
            score += 0.1

        return min(score, 1.0)
```

### 2.2 分层截断

```python
"""
分层内容截断
"""

class LayeredTruncator:
    """分层截断器"""

    def truncate_messages(
        self,
        messages: List[dict],
        max_tokens: int,
        preserve_system: bool = True
    ) -> List[dict]:
        """分层截断消息"""
        result = []
        remaining_tokens = max_tokens

        # 层级1：系统消息（最高优先级）
        if preserve_system:
            system_msgs = [
                m for m in messages
                if m.get("role") == "system"
            ]
            for msg in system_msgs:
                tokens = self._count_tokens(msg["content"])
                if tokens <= remaining_tokens:
                    result.append(msg)
                    remaining_tokens -= tokens

        # 层级2：最近的用户消息
        user_msgs = [
            m for m in messages
            if m.get("role") == "user"
        ]
        for msg in reversed(user_msgs):
            tokens = self._count_tokens(msg["content"])
            if tokens <= remaining_tokens:
                result.insert(
                    len([m for m in result if m.get("role") == "system"]),
                    msg
                )
                remaining_tokens -= tokens

        # 层级3：最近的助手回复
        assistant_msgs = [
            m for m in messages
            if m.get("role") == "assistant"
        ]
        for msg in reversed(assistant_msgs):
            tokens = self._count_tokens(msg["content"])
            if tokens <= remaining_tokens:
                result.append(msg)
                remaining_tokens -= tokens

        # 层级4：其他消息
        other_msgs = [
            m for m in messages
            if m.get("role") not in ["system", "user", "assistant"]
        ]
        for msg in reversed(other_msgs):
            tokens = self._count_tokens(msg["content"])
            if tokens <= remaining_tokens:
                result.append(msg)
                remaining_tokens -= tokens

        return result

    def _count_tokens(self, text: str) -> int:
        """计算Token"""
        return len(tiktoken.encoding_for_model("gpt-4").encode(text))
```

## 三、上下文压缩

### 3.1 滑动窗口

```python
"""
滑动窗口上下文管理
"""

class SlidingWindowContext:
    """滑动窗口"""

    def __init__(self, window_size=10, overlap=2):
        self.window_size = window_size
        self.overlap = overlap
        self.messages = []

    def add_message(self, message: dict):
        """添加消息"""
        self.messages.append(message)

        # 检查窗口大小
        if len(self.messages) > self.window_size + self.overlap:
            # 移除旧消息，保留重叠部分
            remove_count = len(self.messages) - self.window_size
            self.messages = self.messages[remove_count:]

    def get_context(self) -> List[dict]:
        """获取当前上下文"""
        return self.messages.copy()

    def get_context_with_summary(
        self,
        summary: str = None
    ) -> List[dict]:
        """带摘要的上下文"""
        context = []

        if summary:
            context.append({
                "role": "system",
                "content": f"之前的对话摘要：{summary}"
            })

        context.extend(self.messages)

        return context
```

### 3.2 分块处理

```python
"""
大文档分块处理
"""

class ChunkedContextProcessor:
    """分块上下文处理器"""

    def __init__(self, chunk_size=2000, overlap=200):
        self.chunk_size = chunk_size
        self.overlap = overlap

    def process_large_document(
        self,
        document: str,
        query: str
    ) -> str:
        """处理大文档"""
        # 分块
        chunks = self._chunk_document(document)

        # 找到相关块
        relevant_chunks = self._find_relevant_chunks(chunks, query)

        # 合并相关块
        context = "\n\n".join(relevant_chunks)

        return context

    def _chunk_document(self, document: str) -> List[str]:
        """分块"""
        chunks = []
        start = 0

        while start < len(document):
            end = start + self.chunk_size
            chunk = document[start:end]
            chunks.append(chunk)
            start = end - self.overlap

        return chunks

    def _find_relevant_chunks(
        self,
        chunks: List[str],
        query: str
    ) -> List[str]:
        """找到相关块"""
        # 使用向量相似度
        from langchain_openai import OpenAIEmbeddings
        import numpy as np

        embeddings = OpenAIEmbeddings()

        query_vec = embeddings.embed_query(query)
        chunk_vecs = embeddings.embed_documents(chunks)

        # 计算相似度
        scores = []
        for i, chunk_vec in enumerate(chunk_vecs):
            similarity = np.dot(query_vec, chunk_vec) / (
                np.linalg.norm(query_vec) * np.linalg.norm(chunk_vec)
            )
            scores.append((chunks[i], similarity))

        # 排序并选择
        scores.sort(key=lambda x: x[1], reverse=True)
        top_chunks = [c for c, s in scores[:3]]

        return top_chunks
```

## 四、动态加载

### 4.1 按需加载

```python
"""
动态上下文加载
"""

class DynamicContextLoader:
    """动态加载器"""

    def __init__(self, vector_store, llm):
        self.vector_store = vector_store
        self.llm = llm

    def build_context(
        self,
        query: str,
        max_tokens: int = 3000
    ) -> str:
        """动态构建上下文"""
        context_parts = []
        current_tokens = 0

        # 1. 检索相关文档
        docs = self.vector_store.similarity_search(
            query,
            k=10  # 多检索一些
        )

        # 2. 按相关性添加
        for doc in docs:
            doc_tokens = self._count_tokens(doc.page_content)

            if current_tokens + doc_tokens <= max_tokens:
                context_parts.append(doc.page_content)
                current_tokens += doc_tokens
            else:
                # 尝试截断
                truncated = self._truncate(doc.page_content, max_tokens - current_tokens)
                if truncated:
                    context_parts.append(truncated)
                break

        return "\n\n".join(context_parts)

    def _count_tokens(self, text: str) -> int:
        """计算Token"""
        return len(tiktoken.encoding_for_model("gpt-4").encode(text))

    def _truncate(self, text: str, max_tokens: int) -> str:
        """截断文本"""
        if max_tokens < 100:
            return ""

        tokens = tiktoken.encoding_for_model("gpt-4").encode(text)
        if len(tokens) <= max_tokens:
            return text

        return tiktoken.encoding_for_model("gpt-4").decode(tokens[:max_tokens])
```

### 4.2 渐进式加载

```python
"""
渐进式上下文加载
"""

class ProgressiveContextLoader:
    """渐进式加载器"""

    def __init__(self, llm):
        self.llm = llm

    def load_progressive(
        self,
        query: str,
        data_sources: List[dict],
        max_rounds: int = 3
    ) -> str:
        """渐进式加载"""
        context = ""
        satisfied = False

        for round_num in range(max_rounds):
            # 尝试用当前上下文回答
            if context:
                can_answer = self._try_answer(query, context)
                if can_answer:
                    satisfied = True
                    break

            # 加载更多上下文
            new_context = self._load_more(query, data_sources, round_num)
            context = context + "\n\n" + new_context if context else new_context

        return context

    def _try_answer(self, query: str, context: str) -> bool:
        """尝试回答"""
        prompt = f"""
上下文：
{context[:1000]}

问题：{query}

基于现有上下文是否能回答问题？
回答 "可以" 或 "不可以"：
"""
        response = self.llm.invoke(prompt).content
        return "可以" in response

    def _load_more(
        self,
        query: str,
        sources: List[dict],
        round_num: int
    ) -> str:
        """加载更多内容"""
        # 根据轮次选择不同来源
        if round_num < len(sources):
            source = sources[round_num]
            return source.get("content", "")
        return ""
```

## 五、上下文优先级

### 5.1 优先级管理

```python
"""
上下文优先级管理
"""

class ContextPriorityManager:
    """优先级管理器"""

    def __init__(self):
        self.priority_levels = {
            "critical": 100,    # 关键信息
            "high": 80,         # 重要信息
            "medium": 50,       # 一般信息
            "low": 20,          # 次要信息
            "background": 0     # 背景信息
        }

    def prioritize(
        self,
        items: List[dict],
        max_tokens: int
    ) -> List[dict]:
        """按优先级选择"""
        # 计算优先级分数
        for item in items:
            priority = item.get("priority", "medium")
            recency = item.get("recency", 1.0)
            relevance = item.get("relevance", 1.0)

            base_score = self.priority_levels.get(priority, 50)
            item["score"] = base_score * recency * relevance

        # 排序
        items.sort(key=lambda x: x["score"], reverse=True)

        # 选择
        selected = []
        current_tokens = 0

        for item in items:
            tokens = item.get("tokens", len(item.get("content", "")) // 4)

            if current_tokens + tokens <= max_tokens:
                selected.append(item)
                current_tokens += tokens

        return selected
```

## 六、完整优化示例

### 6.1 上下文优化器

```python
"""
完整上下文优化系统
"""

class ContextOptimizer:
    """上下文优化器"""

    def __init__(self, llm, max_tokens=4000):
        self.llm = llm
        self.max_tokens = max_tokens
        self.truncator = SmartTruncator(max_tokens)
        self.priority_manager = ContextPriorityManager()

    def optimize_for_query(
        self,
        query: str,
        history: List[dict],
        documents: List[str],
        system_prompt: str = None
    ) -> List[dict]:
        """为查询优化上下文"""
        messages = []
        remaining_tokens = self.max_tokens

        # 1. 系统提示词（固定）
        if system_prompt:
            messages.append({
                "role": "system",
                "content": system_prompt
            })
            remaining_tokens -= self._count_tokens(system_prompt)

        # 2. 相关文档
        relevant_docs = self._get_relevant_docs(query, documents)
        doc_context = "\n\n".join(relevant_docs)

        if doc_context:
            # 压缩文档
            compressed = self._compress_if_needed(doc_context, remaining_tokens * 0.5)
            messages.append({
                "role": "system",
                "content": f"参考文档：\n{compressed}"
            })
            remaining_tokens -= self._count_tokens(compressed)

        # 3. 对话历史
        truncated_history = self.truncator.truncate_messages(
            history,
            remaining_tokens * 0.3
        )
        messages.extend(truncated_history)

        # 4. 用户查询
        messages.append({
            "role": "user",
            "content": query
        })

        return messages

    def _get_relevant_docs(
        self,
        query: str,
        documents: List[str]
    ) -> List[str]:
        """获取相关文档"""
        if not documents:
            return []

        # 简化：使用关键词匹配
        relevant = []
        query_words = set(query.lower().split())

        for doc in documents:
            doc_words = set(doc.lower().split())
            overlap = len(query_words & doc_words)

            if overlap > 0:
                relevant.append(doc)

        return relevant[:3]

    def _compress_if_needed(
        self,
        text: str,
        max_tokens: int
    ) -> str:
        """需要时压缩"""
        tokens = self._count_tokens(text)

        if tokens <= max_tokens:
            return text

        # 使用LLM压缩
        prompt = f"""
压缩以下内容，保留关键信息，不超过{max_tokens}个token：

{text}
"""
        return self.llm.invoke(prompt).content

    def _count_tokens(self, text: str) -> int:
        """计算Token"""
        return len(tiktoken.encoding_for_model("gpt-4").encode(text))
```

## 总结

### 核心要点

1. **智能截断**：保留关键内容
2. **分层优先级**：系统 > 用户 > 助手 > 其他
3. **滑动窗口**：固定窗口大小
4. **分块处理**：大文档分块检索
5. **动态加载**：按需加载相关内容

### 最佳实践

- 预留响应空间
- 保持关键信息优先
- 使用压缩节省Token
- 动态加载提高相关性
