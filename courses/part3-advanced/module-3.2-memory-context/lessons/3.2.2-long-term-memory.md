# 3.2.2 长期记忆管理

> 构建持久化的Agent记忆系统

## 一、长期记忆概念

### 1.1 什么是长期记忆？

**长期记忆** 是Agent在多个会话间持久保持的知识和信息。

```
长期记忆 vs 短期记忆：

短期记忆：
- 当前会话有效
- 对话历史
- 临时状态

长期记忆：
- 跨会话持久
- 用户偏好
- 知识库
- 历史交互记录
```

### 1.2 长期记忆类型

```
长期记忆类型：

1. 情景记忆（Episodic）
   - 历史事件记录
   - 对话存档
   - 任务执行记录

2. 语义记忆（Semantic）
   - 知识库
   - 概念关系
   - 规则和事实

3. 程序记忆（Procedural）
   - 技能和流程
   - 操作模式
   - 最佳实践
```

## 二、向量存储记忆

### 2.1 基础实现

```python
"""
基于向量存储的长期记忆
"""

from langchain.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain.memory import VectorStoreRetrieverMemory
from datetime import datetime

class VectorMemory:
    """向量记忆系统"""

    def __init__(self, persist_directory="./memory"):
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            persist_directory=persist_directory,
            embedding_function=self.embeddings
        )

    def save(self, content: str, metadata: dict = None):
        """保存记忆"""
        if metadata is None:
            metadata = {}

        metadata["timestamp"] = datetime.now().isoformat()

        self.vectorstore.add_texts(
            texts=[content],
            metadatas=[metadata]
        )

    def recall(self, query: str, k: int = 5) -> list:
        """检索记忆"""
        results = self.vectorstore.similarity_search(
            query,
            k=k
        )
        return results

    def recall_with_scores(self, query: str, k: int = 5):
        """带分数的检索"""
        return self.vectorstore.similarity_search_with_score(
            query,
            k=k
        )

# 使用
memory = VectorMemory()

# 保存记忆
memory.save(
    "用户喜欢Python编程，主要做数据分析工作",
    {"type": "preference", "user_id": "user_123"}
)

# 检索记忆
results = memory.recall("用户的工作是什么？")
print(results[0].page_content)
```

```python
"""
记忆管理器

from typing import List, Dict, Any
from dataclasses import dataclass
from datetime import datetime

@dataclass
class MemoryItem:
    content: str
    memory_type: str
    importance: float
    metadata: Dict
    created_at: datetime

class LongTermMemoryManager:
    """长期记忆管理器"""

    def __init__(self, vector_store):
        self.vector_store = vector_store
        self.memories = []

    def remember(
        self,
        content: str,
        memory_type: str = "general",
        importance: float = 0.5,
        metadata: dict = None
    ):
        """记住信息"""
        item = MemoryItem(
            content=content,
            memory_type=memory_type,
            importance=importance,
            metadata=metadata or {},
            created_at=datetime.now()
        )

        # 保存到向量存储
        self.vector_store.add_texts(
            texts=[content],
            metadatas=[{
                "type": memory_type,
                "importance": importance,
                "timestamp": item.created_at.isoformat(),
                **(metadata or {})
            }]
        )

        self.memories.append(item)

    def recall_relevant(
        self,
        query: str,
        memory_type: str = None,
        min_importance: float = 0.0,
        k: int = 5
    ) -> List[MemoryItem]:
        """回忆相关信息"""
        # 检索
        filter_dict = {}
        if memory_type:
            filter_dict["type"] = memory_type

        results = self.vector_store.similarity_search(
            query,
            k=k * 2,  # 多取一些，再过滤
```

## 三、知识图谱记忆

### 3.1 知识图谱存储

```python
"""
基于知识图谱的记忆
"""

class KnowledgeGraphMemory:
    """知识图谱记忆"""

    def __init__(self):
        self.nodes = {}  # 实体节点
        self.edges = []  # 关系边

    def add_entity(self, entity_id: str, entity_type: str, properties: dict):
        """添加实体"""
        self.nodes[entity_id] = {
            "id": entity_id,
            "type": entity_type,
            "properties": properties
        }

    def add_relation(
        self,
        source: str,
        relation: str,
        target: str,
        properties: dict = None
    ):
        """添加关系"""
        self.edges.append({
            "source": source,
            "relation": relation,
            "target": target,
            "properties": properties or {}
        })

    def query(self, entity_id: str) -> dict:
        """查询实体及其关系"""
        entity = self.nodes.get(entity_id)

        # 查询相关关系
        relations = []
        for edge in self.edges:
            if edge["source"] == entity_id or edge["target"] == entity_id:
                relations.append(edge)

        return {
            "entity": entity,
            "relations": relations
        }

    def find_path(self, from_id: str, to_id: str) -> list:
        """查找路径"""
        # BFS查找路径
        from collections import deque

        queue = deque([(from_id, [from_id])])
        visited = {from_id}

        while queue:
            current, path = queue.popleft()

            if current == to_id:
                return path

            # 找相邻节点
            for edge in self.edges:
                next_node = None
                if edge["source"] == current and edge["target"] not in visited:
                    next_node = edge["target"]
                elif edge["target"] == current and edge["source"] not in visited:
                    next_node = edge["source"]

                if next_node:
                    visited.add(next_node)
                    queue.append((next_node, path + [next_node]))

        return []

# 使用
kg_memory = KnowledgeGraphMemory()

# 添加实体
kg_memory.add_entity("user_123", "User", {"name": "小明", "age": 25})
kg_memory.add_entity("python", "ProgrammingLanguage", {"name": "Python"})
kg_memory.add_entity("ml", "Field", {"name": "机器学习"})

# 添加关系
kg_memory.add_relation("user_123", "knows", "python")
kg_memory.add_relation("user_123", "interested_in", "ml")
kg_memory.add_relation("python", "used_in", "ml")

# 查询
result = kg_memory.query("user_123")
print(result)
```

### 4.1 混合检索

```python
"""
混合记忆检索
"""

class HybridMemoryRetriever:
    """混合检索器"""

    def __init__(self, vector_memory, kg_memory):
        self.vector_memory = vector_memory
        self.kg_memory = kg_memory

    def retrieve(self, query: str, k: int = 5) -> list:
        """检索相关记忆"""
        results = []

        # 1. 向量检索
        vector_results = self.vector_memory.recall(query, k=k)
        for doc in vector_results:
            results.append({
                "content": doc.page_content,
                "source": "vector",
                "score": 1.0  # 简化
            })

        # 2. 知识图谱检索
        # 提取查询中的实体
        entities = self._extract_entities(query)
        for entity in entities:
            kg_result = self.kg_memory.query(entity)
            if kg_result["entity"]:
                results.append({
                    "content": str(kg_result),
                    "source": "knowledge_graph",
                    "score": 0.8
                })

        # 排序和去重
        results.sort(key=lambda x: x["score"], reverse=True)

        return results[:k]

    def _extract_entities(self, query: str) -> list:
        """提取查询中的实体"""
        # 简化实现
        entities = []
        for node_id in self.kg_memory.nodes:
            if node_id.lower() in query.lower():
                entities.append(node_id)
        return entities
```

### 4.2 时间衰减

```python
"""
时间衰减记忆
"""

from datetime import datetime, timedelta

class TimeDecayMemory:
    """时间衰减记忆"""

    def __init__(self, decay_rate=0.1):
        self.decay_rate = decay_rate
        self.memories = []

    def save(self, content: str, importance: float = 0.5):
        """保存记忆"""
        self.memories.append({
            "content": content,
            "importance": importance,
            "timestamp": datetime.now()
        })

    def recall(self, query: str, k: int = 5) -> list:
        """检索（考虑时间衰减）"""
        # 计算每个记忆的有效分数
        scored_memories = []
        for mem in self.memories:
            # 时间衰减
            age = (datetime.now() - mem["timestamp"]).total_seconds() / 3600  # 小时
            decay = 1 / (1 + self.decay_rate * age)

            # 有效重要性
            effective_importance = mem["importance"] * decay

            scored_memories.append({
                **mem,
                "effective_importance": effective_importance
            })

        # 按有效重要性排序
        scored_memories.sort(
            key=lambda x: x["effective_importance"],
            reverse=True
        )

        return scored_memories[:k]

    def cleanup(self, threshold: float = 0.1):
        """清理低效记忆"""
        self.memories = [
            m for m in self.memories
            if self._calculate_effective_importance(m) > threshold
        ]

    def _calculate_effective_importance(self, memory: dict) -> float:
        """计算有效重要性"""
        age = (datetime.now() - memory["timestamp"]).total_seconds() / 3600
        return memory["importance"] / (1 + self.decay_rate * age)
```

## 五、持久化存储

### 5.1 数据库存储

```python
"""
数据库持久化
"""

import sqlite3
import json

class DatabaseMemoryStore:
    """数据库记忆存储"""

    def __init__(self, db_path="memory.db"):
        self.conn = sqlite3.connect(db_path)
        self._create_tables()

    def _create_tables(self):
        """创建表"""
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS memories (
                id INTEGER PRIMARY KEY,
                user_id TEXT,
                content TEXT,
                memory_type TEXT,
                importance REAL,
                metadata TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.conn.commit()

    def save(
        self,
        user_id: str,
        content: str,
        memory_type: str,
        importance: float = 0.5,
        metadata: dict = None
    ):
        """保存记忆"""
        self.conn.execute(
            """
            INSERT INTO memories (user_id, content, memory_type, importance, metadata)
            VALUES (?, ?, ?, ?, ?)
            """,
            (user_id, content, memory_type, importance, json.dumps(metadata or {}))
        )
        self.conn.commit()

    def load(self, user_id: str, limit: int = 100) -> list:
        """加载用户记忆"""
        cursor = self.conn.execute(
            """
            SELECT content, memory_type, importance, metadata, created_at
            FROM memories
            WHERE user_id = ?
            ORDER BY created_at DESC
            LIMIT ?
            """,
            (user_id, limit)
        )

        return [
            {
                "content": row[0],
                "type": row[1],
                "importance": row[2],
                "metadata": json.loads(row[3]),
                "created_at": row[4]
            }
            for row in cursor.fetchall()
        ]

    def search(self, user_id: str, query: str, limit: int = 10) -> list:
        """搜索记忆"""
        cursor = self.conn.execute(
            """
            SELECT content, memory_type, importance, metadata
            FROM memories
            WHERE user_id = ? AND content LIKE ?
            ORDER BY importance DESC
            LIMIT ?
            """,
            (user_id, f"%{query}%", limit)
        )

        return cursor.fetchall()

    def close(self):
        """关闭连接"""
        self.conn.close()
```

```python
"""
用户画像长期记忆
"""

class UserProfileMemory:
    """用户画像记忆系统"""

    def __init__(self, db_store, vector_store):
        self.db = db_store
        self.vector = vector_store

    def learn_about_user(
        self,
        user_id: str,
        interaction: str
    ):
        """从交互中学习用户信息"""
        # 提取信息
        info = self._extract_user_info(interaction)

        for item in info:
            # 保存到数据库
            self.db.save(
                user_id=user_id,
                content=item["content"],
                memory_type=item["type"],
                importance=item.get("importance", 0.5)
            )

            # 保存到向量存储
            self.vector.save(
                item["content"],
                {"user_id": user_id, "type": item["type"]}
            )

    def get_user_context(self, user_id: str) -> str:
        """获取用户上下文"""
        # 加载用户记忆
        memories = self.db.load(user_id, limit=20)

        # 构建上下文
        context_parts = []

        # 按类型分组
        by_type = {}
        for mem in memories:
            mem_type = mem["type"]
            if mem_type not in by_type:
                by_type[mem_type] = []
            by_type[mem_type].append(mem["content"])

        # 格式化
        type_names = {
            "preference": "偏好",
            "interest": "兴趣",
            "fact": "事实",
            "goal": "目标"
        }

        for mem_type, contents in by_type.items():
            type_name = type_names.get(mem_type, mem_type)
            context_parts.append(
                f"用户{type_name}: " + ", ".join(contents[:3])
            )

        return "\n".join(context_parts)

    def _extract_user_info(self, interaction: str) -> list:
        """从交互中提取用户信息"""
        prompt = f"""
从用户交互中提取用户信息：

交互内容：{interaction}

提取的信息（JSON列表）：
[
    {{"type": "preference/interest/fact/goal", "content": "xxx", "importance": 0.0-1.0}}
]
"""
        response = llm.invoke(prompt)
        import json
        return json.loads(response.content)
```

## 总结

### 核心要点

1. **长期记忆类型**：情景、语义、程序
2. **向量存储**：相似度检索记忆
3. **知识图谱**：实体关系存储
4. **混合检索**：结合多种检索方式
5. **持久化**：数据库存储跨会话记忆

### 最佳实践

- 选择合适的存储方式
- 实现记忆重要性评估
- 定期清理过期记忆
- 保护用户隐私数据
