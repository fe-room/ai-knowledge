# 3.2.1 短期记忆管理

> 管理Agent的对话历史和临时信息

## 一、短期记忆概念

### 1.1 什么是短期记忆？

**短期记忆（Working Memory）** 是Agent在当前会话中保持的临时信息。

```
短期记忆内容：

1. 对话历史
   - 用户消息
   - Agent回复
   - 系统消息

2. 当前状态
   - 任务进度
   - 临时变量
   - 上下文信息

3. 工作空间
   - 中间结果
   - 待处理数据
```

### 1.2 挑战

```
短期记忆挑战：

1. Token限制
   - 模型上下文窗口有限
   - 历史消息占用大量Token

2. 信息衰减
   - 早期信息可能丢失
   - 需要策略保留关键信息

3. 效率问题
   - 过多历史影响响应速度
   - 需要平衡完整性和效率
```

```python
LangChain对话记忆
"""

from langchain.memory import ConversationBufferMemory

# 查看记忆
print(memory.load_memory_variables({}))
```

```python
"""
不同类型的对话记忆
"""

from langchain.memory import (
    ConversationBufferMemory,
    ConversationBufferWindowMemory,
    ConversationSummaryMemory,
    ConversationSummaryBufferMemory

# 1. 缓冲记忆 - 保存所有历史
buffer_memory = ConversationBufferMemory(

# 缺点：Token消耗大

    k=5,  # 保留最近5轮对话
    return_messages=True
)
# 优点：Token消耗可控
# 缺点：早期信息丢失

# 3. 摘要记忆 - 压缩历史
summary_memory = ConversationSummaryMemory(
    llm=ChatOpenAI(),
    return_messages=True
)
# 优点：保留关键信息，节省Token
# 缺点：可能丢失细节

# 4. 混合记忆 - 窗口+摘要
hybrid_memory = ConversationSummaryBufferMemory(
    llm=ChatOpenAI(),
    max_token_limit=1000,
    return_messages=True
)
# 优点：平衡完整性和效率
# 缺点：配置复杂
```

### 2.3 自定义记忆

```python
"""
自定义记忆实现
"""

from typing import List, Dict
from dataclasses import dataclass, field

@dataclass
class Message:
    role: str
    content: str
    timestamp: float = field(default_factory=time.time)

class CustomMemory:
    """自定义记忆"""

    def __init__(self, max_messages=50):
        self.messages: List[Message] = []
        self.max_messages = max_messages
        self.important_facts = []

    def add_message(self, role: str, content: str):
        """添加消息"""
        self.messages.append(Message(role=role, content=content))

        # 超出限制时清理
        if len(self.messages) > self.max_messages:
            self._cleanup()

    def get_history(self, limit=None) -> List[Dict]:
        """获取历史"""
        messages = self.messages
        if limit:
            messages = messages[-limit:]

        return [
            {"role": m.role, "content": m.content}
            for m in messages
        ]

    def add_important_fact(self, fact: str):
        """记录重要事实"""
        self.important_facts.append(fact)

    def get_important_facts(self) -> List[str]:
        """获取重要事实"""
        return self.important_facts

    def _cleanup(self):
        """清理旧消息"""
        # 保留最近的消息
        self.messages = self.messages[-self.max_messages:]

    def summarize(self) -> str:
        """生成摘要"""
        # 使用LLM总结
        history = self.get_history()
        summary = llm.invoke(f"总结对话：{history}")
        return summary.content

    def clear(self):
        """清除记忆"""
        self.messages = []
        # 可选择保留重要事实
```

## 三、Token管理

### 3.1 Token计数

```python
"""
Token计数和管理
"""

import tiktoken

class TokenCounter:
    """Token计数器"""

    def __init__(self, model="gpt-4"):
        self.encoding = tiktoken.encoding_for_model(model)

    def count_tokens(self, text: str) -> int:
        """计算Token数量"""
        return len(self.encoding.encode(text))

    def count_messages(self, messages: List[dict]) -> int:
        """计算消息列表Token"""
        total = 0
        for msg in messages:
            total += self.count_tokens(msg.get("content", ""))
            total += 4  # 消息格式开销
        return total

# 使用
counter = TokenCounter()
tokens = counter.count_tokens("你好，这是一个测试")
print(f"Token数量: {tokens}")
```

### 3.2 Token限制处理

```python
"""
Token限制处理策略
"""

class TokenLimiter:
    """Token限制器"""

    def __init__(self, max_tokens=4000):
        self.max_tokens = max_tokens
        self.counter = TokenCounter()

    def truncate_history(
        self,
        messages: List[dict],
        preserve_system: bool = True
    ) -> List[dict]:
        """截断历史以符合Token限制"""
        result = []

        # 保留系统消息
        if preserve_system:
            system_msgs = [
                m for m in messages
                if m.get("role") == "system"
            ]
            result.extend(system_msgs)

        # 添加消息直到达到限制
        current_tokens = self.counter.count_messages(result)
        user_msgs = [
            m for m in messages
            if m.get("role") != "system"
        ]

        for msg in reversed(user_msgs):  # 从最新开始
            msg_tokens = self.counter.count_messages([msg])
            if current_tokens + msg_tokens < self.max_tokens - 500:  # 留出响应空间
                result.insert(len(system_msgs) if preserve_system else 0, msg)
                current_tokens += msg_tokens
            else:
                break

        return result

    def compress_history(self, messages: List[dict]) -> List[dict]:
        """压缩历史"""
        # 保留重要信息
        # 使用LLM生成摘要
        old_messages = messages[:-10]  # 除了最近10条
        recent_messages = messages[-10:]

        # 压缩旧消息
        summary = self._summarize_messages(old_messages)

        return [
            {"role": "system", "content": f"历史摘要: {summary}"}
        ] + recent_messages

    def _summarize_messages(self, messages: List[dict]) -> str:
        """总结消息"""
        content = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages
        ])
        # 使用LLM总结
        return llm.invoke(f"简要总结：{content}").content
```

## 四、消息历史管理

### 4.1 消息优先级

```python
"""
消息优先级管理
"""

class PriorityMessageManager:
    """优先级消息管理"""

    def __init__(self):
        self.messages = []
        self.priorities = {
            "system": 10,    # 最高优先级
            "important": 8,
            "user": 5,
            "assistant": 3,
            "normal": 1
        }

    def add_message(
        self,
        role: str,
        content: str,
        priority: str = None
    ):
        """添加消息"""
        if priority is None:
            priority = role

        self.messages.append({
            "role": role,
            "content": content,
            "priority": self.priorities.get(priority, 1),
            "timestamp": time.time()
        })

    def get_messages(
        self,
        max_tokens=4000,
        preserve_recent=5
    ) -> List[dict]:
        """获取消息（按优先级和最近性）"""
        # 分离最近消息和旧消息
        recent = self.messages[-preserve_recent:]
        older = self.messages[:-preserve_recent]

        # 计算Token
        counter = TokenCounter()
        available_tokens = max_tokens - counter.count_messages(recent)

        # 按优先级选择旧消息
        older_sorted = sorted(
            older,
            key=lambda x: (x["priority"], x["timestamp"]),
            reverse=True
        )

        selected = []
        for msg in older_sorted:
            msg_tokens = counter.count_messages([msg])
            if available_tokens >= msg_tokens:
                selected.append(msg)
                available_tokens -= msg_tokens

        # 合并并按时间排序
        result = sorted(
            selected + recent,
            key=lambda x: x["timestamp"]
        )

        return [
            {"role": m["role"], "content": m["content"]}
            for m in result
        ]
```

```python
"""
提取和保留关键信息
"""

class KeyInformationExtractor:
    """关键信息提取器"""

    def __init__(self, llm):
        self.llm = llm

    def extract(self, messages: List[dict]) -> List[str]:
        """提取关键信息"""
        history = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages
        ])

        prompt = f"""
从以下对话中提取关键信息：
- 用户偏好
- 重要事实
- 任务要求
- 未完成事项

对话内容：
{history}

关键信息（每行一条）：
"""
        response = self.llm.invoke(prompt)
        return response.content.strip().split("\n")

    def create_summary_context(
        self,
        key_info: List[str]
    ) -> str:
        """创建摘要上下文"""
        return "关键信息:\n" + "\n".join([
            f"- {info}"
            for info in key_info
        ])

# 使用
extractor = KeyInformationExtractor(llm)
key_info = extractor.extract(messages)
print(key_info)
```

## 五、实现示例

### 5.1 完整的记忆管理器

```python
"""
完整的短期记忆管理器
"""

class ShortTermMemoryManager:
    """短期记忆管理器"""

    def __init__(
        self,
        max_tokens=4000,
        preserve_recent=5,
        llm=None
    ):
        self.max_tokens = max_tokens
        self.preserve_recent = preserve_recent
        self.llm = llm or ChatOpenAI()

        self.messages = []
        self.key_info = []
        self.counter = TokenCounter()

    def add_user_message(self, content: str):
        """添加用户消息"""
        self.messages.append({
            "role": "user",
            "content": content
        })
        self._check_and_compress()

    def add_assistant_message(self, content: str):
        """添加助手消息"""
        self.messages.append({
            "role": "assistant",
            "content": content
        })

    def get_context(self) -> List[dict]:
        """获取上下文"""
        # 检查Token
        current_tokens = self.counter.count_messages(self.messages)

        if current_tokens <= self.max_tokens:
            return self.messages

        # 需要压缩
        return self._compress()

    def _check_and_compress(self):
        """检查并压缩"""
        current_tokens = self.counter.count_messages(self.messages)

        if current_tokens > self.max_tokens * 0.8:
            # 提取关键信息
            self._extract_key_info()
            # 压缩历史
            self.messages = self._compress()

    def _extract_key_info(self):
        """提取关键信息"""
        extractor = KeyInformationExtractor(self.llm)
        new_info = extractor.extract(self.messages)
        self.key_info.extend(new_info)
        # 去重
        self.key_info = list(set(self.key_info))

    def _compress(self) -> List[dict]:
        """压缩消息"""
        # 保留最近消息
        recent = self.messages[-self.preserve_recent:]

        # 生成摘要
        older = self.messages[:-self.preserve_recent]
        if older:
            summary = self._summarize(older)
            return [
                {"role": "system", "content": f"对话历史摘要: {summary}"},
                {"role": "system", "content": f"关键信息: {', '.join(self.key_info)}"}
            ] + recent

        return recent

    def _summarize(self, messages: List[dict]) -> str:
        """生成摘要"""
        content = "\n".join([
            f"{m['role']}: {m['content'][:200]}"
            for m in messages
        ])
        return self.llm.invoke(
            f"简要总结：{content}"
        ).content

    def clear(self, keep_key_info=True):
        """清除记忆"""
        self.messages = []
        if not keep_key_info:
            self.key_info = []
```

## 总结

### 核心要点

1. **记忆类型**：缓冲、窗口、摘要、混合
2. **Token管理**：计数、限制、截断
3. **优先级管理**：保留重要消息
4. **关键信息提取**：压缩同时保留关键内容

### 最佳实践

- 根据场景选择记忆类型
- 设置合理的Token限制
- 提取关键信息作为摘要
- 定期清理过期内容
