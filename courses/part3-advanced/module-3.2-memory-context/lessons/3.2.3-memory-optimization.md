# 3.2.3 记忆优化策略

> 压缩、提取和遗忘机制的实现

## 一、记忆优化概述

### 1.1 为什么需要优化？

```
记忆问题：

1. 容量问题
   - 存储空间有限
   - Token限制

2. 检索效率
   - 记忆过多影响检索速度
   - 噪声信息干扰

3. 信息质量
   - 过时信息
   - 冗余信息
   - 错误信息
```

### 1.2 优化策略

```
记忆优化策略：

1. 压缩（Compression）
   - 摘要压缩
   - 关键信息提取

2. 更新（Update）
   - 信息修正
   - 重要性调整

3. 遗忘（Forgetting）
   - 时间衰减
   - 相关性过滤
```

## 二、记忆压缩

### 2.1 摘要压缩

```python
"""
记忆摘要压缩
"""

from langchain_openai import ChatOpenAI
from typing import List

class MemoryCompressor:
    """记忆压缩器"""

    def __init__(self, llm=None):
        self.llm = llm or ChatOpenAI()

    def compress_messages(
        self,
        messages: List[dict],
        target_length: int = 500
    ) -> str:
        """压缩消息列表"""
        # 合并消息
        text = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages
        ])

        # 生成摘要
        prompt = f"""
请将以下对话压缩为简洁的摘要，保留关键信息：

{text}

摘要（不超过{target_length}字）：
"""
        summary = self.llm.invoke(prompt).content

        return summary

    def compress_with_structure(
        self,
        messages: List[dict]
    ) -> dict:
        """结构化压缩"""
        text = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages
        ])

        prompt = f"""
分析对话并提取结构化信息：

{text}

输出JSON：
{{
    "topics": ["主题1", "主题2"],
    "key_facts": ["事实1", "事实2"],
    "decisions": ["决定1"],
    "action_items": ["待办1"],
    "summary": "简要摘要"
}}
"""
        response = self.llm.invoke(prompt)
        import json
        return json.loads(response.content)

# 使用
compressor = MemoryCompressor()

messages = [
    {"role": "user", "content": "我想学习Python"},
    {"role": "assistant", "content": "好的，Python是一门很好的编程语言..."},
    # ... 更多消息
]

summary = compressor.compress_messages(messages)
print(summary)

structured = compressor.compress_with_structure(messages)
print(structured)
```

### 2.2 增量摘要

```python
"""
增量摘要更新
"""

class IncrementalSummarizer:
    """增量摘要器"""

    def __init__(self, llm):
        self.llm = llm
        self.current_summary = ""

    def update(self, new_messages: List[dict]):
        """增量更新摘要"""
        if not self.current_summary:
            # 首次生成摘要
            self.current_summary = self._generate_summary(new_messages)
        else:
            # 增量更新
            new_text = "\n".join([
                f"{m['role']}: {m['content']}"
                for m in new_messages
            ])

            prompt = f"""
当前摘要：
{self.current_summary}

新对话内容：
{new_text}

请更新摘要，整合新信息：
"""
            self.current_summary = self.llm.invoke(prompt).content

        return self.current_summary

    def _generate_summary(self, messages: List[dict]) -> str:
        """生成初始摘要"""
        text = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in messages
        ])
        return self.llm.invoke(f"总结对话：{text}").content

    def get_summary(self) -> str:
        """获取当前摘要"""
        return self.current_summary
```

## 三、重要信息提取

### 3.1 信息重要性评估

```python
"""
评估信息重要性
"""

class ImportanceEvaluator:
    """重要性评估器"""

    def __init__(self, llm):
        self.llm = llm

    def evaluate(self, content: str, context: str = "") -> float:
        """评估重要性（0-1）"""
        prompt = f"""
评估以下信息的重要性（对用户长期记忆的价值）：

信息：{content}
上下文：{context}

重要性评分（0-1，保留两位小数）：
- 0.0-0.3: 临时信息，无长期价值
- 0.3-0.6: 一般信息，可能有用
- 0.6-0.8: 重要信息，值得记住
- 0.8-1.0: 关键信息，必须记住

只输出分数：
"""
        response = self.llm.invoke(prompt)
        try:
            return float(response.content.strip())
        except:
            return 0.5

    def batch_evaluate(
        self,
        items: List[str],
        context: str = ""
    ) -> List[float]:
        """批量评估"""
        return [
            self.evaluate(item, context)
            for item in items
        ]

# 使用
evaluator = ImportanceEvaluator(llm)
importance = evaluator.evaluate("用户喜欢使用Python进行数据分析")
print(f"重要性: {importance}")
```

### 3.2 关键信息提取

```python
"""
提取关键信息
"""

class KeyInfoExtractor:
    """关键信息提取器"""

    def __init__(self, llm):
        self.llm = llm

    def extract(self, text: str) -> dict:
        """提取关键信息"""
        prompt = f"""
从以下文本中提取关键信息：

{text}

提取以下类型的信息（JSON格式）：
{{
    "user_preferences": ["偏好1", "偏好2"],
    "user_facts": ["事实1", "事实2"],
    "user_goals": ["目标1"],
    "important_events": ["事件1"],
    "constraints": ["约束1"]
}}
"""
        response = self.llm.invoke(prompt)
        import json
        return json.loads(response.content)

    def extract_and_prioritize(
        self,
        text: str,
        evaluator: ImportanceEvaluator
    ) -> List[dict]:
        """提取并按重要性排序"""
        key_info = self.extract(text)

        prioritized = []
        for info_type, items in key_info.items():
            for item in items:
                importance = evaluator.evaluate(item, text)
                prioritized.append({
                    "type": info_type,
                    "content": item,
                    "importance": importance
                })

        # 按重要性排序
        prioritized.sort(key=lambda x: x["importance"], reverse=True)

        return prioritized
```

## 四、遗忘机制

### 4.1 时间衰减遗忘

```python
"""
时间衰减遗忘
"""

from datetime import datetime, timedelta

class TimeBasedForgetting:
    """时间衰减遗忘"""

    def __init__(self, half_life_days=30):
        """
        half_life_days: 半衰期（天）
        """
        self.half_life = half_life_days
        self.decay_rate = 0.693 / half_life_days  # ln(2) / half_life

    def calculate_retention(
        self,
        created_at: datetime,
        importance: float = 0.5
    ) -> float:
        """计算保留概率"""
        age_days = (datetime.now() - created_at).days
        retention = importance * (0.5 ** (age_days / self.half_life))
        return retention

    def should_forget(
        self,
        memory: dict,
        threshold: float = 0.1
    ) -> bool:
        """判断是否应该遗忘"""
        retention = self.calculate_retention(
            memory["created_at"],
            memory.get("importance", 0.5)
        )
        return retention < threshold

    def apply_forgetting(
        self,
        memories: List[dict],
        threshold: float = 0.1
    ) -> List[dict]:
        """应用遗忘机制"""
        retained = []

        for mem in memories:
            if not self.should_forget(mem, threshold):
                # 更新重要性
                mem["effective_importance"] = self.calculate_retention(
                    mem["created_at"],
                    mem.get("importance", 0.5)
                )
                retained.append(mem)

        return retained
```

```python
"""
主动遗忘策略
"""

class ActiveForgetting:
    """主动遗忘"""

    def __init__(self, llm):
        self.llm = llm

    def identify_obsolete(
        self,
        memories: List[dict],
        current_context: str
    ) -> List[dict]:
        """识别过时的记忆"""
        prompt = f"""
当前上下文：
{current_context}

记忆列表：
{chr(10).join([f"{i}. {m['content']}" for i, m in enumerate(memories)])}

请识别哪些记忆已经过时或不再相关。
输出过时记忆的编号列表（JSON）：
{{"obsolete": [编号1, 编号2, ...]}}
"""
        response = self.llm.invoke(prompt)
        import json
        result = json.loads(response.content)

        obsolete_indices = result["obsolete"]
        return [memories[i] for i in obsolete_indices if i < len(memories)]

    def identify_redundant(
        self,
        memories: List[dict]
    ) -> List[dict]:
        """识别冗余记忆"""
        redundant = []
        seen_content = set()

        for mem in memories:
            content_lower = mem["content"].lower()

            # 检查是否重复
            if content_lower in seen_content:
                redundant.append(mem)
            else:
                # 检查相似内容
                for seen in seen_content:
                    if self._is_similar(content_lower, seen):
                        redundant.append(mem)
                        break

            seen_content.add(content_lower)

        return redundant

    def _is_similar(self, text1: str, text2: str, threshold: float = 0.8) -> bool:
        """判断相似性"""
        # 简化实现
        words1 = set(text1.split())
        words2 = set(text2.split())
        if not words1 or not words2:
            return False
        overlap = len(words1 & words2) / max(len(words1), len(words2))
        return overlap > threshold
```

### 5.1 信息修正

```python
"""
记忆更新和修正
"""

class MemoryUpdater:
    """记忆更新器"""

    def __init__(self, llm):
        self.llm = llm

    def update_memory(
        self,
        old_memory: str,
        new_info: str
    ) -> str:
        """更新记忆"""
        prompt = f"""
原有记忆：{old_memory}
新信息：{new_info}

请整合更新记忆：
- 如果新信息修正了旧信息，以新信息为准
- 如果新信息补充了旧信息，合并保存
- 如果冲突，保留更具体准确的信息

更新后的记忆：
"""
        return self.llm.invoke(prompt).content

    def merge_memories(
        self,
        memories: List[str]
    ) -> str:
        """合并多条记忆"""
        prompt = f"""
合并以下记忆，去除冗余，保留关键信息：

{chr(10).join([f"- {m}" for m in memories])}

合并结果：
"""
        return self.llm.invoke(prompt).content

    def correct_memory(
        self,
        memory: str,
        correction: str
    ) -> str:
        """修正记忆"""
        prompt = f"""
记忆：{memory}
修正：{correction}

请根据修正信息更新记忆：

修正后的记忆：
"""
        return self.llm.invoke(prompt).content
```

### 5.2 重要性调整

```python
"""
动态调整记忆重要性
"""

class ImportanceAdjuster:
    """重要性调整器"""

    def __init__(self):
        self.access_counts = {}  # 访问次数
        self.access_history = {}  # 访问历史

    def record_access(self, memory_id: str):
        """记录访问"""
        if memory_id not in self.access_counts:
            self.access_counts[memory_id] = 0
            self.access_history[memory_id] = []

        self.access_counts[memory_id] += 1
        self.access_history[memory_id].append(datetime.now())

    def adjust_importance(
        self,
        memory_id: str,
        base_importance: float
    ) -> float:
        """调整重要性"""
        # 访问频率加成
        access_count = self.access_counts.get(memory_id, 0)
        frequency_bonus = min(access_count * 0.05, 0.3)

        # 最近访问加成
        history = self.access_history.get(memory_id, [])
        if history:
            last_access = history[-1]
            hours_ago = (datetime.now() - last_access).total_seconds() / 3600
            recency_bonus = max(0, 0.2 - hours_ago * 0.01)
        else:
            recency_bonus = 0

        # 计算新重要性
        new_importance = min(
            base_importance + frequency_bonus + recency_bonus,
            1.0
        )

        return new_importance
```

## 六、综合优化系统

### 6.1 完整优化器

```python
"""
综合记忆优化系统
"""

class MemoryOptimizer:
    """记忆优化器"""

    def __init__(self, llm):
        self.llm = llm
        self.compressor = MemoryCompressor(llm)
        self.evaluator = ImportanceEvaluator(llm)
        self.extractor = KeyInfoExtractor(llm)
        self.forgetting = TimeBasedForgetting()
        self.updater = MemoryUpdater(llm)

    def optimize(
        self,
        memories: List[dict],
        max_count: int = 100
    ) -> List[dict]:
        """优化记忆"""
        # 1. 应用遗忘
        retained = self.forgetting.apply_forgetting(memories)

        # 2. 移除冗余
        unique = self._remove_duplicates(retained)

        # 3. 评估重要性
        for mem in unique:
            if "importance" not in mem:
                mem["importance"] = self.evaluator.evaluate(mem["content"])

        # 4. 按重要性排序
        unique.sort(key=lambda x: x["importance"], reverse=True)

        # 5. 保留最重要的
        top_memories = unique[:max_count]

        # 6. 压缩相似记忆
        final = self._compress_similar(top_memories)

        return final

    def _remove_duplicates(self, memories: List[dict]) -> List[dict]:
        """移除重复"""
        seen = set()
        unique = []

        for mem in memories:
            content_hash = hash(mem["content"][:100])
            if content_hash not in seen:
                seen.add(content_hash)
                unique.append(mem)

        return unique

    def _compress_similar(
        self,
        memories: List[dict],
        similarity_threshold: float = 0.8
    ) -> List[dict]:
        """压缩相似记忆"""
        if len(memories) <= 1:
            return memories

        compressed = []
        used = set()

        for i, mem1 in enumerate(memories):
            if i in used:
                continue

            similar = [mem1]
            for j, mem2 in enumerate(memories[i+1:], i+1):
                if j in used:
                    continue
                if self._is_similar(mem1["content"], mem2["content"]):
                    similar.append(mem2)
                    used.add(j)

            if len(similar) > 1:
                # 合并相似记忆
                merged = self.updater.merge_memories(
                    [m["content"] for m in similar]
                )
                compressed.append({
                    "content": merged,
                    "importance": max(m["importance"] for m in similar),
                    "merged_from": len(similar)
                })
            else:
                compressed.append(mem1)

        return compressed

    def _is_similar(self, text1: str, text2: str) -> bool:
        """判断相似"""
        # 使用向量相似度
        from langchain_openai import OpenAIEmbeddings
        import numpy as np

        embeddings = OpenAIEmbeddings()
        vec1 = embeddings.embed_query(text1)
        vec2 = embeddings.embed_query(text2)

        similarity = np.dot(vec1, vec2) / (
            np.linalg.norm(vec1) * np.linalg.norm(vec2)
        )

        return similarity > 0.8
```

## 总结

### 核心要点

1. **记忆压缩**：摘要、增量更新、结构化
2. **重要性评估**：自动评分、动态调整
3. **关键信息提取**：分类提取、优先级排序
4. **遗忘机制**：时间衰减、主动遗忘
5. **记忆更新**：修正、合并、调整

### 最佳实践

- 定期压缩长对话
- 评估信息长期价值
- 实现合理的遗忘机制
- 保持记忆质量而非数量
