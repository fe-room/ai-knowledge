# 3.1.1 多Agent协作原理

> 理解多Agent系统的设计基础

## 一、为什么需要多Agent？

### 1.1 单Agent的局限

```
单Agent面临的挑战：

1. 能力限制
   - 单一模型能力有限
   - 无法精通所有领域

2. 任务复杂
   - 复杂任务需要多种能力
   - 单一Agent难以完成

3. 效率问题
   - 串行处理效率低
   - 无法并行执行

4. 可靠性
   - 单点故障
   - 缺乏验证机制
```

### 1.2 多Agent优势

```
多Agent系统优势：

1. 专业分工
   ┌─────────┐   ┌─────────┐   ┌─────────┐
   │ Agent 1 │   │ Agent 2 │   │ Agent 3 │
   │ (搜索)  │   │ (分析)  │   │ (写作)  │
   └────┬────┘   └────┬────┘   └────┬────┘
        │             │             │
        └─────────────┼─────────────┘
                      ↓
              ┌──────────────┐
              │   整合输出    │
              └──────────────┘

2. 并行处理
   任务 → [Agent1, Agent2, Agent3] → 并行执行 → 合并结果

3. 互相验证
   任务 → Agent1 → 结果1
        → Agent2 → 结果2
        → 验证Agent → 最终结果

4. 可扩展性
   - 易于添加新能力
   - 灵活调整规模
```

## 二、协作模式

### 2.1 主从模式（Orchestrator）

```python
"""
主从模式：一个协调者分配任务给多个工作Agent
"""

class OrchestratorPattern:
    """主从模式"""

    def __init__(self):
        self.orchestrator = None
        self.workers = []

    def execute(self, task):
        """执行任务"""
        # 1. 协调者分解任务
        subtasks = self.orchestrator.decompose(task)

        # 2. 分配给工作Agent
        results = []
        for subtask, worker in zip(subtasks, self.workers):
            result = worker.execute(subtask)
            results.append(result)

        # 3. 协调者整合结果
        final = self.orchestrator.integrate(results)

        return final

# 示例流程
"""
用户：帮我写一篇关于AI的文章

Orchestrator:
  - 分解任务：
    1. 搜索AI相关资料（Agent: 研究员）
    2. 分析AI趋势（Agent: 分析师）
    3. 撰写文章（Agent: 作家）
    4. 审核文章（Agent: 编辑）

  - 并行执行子任务1和2
  - 顺序执行子任务3（依赖1,2的结果）
  - 执行子任务4审核
  - 返回最终文章
"""
```

### 2.2 对等模式（Peer-to-Peer）

```python
"""
对等模式：多个Agent平等协作
"""

class PeerToPeerPattern:
    """对等模式"""

    def __init__(self, agents):
        self.agents = agents

    def discuss(self, topic, rounds=3):
        """讨论模式"""
        messages = []

        for round_num in range(rounds):
            for agent in self.agents:
                # 每个Agent发言
                response = agent.respond(
                    topic,
                    context=messages
                )
                messages.append({
                    "agent": agent.name,
                    "message": response
                })

        # 总结
        return self.summarize(messages)

# 示例流程
"""
用户：讨论AI是否会取代人类工作

Round 1:
  Agent 1 (乐观派): AI会创造新工作...
  Agent 2 (悲观派): AI会取代重复性工作...
  Agent 3 (中立派): 需要平衡看待...

Round 2:
  Agent 1: 回应悲观派观点...
  Agent 2: 回应乐观派观点...
  Agent 3: 综合两方观点...

Round 3:
  ... 继续讨论 ...

最终：生成综合分析报告
"""
```

### 2.3 层级模式（Hierarchical）

```python
"""
层级模式：多层Agent结构
"""

class HierarchicalPattern:
    """层级模式"""

    def __init__(self):
        # 顶层：管理者
        self.manager = ManagerAgent()

        # 中层：专家
        self.experts = {
            "tech": TechExpert(),
            "business": BusinessExpert(),
            "design": DesignExpert()
        }

        # 底层：执行者
        self.executors = [
            ExecutorAgent() for _ in range(5)
        ]

    def execute(self, task):
        """执行任务"""
        # 1. 管理者决策
        decision = self.manager.analyze(task)

        # 2. 分配给专家
        expert_results = {}
        for area in decision.areas:
            expert = self.experts[area]
            expert_results[area] = expert.plan(task)

        # 3. 执行者执行
        execution_results = []
        for area, plan in expert_results.items():
            for step in plan.steps:
                executor = self.select_executor()
                result = executor.execute(step)
                execution_results.append(result)

        return execution_results
```

### 2.4 流水线模式（Pipeline）

```python
"""
流水线模式：顺序处理
"""

class PipelinePattern:
    """流水线模式"""

    def __init__(self, stages):
        self.stages = stages  # [(Agent, task_type), ...]

    def process(self, input_data):
        """流水线处理"""
        data = input_data

        for agent, task_type in self.stages:
            data = agent.process(data, task_type)

        return data

# 示例
pipeline = PipelinePattern([
    (ResearcherAgent(), "research"),
    (WriterAgent(), "write"),
    (EditorAgent(), "edit"),
    (ReviewerAgent(), "review")
])

result = pipeline.process("写一篇AI文章")
```

## 四、任务分配

### 4.1 分配策略

```python
"""
任务分配策略
"""

class TaskAllocator:
    """任务分配器"""

    def __init__(self, agents):
        self.agents = agents

    def allocate(self, task):
        """分配任务"""
        # 策略1：能力匹配
        best_agent = self._match_capability(task)

        # 策略2：负载均衡
        least_busy = self._find_least_busy()

        # 策略3：优先级
        high_priority = self._get_priority_agent(task)

        return best_agent

    def _match_capability(self, task):
        """能力匹配"""
        for agent in self.agents:
            if self._can_handle(agent, task):
                return agent
        return None

    def _find_least_busy(self):
        """找最空闲的Agent"""
        return min(self.agents, key=lambda a: a.workload)

    def _can_handle(self, agent, task):
        """检查是否能处理"""
        required = task.required_capabilities
        available = agent.capabilities
        return set(required).issubset(set(available))
```

### 4.2 任务分解

```python
"""
任务分解
"""

class TaskDecomposer:
    """任务分解器"""

    def decompose(self, task):
        """分解复杂任务"""
        # 分析任务复杂度
        complexity = self._analyze_complexity(task)

        if complexity == "simple":
            return [task]

        # 分解为子任务
        subtasks = []

        # 按步骤分解
        steps = self._identify_steps(task)
        for step in steps:
            subtask = SubTask(
                description=step,
                dependencies=step.dependencies
            )
            subtasks.append(subtask)

        return subtasks

    def _identify_steps(self, task):
        """识别步骤"""
        # 使用LLM分解
        prompt = f"""
任务：{task.description}

请将此任务分解为具体步骤：
1. 每个步骤清晰明确
2. 标注步骤间的依赖关系
3. 指定每个步骤需要的Agent类型
"""
        # 解析LLM输出
        steps = self._parse_steps(llm.invoke(prompt))
        return steps
```

## 五、通信机制

### 5.1 Agent通信

```python
"""
Agent间通信
"""

from dataclasses import dataclass
from typing import Any, Optional

@dataclass
class Message:
    """Agent消息"""
    sender: str
    receiver: str
    content: Any
    message_type: str  # request, response, notification
    metadata: dict = None

class CommunicationBus:
    """通信总线"""

    def __init__(self):
        self.agents = {}
        self.message_queue = []

    def register(self, agent):
        """注册Agent"""
        self.agents[agent.name] = agent

    def send(self, message: Message):
        """发送消息"""
        receiver = self.agents.get(message.receiver)
        if receiver:
            receiver.receive(message)

    def broadcast(self, sender, content):
        """广播消息"""
        for name, agent in self.agents.items():
            if name != sender:
                message = Message(
                    sender=sender,
                    receiver=name,
                    content=content,
                    message_type="notification"
                )
                agent.receive(message)
```

### 5.2 共享状态

```python
"""
共享状态管理
"""

class SharedState:
    """共享状态"""

    def __init__(self):
        self.state = {}
        self.lock = threading.Lock()

    def set(self, key, value):
        """设置状态"""
        with self.lock:
            self.state[key] = value

    def get(self, key):
        """获取状态"""
        with self.lock:
            return self.state.get(key)

    def update(self, updates):
        """批量更新"""
        with self.lock:
            self.state.update(updates)

# 使用
shared = SharedState()

# Agent写入
shared.set("research_result", {...})

# Agent读取
result = shared.get("research_result")
```

## 六、协作示例

### 6.1 完整协作流程

```python
"""
完整的多Agent协作示例
"""

class MultiAgentSystem:
    """多Agent系统"""

    def __init__(self):
        # 创建Agent
        self.coordinator = CoordinatorAgent("coordinator")
        self.researcher = ResearcherAgent("researcher")
        self.writer = WriterAgent("writer")
        self.reviewer = ReviewerAgent("reviewer")

        # 通信
        self.bus = CommunicationBus()
        self.bus.register(self.coordinator)
        self.bus.register(self.researcher)
        self.bus.register(self.writer)
        self.bus.register(self.reviewer)

        # 状态
        self.state = SharedState()

    def execute(self, task):
        """执行任务"""
        # 1. 协调者分解任务
        subtasks = self.coordinator.decompose(task)
        self.state.set("subtasks", subtasks)

        # 2. 研究员收集信息
        research = self.researcher.execute(subtasks[0])
        self.state.set("research", research)

        # 3. 写作者撰写内容
        draft = self.writer.execute(
            subtasks[1],
            context=self.state.get("research")
        )
        self.state.set("draft", draft)

        # 4. 审核员检查
        review = self.reviewer.execute(
            subtasks[2],
            content=self.state.get("draft")
        )

        # 5. 根据反馈修改
        if review.needs_revision:
            final = self.writer.revise(
                draft,
                feedback=review.feedback
            )
        else:
            final = draft

        return final
```
