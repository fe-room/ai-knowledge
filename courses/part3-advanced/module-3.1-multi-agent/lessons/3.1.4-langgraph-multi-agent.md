# 3.1.4 LangGraph多Agent

> 使用LangGraph构建复杂多Agent系统

## 一、LangGraph多Agent概述

### 1.1 为什么用LangGraph？

```
LangGraph多Agent优势：

1. 状态管理
   - 清晰的状态流转
   - 可持久化状态

2. 循环控制
   - Agent间循环交互
   - 条件分支

3. 可视化
   - 图结构可视化
   - 执行流程清晰

4. 灵活性
   - 自定义节点和边
   - 复杂协作模式
```

### 1.2 架构设计

```
LangGraph多Agent架构：

┌────────────────────────────────────┐
│           StateGraph               │
├────────────────────────────────────┤
│                                    │
│  ┌─────────┐     ┌─────────┐      │
│  │ Agent 1 │ ──→ │ Agent 2 │      │
│  └─────────┘     └────┬────┘      │
│                       │            │
│       ┌───────────────┤            │
│       ↓               ↓            │
│  ┌─────────┐    ┌─────────┐       │
│  │ Agent 3 │    │ Agent 4 │       │
│  └────┬────┘    └────┬────┘       │
│       │              │             │
```

## 二、状态定义

### 2.1 共享状态

```python
"""
多Agent共享状态
"""

from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph

class AgentState(TypedDict):
    """Agent共享状态"""
    # 输入
    task: str
    input_data: dict

    # 各Agent输出
    research_result: str
    analysis_result: str
    writing_result: str

    # 协作信息
    messages: List[dict]
    current_agent: str
    iteration: int

    # 控制
    next_agent: str
    is_complete: bool
```

### 2.2 状态更新

```python
"""
状态更新机制
"""

from langgraph.graph.message import add_messages

class AgentState(TypedDict):
    # 使用annotated定义更新策略
    messages: Annotated[List[dict], add_messages]

def update_state(state: AgentState, updates: dict) -> AgentState:
    """更新状态"""
    return {**state, **updates}
```

## 三、Agent节点

### 3.1 定义Agent节点

```python
"""
定义Agent节点函数
"""

from langchain_openai import ChatOpenAI

llm = ChatOpenAI(model="gpt-4")

def researcher_node(state: AgentState) -> AgentState:
    """研究员节点"""
    task = state["task"]

    # 执行研究
    prompt = f"研究任务：{task}"
    result = llm.invoke(prompt).content

    # 更新状态
    return {
        **state,
        "research_result": result,
        "current_agent": "researcher",
        "messages": [{"role": "researcher", "content": result}]
    }

def analyst_node(state: AgentState) -> AgentState:
    """分析师节点"""
    research = state["research_result"]

    # 执行分析
    prompt = f"分析研究结果：{research}"
    result = llm.invoke(prompt).content

    return {
        **state,
        "analysis_result": result,
        "current_agent": "analyst",
        "messages": [{"role": "analyst", "content": result}]
    }

def writer_node(state: AgentState) -> AgentState:
    """作家节点"""
    analysis = state["analysis_result"]

    # 写作
    prompt = f"基于分析撰写报告：{analysis}"
    result = llm.invoke(prompt).content

    return {
        **state,
        "writing_result": result,
        "current_agent": "writer",
        "is_complete": True
    }
```

### 3.2 带工具的Agent

```python
"""
带工具的Agent节点
"""

from langchain.tools import Tool

def tool_agent_node(state: AgentState, tools: list) -> AgentState:
    """带工具的Agent节点"""
    from langchain.agents import create_tool_calling_agent, AgentExecutor
    from langchain.prompts import ChatPromptTemplate

    # 创建Agent
    prompt = ChatPromptTemplate.from_messages([
        ("system", "你是一个有用的助手，可以使用工具。"),
        ("human", "{input}")
    ])

    agent = create_tool_calling_agent(llm, tools, prompt)
    executor = AgentExecutor(agent=agent, tools=tools)

    # 执行
    result = executor.invoke({"input": state["task"]})

    return {
        **state,
        "result": result["output"]
    }
```

## 四、构建多Agent图

### 4.1 顺序执行

```python
"""
顺序执行的多Agent
"""

from langgraph.graph import StateGraph, END

# 创建图
workflow = StateGraph(AgentState)

# 添加节点
workflow.add_node("researcher", researcher_node)
workflow.add_node("analyst", analyst_node)
workflow.add_node("writer", writer_node)

# 添加边（顺序执行）
workflow.add_edge("researcher", "analyst")
workflow.add_edge("analyst", "writer")
workflow.add_edge("writer", END)

# 设置入口
workflow.set_entry_point("researcher")

# 编译
app = workflow.compile()

# 执行
initial_state = {
    "task": "研究AI发展趋势",
    "messages": [],
    "is_complete": False
}

result = app.invoke(initial_state)
print(result["writing_result"])
```

### 4.2 条件分支

```python
"""
条件分支的多Agent
"""

def router(state: AgentState) -> str:
    """路由函数"""
    # 根据状态决定下一个Agent
    if state.get("needs_research"):
        return "researcher"
    elif state.get("needs_analysis"):
        return "analyst"
    else:
        return "writer"

# 创建图
workflow = StateGraph(AgentState)

# 添加节点
workflow.add_node("researcher", researcher_node)
workflow.add_node("analyst", analyst_node)
workflow.add_node("writer", writer_node)

# 添加条件边
workflow.add_conditional_edges(
    "researcher",
    router,
    {
        "researcher": "researcher",
        "analyst": "analyst",
        "writer": "writer"
    }
)

workflow.add_conditional_edges(
    "analyst",
    router,
    {
        "researcher": "researcher",
        "analyst": "analyst",
        "writer": "writer"
    }
)

workflow.add_edge("writer", END)

workflow.set_entry_point("researcher")
```

### 4.3 循环交互

```python
"""
Agent循环交互
"""

def should_continue(state: AgentState) -> str:
    """判断是否继续"""
    if state["is_complete"]:
        return "end"
    if state["iteration"] >= 5:
        return "end"
    return "continue"

def reviewer_node(state: AgentState) -> AgentState:
    """审核员节点"""
    writing = state["writing_result"]

    # 审核
    prompt = f"审核内容：{writing}\n给出修改建议或确认完成"
    result = llm.invoke(prompt).content

    # 判断是否需要修改
    needs_revision = "修改" in result

    return {
        **state,
        "review_feedback": result,
        "needs_revision": needs_revision,
        "iteration": state["iteration"] + 1
    }

# 创建循环图
workflow = StateGraph(AgentState)

workflow.add_node("researcher", researcher_node)
workflow.add_node("writer", writer_node)
workflow.add_node("reviewer", reviewer_node)

# 循环结构
workflow.add_edge("researcher", "writer")
workflow.add_edge("writer", "reviewer")

workflow.add_conditional_edges(
    "reviewer",
    should_continue,
    {
        "continue": "writer",  # 需要修改，回到writer
        "end": END
    }
)

workflow.set_entry_point("researcher")
```

### 5.1 消息传递

```python
"""
Agent间消息传递
"""

class AgentState(TypedDict):
    messages: Annotated[List[dict], add_messages]

def agent_with_communication(
    state: AgentState,
    agent_name: str,
    system_prompt: str
) -> AgentState:
    """支持通信的Agent"""
    # 获取消息历史
    history = state["messages"]

    # 构建提示
    prompt = f"{system_prompt}\n\n对话历史：\n{history}\n\n请回复："

    # 生成回复
    response = llm.invoke(prompt).content

    # 添加消息
    new_message = {
        "role": agent_name,
        "content": response
    }

    return {
        **state,
        "messages": [new_message]
    }
```

## 六、完整示例

### 6.1 研究团队

```python
"""
LangGraph完整示例：研究团队
"""

from typing import TypedDict, List
from langgraph.graph import StateGraph, END
from langchain_openai import ChatOpenAI

# 定义状态
class ResearchState(TypedDict):
    topic: str
    research_notes: str
    analysis: str
    report: str
    feedback: str
    iteration: int
    is_approved: bool

# 初始化LLM
llm = ChatOpenAI(model="gpt-4")

# 定义节点
def researcher(state: ResearchState) -> ResearchState:
    """研究节点"""
    topic = state["topic"]

    prompt = f"""
    研究主题：{topic}

    请收集相关信息并总结。
    """
    notes = llm.invoke(prompt).content

    return {**state, "research_notes": notes}

def analyst(state: ResearchState) -> ResearchState:
    """分析节点"""
    notes = state["research_notes"]

    prompt = f"""
    研究笔记：{notes}

    请分析关键趋势和洞察。
    """
    analysis = llm.invoke(prompt).content

    return {**state, "analysis": analysis}

def writer(state: ResearchState) -> ResearchState:
    """写作节点"""
    analysis = state["analysis"]

    prompt = f"""
    分析结果：{analysis}

    请撰写一份专业报告。
    """
    report = llm.invoke(prompt).content

    return {**state, "report": report}

def reviewer(state: ResearchState) -> ResearchState:
    """审核节点"""
    report = state["report"]

    prompt = f"""
    报告内容：{report}

    请审核报告质量。如果满意，回复"APPROVED"。
    如果需要修改，提供具体建议。
    """
    feedback = llm.invoke(prompt).content

    is_approved = "APPROVED" in feedback

    return {
        **state,
        "feedback": feedback,
        "is_approved": is_approved,
        "iteration": state["iteration"] + 1
    }

# 路由函数
def route_after_review(state: ResearchState) -> str:
    """审核后路由"""
    if state["is_approved"]:
        return "end"
    if state["iteration"] >= 3:
        return "end"
    return "revise"

# 构建图
workflow = StateGraph(ResearchState)

# 添加节点
workflow.add_node("researcher", researcher)
workflow.add_node("analyst", analyst)
workflow.add_node("writer", writer)
workflow.add_node("reviewer", reviewer)

# 添加边
workflow.add_edge("researcher", "analyst")
workflow.add_edge("analyst", "writer")
workflow.add_edge("writer", "reviewer")

workflow.add_conditional_edges(
    "reviewer",
    route_after_review,
    {
        "revise": "writer",
        "end": END
    }
)

# 入口
workflow.set_entry_point("researcher")

# 编译
app = workflow.compile()

# 执行
initial_state = {
    "topic": "2024年AI发展趋势",
    "research_notes": "",
    "analysis": "",
    "report": "",
    "feedback": "",
    "iteration": 0,
    "is_approved": False
}

result = app.invoke(initial_state)
print(result["report"])
```

### 6.2 并行处理

```python
"""
并行Agent处理
"""

from concurrent.futures import ThreadPoolExecutor

def parallel_agents(state: AgentState) -> AgentState:
    """并行执行多个Agent"""
    tasks = [
        ("researcher", researcher_node),
        ("analyst", analyst_node),
        ("writer", writer_node)
    ]

    results = {}
    with ThreadPoolExecutor() as executor:
        futures = {
            executor.submit(func, state): name
            for name, func in tasks
        }

        for future in futures:
            name = futures[future]
            results[name] = future.result()

    return {
        **state,
        "results": results
    }
```

## 总结

### 核心要点

1. **状态管理**：定义共享状态类型
2. **Agent节点**：每个Agent是一个节点函数
3. **条件分支**：根据状态动态路由
4. **循环交互**：Agent可以循环协作
5. **消息传递**：通过状态共享消息

### 最佳实践

- 状态定义要清晰完整
- 节点函数职责单一
- 合理设置循环终止条件
- 使用条件边实现复杂流程

### 框架对比

| 框架 | 优势 | 适用场景 |
|------|------|---------|
| CrewAI | 简单易用 | 任务驱动团队 |
| AutoGen | 对话式 | 需要人类参与 |
| LangGraph | 灵活可控 | 复杂状态流程 |
