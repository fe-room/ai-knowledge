### 2.1 关键指标

```python
"""
Agent性能指标监控
"""

from dataclasses import dataclass
from typing import Dict, List
import time

@dataclass
class PerformanceMetrics:
    """性能指标"""
    # 响应时间
    avg_latency: float
    p50_latency: float
    p95_latency: float
    p99_latency: float

    # 吞吐量
    requests_per_second: float

    # 错误率
    error_rate: float

    # 资源使用
    cpu_usage: float
    memory_usage: float

    # Agent特定指标
    avg_iterations: float
    tool_call_success_rate: float
    avg_tokens_per_request: float

class PerformanceMonitor:
    """性能监控器"""

    def __init__(self):
        self.metrics_history = []

    def collect_metrics(self) -> PerformanceMetrics:
        """收集指标"""
        # 收集系统指标
        cpu = self._get_cpu_usage()
        memory = self._get_memory_usage()

        # 收集应用指标
        latencies = self._get_latencies()
        errors = self._get_error_count()
        total = self._get_total_requests()

        # 计算指标
        metrics = PerformanceMetrics(
            avg_latency=self._calculate_avg(latencies),
            p50_latency=self._calculate_percentile(latencies, 50),
            p95_latency=self._calculate_percentile(latencies, 95),
            p99_latency=self._calculate_percentile(latencies, 99),
            requests_per_second=self._calculate_rps(total),
            error_rate=errors / total if total > 0 else 0,
            cpu_usage=cpu,
            memory_usage=memory,
            avg_iterations=self._get_avg_iterations(),
            tool_call_success_rate=self._get_tool_success_rate(),
            avg_tokens_per_request=self._get_avg_tokens()
        )

        self.metrics_history.append(metrics)
        return metrics

    def _calculate_percentile(self, values: List[float], percentile: int) -> float:
        """计算百分位数"""
        if not values:
            return 0
        import numpy as np
        return np.percentile(values, percentile)

    def get_performance_report(self) -> Dict:
        """生成性能报告"""
        if not self.metrics_history:
            return {}

        recent = self.metrics_history[-100:]  # 最近100个数据点

        return {
            "latency_trend": [m.avg_latency for m in recent],
            "error_rate_trend": [m.error_rate for m in recent],
            "resource_usage": {
                "cpu": [m.cpu_usage for m in recent],
                "memory": [m.memory_usage for m in recent]
            }
        }
```

```python
"""

from fastapi import FastAPI
from fastapi.responses import HTMLResponse

app = FastAPI()

class DashboardAPI:
    """仪表板API"""

    def __init__(self, monitor):
        self.monitor = monitor

    def get_system_status(self) -> Dict:
        """获取系统状态"""
        return {
            "status": "healthy",
            "uptime": self._get_uptime(),
            "active_connections": self._get_active_connections(),
            "queue_size": self._get_queue_size()
        }

    def get_performance_summary(self) -> Dict:
        """获取性能摘要"""
        metrics = self.monitor.collect_metrics()

        return {
            "latency": {
                "avg": f"{metrics.avg_latency:.2f}ms",
                "p95": f"{metrics.p95_latency:.2f}ms",
                "status": self._get_status(metrics.avg_latency, 1000)
            },
            "throughput": {
                "rps": f"{metrics.requests_per_second:.2f}",
                "status": self._get_status(metrics.requests_per_second, 10, reverse=True)
            },
            "errors": {
                "rate": f"{metrics.error_rate:.2%}",
                "status": self._get_status(metrics.error_rate, 0.05)
            },
            "resources": {
                "cpu": f"{metrics.cpu_usage:.1f}%",
                "memory": f"{metrics.memory_usage:.1f}%",
                "status": self._get_status(max(metrics.cpu_usage, metrics.memory_usage), 80)
            }
        }

    def _get_status(self, value: float, threshold: float, reverse: bool = False) -> str:
        """获取状态"""
        if reverse:
            return "good" if value >= threshold else "warning"
        return "good" if value <= threshold else "warning"

@app.get("/dashboard/status")
async def get_status():
    """获取系统状态"""
    dashboard = DashboardAPI(monitor)
    return dashboard.get_system_status()

@app.get("/dashboard/performance")
async def get_performance():
    """获取性能摘要"""
    dashboard = DashboardAPI(monitor)
    return dashboard.get_performance_summary()
```

```python
"""
故障诊断系统
"""

from typing import List, Dict
from dataclasses import dataclass

@dataclass
class DiagnosticResult:
    """诊断结果"""
    issue_type: str
    severity: str
    root_cause: str
    recommendations: List[str]

class FaultDiagnoser:
    """故障诊断器"""

    def diagnose(self, symptoms: Dict) -> DiagnosticResult:
        """诊断故障"""
        # 检查常见问题
        checks = [
            self._check_high_latency,
            self._check_high_error_rate,
            self._check_resource_exhaustion,
            self._check_llm_issues,
            self._check_tool_failures
        ]

        for check in checks:
            result = check(symptoms)
            if result:
                return result

        return DiagnosticResult(
            issue_type="unknown",
            severity="low",
            root_cause="未识别的问题",
            recommendations=["检查日志", "联系支持"]
        )

    def _check_high_latency(self, symptoms: Dict) -> DiagnosticResult:
        """检查高延迟"""
        avg_latency = symptoms.get("avg_latency", 0)

        if avg_latency > 5000:
            return DiagnosticResult(
                issue_type="high_latency",
                severity="high",
                root_cause="响应时间过长",
                recommendations=[
                    "检查LLM API响应时间",
                    "检查网络连接",
                    "考虑使用缓存",
                    "优化提示词长度"
                ]
            )

        return None

    def _check_high_error_rate(self, symptoms: Dict) -> DiagnosticResult:
        """检查高错误率"""
        error_rate = symptoms.get("error_rate", 0)

        if error_rate > 0.1:
            errors = symptoms.get("recent_errors", [])

            return DiagnosticResult(
                issue_type="high_error_rate",
                severity="critical",
                root_cause=self._analyze_errors(errors),
                recommendations=[
                    "检查错误日志",
                    "验证API密钥",
                    "检查配额限制",
                    "重试机制"
                ]
            )

        return None

    def _check_resource_exhaustion(self, symptoms: Dict) -> DiagnosticResult:
        """检查资源耗尽"""
        cpu = symptoms.get("cpu_usage", 0)
        memory = symptoms.get("memory_usage", 0)

        if cpu > 90 or memory > 90:
            return DiagnosticResult(
                issue_type="resource_exhaustion",
                severity="high",
                root_cause=f"资源使用过高: CPU {cpu}%, 内存 {memory}%",
                recommendations=[
                    "扩容实例",
                    "优化资源使用",
                    "检查内存泄漏"
                ]
            )

        return None
```

### 3.2 故障处理脚本

```bash
#!/bin/bash
# incident-response.sh

# 故障响应脚本

INCIDENT_ID=$1
ACTION=$2

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

case $ACTION in
    "diagnose")
        log "开始诊断故障: $INCIDENT_ID"

        # 收集日志
        kubectl logs -l app=agent-api --tail=1000 > /tmp/incident-$INCIDENT_ID.log

        # 检查资源
        kubectl top pods -l app=agent-api

        # 检查事件
        kubectl get events --sort-by='.lastTimestamp'

        log "诊断完成，日志保存在 /tmp/incident-$INCIDENT_ID.log"
        ;;

    "restart")
        log "重启服务: $INCIDENT_ID"

        kubectl rollout restart deployment/agent-api

        log "重启完成"
        ;;

    "scale-up")
        log "扩容服务: $INCIDENT_ID"

        kubectl scale deployment/agent-api --replicas=5

        log "扩容完成"
        ;;

    "rollback")
        log "回滚服务: $INCIDENT_ID"

        kubectl rollout undo deployment/agent-api

        log "回滚完成"
        ;;

    *)
        echo "用法: $0 <incident_id> <diagnose|restart|scale-up|rollback>"
        exit 1
        ;;
esac
```

```python
"""
备份管理
"""

import os
import json
from datetime import datetime
from typing import List
import boto3

class BackupManager:
    """备份管理器"""

    def __init__(self, s3_bucket: str):
        self.s3 = boto3.client('s3')
        self.bucket = s3_bucket

    def backup_data(self, data_type: str, data: dict):
        """备份数据"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        key = f"backups/{data_type}/{timestamp}.json"

        self.s3.put_object(
            Bucket=self.bucket,
            Key=key,
            Body=json.dumps(data),
            ContentType='application/json'
        )

        return key

    def backup_database(self, db_url: str):
        """备份数据库"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_file = f"/tmp/db_backup_{timestamp}.sql"

        # 执行备份
        os.system(f"pg_dump {db_url} > {backup_file}")

        # 上传到S3
        key = f"backups/database/{timestamp}.sql"
        self.s3.upload_file(backup_file, self.bucket, key)

        # 清理本地文件
        os.remove(backup_file)

        return key

    def restore_data(self, backup_key: str) -> dict:
        """恢复数据"""
        response = self.s3.get_object(
            Bucket=self.bucket,
            Key=backup_key
        )

        return json.loads(response['Body'].read())

    def list_backups(self, data_type: str = None) -> List[dict]:
        """列出备份"""
        prefix = f"backups/{data_type}/" if data_type else "backups/"

        response = self.s3.list_objects_v2(
            Bucket=self.bucket,
            Prefix=prefix
        )

        return [
            {
                "key": obj['Key'],
                "size": obj['Size'],
                "last_modified": obj['LastModified'].isoformat()
            }
            for obj in response.get('Contents', [])
        ]

    def cleanup_old_backups(self, retention_days: int = 30):
        """清理旧备份"""
        cutoff = datetime.now() - timedelta(days=retention_days)

        for backup in self.list_backups():
            if datetime.fromisoformat(backup['last_modified']) < cutoff:
                self.s3.delete_object(
                    Bucket=self.bucket,
                    Key=backup['key']
                )
```

### 5.2 回滚机制

```python
"""
回滚管理
"""

class RollbackManager:
    """回滚管理器"""

    def __init__(self, k8s_api):
        self.k8s = k8s_api

    def rollback_deployment(
        self,
        deployment_name: str,
        revision: int = None
    ):
        """回滚部署"""
        if revision:
            # 回滚到指定版本
            self.k8s.rollback_to_revision(deployment_name, revision)
        else:
            # 回滚到上一版本
            self.k8s.rollback(deployment_name)

    def list_revisions(self, deployment_name: str) -> List[dict]:
        """列出历史版本"""
        history = self.k8s.get_deployment_history(deployment_name)

        return [
            {
                "revision": h.revision,
                "image": h.image,
                "timestamp": h.timestamp,
                "reason": h.reason
            }
            for h in history
        ]

    def verify_rollback(self, deployment_name: str) -> bool:
        """验证回滚"""
        # 检查Pod状态
        pods = self.k8s.get_pods(deployment_name)

        # 检查健康状态
        healthy = all(pod.status == "Running" for pod in pods)

        return healthy
```
