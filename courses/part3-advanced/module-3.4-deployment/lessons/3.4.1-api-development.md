## 三、Express/Node.js服务

### 3.1 基础服务

```javascript
/**
 * Express Agent服务
 */

const express = require('express');
const { OpenAI } = require('openai');

const app = express();

// 中间件
app.use(express.json());
app.use(cors());

// OpenAI客户端
const openai = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY
});

// Agent服务
class AgentService {
    constructor() {
        this.conversations = new Map();
    }

    async chat(message, sessionId) {
        const history = this.conversations.get(sessionId) || [];
        history.push({ role: 'user', content: message });

        const response = await openai.chat.completions.create({
            model: 'gpt-3.5-turbo',
            messages: history
        });

        const assistantMessage = response.choices[0].message;
        history.push(assistantMessage);

        this.conversations.set(sessionId, history);

        return {
            response: assistantMessage.content,
            sessionId,
            tokensUsed: response.usage.total_tokens
        };
    }
}

const agentService = new AgentService();

// 路由
app.post('/api/v1/chat', async (req, res) => {
    try {
        const { message, session_id } = req.body;
        const result = await agentService.chat(message, session_id);
        res.json(result);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// 流式响应
app.post('/api/v1/chat/stream', async (req, res) => {
    const { message, session_id } = req.body;

    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');

    const stream = await openai.chat.completions.create({
        model: 'gpt-3.5-turbo',
        messages: [{ role: 'user', content: message }],
        stream: true
    });

    for await (const chunk of stream) {
        const content = chunk.choices[0]?.delta?.content || '';
        res.write(`data: ${content}\n\n`);
    }

    res.write('data: [DONE]\n\n');
    res.end();
});

// 健康检查
app.get('/health', (req, res) => {
    res.json({ status: 'healthy' });
});

// 启动
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
```

### 4.1 错误处理

```python
"""
统一错误处理
"""

from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError

class AgentError(Exception):
    """Agent错误"""
    def __init__(self, code: str, message: str, details: dict = None):
        self.code = code
        self.message = message
        self.details = details

# 错误码定义
ERROR_CODES = {
    "AUTH_FAILED": {"status": 401, "message": "认证失败"},
    "RATE_LIMIT": {"status": 429, "message": "请求过于频繁"},
    "INVALID_INPUT": {"status": 400, "message": "输入参数无效"},
    "AGENT_ERROR": {"status": 500, "message": "Agent执行错误"},
    "TIMEOUT": {"status": 504, "message": "请求超时"}
}

@app.exception_handler(AgentError)
async def agent_error_handler(request: Request, exc: AgentError):
    """Agent错误处理"""
    error_info = ERROR_CODES.get(exc.code, {"status": 500, "message": "未知错误"})

    return JSONResponse(
        status_code=error_info["status"],
        content={
            "error": {
                "code": exc.code,
                "message": exc.message,
                "details": exc.details
            }
        }
    )

@app.exception_handler(RequestValidationError)
async def validation_error_handler(request: Request, exc: RequestValidationError):
    """验证错误处理"""
    return JSONResponse(
        status_code=422,
        content={
            "error": {
                "code": "VALIDATION_ERROR",
                "message": "请求参数验证失败",
                "details": exc.errors()
            }
        }
    )
```

### 4.2 请求验证

```python
"""
请求验证
"""

from pydantic import BaseModel, Field, validator
from typing import List, Optional

class ChatRequest(BaseModel):
    """聊天请求验证"""
    message: str = Field(
        ...,
        min_length=1,
        max_length=10000,
        description="用户消息"
    )
    session_id: Optional[str] = Field(
        None,
        regex=r"^[a-zA-Z0-9_-]{8,36}$",
        description="会话ID"
    )
    model: str = Field(
        "gpt-3.5-turbo",
        description="模型名称"
    )
    temperature: float = Field(
        0.7,
        ge=0,
        le=2,
        description="温度参数"
    )
    max_tokens: int = Field(
        1000,
        ge=1,
        le=4000,
        description="最大Token数"
    )

    @validator('message')
    def validate_message(cls, v):
        """验证消息"""
        if not v.strip():
            raise ValueError('消息不能为空')
        return v.strip()

    @validator('model')
    def validate_model(cls, v):
        """验证模型"""
        allowed_models = [
            "gpt-3.5-turbo",
            "gpt-4",
            "claude-3-opus",
            "claude-3-sonnet"
        ]
        if v not in allowed_models:
            raise ValueError(f'不支持的模型: {v}')
        return v
```

### 4.3 日志记录

```python
"""
请求日志中间件
"""

import time
from fastapi import Request
import logging

logger = logging.getLogger("api")

@app.middleware("http")
async def log_requests(request: Request, call_next):
    """记录请求日志"""
    start_time = time.time()

    # 请求信息
    request_id = request.headers.get("X-Request-ID", "-")

    logger.info(
        f"Request started: {request.method} {request.url.path}",
        extra={
            "request_id": request_id,
            "method": request.method,
            "path": request.url.path
        }
    )

    # 执行
    response = await call_next(request)

    # 计算时长
    duration = time.time() - start_time

    # 响应信息
    logger.info(
        f"Request completed: {request.method} {request.url.path} "
        f"- Status: {response.status_code} - Duration: {duration:.3f}s",
        extra={
            "request_id": request_id,
            "status_code": response.status_code,
            "duration": duration
        }
    )

    # 添加响应头
    response.headers["X-Request-ID"] = request_id
    response.headers["X-Response-Time"] = f"{duration:.3f}s"

    return response
```
