## 一、工具设计原则

### 1.1 好工具的特征

```
好的工具设计：

1. 单一职责
   - 一个工具只做一件事
   - 功能明确清晰

2. 清晰描述
   - 描述让LLM理解工具用途
   - 参数说明详细

3. 安全可控
   - 验证输入参数
   - 限制执行范围

4. 错误处理
   - 优雅处理错误
   - 返回有意义的错误信息

5. 可观测
   - 记录执行日志
   - 便于调试和监控
```

### 1.2 工具设计模板

```python
"""
自定义工具模板
"""

from dataclasses import dataclass
from typing import Any, Dict, Optional

@dataclass
class ToolResult:
    """工具执行结果"""
    success: bool
    data: Any
    error: Optional[str] = None
    metadata: Optional[Dict] = None

class CustomTool:
    """自定义工具基类"""

    name = "custom_tool"
    description = "工具描述"

    def get_definition(self):
        """获取工具定义"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.get_parameters()
            }
        }

    def get_parameters(self):
        """获取参数定义"""
        return {
            "type": "object",
            "properties": {},
            "required": []
        }

    def validate(self, **kwargs) -> tuple[bool, str]:
        """验证参数"""
        return True, ""

    def execute(self, **kwargs) -> ToolResult:
        """执行工具"""
        raise NotImplementedError

    def __call__(self, **kwargs):
        """调用工具"""
        # 验证
        valid, error = self.validate(**kwargs)
        if not valid:
            return ToolResult(success=False, data=None, error=error)

        # 执行
        try:
            result = self.execute(**kwargs)
            return result
        except Exception as e:
            return ToolResult(success=False, data=None, error=str(e))
```

### 2.1 文本分析工具

```python
"""
文本分析工具
"""

class TextAnalyzerTool(CustomTool):
    """文本分析工具"""

    name = "analyze_text"
    description = "分析文本的统计信息，包括字数、句子数等"

    def get_parameters(self):
        return {
            "type": "object",
            "properties": {
                "text": {
                    "type": "string",
                    "description": "要分析的文本"
                },
                "analysis_type": {
                    "type": "string",
                    "enum": ["basic", "detailed", "sentiment"],
                    "description": "分析类型"
                }
            },
            "required": ["text"]
        }

    def validate(self, **kwargs):
        text = kwargs.get("text")
        if not text or len(text.strip()) == 0:
            return False, "文本不能为空"
        if len(text) > 100000:
            return False, "文本过长，最大100000字符"
        return True, ""

    def execute(self, **kwargs):
        text = kwargs["text"]
        analysis_type = kwargs.get("analysis_type", "basic")

        if analysis_type == "basic":
            result = self._basic_analysis(text)
        elif analysis_type == "detailed":
            result = self._detailed_analysis(text)
        elif analysis_type == "sentiment":
            result = self._sentiment_analysis(text)
        else:
            result = self._basic_analysis(text)

        return ToolResult(
            success=True,
            data=result,
            metadata={"analysis_type": analysis_type}
        )

    def _basic_analysis(self, text):
        """基础分析"""
        import re
        return {
            "char_count": len(text),
            "word_count": len(text.split()),
            "sentence_count": len(re.split(r'[.!?。！？]', text)),
            "paragraph_count": len(text.split('\n\n'))
        }

    def _detailed_analysis(self, text):
        """详细分析"""
        basic = self._basic_analysis(text)
        words = text.split()

        return {
            **basic,
            "avg_word_length": sum(len(w) for w in words) / len(words) if words else 0,
            "longest_word": max(words, key=len) if words else "",
            "unique_words": len(set(words))
        }

    def _sentiment_analysis(self, text):
        """情感分析"""
        # 简化示例
        positive_words = ["好", "优秀", "棒", "喜欢", "开心"]
        negative_words = ["差", "糟糕", "讨厌", "难过", "失望"]

        score = 0
        for word in positive_words:
            score += text.count(word)
        for word in negative_words:
            score -= text.count(word)

        if score > 0:
            sentiment = "positive"
        elif score < 0:
            sentiment = "negative"
        else:
            sentiment = "neutral"

        return {
            "sentiment": sentiment,
            "score": score
        }
```

### 2.2 数据处理工具

```python
"""
数据处理工具
"""

class DataProcessorTool(CustomTool):
    """数据处理工具"""

    name = "process_data"
    description = "处理JSON或CSV格式的数据"

    def get_parameters(self):
        return {
            "type": "object",
            "properties": {
                "data": {
                    "type": "string",
                    "description": "数据内容（JSON或CSV格式）"
                },
                "format": {
                    "type": "string",
                    "enum": ["json", "csv"],
                    "description": "数据格式"
                },
                "operation": {
                    "type": "string",
                    "enum": ["filter", "sort", "aggregate", "transform"],
                    "description": "操作类型"
                },
                "params": {
                    "type": "object",
                    "description": "操作参数"
                }
            },
            "required": ["data", "format", "operation"]
        }

    def execute(self, **kwargs):
        data = kwargs["data"]
        format = kwargs["format"]
        operation = kwargs["operation"]
        params = kwargs.get("params", {})

        # 解析数据
        if format == "json":
            import json
            parsed = json.loads(data)
        else:  # csv
            parsed = self._parse_csv(data)

        # 执行操作
        if operation == "filter":
            result = self._filter_data(parsed, params)
        elif operation == "sort":
            result = self._sort_data(parsed, params)
        elif operation == "aggregate":
            result = self._aggregate_data(parsed, params)
        else:
            result = parsed

        return ToolResult(success=True, data=result)

    def _parse_csv(self, csv_data):
        """解析CSV"""
        import csv
        from io import StringIO
        reader = csv.DictReader(StringIO(csv_data))
        return list(reader)

    def _filter_data(self, data, params):
        """过滤数据"""
        field = params.get("field")
        value = params.get("value")
        operator = params.get("operator", "eq")

        if operator == "eq":
            return [d for d in data if d.get(field) == value]
        elif operator == "gt":
            return [d for d in data if d.get(field) > value]
        elif operator == "lt":
            return [d for d in data if d.get(field) < value]

        return data

    def _sort_data(self, data, params):
        """排序数据"""
        field = params.get("field")
        reverse = params.get("reverse", False)
        return sorted(data, key=lambda x: x.get(field, ""), reverse=reverse)

    def _aggregate_data(self, data, params):
        """聚合数据"""
        field = params.get("field")
        agg_type = params.get("type", "sum")

        values = [float(d.get(field, 0)) for d in data]

        if agg_type == "sum":
            return sum(values)
        elif agg_type == "avg":
            return sum(values) / len(values) if values else 0
        elif agg_type == "max":
            return max(values) if values else 0
        elif agg_type == "min":
            return min(values) if values else 0

        return None
```

## 三、工具组合

### 3.1 工具链

```python
"""
工具链：串联多个工具
"""

class ToolChain:
    """工具链"""

    def __init__(self):
        self.steps = []

    def add_step(self, tool, params=None, output_key=None):
        """添加步骤"""
        self.steps.append({
            "tool": tool,
            "params": params or {},
            "output_key": output_key
        })
        return self

    def execute(self, initial_input=None):
        """执行工具链"""
        context = initial_input or {}

        for step in self.steps:
            tool = step["tool"]
            params = {**step["params"]}

            # 从上下文获取参数
            for key, value in params.items():
                if isinstance(value, str) and value.startswith("$"):
                    # 引用上下文变量
                    context_key = value[1:]
                    params[key] = context.get(context_key)

            # 执行工具
            result = tool(**params)

            # 保存结果到上下文
            if step["output_key"]:
                context[step["output_key"]] = result

        return context

# 使用示例
search_tool = WebSearchTool()
analyze_tool = TextAnalyzerTool()

chain = ToolChain()
chain.add_step(search_tool, {"query": "AI trends 2024"}, output_key="search_result")
chain.add_step(analyze_tool, {"text": "$search_result"}, output_key="analysis")

result = chain.execute()
print(result["analysis"])
```

### 3.2 并行工具

```python
"""
并行执行多个工具
"""

import concurrent.futures

class ParallelTools:
    """并行工具执行"""

    def __init__(self, tools):
        self.tools = tools

    def execute_all(self, **kwargs):
        """并行执行所有工具"""
        results = {}

        with concurrent.futures.ThreadPoolExecutor() as executor:
            futures = {}
            for name, tool in self.tools.items():
                future = executor.submit(tool, **kwargs)
                futures[future] = name

            for future in concurrent.futures.as_completed(futures):
                name = futures[future]
                try:
                    results[name] = future.result()
                except Exception as e:
                    results[name] = {"error": str(e)}

        return results

# 使用示例
tools = {
    "search": WebSearchTool(),
    "weather": WeatherTool(),
    "news": NewsTool()
}

parallel = ParallelTools(tools)
results = parallel.execute_all(query="北京")
```

## 四、错误处理

### 4.1 错误处理策略

```python
"""
工具错误处理
"""

from enum import Enum
from typing import Optional

class ErrorType(Enum):
    VALIDATION_ERROR = "validation_error"
    EXECUTION_ERROR = "execution_error"
    TIMEOUT_ERROR = "timeout_error"
    PERMISSION_ERROR = "permission_error"

class ToolErrorHandler:
    """工具错误处理器"""

    def handle(self, error_type: ErrorType, error: Exception, context: dict):
        """处理错误"""
        handlers = {
            ErrorType.VALIDATION_ERROR: self._handle_validation,
            ErrorType.EXECUTION_ERROR: self._handle_execution,
            ErrorType.TIMEOUT_ERROR: self._handle_timeout,
            ErrorType.PERMISSION_ERROR: self._handle_permission
        }

        handler = handlers.get(error_type, self._handle_unknown)
        return handler(error, context)

    def _handle_validation(self, error, context):
        """处理验证错误"""
        return {
            "success": False,
            "error_type": "validation",
            "message": str(error),
            "suggestion": "请检查输入参数是否正确"
        }

    def _handle_execution(self, error, context):
        """处理执行错误"""
        return {
            "success": False,
            "error_type": "execution",
            "message": str(error),
            "suggestion": "工具执行失败，请稍后重试"
        }

    def _handle_timeout(self, error, context):
        """处理超时错误"""
        return {
            "success": False,
            "error_type": "timeout",
            "message": "操作超时",
            "suggestion": "请尝试减少数据量或增加超时时间"
        }

    def _handle_permission(self, error, context):
        """处理权限错误"""
        return {
            "success": False,
            "error_type": "permission",
            "message": "权限不足",
            "suggestion": "请检查权限配置"
        }

    def _handle_unknown(self, error, context):
        """处理未知错误"""
        return {
            "success": False,
            "error_type": "unknown",
            "message": str(error)
        }
```

### 4.2 重试机制

```python
"""
工具重试机制
"""

import time
from functools import wraps

def retry(max_attempts=3, delay=1, backoff=2):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_error = None
            current_delay = delay

            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_error = e
                    if attempt < max_attempts - 1:
                        time.sleep(current_delay)
                        current_delay *= backoff

            return {
                "success": False,
                "error": str(last_error),
                "attempts": max_attempts
            }

        return wrapper
    return decorator

# 使用示例
class ReliableTool(CustomTool):
    """可靠工具"""

    @retry(max_attempts=3, delay=1)
    def execute(self, **kwargs):
        # 可能失败的操作
        result = self._do_work(**kwargs)
        return ToolResult(success=True, data=result)
```

工具测试

import unittest

class TestTextAnalyzerTool(unittest.TestCase):

    def test_long_text(self):
        """测试长文本"""

## 总结

### 核心要点

1. **设计原则**：单一职责、清晰描述、安全可控
2. **开发模板**：基类、验证、执行、结果
3. **工具组合**：链式执行、并行执行
4. **错误处理**：分类处理、重试机制
5. **测试验证**：单元测试、集成测试

### 最佳实践

- 保持工具功能单一
- 提供详细的参数描述
- 实现完善的错误处理
- 编写测试用例验证
