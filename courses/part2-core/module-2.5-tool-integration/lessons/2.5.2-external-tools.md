```python
"""
网页搜索工具
"""

from langchain.tools import Tool
from langchain.utilities import GoogleSearchAPIWrapper, BingSearchAPIWrapper

# Google搜索
google_search = GoogleSearchAPIWrapper()

google_tool = Tool(
    name="google_search",
    description="搜索Google获取信息",
    func=lambda q: google_search.run(q)
)

# Bing搜索
bing_search = BingSearchAPIWrapper()

bing_tool = Tool(
    name="bing_search",
    description="搜索Bing获取信息",
    func=lambda q: bing_search.run(q)
)

# 自定义搜索实现
import requests

def search_duckduckgo(query, max_results=5):
    """DuckDuckGo搜索"""
    url = f"https://api.duckduckgo.com/?q={query}&format=json"
    response = requests.get(url)
    data = response.json()

    results = []
    if data.get("AbstractText"):
        results.append({
            "title": "摘要",
            "snippet": data["AbstractText"],
            "url": data.get("AbstractURL", "")
        })

    for result in data.get("RelatedTopics", [])[:max_results]:
        if isinstance(result, dict) and "Text" in result:
            results.append({
                "title": result.get("FirstURL", ""),
                "snippet": result["Text"],
                "url": result.get("FirstURL", "")
            })

    return results

# 注册工具
search_tool = Tool(
    name="web_search",
    description="搜索网络获取信息",
    func=search_duckduckgo
)
```

### 2.2 知识搜索

```python
"""
知识图谱搜索
"""

import wikipedia

def wikipedia_search(query, sentences=3):
    """Wikipedia搜索"""
    try:
        result = wikipedia.summary(query, sentences=sentences)
        return {
            "title": query,
            "summary": result,
            "url": wikipedia.page(query).url
        }
    except wikipedia.DisambiguationError as e:
        return {"error": f"歧义词，可选: {e.options[:5]}"}
    except wikipedia.PageError:
        return {"error": "未找到页面"}

wiki_tool = Tool(
    name="wikipedia",
    description="在维基百科搜索信息",
    func=wikipedia_search
)
```

```python
"""
SQL数据库工具

from langchain.tools import SQLDatabaseTool
from langchain.sql_database import SQLDatabase

# 连接数据库
db = SQLDatabase.from_uri("sqlite:///example.db")

# 创建工具
sql_tool = SQLDatabaseTool(db=db)

# 自定义SQL工具
class DatabaseTool:
    """数据库工具"""

    def __init__(self, connection_string):
        import sqlite3
        self.conn = sqlite3.connect(connection_string)

    def query(self, sql):
        """执行SQL查询"""
        try:
            cursor = self.conn.cursor()
            cursor.execute(sql)
            columns = [desc[0] for desc in cursor.description]
            rows = cursor.fetchall()

            result = []
            for row in rows:
                result.append(dict(zip(columns, row)))

            return result

        except Exception as e:
            return {"error": str(e)}

    def get_schema(self):
        """获取数据库结构"""
        cursor = self.conn.cursor()
        cursor.execute("SELECT name FROM sqlite_master WHERE type='table'")
        tables = cursor.fetchall()

        schema = {}
        for (table_name,) in tables:
            cursor.execute(f"PRAGMA table_info({table_name})")
            columns = cursor.fetchall()
            schema[table_name] = [
                {"name": col[1], "type": col[2]}
                for col in columns
            ]

        return schema

# 使用
db_tool = DatabaseTool("example.db")

# 定义Function Calling工具

                "sql": {
                    "type": "string",
```

### 3.2 NoSQL数据库

```python
"""
MongoDB工具
"""

from pymongo import MongoClient

class MongoDBTool:
    """MongoDB工具"""

    def __init__(self, uri, database):
        self.client = MongoClient(uri)
        self.db = self.client[database]

    def find(self, collection, query, limit=10):
        """查询文档"""
        results = self.db[collection].find(query).limit(limit)
        return list(results)

    def insert(self, collection, document):
        """插入文档"""
        result = self.db[collection].insert_one(document)
        return {"inserted_id": str(result.inserted_id)}

    def update(self, collection, query, update):
        """更新文档"""
        result = self.db[collection].update_many(query, {"$set": update})
        return {"modified_count": result.modified_count}

# 工具定义

                    "type": "object",
```

### 4.1 REST API调用

```python
"""
REST API工具
"""

import requests

class APITool:
    """API调用工具"""

    def __init__(self, base_url, headers=None):
        self.base_url = base_url
        self.headers = headers or {}

    def get(self, endpoint, params=None):
        """GET请求"""
        response = requests.get(
            f"{self.base_url}{endpoint}",
            params=params,
            headers=self.headers
        )
        return response.json()

    def post(self, endpoint, data=None):
        """POST请求"""
        response = requests.post(
            f"{self.base_url}{endpoint}",
            json=data,
            headers=self.headers
        )
        return response.json()

# 天气API工具
class WeatherAPI:
    """天气API"""

    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://api.openweathermap.org/data/2.5"

    def get_weather(self, city):
        """获取天气"""
        params = {
            "q": city,
            "appid": self.api_key,
            "units": "metric"
        }
        response = requests.get(
            f"{self.base_url}/weather",
            params=params
        )
        data = response.json()

        return {
            "city": city,
            "temperature": data["main"]["temp"],
            "description": data["weather"][0]["description"],
            "humidity": data["main"]["humidity"]
        }

# 工具定义
weather_tool_definition = {
```

```python
# 新闻API
class NewsAPI:
    """新闻API"""

    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://newsapi.org/v2"

    def get_top_headlines(self, country="us", category=None):
        """获取头条新闻"""
        params = {
            "apiKey": self.api_key,
            "country": country
        }
        if category:
            params["category"] = category

        response = requests.get(
            f"{self.base_url}/top-headlines",
            params=params
        )
        return response.json()

    def search_news(self, query):
        """搜索新闻"""
        params = {
            "apiKey": self.api_key,
            "q": query
        }
        response = requests.get(
            f"{self.base_url}/everything",
            params=params
        )
        return response.json()

# 汇率API
class ExchangeRateAPI:
    """汇率API"""

    def __init__(self, api_key):
        self.api_key = api_key

    def get_rate(self, from_currency, to_currency):
        """获取汇率"""
        url = f"https://api.exchangerate-api.com/v4/latest/{from_currency}"
        response = requests.get(url)
        data = response.json()
        return {
            "from": from_currency,
            "to": to_currency,
            "rate": data["rates"].get(to_currency)
        }

    def convert(self, amount, from_currency, to_currency):
        """货币转换"""
        rate_info = self.get_rate(from_currency, to_currency)
        converted = amount * rate_info["rate"]
        return {
            "amount": amount,
            "from": from_currency,
            "to": to_currency,
            "result": round(converted, 2)
        }
```

```python
import math

def calculate(expression):
    """数学计算"""
    try:
        # 安全计算
        allowed_names = {
            "abs": abs,
            "round": round,
            "min": min,
            "max": max,
            "sum": sum,
            "sqrt": math.sqrt,
            "sin": math.sin,

            "log": math.log,
            "exp": math.exp,
            "pi": math.pi,
            "e": math.e
        }

        return {"error": str(e)}
```

### 5.2 代码执行

```python
"""
代码执行工具
"""

import subprocess
import tempfile

class CodeExecutor:
    """代码执行器"""

    def execute_python(self, code, timeout=10):
        """执行Python代码"""
        try:
            with tempfile.NamedTemporaryFile(
                mode='w',
                suffix='.py',
                delete=False
            ) as f:
                f.write(code)
                temp_file = f.name

            result = subprocess.run(
                ['python', temp_file],
                capture_output=True,
                text=True,
                timeout=timeout
            )

            import os
            os.unlink(temp_file)

            return {
                "stdout": result.stdout,
                "stderr": result.stderr,
                "returncode": result.returncode
            }

        except subprocess.TimeoutExpired:
            return {"error": "执行超时"}
        except Exception as e:
            return {"error": str(e)}

    def execute_javascript(self, code, timeout=10):
        """执行JavaScript代码"""
        try:
            result = subprocess.run(
                ['node', '-e', code],
                capture_output=True,
                text=True,
                timeout=timeout
            )

            return {
                "stdout": result.stdout,
                "stderr": result.stderr,
                "returncode": result.returncode
            }

        except Exception as e:
            return {"error": str(e)}

# 工具定义
code_executor_tool = {
    "type": "function",
    "function": {
        "name": "execute_python",
        "description": "执行Python代码并返回结果",

            "type": "object",

                "code": {
                    "type": "string",
                    "description": "要执行的Python代码"
                }
            },
            "required": ["code"]
```

```python
"""
文件操作工具
"""

import os

class FileTool:
    """文件工具"""

    def read_file(self, filepath):
        """读取文件"""
        try:
            with open(filepath, 'r', encoding='utf-8') as f:
                return f.read()
        except Exception as e:
            return {"error": str(e)}

    def write_file(self, filepath, content):
        """写入文件"""
        try:
            with open(filepath, 'w', encoding='utf-8') as f:
                f.write(content)
            return {"success": True, "path": filepath}
        except Exception as e:
            return {"error": str(e)}

    def list_directory(self, path="."):
        """列出目录"""
        try:
            items = os.listdir(path)
            return {
                "path": os.path.abspath(path),
                "items": items
            }
        except Exception as e:
            return {"error": str(e)}

    def get_file_info(self, filepath):
        """获取文件信息"""
        try:
            stat = os.stat(filepath)
            return {
                "path": filepath,
                "size": stat.st_size,
                "modified": stat.st_mtime,
                "is_file": os.path.isfile(filepath),
                "is_dir": os.path.isdir(filepath)
            }
        except Exception as e:
            return {"error": str(e)}

# 工具定义
file_tools = [
    {
        "type": "function",
        "function": {
            "name": "read_file",
```

## 七、工具注册与使用

### 7.1 工具管理器

```python
"""
工具管理器
"""

class ToolManager:
    """工具管理器"""

    def __init__(self):
        self.tools = {}
        self.definitions = []

    def register_tool(self, name, function, definition):
        """注册工具"""
        self.tools[name] = function
        self.definitions.append(definition)

    def execute(self, name, **kwargs):
        """执行工具"""
        if name not in self.tools:
            return {"error": f"工具 {name} 不存在"}

        try:
            result = self.tools[name](**kwargs)
            return result
        except Exception as e:
            return {"error": str(e)}

    def get_definitions(self):
        """获取工具定义列表"""
        return self.definitions

# 使用
manager = ToolManager()

# 注册搜索工具
manager.register_tool(
    "web_search",
    search_duckduckgo,
    {
        "type": "function",
        "function": {
            "name": "web_search",
            "description": "搜索网络获取信息",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "搜索查询"}
                },
                "required": ["query"]
            }
        }
    }
)

# 注册天气工具
weather_api = WeatherAPI("your-api-key")

    "get_weather",
```
