```python
"""
工具权限管理
"""

from enum import Enum
from typing import Set, List

class Permission(Enum):
    """权限类型"""
    READ = "read"
    WRITE = "write"
    EXECUTE = "execute"
    DELETE = "delete"

class Role:
    """角色定义"""

    def __init__(self, name: str, permissions: Set[Permission]):
        self.name = name
        self.permissions = permissions

# 预定义角色
ROLES = {
    "guest": Role("guest", {Permission.READ}),
    "user": Role("user", {Permission.READ, Permission.EXECUTE}),
    "admin": Role("admin", {Permission.READ, Permission.WRITE, Permission.EXECUTE, Permission.DELETE})
}

class ToolPermissionManager:
    """工具权限管理器"""

    def __init__(self):
        self.tool_permissions = {}
        self.user_roles = {}

    def set_tool_permission(self, tool_name: str, required_permission: Permission):
        """设置工具所需权限"""
        self.tool_permissions[tool_name] = required_permission

    def assign_role(self, user_id: str, role_name: str):
        """分配用户角色"""
        self.user_roles[user_id] = ROLES.get(role_name, ROLES["guest"])

    def check_permission(self, user_id: str, tool_name: str) -> bool:
        """检查权限"""
        role = self.user_roles.get(user_id, ROLES["guest"])
        required = self.tool_permissions.get(tool_name)

        if not required:
            return True  # 无权限要求的工具

        return required in role.permissions

# 使用
manager = ToolPermissionManager()
manager.set_tool_permission("delete_file", Permission.DELETE)
manager.set_tool_permission("execute_code", Permission.EXECUTE)
manager.assign_role("user_123", "user")

# 检查权限
can_execute = manager.check_permission("user_123", "execute_code")  # True
can_delete = manager.check_permission("user_123", "delete_file")    # False
```

### 2.2 访问控制

```python
"""
访问控制列表（ACL）
"""

class AccessControlList:
    """访问控制列表"""

    def __init__(self):
        self.acl = {}

    def grant(self, user_id: str, tool_name: str, conditions: dict = None):
        """授权"""
        if tool_name not in self.acl:
            self.acl[tool_name] = {}

        self.acl[tool_name][user_id] = {
            "granted": True,
            "conditions": conditions or {}
        }

    def deny(self, user_id: str, tool_name: str):
        """拒绝"""
        if tool_name not in self.acl:
            self.acl[tool_name] = {}

        self.acl[tool_name][user_id] = {"granted": False}

    def check_access(self, user_id: str, tool_name: str, context: dict = None) -> bool:
        """检查访问权限"""
        if tool_name not in self.acl:
            return False

        user_acl = self.acl[tool_name].get(user_id)
        if not user_acl:
            return False

        if not user_acl["granted"]:
            return False

        # 检查条件
        conditions = user_acl.get("conditions", {})
        if conditions and context:
            return self._check_conditions(conditions, context)

        return True

    def _check_conditions(self, conditions: dict, context: dict) -> bool:
        """检查条件"""
        for key, value in conditions.items():
            if context.get(key) != value:
                return False
        return True

# 使用
acl = AccessControlList()
acl.grant("user_123", "database_query", {"allowed_tables": ["public_data"]})
acl.deny("user_123", "admin_tools")

# 检查
can_access = acl.check_access("user_123", "database_query", {"allowed_tables": ["public_data"]})
```

### 3.1 参数验证

```python
"""
工具参数验证
"""

from typing import Any, Dict, List
import re

class ParameterValidator:
    """参数验证器"""

    def __init__(self, schema: Dict):
        self.schema = schema

    def validate(self, params: Dict) -> tuple[bool, str]:
        """验证参数"""
        # 检查必填参数
        required = self.schema.get("required", [])
        for field in required:
            if field not in params:
                return False, f"缺少必填参数: {field}"

        # 检查参数类型和值
        properties = self.schema.get("properties", {})
        for key, value in params.items():
            if key in properties:
                prop_schema = properties[key]
                valid, error = self._validate_property(key, value, prop_schema)
                if not valid:
                    return False, error

        return True, ""

    def _validate_property(self, key: str, value: Any, schema: Dict) -> tuple[bool, str]:
        """验证单个属性"""
        # 类型检查
        expected_type = schema.get("type")
        if expected_type:
            type_valid, error = self._check_type(value, expected_type)
            if not type_valid:
                return False, f"参数 '{key}' {error}"

        # 枚举检查
        if "enum" in schema:
            if value not in schema["enum"]:
                return False, f"参数 '{key}' 必须是: {schema['enum']}"

        # 范围检查
        if "minimum" in schema and value < schema["minimum"]:
            return False, f"参数 '{key}' 不能小于 {schema['minimum']}"
        if "maximum" in schema and value > schema["maximum"]:
            return False, f"参数 '{key}' 不能大于 {schema['maximum']}"

        # 长度检查
        if "minLength" in schema and len(str(value)) < schema["minLength"]:
            return False, f"参数 '{key}' 长度不足"
        if "maxLength" in schema and len(str(value)) > schema["maxLength"]:
            return False, f"参数 '{key}' 长度超限"

        # 正则检查
        if "pattern" in schema:
            if not re.match(schema["pattern"], str(value)):
                return False, f"参数 '{key}' 格式不正确"

        return True, ""

    def _check_type(self, value: Any, expected_type: str) -> tuple[bool, str]:
        """检查类型"""
        type_checks = {
            "string": lambda v: isinstance(v, str),
            "integer": lambda v: isinstance(v, int) and not isinstance(v, bool),
            "number": lambda v: isinstance(v, (int, float)),
            "boolean": lambda v: isinstance(v, bool),
            "array": lambda v: isinstance(v, list),
            "object": lambda v: isinstance(v, dict)
        }

        checker = type_checks.get(expected_type)
        if checker and not checker(value):
            return False, f"应为 {expected_type} 类型"

        return True, ""

# 使用示例
schema = {
    "type": "object",
    "properties": {
        "query": {
            "type": "string",
            "minLength": 1,
            "maxLength": 1000
        },
        "limit": {
            "type": "integer",
            "minimum": 1,
            "maximum": 100
        },
        "format": {
            "type": "string",
            "enum": ["json", "csv"]
        }
    },
    "required": ["query"]
}

validator = ParameterValidator(schema)
valid, error = validator.validate({"query": "test", "limit": 50})
```

### 3.2 注入防护

```python
"""
注入攻击防护
"""

class InjectionProtector:
    """注入防护器"""

    # SQL注入关键词
    SQL_KEYWORDS = [
        "SELECT", "INSERT", "UPDATE", "DELETE", "DROP",
        "UNION", "EXEC", "EXECUTE", "xp_", "sp_"
    ]

    # 命令注入字符
    COMMAND_CHARS = ["|", ";", "&", "$", "`", "\n", "\r"]

    def sanitize_sql(self, value: str) -> str:
        """清理SQL输入"""
        # 转义单引号
        value = value.replace("'", "''")

        # 检查危险关键词
        upper_value = value.upper()
        for keyword in self.SQL_KEYWORDS:
            if keyword in upper_value:
                raise ValueError(f"检测到潜在的SQL注入: {keyword}")

        return value

    def sanitize_command(self, value: str) -> str:
        """清理命令输入"""
        for char in self.COMMAND_CHARS:
            if char in value:
                raise ValueError(f"检测到潜在的命令注入字符: {char}")

        return value

    def sanitize_path(self, path: str) -> str:
        """清理路径输入"""
        # 防止路径遍历
        if ".." in path:
            raise ValueError("检测到路径遍历尝试")

        # 防止绝对路径
        if path.startswith("/") or ":" in path:
            raise ValueError("不允许使用绝对路径")

        return path

    def sanitize_json(self, value: str) -> str:
        """清理JSON输入"""
        import json
        try:
            # 验证JSON格式
            json.loads(value)
            return value
        except json.JSONDecodeError:
            raise ValueError("无效的JSON格式")

# 使用
protector = InjectionProtector()

try:
    safe_value = protector.sanitize_sql(user_input)
except ValueError as e:
    print(f"安全错误: {e}")
```

### 4.1 资源限制

```python
"""
资源限制控制
"""

import resource
import signal
from contextlib import contextmanager

class ResourceLimiter:
    """资源限制器"""

    def __init__(self, max_memory_mb=100, max_cpu_seconds=10, max_file_size_mb=10):
        self.max_memory = max_memory_mb * 1024 * 1024
        self.max_cpu = max_cpu_seconds
        self.max_file_size = max_file_size_mb * 1024 * 1024

    @contextmanager
    def limits(self):
        """设置资源限制"""
        # 设置CPU时间限制
        resource.setrlimit(resource.RLIMIT_CPU, (self.max_cpu, self.max_cpu))

        # 设置内存限制
        resource.setrlimit(resource.RLIMIT_AS, (self.max_memory, self.max_memory))

        # 设置文件大小限制
        resource.setrlimit(resource.RLIMIT_FSIZE, (self.max_file_size, self.max_file_size))

        try:
            yield
        finally:
            pass

    def limit_execution(self, func, *args, **kwargs):
        """限制执行"""
        with self.limits():
            return func(*args, **kwargs)

# 超时控制
class TimeoutError(Exception):
    pass

@contextmanager
def timeout(seconds):
    """超时上下文管理器"""
    def handler(signum, frame):
        raise TimeoutError("执行超时")

    original_handler = signal.signal(signal.SIGALRM, handler)
    signal.alarm(seconds)

    try:
        yield
    finally:
        signal.alarm(0)
        signal.signal(signal.SIGALRM, original_handler)

# 使用
limiter = ResourceLimiter(max_memory_mb=100, max_cpu_seconds=5)

try:
    with timeout(10):
        with limiter.limits():
            result = execute_tool(**params)
except TimeoutError:
    print("工具执行超时")
except resource.error as e:
    print(f"资源限制: {e}")
```

### 4.2 执行监控

```python
"""
执行监控
"""

import time
import logging

class ExecutionMonitor:
    """执行监控器"""

    def __init__(self):
        self.logger = logging.getLogger("tool_monitor")
        self.metrics = {}

    def start_execution(self, tool_name: str, params: dict):
        """开始执行"""
        execution_id = f"{tool_name}_{time.time()}"
        self.metrics[execution_id] = {
            "tool": tool_name,
            "start_time": time.time(),
            "params": params
        }

        self.logger.info(f"开始执行: {tool_name}")
        return execution_id

    def end_execution(self, execution_id: str, success: bool, result: any):
        """结束执行"""
        if execution_id not in self.metrics:
            return

        metrics = self.metrics[execution_id]
        metrics["end_time"] = time.time()
        metrics["duration"] = metrics["end_time"] - metrics["start_time"]
        metrics["success"] = success
        metrics["result_size"] = len(str(result))

        self.logger.info(
            f"执行完成: {metrics['tool']}, "
            f"耗时: {metrics['duration']:.2f}s, "
            f"成功: {success}"
        )

        return metrics

    def check_anomalies(self):
        """检查异常"""
        anomalies = []

        for exec_id, metrics in self.metrics.items():
            # 检查执行时间过长
            if metrics.get("duration", 0) > 30:
                anomalies.append({
                    "type": "long_execution",
                    "execution_id": exec_id,
                    "duration": metrics["duration"]
                })

            # 检查频繁失败
            if not metrics.get("success", True):
                anomalies.append({
                    "type": "execution_failure",
                    "execution_id": exec_id
                })

        return anomalies

# 使用
monitor = ExecutionMonitor()

exec_id = monitor.start_execution("web_search", {"query": "test"})
try:
    result = execute_search("test")
    monitor.end_execution(exec_id, True, result)
except Exception as e:
    monitor.end_execution(exec_id, False, str(e))
```

### 5.1 审计记录

```python
"""
审计日志系统
"""

import json
from datetime import datetime

class AuditLogger:
    """审计日志"""

    def __init__(self, log_file="audit.log"):
        self.log_file = log_file

    def log_tool_call(
        self,
        user_id: str,
        tool_name: str,
        params: dict,
        result: any,
        success: bool
    ):
        """记录工具调用"""
        entry = {
            "timestamp": datetime.now().isoformat(),
            "event": "tool_call",
            "user_id": user_id,
            "tool_name": tool_name,
            "params": self._sanitize_params(params),
            "result_preview": str(result)[:200] if result else None,
            "success": success
        }

        self._write_log(entry)

    def log_permission_denied(self, user_id: str, tool_name: str, reason: str):
        """记录权限拒绝"""
        entry = {
            "timestamp": datetime.now().isoformat(),
            "event": "permission_denied",
            "user_id": user_id,
            "tool_name": tool_name,
            "reason": reason
        }

        self._write_log(entry)

    def log_security_event(self, event_type: str, details: dict):
        """记录安全事件"""
        entry = {
            "timestamp": datetime.now().isoformat(),
            "event": "security_event",
            "event_type": event_type,
            "details": details
        }

        self._write_log(entry)

    def _sanitize_params(self, params: dict) -> dict:
        """清理敏感参数"""
        sensitive_keys = ["password", "token", "api_key", "secret"]
        sanitized = {}

        for key, value in params.items():
            if key.lower() in sensitive_keys:
                sanitized[key] = "***REDACTED***"
            else:
                sanitized[key] = value

        return sanitized

    def _write_log(self, entry: dict):
        """写入日志"""
        with open(self.log_file, "a") as f:
            f.write(json.dumps(entry) + "\n")

# 使用
audit = AuditLogger()

audit.log_tool_call(
    user_id="user_123",
    tool_name="database_query",
    params={"table": "users", "filter": {"id": 1}},
    result=[{"id": 1, "name": "John"}],
    success=True
)

audit.log_permission_denied(
    user_id="user_123",
    tool_name="admin_action",
    reason="需要管理员权限"
)
```

```python
        except Exception as e:
            self.monitor.end_execution(exec_id, False, str(e))

                user_id, self.tool.name, kwargs, str(e), False
            )
            return {"error": str(e)}

        """检查权限"""
        # 实现权限检查逻辑
        return True

    def _sanitize_inputs(self, params: dict) -> dict:
        """清理输入"""
        sanitized = {}
        for key, value in params.items():
            if isinstance(value, str):
                # 根据参数类型选择清理方法
                if "sql" in key.lower():
                    value = self.protector.sanitize_sql(value)
                elif "path" in key.lower():
                    value = self.protector.sanitize_path(value)
                elif "command" in key.lower():
```

## 总结

### 核心要点

1. **权限管理**：角色权限、访问控制
2. **输入验证**：参数验证、注入防护
3. **执行控制**：资源限制、超时控制
4. **监控审计**：执行监控、审计日志
5. **最佳实践**：安全检查清单、安全包装

### 安全原则

- 最小权限原则
- 深度防御
- 输入验证
- 失败安全
- 审计追踪
