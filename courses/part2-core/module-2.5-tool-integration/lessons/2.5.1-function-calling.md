# 2.5.1 函数调用基础

> 掌握Function Calling，让LLM调用外部函数

## 一、什么是Function Calling？

### 1.1 定义

**Function Calling** 是让LLM能够调用预定义函数的能力。

**工作原理**：
1. 定义可用函数
2. LLM分析用户输入
3. LLM决定是否调用函数
4. LLM生成函数调用参数
5. 执行函数并返回结果
6. LLM基于结果生成回答

### 1.2 与普通对话对比

```
普通对话：
用户: "北京今天天气怎么样？"
LLM: "抱歉，我无法获取实时天气信息。"

Function Calling：
用户: "北京今天天气怎么样？"
LLM: 调用 get_weather("北京")
函数返回: {"temp": 25, "weather": "晴"}
LLM: "北京今天天气晴朗，温度25°C。"
```

## 三、函数调用流程

### 3.1 完整流程示例

```python
from openai import OpenAI
import json

client = OpenAI()

def chat_with_tools(user_message):
    """带工具调用的对话"""

    # 第一次调用：让LLM决定是否调用工具
    response = client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "user", "content": user_message}
        ],
        tools=tools,
        tool_choice="auto"  # 让模型自动决定
    )

    message = response.choices[0].message

    # 检查是否需要调用工具
    if message.tool_calls:
        print(f"LLM决定调用工具: {message.tool_calls[0].function.name}")

        # 准备消息历史
        messages = [
            {"role": "user", "content": user_message},
            message  # 第一次的响应
        ]

        # 处理每个工具调用
        for tool_call in message.tool_calls:
            function_name = tool_call.function.name
            function_args = json.loads(tool_call.function.arguments)

            print(f"参数: {function_args}")

            # 执行函数
            result = execute_function(function_name, function_args)
            print(f"结果: {result}")

            # 添加工具响应
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": str(result)
            })

        # 第二次调用：让LLM基于工具结果生成回答
        final_response = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=messages
        )

        return final_response.choices[0].message.content

    # 不需要调用工具，直接返回
    return message.content

def execute_function(name, args):
    """执行函数"""
    if name == "get_weather":
        return get_weather(**args)
    elif name == "search_web":
        return search_web(**args)
    elif name == "calculate":
        return calculate(**args)
    else:
        return f"未知函数: {name}"
```

```python
def handle_multiple_tool_calls(user_message):
    """处理多个工具调用"""

    response = client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[{"role": "user", "content": user_message}],
        tools=tools
    )

    message = response.choices[0].message

    if message.tool_calls:
        messages = [
            {"role": "user", "content": user_message},
            message
        ]

        # 并行执行所有工具调用
        for tool_call in message.tool_calls:
            result = execute_function(
                tool_call.function.name,
                json.loads(tool_call.function.arguments)
            )

            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": str(result)
            })

        # 生成最终回答
        final = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=messages
        )

        return final.choices[0].message.content

    return message.content
```

## 四、参数验证

### 4.1 类型验证

```python
from pydantic import BaseModel, validator
from typing import Optional, Literal

class WeatherParams(BaseModel):
    """天气查询参数"""
    city: str
    unit: Optional[Literal["celsius", "fahrenheit"]] = "celsius"

    @validator('city')
    def validate_city(cls, v):
        if not v or not v.strip():
            raise ValueError('城市名称不能为空')
        return v.strip()

# 使用
def validate_and_execute(function_name, arguments):
    """验证参数并执行"""
    try:
        if function_name == "get_weather":
            params = WeatherParams(**arguments)
            return get_weather(params.city, params.unit)
    except Exception as e:
        return f"参数验证失败: {str(e)}"
```

### 4.2 完整验证流程

```python
import json
from typing import Any, Dict

def validate_tool_arguments(
    tool_name: str,
    arguments: Dict[str, Any],
    tool_definitions: list
) -> tuple[bool, str, Dict]:
    """
    验证工具参数

    Returns:
        (是否有效, 错误信息, 验证后的参数)
    """
    # 查找工具定义
    tool_def = None
    for tool in tool_definitions:
        if tool["function"]["name"] == tool_name:
            tool_def = tool["function"]
            break

    if not tool_def:
        return False, f"未知工具: {tool_name}", {}

    params_schema = tool_def.get("parameters", {})
    required = params_schema.get("required", [])
    properties = params_schema.get("properties", {})

    validated = {}

    # 检查必填参数
    for param in required:
        if param not in arguments:
            return False, f"缺少必填参数: {param}", {}

    # 验证参数类型
    for param, value in arguments.items():
        if param not in properties:
            continue  # 允许额外参数

        prop_def = properties[param]
        expected_type = prop_def.get("type")

        # 类型检查
        if expected_type == "string" and not isinstance(value, str):
            return False, f"参数 '{param}' 应为字符串", {}
        elif expected_type == "integer" and not isinstance(value, int):
            return False, f"参数 '{param}' 应为整数", {}
        elif expected_type == "number" and not isinstance(value, (int, float)):
            return False, f"参数 '{param}' 应为数字", {}
        elif expected_type == "boolean" and not isinstance(value, bool):
            return False, f"参数 '{param}' 应为布尔值", {}

        # 枚举检查
        if "enum" in prop_def and value not in prop_def["enum"]:
            return False, f"参数 '{param}' 应为: {prop_def['enum']}", {}

        validated[param] = value

    return True, "", validated
```

## 六、实战示例

### 6.1 完整的Agent工具调用

```python
"""
完整的Agent工具调用示例
"""

from openai import OpenAI
import json

class ToolAgent:
    def __init__(self):
        self.client = OpenAI()
        self.tools = []
        self.functions = {}

    def register_tool(self, tool_definition, function):
        """注册工具"""
        self.tools.append(tool_definition)
        self.functions[tool_definition["function"]["name"]] = function

    def run(self, user_input):
        """运行Agent"""
        messages = [
            {"role": "system", "content": "你是一个有用的助手，可以使用工具完成任务。"},
            {"role": "user", "content": user_input}
        ]

        while True:
            # 调用LLM
            response = self.client.chat.completions.create(
                model="gpt-3.5-turbo",
                messages=messages,
                tools=self.tools if self.tools else None,
                tool_choice="auto" if self.tools else "none"
            )

            message = response.choices[0].message
            messages.append(message)

            # 检查是否需要调用工具
            if message.tool_calls:
                for tool_call in message.tool_calls:
                    func_name = tool_call.function.name
                    func_args = json.loads(tool_call.function.arguments)

                    print(f"[调用工具] {func_name}({func_args})")

                    # 执行工具
                    if func_name in self.functions:
                        result = self.functions[func_name](**func_args)
                    else:
                        result = f"未知工具: {func_name}"

                    # 添加结果
                    messages.append({
                        "role": "tool",
                        "tool_call_id": tool_call.id,
                        "content": str(result)
                    })

                # 继续循环，让LLM处理工具结果

            else:
                # 没有工具调用，返回最终回答
                return message.content

# 使用示例
agent = ToolAgent()

# 注册工具
agent.register_tool(
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "获取天气",
            "parameters": {
                "type": "object",
                "properties": {
                    "city": {"type": "string", "description": "城市"}
                },
                "required": ["city"]
            }
        }
    },
    lambda city: f"{city}的天气：晴，25°C"
)

# 运行
result = agent.run("北京今天天气怎么样？")
print(result)
```

## 总结

### 核心要点

1. **Function Calling** = 让LLM调用预定义函数
2. **工具定义**：name + description + parameters
3. **调用流程**：LLM决定 → 执行函数 → LLM生成回答
4. **参数验证**：类型检查、必填检查、枚举检查
5. **工具选择**：auto/none/强制特定工具
