# 2.1.3 提示词模板管理

> 构建可复用、可维护的提示词模板系统

## 一、为什么需要模板管理？

### 1.1 问题场景

**没有模板的问题**：
- 提示词散落在代码各处
- 难以维护和更新
- 格式不统一
- 无法复用

### 1.2 模板管理的优势

**使用模板的好处**：
- ✅ 统一管理，易于维护
- ✅ 可复用，提高效率
- ✅ 版本控制，追溯变更
- ✅ 动态生成，灵活适配

### 2.3 模板类设计

```python
class PromptTemplate:
    """提示词模板类"""

    def __init__(self, template, input_variables):
        """
        初始化模板

        Args:
            template: 模板字符串
            input_variables: 输入变量列表
        """
        self.template = template
        self.input_variables = input_variables

    def format(self, **kwargs):
        """填充模板"""
        # 验证变量
        missing = set(self.input_variables) - set(kwargs.keys())
        if missing:
            raise ValueError(f"缺少变量: {missing}")

        return self.template.format(**kwargs)

    def partial(self, **kwargs):
        """部分填充，返回新模板"""
        new_template = self.template.format(**kwargs)
        new_variables = [v for v in self.input_variables if v not in kwargs]
        return PromptTemplate(new_template, new_variables)

# 使用示例
template = PromptTemplate(
    template="将'{text}'翻译成{language}",
    input_variables=["text", "language"]
)

# 完整填充
prompt = template.format(text="Hello", language="中文")

# 部分填充
chinese_template = template.partial(language="中文")
prompt = chinese_template.format(text="Hello")
```

## 三、动态提示词构建

### 3.1 条件模板

```python
class DynamicPromptTemplate:
    """动态提示词模板"""

    def __init__(self):
        self.sections = {}

    def add_section(self, name, content, condition=None):
        """
        添加段落

        Args:
            name: 段落名称
            content: 内容
            condition: 条件函数
        """
        self.sections[name] = {
            "content": content,
            "condition": condition
        }

    def build(self, context):
        """构建提示词"""
        parts = []

        for name, section in self.sections.items():
            # 检查条件
            if section["condition"] is None or section["condition"](context):
                content = section["content"]
                if callable(content):
                    content = content(context)
                parts.append(content)

        return "\n\n".join(parts)

# 使用示例
def build_analysis_prompt():
    template = DynamicPromptTemplate()

    # 添加角色段落（始终包含）
    template.add_section(
        "role",
        "你是一个专业的数据分析师。"
    )

    # 添加背景段落（条件包含）
    template.add_section(
        "context",
        lambda ctx: f"背景信息：{ctx.get('background')}",
        condition=lambda ctx: ctx.get("background") is not None
    )

    # 添加任务段落
    template.add_section(
        "task",
        "请分析以下数据：\n{data}"
    )

    # 添加输出要求
    template.add_section(
        "output",
        "请以Markdown格式输出分析结果。"
    )

    return template

# 构建
template = build_analysis_prompt()
prompt = template.build({
    "background": "这是一个销售数据分析任务",
    "data": "..."
})
```

### 3.2 模板组合

```python
class PromptComposer:
    """提示词组合器"""

    def __init__(self):
        self.templates = {}

    def register(self, name, template):
        """注册模板"""
        self.templates[name] = template

    def compose(self, template_names, separator="\n\n"):
        """组合多个模板"""
        parts = []
        for name in template_names:
            if name in self.templates:
                parts.append(self.templates[name])
        return separator.join(parts)

# 使用示例
composer = PromptComposer()

# 注册基础模板
composer.register("role", "你是一个AI助手。")
composer.register("task", "请完成以下任务：{task}")
composer.register("format", "以JSON格式输出。")
composer.register("constraint", "回答不超过500字。")

# 组合模板
full_prompt = composer.compose(["role", "task", "format", "constraint"])
```

### 4.3 模板管理器

```python
import yaml
import os
from pathlib import Path

class PromptManager:
    """提示词模板管理器"""

    def __init__(self, templates_dir="prompts/templates"):
        self.templates_dir = Path(templates_dir)
        self.templates = {}
        self._load_all_templates()

    def _load_all_templates(self):
        """加载所有模板文件"""
        for yaml_file in self.templates_dir.rglob("*.yaml"):
            self._load_template(yaml_file)

    def _load_template(self, file_path):
        """加载单个模板"""
        with open(file_path, 'r', encoding='utf-8') as f:
            template_data = yaml.safe_load(f)

        name = template_data['name']
        self.templates[name] = {
            'template': template_data['template'],
            'variables': template_data.get('input_variables', []),
            'defaults': {
                v['name']: v.get('default')
                for v in template_data.get('input_variables', [])
                if 'default' in v
            }
        }

    def get(self, name):
        """获取模板"""
        if name not in self.templates:
            raise ValueError(f"模板 '{name}' 不存在")
        return self.templates[name]

    def render(self, name, **kwargs):
        """渲染模板"""
        template_data = self.get(name)

        # 合并默认值
        variables = {**template_data['defaults'], **kwargs}

        # 验证必填变量
        for var in template_data['variables']:
            if var.get('required', False) and var['name'] not in kwargs:
                raise ValueError(f"缺少必填变量: {var['name']}")

        # 渲染模板（简单实现）
        template = template_data['template']
        for key, value in variables.items():
            placeholder = "{{ " + key + " }}"
            if isinstance(value, list):
                value = "\n".join(f"- {v}" for v in value)
            template = template.replace(placeholder, str(value))

        return template

    def list_templates(self):
        """列出所有模板"""
        return list(self.templates.keys())

# 使用示例
manager = PromptManager()

# 渲染模板
prompt = manager.render(
    "assistant",
    role="Python编程专家",
    task="写一个计算斐波那契数列的函数"
)

print(prompt)
```

## 五、版本管理

### 5.1 版本控制策略

```python
class VersionedPromptTemplate:
    """版本化提示词模板"""

    def __init__(self, name):
        self.name = name
        self.versions = {}
        self.current_version = None

    def add_version(self, version, template, metadata=None):
        """添加版本"""
        self.versions[version] = {
            'template': template,
            'metadata': metadata or {},
            'created_at': datetime.now()
        }

        if self.current_version is None:
            self.current_version = version

    def get_version(self, version=None):
        """获取指定版本"""
        version = version or self.current_version
        if version not in self.versions:
            raise ValueError(f"版本 {version} 不存在")
        return self.versions[version]['template']

    def set_current_version(self, version):
        """设置当前版本"""
        if version not in self.versions:
            raise ValueError(f"版本 {version} 不存在")
        self.current_version = version

    def list_versions(self):
        """列出所有版本"""
        return [
            {
                'version': v,
                'metadata': data['metadata'],
                'created_at': data['created_at']
            }
            for v, data in self.versions.items()
        ]

# 使用示例
versioned_template = VersionedPromptTemplate("translator")

# 添加版本
versioned_template.add_version(
    version="1.0",
    template="翻译：{text}",
    metadata={"author": "Alice", "description": "基础翻译模板"}
)

versioned_template.add_version(
    version="2.0",
    template="请将以下文本翻译成{language}：\n{text}\n\n要求：保持原意，语言流畅。",
    metadata={"author": "Bob", "description": "增强版翻译模板"}
)

# 使用当前版本
prompt = versioned_template.get_version().format(text="Hello", language="中文")
```
