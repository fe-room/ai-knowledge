## 一、向量表示

### 1.1 什么是向量？

**向量（Embedding）** = 将文本转换为数值向量，捕获语义信息。

```
文本 → Embedding模型 → 向量

例如：
"苹果是一种水果" → [0.1, 0.2, 0.3, ...]
"香蕉也是一种水果" → [0.1, 0.25, 0.28, ...]

相似文本 → 相似向量
```

### 1.2 向量的意义

```
向量维度：通常为384、768、1024、1536等

向量特性：
1. 相似语义 = 相似向量
2. 可以计算相似度
3. 支持向量搜索
4. 可以用于聚类
```

```python
"""
OpenAI Embedding
"""

from openai import OpenAI
```

## 三、向量相似度

### 3.1 相似度计算

```python
"""
向量相似度计算
"""

import numpy as np

def cosine_similarity(vec1, vec2):
    """余弦相似度"""
    dot_product = np.dot(vec1, vec2)
    norm1 = np.linalg.norm(vec1)
    norm2 = np.linalg.norm(vec2)
    return dot_product / (norm1 * norm2)

def euclidean_distance(vec1, vec2):
    """欧几里得距离"""
    return np.linalg.norm(vec1 - vec2)

def dot_product(vec1, vec2):
    """点积"""
    return np.dot(vec1, vec2)

# 使用示例
vec1 = np.array([0.1, 0.2, 0.3])
vec2 = np.array([0.15, 0.22, 0.31])

print(f"余弦相似度: {cosine_similarity(vec1, vec2)}")
print(f"欧几里得距离: {euclidean_distance(vec1, vec2)}")
print(f"点积: {dot_product(vec1, vec2)}")

# 文本相似度计算
def text_similarity(text1, text2, embedding_func):
    """文本相似度"""
    emb1 = embedding_func(text1)
    emb2 = embedding_func(text2)
    return cosine_similarity(emb1, emb2)

similarity = text_similarity(
    "苹果是水果",
    "香蕉是水果",
    get_embedding
)
print(f"文本相似度: {similarity}")
```

## 四、向量数据库

### 4.1 数据库选择

```
主流向量数据库：

1. Chroma（轻量级，易用）
2. Pinecone（云端托管）
3. Weaviate（开源，功能丰富）
4. Milvus（高性能，云原生）
5. FAISS（Facebook，纯向量索引）
6. Qdrant（开源，Rust实现）

选择建议：
- 开发测试：Chroma
- 生产环境：Pinecone、Weaviate、Milvus
- 本地部署：FAISS、Qdrant
```

### 4.2 Chroma

```python
"""
Chroma向量数据库
"""

from langchain.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

# 创建向量存储
embeddings = OpenAIEmbeddings()

# 从文档创建
from langchain.documents import Document

docs = [
    Document(page_content="Python是一种编程语言"),
    Document(page_content="JavaScript是一种脚本语言"),
    Document(page_content="向量数据库存储向量数据")
]

vectorstore = Chroma.from_documents(
    documents=docs,
    embedding=embeddings,
    persist_directory="./chroma_db"  # 持久化目录
)

# 检索
results = vectorstore.similarity_search("编程语言", k=2)
for doc in results:
    print(doc.page_content)

# 检索带分数
results_with_scores = vectorstore.similarity_search_with_score("编程", k=2)
for doc, score in results_with_scores:
    print(f"分数: {score}, 内容: {doc.page_content}")

# 添加新文档
vectorstore.add_documents([
    Document(page_content="机器学习是AI的一个分支")
])

# 删除文档
vectorstore.delete(ids=["doc_id_1"])
```

## 五、向量检索

### 5.1 基础检索

```python
"""
向量检索方式
"""

# 1. 相似度搜索
results = vectorstore.similarity_search(
    query="查询文本",
    k=5  # 返回前5个结果
)

# 2. 带分数的搜索
results = vectorstore.similarity_search_with_score(
    query="查询文本",
    k=5
)

# 3. 基于向量搜索
query_embedding = embeddings.embed_query("查询文本")
results = vectorstore.similarity_search_by_vector(
    embedding=query_embedding,
    k=5
)

# 4. 最大边际相关性（MMR）搜索
# 避免返回过于相似的结果
results = vectorstore.max_marginal_relevance_search(
    query="查询文本",
    k=5,
    fetch_k=20  # 先获取20个，再从中选择5个
)
```

### 5.2 过滤检索

```python
"""
带过滤条件的检索
"""

# Chroma过滤
results = vectorstore.similarity_search(
    query="查询",
    k=5,
    filter={
        "source": "document.pdf",
        "page": {"$gt": 10}
    }

# Pinecone过滤

    query="查询",
    k=5,
    filter={
        "category": "tech",
        "date": {"$gte": "2024-01-01"}

)
```

## 六、自定义向量存储

### 6.1 简单实现

```python
"""
简单的内存向量存储
"""

import numpy as np

class SimpleVectorStore:
    """简单向量存储"""

    def __init__(self):
        self.documents = []
        self.embeddings = []
        self.ids = []

    def add(self, doc_id, document, embedding):
        """添加文档"""
        self.ids.append(doc_id)
        self.documents.append(document)
        self.embeddings.append(embedding)

    def search(self, query_embedding, k=5):
        """检索"""
        similarities = [
            cosine_similarity(query_embedding, emb)
            for emb in self.embeddings
        ]

        # 排序获取top-k
        top_indices = np.argsort(similarities)[-k:][::-1]

        return [
            {
                "id": self.ids[i],
                "document": self.documents[i],
                "score": similarities[i]
            }
            for i in top_indices
        ]

# 使用
store = SimpleVectorStore()

docs = ["文档1", "文档2", "文档3"]
for i, doc in enumerate(docs):
    emb = get_embedding(doc)
    store.add(f"doc_{i}", doc, emb)

# 检索
query_emb = get_embedding("查询")
results = store.search(query_emb, k=2)
for r in results:
    print(f"分数: {r['score']:.4f}, 内容: {r['document']}")
```

## 七、向量数据库管理

### 7.1 索引管理

```python
"""
向量索引管理
"""

class VectorIndexManager:
    """向量索引管理"""

    def __init__(self, vectorstore):
        self.vectorstore = vectorstore

    def update_index(self, new_docs):
        """更新索引"""
        self.vectorstore.add_documents(new_docs)

    def delete_by_filter(self, filter_dict):
        """删除文档"""
        self.vectorstore.delete(filter_dict)

    def rebuild_index(self, all_docs):
        """重建索引"""
        # 清空旧索引
        self.vectorstore.delete_all()
        # 重新添加
        self.vectorstore.add_documents(all_docs)

    def get_stats(self):
        """获取统计"""
        return {
            "total_documents": len(self.vectorstore.get()["ids"]),
            "last_update": "..."
        }
```
