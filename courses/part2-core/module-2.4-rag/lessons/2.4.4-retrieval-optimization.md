## 二、查询优化

### 2.1 查询扩展

```python
"""
查询扩展策略
"""

class QueryExpansion:
    """查询扩展"""

    def __init__(self, llm):
        self.llm = llm

    def expand_query(self, original_query):
        """扩展查询"""
        prompt = f"""
原始查询：{original_query}

请生成3个相关但不同的查询，帮助搜索更多信息。

输出JSON格式：
```json
{{"expanded_queries": ["查询1", "查询2", "查询3"]}}
```
"""
        response = self.llm.invoke(prompt)
        import json
        expanded = json.loads(response.content)
        return expanded["expanded_queries"]

    def multi_query_search(self, vectorstore, query, k=3):
        """多查询检索"""
        expanded_queries = self.expand_query(query)

        all_results = []
        for q in expanded_queries:
            results = vectorstore.similarity_search(q, k=k)
            all_results.extend(results)

        # 去重
        unique_results = self._deduplicate(all_results)
        return unique_results[:k*3]

    def _deduplicate(self, results):
        """去重"""
        seen = set()
        unique = []
        for doc in results:
            content_hash = hash(doc.page_content)
            if content_hash not in seen:
                seen.add(content_hash)
                unique.append(doc)
        return unique
```

class QueryRewriter:
    """查询改写"""

    def rewrite_for_search(self, query):

上下文：{context[:500]}

"""
        else:
            prompt = f"""
问题：{query}

改写问题使其更清晰、具体：
"""

        return self.llm.invoke(prompt).content

### 2.3 HyDE（假设文档）

```python
"""
HyDE - Hypothetical Document Embeddings
"""

class HyDE:
    """假设文档检索"""

    def __init__(self, llm, embeddings):
        self.llm = llm
        self.embeddings = embeddings

    def generate_hypothetical_doc(self, query):
        """生成假设文档"""
        prompt = f"""
问题：{query}

请生成一个可能包含答案的文档片段：
"""
        response = self.llm.invoke(prompt)
        return response.content

    def search_with_hyde(self, vectorstore, query, k=5):
        """使用假设文档检索"""
        # 生成假设文档
        hypo_doc = self.generate_hypothetical_doc(query)

        # 用假设文档的向量检索
        hypo_embedding = self.embeddings.embed_query(hypo_doc)
        results = vectorstore.similarity_search_by_vector(
            hypo_embedding,
            k=k
        )

        return results, hypo_doc
```

## 三、混合检索

### 3.1 关键词+向量

```python
"""
混合检索：关键词 + 向量
"""

from rank_bm25 import BM25Okapi

class HybridSearch:
    """混合检索"""

    def __init__(self, documents, vectorstore):
        self.documents = documents
        self.vectorstore = vectorstore

        # 构建BM25索引
        tokenized = [doc.split() for doc in documents]
        self.bm25 = BM25Okapi(tokenized)

    def search(self, query, k=5, alpha=0.5):
        """
        alpha: 向量检索权重（1-alpha为关键词权重）
        """
        # 向量检索
        vector_results = self.vectorstore.similarity_search_with_score(query, k=k*2)

        # 关键词检索
        tokenized_query = query.split()
        bm25_scores = self.bm25.get_scores(tokenized_query)
        bm25_results = sorted(
            zip(self.documents, bm25_scores),
            key=lambda x: x[1],
            reverse=True
        )[:k*2]

        # 合并结果
        combined = self._combine_results(
            vector_results,
            bm25_results,
            alpha
        )

        return combined[:k]

    def _combine_results(self, vector_results, keyword_results, alpha):
        """合并和排序"""
        scores = {}

        # 向量结果
        for doc, score in vector_results:
            key = hash(doc.page_content)
            scores[key] = scores.get(key, 0) + alpha * (1 - score)  # 向量分数转换

        # 关键词结果
        for doc, score in keyword_results:
            key = hash(doc)
            scores[key] = scores.get(key, 0) + (1-alpha) * score

        # 排序
        sorted_items = sorted(scores.items(), key=lambda x: x[1], reverse=True)

        results = []
        for key, score in sorted_items:
            # 找到对应文档
            doc = self._find_doc_by_hash(key)
            if doc:
                results.append((doc, score))

        return results

    def _find_doc_by_hash(self, key):
        """根据hash找文档"""
        for doc in self.documents:
            if hash(doc) == key:
                return doc
        return None
```

## 四、重排序

### 4.1 基于LLM重排序

```python
"""
LLM重排序
"""

class LLMReranker:
    """LLM重排序"""

    def __init__(self, llm):
        self.llm = llm

    def rerank(self, query, documents, k=5):
        """重排序"""
        prompt = f"""
问题：{query}

文档列表：
{chr(10).join([f"{i+1}. {doc[:200]}..." for i, doc in enumerate(documents)])}

请根据与问题的相关性，重新排序文档。
输出排序后的文档编号，从最相关到最不相关。

排序结果（JSON）：
```json
{{"ranking": [编号1, 编号2, ...]}}
```
"""
        response = self.llm.invoke(prompt)
        import json
        ranking = json.loads(response.content)["ranking"]

        # 重新排序
        reranked = [documents[i-1] for i in ranking[:k]]
        return reranked
```

### 4.2 Cross-Encoder重排序

```python
"""
Cross-Encoder重排序
"""

from sentence_transformers import CrossEncoder

class CrossEncoderReranker:
    """Cross-Encoder重排序"""

    def __init__(self, model_name="cross-encoder/ms-marco-MiniLM-L-6-v2"):
        self.model = CrossEncoder(model_name)

    def rerank(self, query, documents, k=5):
        """重排序"""
        # 构建查询-文档对
        pairs = [(query, doc) for doc in documents]

        # 计算分数
        scores = self.model.predict(pairs)

        # 排序
        ranked_indices = np.argsort(scores)[-k:][::-1]
        reranked = [documents[i] for i in ranked_indices]

        return reranked, [scores[i] for i in ranked_indices]
```

## 五、上下文优化

### 5.1 上下文压缩

```python
"""
上下文压缩
"""

class ContextCompressor:
    """上下文压缩"""

    def __init__(self, llm):
        self.llm = llm

    def compress(self, query, documents):
        """压缩上下文"""
        context = "\n\n".join([doc[:500] for doc in documents])

        prompt = f"""
问题：{query}

原始上下文：
{context}

请压缩上下文，只保留与问题相关的信息：
"""

        response = self.llm.invoke(prompt)
        return response.content

    def compress_documents(self, query, documents):
        """压缩每个文档"""
        compressed_docs = []

        for doc in documents:
            prompt = f"""
问题：{query}

文档片段：
{doc}

提取与问题相关的关键信息：
"""
            compressed = self.llm.invoke(prompt).content
            compressed_docs.append(compressed)

        return compressed_docs
```

### 5.2 相关片段选择

```python
"""
选择最相关的片段
"""

class RelevantSnippetSelector:
    """相关片段选择"""

    def select_snippets(self, query, document, max_length=500):
        """从文档中选择相关片段"""
        # 分段
        paragraphs = document.split('\n\n')

        # 计算每段相关性
        relevant_snippets = []
        current_length = 0

        for para in paragraphs:
            # 简化判断：包含关键词
            if self._is_relevant(query, para):
                if current_length + len(para) <= max_length:
                    relevant_snippets.append(para)
                    current_length += len(para)

        return "\n\n".join(relevant_snippets)

    def _is_relevant(self, query, paragraph):
        """判断相关性"""
        query_words = set(query.lower().split())
        para_words = set(paragraph.lower().split())
        overlap = len(query_words & para_words)
        return overlap > 0
```

## 六、完整优化流程

### 6.1 优化检索Pipeline

```python
"""
完整的检索优化流程
"""

class OptimizedRetrievalPipeline:
    """优化检索Pipeline"""

    def __init__(self, vectorstore, llm, embeddings):
        self.vectorstore = vectorstore
        self.llm = llm
        self.embeddings = embeddings

        self.query_expander = QueryExpansion(llm)
        self.hyde = HyDE(llm, embeddings)
        self.reranker = CrossEncoderReranker()

    def retrieve(self, query, k=5):
        """优化检索"""
        # Step 1: 查询扩展
        expanded_queries = self.query_expander.expand_query(query)

        # Step 2: 多路检索
        all_results = []

        # 原始查询
        all_results.extend(
            self.vectorstore.similarity_search(query, k=k*2)
        )

        # 扩展查询
        for q in expanded_queries:
            all_results.extend(
                self.vectorstore.similarity_search(q, k=k)
            )

        # HyDE检索
        hypo_results, _ = self.hyde.search_with_hyde(query, k=k)
        all_results.extend(hypo_results)

        # Step 3: 去重
        unique_results = self._deduplicate(all_results)

        # Step 4: 重排序
        final_results = self.reranker.rerank(
            query,
            [r.page_content for r in unique_results],
            k=k
        )

        return final_results

    def _deduplicate(self, results):
        """去重"""
        seen = set()
        unique = []
        for doc in results:
            key = hash(doc.page_content)
            if key not in seen:
                seen.add(key)
                unique.append(doc)
        return unique
```

## 总结

### 核心要点

1. **查询优化**：扩展、改写、HyDE
2. **混合检索**：关键词+向量结合
3. **重排序**：LLM重排序、Cross-Encoder
4. **上下文优化**：压缩、片段选择
5. **完整流程**：多路检索→去重→重排序

### 最佳实践

- 先检索更多候选（k*2），再重排序选最优
- 使用HyDE提升语义匹配
- 长文档使用压缩减少噪声
- 混合检索平衡关键词和语义
