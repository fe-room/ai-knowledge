```python
"""
文档收集与管理
"""

class DocumentCollector:
    """文档收集器"""

    def collect_from_sources(self, sources):
        """从多个来源收集"""
        documents = []

        for source in sources:
            if source["type"] == "file":
                docs = self._collect_files(source["path"])
            elif source["type"] == "web":
                docs = self._collect_web(source["urls"])
            elif source["type"] == "database":
                docs = self._collect_db(source["query"])

            documents.extend(docs)

        return documents

    def _collect_files(self, path):
        """收集文件"""
        import os
        docs = []

        for root, dirs, files in os.walk(path):
            for file in files:
                filepath = os.path.join(root, file)
                if file.endswith('.pdf'):
                    docs.extend(self._load_pdf(filepath))
                elif file.endswith('.txt'):
                    docs.extend(self._load_text(filepath))

        return docs
```

### 2.2 索引Pipeline

```python
"""
索引Pipeline
"""

class IndexingPipeline:
    """索引Pipeline"""

    def __init__(self, config):
        self.config = config
        self.parser = DocumentParser()
        self.chunker = SmartChunker(
            chunk_size=config.get("chunk_size", 500)
        )
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            persist_directory=config.get("persist_dir", "./index")
        )

    def build_index(self, documents):
        """构建索引"""
        # Step 1: 解析文档
        parsed_docs = []
        for doc in documents:
            parsed = self.parser.parse(doc)
            parsed_docs.extend(parsed)

        # Step 2: 分块
        chunks = self.chunker.chunk_documents(parsed_docs)

        # Step 3: 向量化并存储
        self.vectorstore.add_documents(chunks)

        # Step 4: 持久化
        self.vectorstore.persist()

        return {
            "total_docs": len(parsed_docs),
            "total_chunks": len(chunks)
        }

    def update_index(self, new_docs):
        """更新索引"""
        # 解析和分块
        parsed = [self.parser.parse(doc) for doc in new_docs]
        chunks = self.chunker.chunk_documents(parsed)

        # 添加到向量存储
        self.vectorstore.add_documents(chunks)
        self.vectorstore.persist()

    def rebuild_index(self):
        """重建索引"""
        self.vectorstore.delete_all()
        # 重新索引所有文档
```

## 三、查询阶段

### 3.1 查询Pipeline

```python
"""
查询Pipeline
"""

class QueryPipeline:
    """查询Pipeline"""

    def __init__(self, vectorstore, llm, config):
        self.vectorstore = vectorstore
        self.llm = llm
        self.config = config

        # 各阶段处理器
        self.query_processor = QueryProcessor()
        self.retriever = OptimizedRetriever(vectorstore)
        self.post_processor = PostProcessor(llm)
        self.generator = Generator(llm)

    def process_query(self, query):
        """处理查询"""
        # Step 1: 查询预处理
        processed_query = self.query_processor.process(query)

        # Step 2: 检索
        retrieved_docs = self.retriever.retrieve(
            processed_query,
            k=self.config.get("top_k", 5)
        )

        # Step 3: 后处理
        processed_docs = self.post_processor.process(
            query,
            retrieved_docs
        )

        # Step 4: 生成回答
        answer = self.generator.generate(
            query,
            processed_docs
        )

        return {
            "query": query,
            "processed_query": processed_query,
            "retrieved_docs": retrieved_docs,
            "answer": answer
        }
```

### 3.2 查询处理器

```python
"""
查询预处理
"""

class QueryProcessor:
    """查询处理器"""

    def process(self, query):
        """预处理查询"""
        # 清理
        cleaned = self._clean_query(query)

        # 扩展
        expanded = self._expand_query(cleaned)

        # 分类
        query_type = self._classify_query(cleaned)

        return {
            "original": query,
            "cleaned": cleaned,
            "expanded": expanded,
            "type": query_type
        }

    def _clean_query(self, query):
        """清理查询"""
        import re
        query = query.strip()
        query = re.sub(r'[^\w\s\u4e00-\u9fff]', '', query)
        return query

    def _expand_query(self, query):
        """扩展查询"""
        # 简化：添加同义词
        synonyms = {
            "AI": ["人工智能", "机器智能"],
            "Python": ["py", "python语言"]
        }

        expanded = query
        for term, syns in synonyms.items():
            if term in query:
                expanded = query.replace(term, f"{term} {syns[0]}")

        return expanded

    def _classify_query(self, query):
        """分类查询"""
        if "是什么" in query or "定义" in query:
            return "definition"
        elif "如何" in query or "怎么" in query:
            return "how-to"
        elif "为什么" in query:
            return "explanation"
        else:
            return "general"
```

## 四、生成器

### 4.1 回答生成

```python
"""
答案生成器
"""

class Generator:
    """答案生成器"""

    def __init__(self, llm):
        self.llm = llm

    def generate(self, query, documents):
        """生成答案"""
        # 构建提示词
        prompt = self._build_prompt(query, documents)

        # 生成
        response = self.llm.invoke(prompt)

        return {
            "answer": response.content,
            "sources": self._extract_sources(documents)
        }

    def _build_prompt(self, query, documents):
        """构建提示词"""
        context = "\n\n".join([
            f"[文档{i+1}]\n{doc.page_content}"
            for i, doc in enumerate(documents)
        ])

        prompt = f"""
基于以下参考文档回答问题。如果文档中没有相关信息，请说明。

参考文档：
{context}

问题：{query}

回答格式：
1. 直接回答问题
2. 如果引用文档内容，标注来源
3. 如果信息不确定，说明原因

回答：
"""
        return prompt

    def _extract_sources(self, documents):
        """提取来源"""
        return [
            {
                "content": doc.page_content[:100],
                "metadata": doc.metadata
            }
            for doc in documents
        ]
```

## 五、完整RAG系统

### 5.1 RAG系统类

```python
"""
完整的RAG系统
"""

class RAGSystem:
    """RAG系统"""

    def __init__(self, config):
        # 初始化组件
        self.llm = ChatOpenAI(
            model=config.get("model", "gpt-3.5-turbo")
        )
        self.embeddings = OpenAIEmbeddings()

        # 索引Pipeline
        self.indexing_pipeline = IndexingPipeline(config)

        # 查询Pipeline
        vectorstore = Chroma(
            persist_directory=config.get("persist_dir"),
            embedding_function=self.embeddings
        )
        self.query_pipeline = QueryPipeline(
            vectorstore, self.llm, config
        )

    def index_documents(self, documents):
        """索引文档"""
        return self.indexing_pipeline.build_index(documents)

    def query(self, question):
        """查询"""
        return self.query_pipeline.process_query(question)

    def update_knowledge(self, new_docs):
        """更新知识库"""
        self.indexing_pipeline.update_index(new_docs)

# 使用示例
config = {
    "persist_dir": "./rag_index",
    "chunk_size": 500,
    "top_k": 5,
    "model": "gpt-3.5-turbo"
}

rag = RAGSystem(config)

# 索引文档
documents = [
    {"content": "Python是一种编程语言...", "metadata": {"source": "python.txt"}},
    {"content": "机器学习是AI的一个分支...", "metadata": {"source": "ai.txt"}}
]
rag.index_documents(documents)

# 查询
result = rag.query("什么是Python？")
print(result["answer"])
```

## 七、增量更新

### 7.1 增量索引

```python
"""
增量更新策略
"""

class IncrementalIndexer:
    """增量索引器"""

    def __init__(self, vectorstore):
        self.vectorstore = vectorstore
        self.file_hashes = {}

    def check_and_update(self, filepath):
        """检查并更新"""
        current_hash = self._get_file_hash(filepath)
        old_hash = self.file_hashes.get(filepath)

        if current_hash != old_hash:
            # 文件已变更
            self._update_index(filepath, current_hash)
            return True

        return False

    def _get_file_hash(self, filepath):
        """计算文件hash"""
        import hashlib
        with open(filepath, 'rb') as f:
            return hashlib.md5(f.read()).hexdigest()

    def _update_index(self, filepath, new_hash):
        """更新索引"""
        # 删除旧索引
        if filepath in self.file_hashes:
            self.vectorstore.delete(
                filter={"source": filepath}
            )

        # 添加新索引
        docs = self._load_and_chunk(filepath)
        self.vectorstore.add_documents(docs)

        # 更新hash记录
        self.file_hashes[filepath] = new_hash
```

## 总结

### 核心要点

1. **索引阶段**：文档收集→解析→分块→向量化→存储
2. **查询阶段**：预处理→检索→后处理→生成
3. **知识库管理**：添加、删除、更新、同步
4. **增量更新**：检测变更，智能更新索引
5. **Pipeline设计**：模块化、可配置、易扩展

### 最佳实践

- 定期同步知识库与源数据
- 使用增量更新减少重建成本
- 监控索引质量和查询效果
- 保存来源元数据便于追溯
