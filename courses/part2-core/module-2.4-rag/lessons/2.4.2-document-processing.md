## 二、文档解析

### 2.1 解析策略

```python
"""
文档解析策略
"""

class DocumentParser:
    """文档解析器"""

    def parse_pdf(self, filepath):
        """解析PDF"""
        loader = PyPDFLoader(filepath)
        docs = loader.load()

        # 清理文本
        for doc in docs:
            doc.page_content = self._clean_text(doc.page_content)

        return docs

    def parse_html(self, filepath):
        """解析HTML"""
        from bs4 import BeautifulSoup

        with open(filepath, 'r') as f:
            soup = BeautifulSoup(f.read(), 'html.parser')

        # 提取正文
        # 移除脚本和样式
        for script in soup(["script", "style", "nav", "footer"]):
            script.decompose()

        text = soup.get_text(separator=' ', strip=True)
        return [{"page_content": text, "metadata": {"source": filepath}}]

    def _clean_text(self, text):
        """清理文本"""
        # 移除多余空白
        import re
        text = re.sub(r'\s+', ' ', text)
        text = text.strip()
        return text
```

### 2.2 元数据提取

```python
"""
提取文档元数据
"""

def extract_metadata(doc):
    """提取元数据"""
    metadata = {
        "source": doc.metadata.get("source", "unknown"),
        "page": doc.metadata.get("page", 0),
        "title": extract_title(doc.page_content),
        "author": extract_author(doc.page_content),
        "date": extract_date(doc.page_content),
        "word_count": len(doc.page_content.split()),
        "char_count": len(doc.page_content)
    }
    return metadata

def extract_title(text):
    """提取标题"""
    lines = text.split('\n')
    if lines:
        return lines[0][:100]  # 第一行作为标题
    return ""

def extract_author(text):
    """提取作者"""
    import re
    match = re.search(r'作者[:：]\s*(.+)', text)
    if match:
        return match.group(1).strip()
    return ""

def extract_date(text):
    """提取日期"""
    import re
    match = re.search(r'\d{4}[-年]\d{1,2}[-月]\d{1,2}', text)
    if match:
        return match.group(0)
    return ""
```

## 三、文本分块

### 3.1 为什么需要分块？

```
分块的原因：

1. Token限制
   - LLM有最大Token限制
   - 大文档无法一次处理

2. 检索效率
   - 小块检索更精准
   - 避免返回过多无关内容

3. 质量保证
   - 每块内容聚焦
   - 更好的语义完整性

分块大小建议：
- 问答场景：200-500 tokens
- 总结场景：1000-2000 tokens
- 一般场景：500-1000 tokens
```

### 3.2 基础分块

```python
"""
基础分块方法
"""

def simple_chunk(text, chunk_size=500, overlap=50):
    """简单固定大小分块"""
    chunks = []
    start = 0

    while start < len(text):
        end = start + chunk_size
        chunk = text[start:end]
        chunks.append(chunk)
        start = end - overlap  # 重叠部分

    return chunks

# 使用示例
text = "很长的文档内容..."
chunks = simple_chunk(text, 500, 50)
print(f"分块数量: {len(chunks)}")
```

### 3.4 智能分块

```python
"""
智能分块策略
"""

class SmartChunker:
    """智能分块"""

    def chunk_by_semantic(self, text, max_size=500):
        """语义分块"""
        # 按段落分割
        paragraphs = text.split('\n\n')

        chunks = []
        current_chunk = ""

        for para in paragraphs:
            if len(current_chunk) + len(para) <= max_size:
                current_chunk += para + "\n\n"
            else:
                if current_chunk:
                    chunks.append(current_chunk.strip())
                current_chunk = para + "\n\n"

        if current_chunk:
            chunks.append(current_chunk.strip())

        return chunks

    def chunk_by_structure(self, text, structure_type="markdown"):
        """按结构分块"""
        if structure_type == "markdown":
            return self._chunk_markdown(text)
        elif structure_type == "html":
            return self._chunk_html(text)
        else:
            return simple_chunk(text)

    def _chunk_markdown(self, text):
        """Markdown结构分块"""
        md_splitter = MarkdownHeaderTextSplitter(
            headers_to_split_on=[
                ("#", "h1"),
                ("##", "h2"),
                ("###", "h3")
            ]
        )
        return md_splitter.split_text(text)

    def chunk_with_metadata(self, text, metadata):
        """带元数据的分块"""
        chunks = simple_chunk(text)

        # 为每个块添加元数据
        chunk_objects = []
        for i, chunk in enumerate(chunks):
            chunk_objects.append({
                "content": chunk,
                "metadata": {
                    **metadata,
                    "chunk_id": i,
                    "total_chunks": len(chunks)
                }
            })

        return chunk_objects
```

### 4.2 分块优化

```python
"""
分块优化技巧
"""

def optimize_chunks(chunks, min_size=50, max_size=1000):
    """优化分块"""
    optimized = []

    for chunk in chunks:
        # 太小的块合并
        if len(chunk) < min_size:
            if optimized:
                # 合并到上一个块
                optimized[-1] += "\n" + chunk
            continue

        # 太大的块拆分
        if len(chunk) > max_size:
            sub_chunks = simple_chunk(chunk, max_size, 50)
            optimized.extend(sub_chunks)
            continue

        optimized.append(chunk)

    return optimized

def chunk_with_context(chunks, window_size=1):
    """添加上下文窗口"""
    context_chunks = []

    for i, chunk in enumerate(chunks):
        # 前后窗口
        before = chunks[max(0, i-window_size):i]
        after = chunks[i+1:min(len(chunks), i+window_size+1)]

        context_chunks.append({
            "content": chunk,
            "before_context": before,
            "after_context": after,
            "position": i
        })

    return context_chunks
```

## 五、完整Pipeline

### 5.1 文档处理Pipeline

```python
"""
完整的文档处理流程
"""

class DocumentPipeline:
    """文档处理管道"""

    def __init__(self, chunk_size=500, chunk_overlap=50):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap
        )

    def process(self, filepath):
        """处理文档"""
        # 1. 加载文档
        docs = self.load_document(filepath)

        # 2. 解析内容
        parsed_docs = self.parse_documents(docs)

        # 3. 分块
        chunks = self.split_documents(parsed_docs)

        # 4. 清理和优化
        optimized_chunks = self.optimize_chunks(chunks)

        # 5. 添加元数据
        final_chunks = self.add_metadata(optimized_chunks, filepath)

        return final_chunks

    def load_document(self, filepath):
        """加载文档"""
        if filepath.endswith('.pdf'):
            loader = PyPDFLoader(filepath)
        elif filepath.endswith('.txt'):
            loader = TextLoader(filepath)
        elif filepath.endswith('.docx'):
            loader = Docx2txtLoader(filepath)
        else:
            loader = TextLoader(filepath)

        return loader.load()

    def parse_documents(self, docs):
        """解析文档"""
        for doc in docs:
            doc.page_content = self._clean_text(doc.page_content)
        return docs

    def split_documents(self, docs):
        """分块"""
        return self.splitter.split_documents(docs)

    def optimize_chunks(self, chunks):
        """优化"""
        return [c for c in chunks if len(c.page_content) > 50]

    def add_metadata(self, chunks, filepath):
        """添加元数据"""
        for i, chunk in enumerate(chunks):
            chunk.metadata.update({
                "chunk_id": i,
                "total_chunks": len(chunks),
                "source_file": filepath
            })
        return chunks

    def _clean_text(self, text):
        """清理文本"""
        import re
        text = re.sub(r'\s+', ' ', text)
        return text.strip()

# 使用
pipeline = DocumentPipeline()
chunks = pipeline.process("document.pdf")
print(f"生成 {len(chunks)} 个文档块")
```

## 总结

### 核心要点

1. **文档加载**：支持多种格式（PDF、Word、HTML等）
2. **文档解析**：清理文本、提取元数据
3. **文本分块**：固定大小、递归、语义、结构化
4. **分块参数**：chunk_size、chunk_overlap的选择
5. **分块优化**：合并小块、拆分大块、添加上下文

### 最佳实践

- Markdown文档使用MarkdownHeaderTextSplitter
- PDF文档使用RecursiveCharacterTextSplitter
- overlap设置为chunk_size的10-20%
- 保持分块的语义完整性
