## 二、检索评估

### 2.1 检索指标

```python
"""
检索质量评估
"""

import numpy as np

class RetrievalEvaluator:
    """检索评估器"""

    def evaluate(self, query, retrieved_docs, ground_truth):
        """
        ground_truth: 正确答案对应的文档ID列表
        """
        retrieved_ids = [doc.metadata.get("id") for doc in retrieved_docs]

        # 计算指标
        precision = self._precision(retrieved_ids, ground_truth)
        recall = self._recall(retrieved_ids, ground_truth)
        mrr = self._mrr(retrieved_ids, ground_truth)
        ndcg = self._ndcg(retrieved_ids, ground_truth)

        return {
            "precision": precision,
            "recall": recall,
            "mrr": mrr,
            "ndcg": ndcg
        }

    def _precision(self, retrieved, relevant):
        """精确率"""
        if not retrieved:
            return 0
        hits = len(set(retrieved) & set(relevant))
        return hits / len(retrieved)

    def _recall(self, retrieved, relevant):
        """召回率"""
        if not relevant:
            return 0
        hits = len(set(retrieved) & set(relevant))
        return hits / len(relevant)

    def _mrr(self, retrieved, relevant):
        """平均倒数排名"""
        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                return 1 / (i + 1)
        return 0

    def _ndcg(self, retrieved, relevant):
        """NDCG"""
        dcg = 0
        for i, doc_id in enumerate(retrieved):
            if doc_id in relevant:
                dcg += 1 / np.log2(i + 2)

        # 理想DCG
        ideal_dcg = sum([1 / np.log2(i + 2) for i in range(len(relevant))])

        return dcg / ideal_dcg if ideal_dcg > 0 else 0
```

### 2.2 检索测试集

```python
"""
构建检索测试集
"""

class RetrievalTestSet:
    """检索测试集"""

    def __init__(self):
        self.test_cases = []

    def add_test_case(self, query, relevant_doc_ids, description=""):
        """添加测试案例"""
        self.test_cases.append({
            "query": query,
            "relevant_ids": relevant_doc_ids,
            "description": description
        })

    def run_evaluation(self, retriever, k=5):
        """运行评估"""
        evaluator = RetrievalEvaluator()
        results = []

        for case in self.test_cases:
            retrieved = retriever.retrieve(case["query"], k=k)
            metrics = evaluator.evaluate(
                case["query"],
                retrieved,
                case["relevant_ids"]
            )
            results.append({
                "query": case["query"],
                "metrics": metrics
            })

        # 计算平均指标
        avg_metrics = self._average_metrics(results)
        return {
            "individual_results": results,
            "average_metrics": avg_metrics
        }

    def _average_metrics(self, results):
        """计算平均"""
        metrics_list = [r["metrics"] for r in results]
        return {
            "precision": np.mean([m["precision"] for m in metrics_list]),
            "recall": np.mean([m["recall"] for m in metrics_list]),
            "mrr": np.mean([m["mrr"] for m in metrics_list]),
            "ndcg": np.mean([m["ndcg"] for m in metrics_list])
        }
```

"""

class AnswerEvaluator:
    """答案评估器"""

    def __init__(self, llm):
        self.llm = llm

    def evaluate_answer(self, question, answer, ground_truth):
        """评估答案"""
        prompt = f"""
问题：{question}
参考答案：{ground_truth}
生成的答案：{answer}

请评估生成答案的质量，按以下维度评分（1-5分）：

1. 准确性：答案是否正确
2. 完整性：是否包含所有关键信息
3. 相关性：是否直接回答问题
4. 可读性：是否易于理解

输出JSON格式：
```json
{{"scores": {"accuracy": X, "completeness": X, "relevance": X, "readability": X}}}
```
"""
        response = self.llm.invoke(prompt)
        import json
        scores = json.loads(response.content)["scores"]
        return scores

    def evaluate_with_ragas(self, question, answer, contexts, ground_truth):
        """使用RAGAS评估框架"""
        # Faithfulness：答案与检索内容的忠实度
        faithfulness = self._evaluate_faithfulness(answer, contexts)

        # Answer Relevancy：答案与问题的相关性
        relevancy = self._evaluate_relevancy(question, answer)

        # Context Relevancy：检索内容与问题的相关性
        context_relevancy = self._evaluate_context_relevancy(question, contexts)

        return {
            "faithfulness": faithfulness,
            "answer_relevancy": relevancy,
            "context_relevancy": context_relevancy
        }

    def _evaluate_faithfulness(self, answer, contexts):
        """评估忠实度"""
        context_text = "\n".join(contexts)
        prompt = f"""
答案：{answer}
参考内容：{context_text}

评估答案是否基于参考内容生成（1-5分）：
"""
        response = self.llm.invoke(prompt)
        return float(response.content)

    def _evaluate_relevancy(self, question, answer):
        """评估相关性"""
        prompt = f"""
问题：{question}
答案：{answer}

评估答案是否直接回答问题（1-5分）：
"""
        response = self.llm.invoke(prompt)
        return float(response.content)

    def _evaluate_context_relevancy(self, question, contexts):
        """评估上下文相关性"""
        prompt = f"""
问题：{question}
检索内容：{chr(10).join(contexts[:3])}

评估检索内容是否与问题相关（1-5分）：
"""
        response = self.llm.invoke(prompt)
        return float(response.content)
```

## 四、端到端评估

### 4.1 RAG系统评估

```python
"""
端到端RAG评估
"""

class RAGEvaluator:
    """RAG系统评估器"""

    def __init__(self, rag_system, llm):
        self.rag = rag_system
        self.llm = llm
        self.retrieval_evaluator = RetrievalEvaluator()
        self.answer_evaluator = AnswerEvaluator(llm)

    def evaluate(self, test_cases):
        """全面评估"""
        results = []

        for case in test_cases:
            # 执行查询
            rag_result = self.rag.query(case["question"])

            # 检索评估
            retrieval_metrics = self.retrieval_evaluator.evaluate(
                case["question"],
                rag_result["retrieved_docs"],
                case["relevant_doc_ids"]
            )

            # 答案评估
            answer_metrics = self.answer_evaluator.evaluate_answer(
                case["question"],
                rag_result["answer"],
                case["ground_truth_answer"]
            )

            results.append({
                "question": case["question"],
                "retrieval_metrics": retrieval_metrics,
                "answer_metrics": answer_metrics,
                "generated_answer": rag_result["answer"],
                "ground_truth": case["ground_truth_answer"]
            })

        # 汇总
        summary = self._summarize(results)
        return {
            "detailed_results": results,
            "summary": summary
        }

    def _summarize(self, results):
        """汇总结果"""
        retrieval_avg = {
            "precision": np.mean([r["retrieval_metrics"]["precision"] for r in results]),
            "recall": np.mean([r["retrieval_metrics"]["recall"] for r in results]),
            "mrr": np.mean([r["retrieval_metrics"]["mrr"] for r in results])
        }

        answer_avg = {
            "accuracy": np.mean([r["answer_metrics"]["accuracy"] for r in results]),
            "completeness": np.mean([r["answer_metrics"]["completeness"] for r in results]),
            "relevance": np.mean([r["answer_metrics"]["relevance"] for r in results])
        }

        return {
            "retrieval_metrics": retrieval_avg,
            "answer_metrics": answer_avg,
            "total_test_cases": len(results)
        }
```

### 5.1 问题诊断

```python
"""
诊断RAG问题
"""

class RAGDiagnoser:
    """RAG诊断器"""

    def diagnose(self, question, result, ground_truth):
        """诊断问题"""
        issues = []

        # 检查检索问题
        if not result["retrieved_docs"]:
            issues.append({
                "type": "retrieval_empty",
                "description": "检索返回空结果",
                "suggestion": "检查分块策略或查询扩展"
            })

        # 检查答案准确性
        if ground_truth and result["answer"] != ground_truth:
            issues.append({
                "type": "answer_inaccurate",
                "description": "答案不准确",
                "suggestion": "检查生成提示词或模型选择"
            })

        # 检查来源缺失
        if not result.get("sources"):
            issues.append({
                "type": "missing_sources",
                "description": "答案未标注来源",
                "suggestion": "优化生成提示词要求标注来源"
            })

        return issues

    def suggest_improvements(self, issues):
        """改进建议"""
        suggestions = []

        for issue in issues:
            if issue["type"] == "retrieval_empty":
                suggestions.extend([
                    "优化分块大小",
                    "使用查询扩展",
                    "检查向量相似度阈值"
                ])
            elif issue["type"] == "answer_inaccurate":
                suggestions.extend([
                    "优化生成提示词",
                    "使用更强的LLM",
                    "增加更多检索结果"
                ])

        return suggestions
```

        performance = self._analyze_performance()

        # 问题检测
        issues = self._detect_issues()

        return {
            "query_frequency": query_freq,
            "performance": performance,
            "issues": issues
        }

    def _analyze_frequency(self):
        """分析查询频率"""
        queries = [log["query"] for log in self.query_logs]
        from collections import Counter
        return Counter(queries).most_common(10)

    def _analyze_performance(self):
        """分析性能"""
        if not self.query_logs:
            return {}

        avg_retrieved = np.mean([log["retrieved_count"] for log in self.query_logs])
        avg_length = np.mean([log["answer_length"] for log in self.query_logs])

        return {
            "avg_retrieved_docs": avg_retrieved,
            "avg_answer_length": avg_length
        }
