## 一、错误处理策略

### 1.1 错误类型

```python
from openai import (
    APIError,           # API通用错误
    APIConnectionError, # 网络连接错误
    RateLimitError,     # 速率限制
    AuthenticationError,# 认证错误
    BadRequestError,    # 请求错误
    ConflictError,      # 冲突错误
    NotFoundError,      # 资源不存在
    PermissionError,    # 权限错误
    UnprocessableEntityError  # 无法处理的实体
)
```

### 1.2 完整错误处理

```python
import time
import logging
from functools import wraps

logger = logging.getLogger(__name__)

def handle_api_errors(max_retries=3, initial_delay=1, max_delay=60):
    """API错误处理装饰器"""

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None

            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)

                except AuthenticationError as e:
                    # 认证错误，不重试
                    logger.error(f"认证失败: {e}")
                    raise

                except RateLimitError as e:
                    # 速率限制，等待后重试
                    delay = min(initial_delay * (2 ** attempt), max_delay)
                    logger.warning(f"速率限制，等待{delay}秒后重试 (尝试 {attempt+1}/{max_retries})")
                    time.sleep(delay)
                    last_exception = e

                except APIConnectionError as e:
                    # 网络错误，重试
                    delay = initial_delay * (2 ** attempt)
                    logger.warning(f"连接错误，等待{delay}秒后重试")
                    time.sleep(delay)
                    last_exception = e

                except BadRequestError as e:
                    # 请求错误，不重试
                    logger.error(f"请求错误: {e}")
                    raise

                except APIError as e:
                    # API错误，可能是临时问题
                    if attempt < max_retries - 1:
                        delay = initial_delay * (2 ** attempt)
                        logger.warning(f"API错误，等待{delay}秒后重试")
                        time.sleep(delay)
                    last_exception = e

            raise last_exception

        return wrapper
    return decorator

# 使用
@handle_api_errors(max_retries=3)
def call_api(messages):
    return client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=messages
    )
```

## 二、重试机制

### 2.1 指数退避

```python
import random

def exponential_backoff(
    func,
    max_retries=5,
    base_delay=1,
    max_delay=60,
    jitter=True
):
    """指数退避重试"""

    for attempt in range(max_retries):
        try:
            return func()

        except Exception as e:
            if attempt == max_retries - 1:
                raise

            # 计算延迟
            delay = min(base_delay * (2 ** attempt), max_delay)

            # 添加随机抖动
            if jitter:
                delay = delay * (0.5 + random.random())

            logger.info(f"重试 {attempt+1}/{max_retries}, 等待 {delay:.2f}秒")
            time.sleep(delay)
```

### 2.2 自适应重试

```python
class AdaptiveRetry:
    """自适应重试策略"""

    def __init__(self):
        self.success_count = 0
        self.failure_count = 0
        self.base_delay = 1

    def get_delay(self):
        """根据历史成功率调整延迟"""
        if self.success_count + self.failure_count == 0:
            return self.base_delay

        success_rate = self.success_count / (self.success_count + self.failure_count)

        # 成功率高，减少延迟
        if success_rate > 0.9:
            return self.base_delay * 0.5
        # 成功率低，增加延迟
        elif success_rate < 0.5:
            return self.base_delay * 2
        else:
            return self.base_delay

    def record_success(self):
        self.success_count += 1

    def record_failure(self):
        self.failure_count += 1
```

## 三、并发控制

### 3.1 速率限制器

```python
import time
from threading import Lock

class RateLimiter:
    """速率限制器"""

    def __init__(self, calls_per_minute=60):
        self.calls_per_minute = calls_per_minute
        self.min_interval = 60.0 / calls_per_minute
        self.last_call_time = 0
        self.lock = Lock()

    def wait_if_needed(self):
        """等待直到可以调用"""
        with self.lock:
            now = time.time()
            time_since_last = now - self.last_call_time

            if time_since_last < self.min_interval:
                sleep_time = self.min_interval - time_since_last
                time.sleep(sleep_time)

            self.last_call_time = time.time()

# 使用
limiter = RateLimiter(calls_per_minute=60)

for prompt in prompts:
    limiter.wait_if_needed()
    response = call_api(prompt)
```

## 四、日志与监控

### 4.1 日志记录

```python
import logging
from datetime import datetime

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)

logger = logging.getLogger('llm_api')

def log_api_call(func):
    """API调用日志装饰器"""

    @wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.time()

        # 记录请求
        logger.info(f"API调用开始: {func.__name__}")
        logger.debug(f"参数: {kwargs}")

        try:
            result = func(*args, **kwargs)

            # 记录成功
            duration = time.time() - start_time
            logger.info(f"API调用成功, 耗时: {duration:.2f}秒")

            # 记录Token使用
            if hasattr(result, 'usage'):
                logger.info(f"Token使用: 输入={result.usage.prompt_tokens}, "
                           f"输出={result.usage.completion_tokens}")

            return result

        except Exception as e:
            # 记录失败
            duration = time.time() - start_time
            logger.error(f"API调用失败, 耗时: {duration:.2f}秒, 错误: {e}")
            raise

    return wrapper
```

### 4.2 监控指标

```python
from dataclasses import dataclass, field
from typing import List
import statistics

@dataclass
class APIMetrics:
    """API监控指标"""

    total_calls: int = 0
    successful_calls: int = 0
    failed_calls: int = 0
    total_tokens: int = 0
    total_cost: float = 0

    response_times: List[float] = field(default_factory=list)

    def record_call(self, success: bool, tokens: int, cost: float, response_time: float):
        self.total_calls += 1
        if success:
            self.successful_calls += 1
        else:
            self.failed_calls += 1

        self.total_tokens += tokens
        self.total_cost += cost
        self.response_times.append(response_time)

    def get_stats(self):
        """获取统计信息"""
        return {
            'total_calls': self.total_calls,
            'success_rate': self.successful_calls / max(self.total_calls, 1),
            'total_tokens': self.total_tokens,
            'total_cost': self.total_cost,
            'avg_response_time': statistics.mean(self.response_times) if self.response_times else 0,
            'p95_response_time': statistics.quantiles(self.response_times, n=20)[18] if len(self.response_times) > 20 else 0
        }

# 使用
metrics = APIMetrics()

def tracked_call(prompt):
    start = time.time()
    try:
        response = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}]
        )

        metrics.record_call(
            success=True,
            tokens=response.usage.total_tokens,
            cost=calculate_cost(response.usage),
            response_time=time.time() - start
        )

        return response

    except Exception as e:
        metrics.record_call(
            success=False,
            tokens=0,
            cost=0,
            response_time=time.time() - start
        )
        raise
```

## 五、成本控制

### 5.1 Token预算控制

```python
class TokenBudget:
    """Token预算管理"""

    def __init__(self, daily_budget=100000):
        self.daily_budget = daily_budget
        self.used_today = 0
        self.last_reset = datetime.now().date()

    def check_and_use(self, estimated_tokens):
        """检查预算并使用"""
        # 检查是否需要重置
        today = datetime.now().date()
        if today > self.last_reset:
            self.used_today = 0
            self.last_reset = today

        # 检查预算
        if self.used_today + estimated_tokens > self.daily_budget:
            raise Exception(f"超出每日Token预算 (已用: {self.used_today}, 预算: {self.daily_budget})")

        self.used_today += estimated_tokens
        return True

    def get_remaining(self):
        """获取剩余预算"""
        return self.daily_budget - self.used_today

# 使用
budget = TokenBudget(daily_budget=100000)

def budget_aware_call(prompt):
    estimated = count_tokens(prompt) + 500  # 估算输出
    budget.check_and_use(estimated)

    return call_api(prompt)
```

### 5.2 成本优化策略

```python
class CostOptimizer:
    """成本优化器"""

    # 缓存
    cache = {}

    # 模型价格
    model_prices = {
        "gpt-3.5-turbo": {"input": 0.0005, "output": 0.0015},
        "gpt-4": {"input": 0.03, "output": 0.06},
    }

    @classmethod
    def cached_call(cls, prompt, model="gpt-3.5-turbo"):
        """带缓存的调用"""
        cache_key = f"{model}:{prompt}"

        if cache_key in cls.cache:
            return cls.cache[cache_key]

        response = call_api(prompt, model=model)
        cls.cache[cache_key] = response
        return response

    @classmethod
    def select_model(cls, task_complexity):
        """根据任务复杂度选择模型"""
        if task_complexity == "simple":
            return "gpt-3.5-turbo"
        elif task_complexity == "medium":
            return "gpt-3.5-turbo"
        else:  # complex
            return "gpt-4"

    @classmethod
    def estimate_cost(cls, input_tokens, output_tokens, model):
        """估算成本"""
        prices = cls.model_prices[model]
        return (
            input_tokens * prices["input"] / 1000 +
            output_tokens * prices["output"] / 1000
        )
```
