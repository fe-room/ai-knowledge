```python
def gemini_chat(prompt):
    """Gemini对话"""
    model = genai.GenerativeModel('gemini-pro')

    response = model.generate_content(prompt)
    return response.text

# 多轮对话
def gemini_conversation():
    """多轮对话"""
    model = genai.GenerativeModel('gemini-pro')
    chat = model.start_chat(history=[])

    # 发送消息
    response1 = chat.send_message("你好")
    print(response1.text)

    response2 = chat.send_message("介绍一下Python")
    print(response2.text)

    return chat.history
```

### 2.3 多模态能力

```python
def analyze_image(image_path, prompt):
    """图像分析"""
    import PIL.Image

    model = genai.GenerativeModel('gemini-pro-vision')
    img = PIL.Image.open(image_path)

    response = model.generate_content([prompt, img])
    return response.text
```

## 三、开源模型API

### 3.1 Llama系列

```python
"""
Llama系列模型特点：
- 开源免费
- 可本地部署
- 多种尺寸（7B/13B/70B）
"""

# 通过HuggingFace使用
from transformers import AutoModelForCausalLM, AutoTokenizer

def use_llama_locally():
    """本地使用Llama"""
    model_name = "meta-llama/Llama-2-7b-chat-hf"

    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(model_name)

    prompt = "你好，请介绍一下你自己"

    inputs = tokenizer(prompt, return_tensors="pt")
    outputs = model.generate(**inputs, max_length=100)

    response = tokenizer.decode(outputs[0])
    return response

# 通过API服务使用（如Together.ai）
import requests

def llama_api_call(prompt):
    """通过API调用Llama"""
    response = requests.post(
        "https://api.together.xyz/inference",
        headers={"Authorization": "Bearer YOUR_KEY"},
        json={
            "model": "meta-llama/Llama-2-70b-chat",
            "prompt": prompt,
            "max_tokens": 100
        }
    )
    return response.json()["output"]["text"]
```

### 3.2 Mistral API

```python
"""
Mistral特点：
- 欧洲开源模型
- 高效推理
- 商业友好
"""

def mistral_api_call(prompt):
    """Mistral API调用"""
    import requests

    response = requests.post(
        "https://api.mistral.ai/v1/chat/completions",
        headers={"Authorization": "Bearer YOUR_KEY"},
        json={
            "model": "mistral-medium",
            "messages": [
                {"role": "user", "content": prompt}
            ]
        }
    )

    return response.json()["choices"][0]["message"]["content"]
```

```python
"""
文心一言特点：
- 中文理解强
- 支持长文本
- 企业应用集成

import requests

def wenxin_chat(prompt):
    """文心一言调用"""
    # 获取access_token
    token_url = f"https://aip.baidubce.com/oauth/2.0/token?grant_type=client_credentials&client_id={AK}&client_secret={SK}"
    token_response = requests.post(token_url)
    access_token = token_response.json()["access_token"]

    # 调用API
    url = f"https://aip.baidubce.com/rpc/2.0/ai_custom/v1/wenxinworkshop/chat/completions?access_token={access_token}"

    response = requests.post(
        url,
        json={
            "messages": [
                {"role": "user", "content": prompt}
            ]
        }
    )

    return response.json()["result"]
```

### 4.3 Kimi（月之暗面）

```python
"""
Kimi特点：
- 200K+超长上下文
- 中文优化
- 文档处理强
"""

def kimi_chat(prompt, document=None):
    """Kimi API调用"""
    import requests

    messages = [{"role": "user", "content": prompt}]

    if document:
        messages[0]["content"] = f"{prompt}\n\n文档内容：\n{document}"

    response = requests.post(
        "https://api.moonshot.cn/v1/chat/completions",
        headers={"Authorization": "Bearer YOUR_KEY"},
        json={
            "model": "moonshot-v1-8k",
            "messages": messages
        }
    )

    return response.json()["choices"][0]["message"]["content"]
```

## 五、API统一封装

### 5.1 多模型适配器

```python
"""
统一API接口设计
"""

class LLMAdapter:
    """LLM统一适配器"""

    def __init__(self, provider="openai"):
        self.provider = provider
        self._setup_client()

    def _setup_client(self):
        """初始化客户端"""
        if self.provider == "openai":
            from openai import OpenAI
            self.client = OpenAI()
        elif self.provider == "anthropic":
            import anthropic
            self.client = anthropic.Anthropic()
        elif self.provider == "gemini":
            import google.generativeai as genai
            genai.configure()
            self.client = genai

    def chat(self, prompt, model=None, **kwargs):
        """统一对话接口"""
        if self.provider == "openai":
            model = model or "gpt-3.5-turbo"
            response = self.client.chat.completions.create(
                model=model,
                messages=[{"role": "user", "content": prompt}],
                **kwargs
            )
            return response.choices[0].message.content

        elif self.provider == "anthropic":
            model = model or "claude-3-haiku-20240307"
            response = self.client.messages.create(
                model=model,
                max_tokens=kwargs.get("max_tokens", 1024),
                messages=[{"role": "user", "content": prompt}]
            )
            return response.content[0].text

        elif self.provider == "gemini":
            model = model or "gemini-pro"
            model_instance = self.client.GenerativeModel(model)
            response = model_instance.generate_content(prompt)
            return response.text

# 使用示例
adapter = LLMAdapter("openai")
print(adapter.chat("你好"))

adapter = LLMAdapter("anthropic")
print(adapter.chat("你好"))
```

## 六、模型选择策略

### 6.1 选择决策树

```python
def select_model_for_task(task_type, language="zh", budget="medium"):
    """
    根据任务选择模型

    参数：
    - task_type: 任务类型（chat/code/document/analysis）
    - language: 语言（zh/en）
    - budget: 预算（low/medium/high）
    """

    recommendations = {
        ("chat", "zh", "high"): "文心一言/Kimi",
        ("chat", "zh", "medium"): "通义千问",
        ("chat", "zh", "low"): "ChatGLM",

        ("chat", "en", "high"): "GPT-4",
        ("chat", "en", "medium"): "Claude-3-Sonnet",
        ("chat", "en", "low"): "GPT-3.5/Llama",

        ("code", "any", "high"): "GPT-4",
        ("code", "any", "medium"): "Claude-3-Sonnet",
        ("code", "any", "low"): "CodeLlama",

        ("document", "zh", "high"): "Kimi",
        ("document", "en", "high"): "Claude-3-Opus",

        ("analysis", "any", "high"): "Gemini Pro Vision",
    }

    return recommendations.get((task_type, language, budget), "GPT-3.5")
```
