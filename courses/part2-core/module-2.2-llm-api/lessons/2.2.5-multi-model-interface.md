## 二、统一接口实现

### 2.1 基类定义

```python
"""
多模型统一接口框架
"""

from abc import ABC, abstractmethod
from typing import List, Dict, Any, Iterator, Optional
from dataclasses import dataclass
import time

@dataclass
class LLMResponse:
    """统一响应结构"""
    content: str
    model: str
    usage: Dict[str, int]
    cost: float
    latency: float
    metadata: Dict[str, Any] = None

class LLMBase(ABC):
    """LLM基类"""

    def __init__(self, api_key: str = None, **kwargs):
        self.api_key = api_key
        self.config = kwargs

    @abstractmethod
    def chat(
        self,
        messages: List[Dict[str, str]],
        **kwargs
    ) -> LLMResponse:
        """对话接口"""
        pass

    @abstractmethod
    def stream(
        self,
        messages: List[Dict[str, str]],
        **kwargs
    ) -> Iterator[str]:
        """流式接口"""
        pass

    @abstractmethod
    def count_tokens(self, text: str) -> int:
        """计算Token"""
        pass

    def calculate_cost(
        self,
        input_tokens: int,
        output_tokens: int
    ) -> float:
        """计算成本"""
        prices = self.get_prices()
        return (
            input_tokens * prices["input"] / 1000 +
            output_tokens * prices["output"] / 1000
        )

    @abstractmethod
    def get_prices(self) -> Dict[str, float]:
        """获取价格"""
        pass
```

### 2.2 OpenAI实现

```python
class OpenAI(LLMBase):
    """OpenAI实现"""

    def __init__(self, api_key: str = None):
        super().__init__(api_key)
        from openai import OpenAI as OpenAIClient
        self.client = OpenAIClient(api_key=api_key)

    def chat(
        self,
        messages: List[Dict[str, str]],
        model: str = "gpt-3.5-turbo",
        **kwargs
    ) -> LLMResponse:
        """对话"""
        start_time = time.time()

        response = self.client.chat.completions.create(
            model=model,
            messages=messages,
            **kwargs
        )

        latency = time.time() - start_time
        usage = response.usage

        return LLMResponse(
            content=response.choices[0].message.content,
            model=model,
            usage={
                "input": usage.prompt_tokens,
                "output": usage.completion_tokens,
                "total": usage.total_tokens
            },
            cost=self.calculate_cost(
                usage.prompt_tokens,
                usage.completion_tokens
            ),
            latency=latency
        )

    def stream(
        self,
        messages: List[Dict[str, str]],
        model: str = "gpt-3.5-turbo",
        **kwargs
    ) -> Iterator[str]:
        """流式输出"""
        stream = self.client.chat.completions.create(
            model=model,
            messages=messages,
            stream=True,
            **kwargs
        )

        for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content

    def count_tokens(self, text: str) -> int:
        """计算Token"""
        # 简化估算
        return len(text) // 4

    def get_prices(self) -> Dict[str, float]:
        """价格"""
        return {
            "gpt-3.5-turbo": {"input": 0.0005, "output": 0.0015},
            "gpt-4": {"input": 0.03, "output": 0.06}
        }
```

class Claude(LLMBase):
    """Claude实现"""

    def __init__(self, api_key: str = None):
        super().__init__(api_key)
        import anthropic
        self.client = anthropic.Anthropic(api_key=api_key)

    def chat(
        self,
        messages: List[Dict[str, str]],
        model: str = "claude-3-haiku-20240307",
        system: str = None,
        **kwargs
    ) -> LLMResponse:
        """对话"""
        start_time = time.time()

        params = {
            "model": model,
            "max_tokens": kwargs.get("max_tokens", 1024),
            "messages": messages
        }
        if system:
            params["system"] = system

        response = self.client.messages.create(**params)

        latency = time.time() - start_time

        return LLMResponse(
            content=response.content[0].text,
            model=model,
            usage={
                "input": response.usage.input_tokens,
                "output": response.usage.output_tokens,
                "total": response.usage.input_tokens + response.usage.output_tokens
            },
            cost=self.calculate_cost(
                response.usage.input_tokens,
                response.usage.output_tokens
            ),
            latency=latency
        )

    def stream(
        self,
        messages: List[Dict[str, str]],
        model: str = "claude-3-haiku-20240307",
        **kwargs
    ) -> Iterator[str]:
        """流式输出"""
        with self.client.messages.stream(
            model=model,
            max_tokens=kwargs.get("max_tokens", 1024),
            messages=messages
        ) as stream:
            for text in stream.text_stream:
                yield text

    def count_tokens(self, text: str) -> int:
        """计算Token"""
        return len(text) // 4

    def get_prices(self) -> Dict[str, float]:
        """价格"""
        return {"input": 0.00025, "output": 0.00125}

## 三、统一工厂类

### 3.1 LLM工厂

```python
class LLMFactory:
    """LLM工厂"""

    _providers = {
        "openai": OpenAI,
        "claude": Claude,
        # 可扩展其他模型
    }

    @classmethod
    def create(
        cls,
        provider: str,
        api_key: str = None,
        **kwargs
    ) -> LLMBase:
        """创建LLM实例"""
        if provider not in cls._providers:
            raise ValueError(f"不支持的服务商: {provider}")

        return cls._providers[provider](api_key=api_key, **kwargs)

    @classmethod
    def register(cls, provider: str, llm_class: type):
        """注册新服务商"""
        cls._providers[provider] = llm_class

# 使用示例
openai = LLMFactory.create("openai")
claude = LLMFactory.create("claude")

response = openai.chat([{"role": "user", "content": "你好"}])
print(response.content)
```

## 四、统一聊天类

### 4.1 高级封装

```python
class UnifiedChat:
    """统一聊天类"""

    def __init__(
        self,
        provider: str = "openai",
        model: str = None,
        api_key: str = None
    ):
        self.llm = LLMFactory.create(provider, api_key)
        self.model = model or self._get_default_model(provider)
        self.provider = provider
        self.history = []

    def _get_default_model(self, provider: str) -> str:
        """获取默认模型"""
        defaults = {
            "openai": "gpt-3.5-turbo",
            "claude": "claude-3-haiku-20240307"
        }
        return defaults.get(provider)

    def chat(
        self,
        user_input: str,
        stream: bool = False,
        **kwargs
    ) -> str:
        """对话"""
        # 添加用户消息
        self.history.append({
            "role": "user",
            "content": user_input
        })

        if stream:
            # 流式输出
            full_response = ""
            for text in self.llm.stream(
                self.history,
                model=self.model,
                **kwargs
            ):
                print(text, end="", flush=True)
                full_response += text
            print()

            response_content = full_response
        else:
            # 普通输出
            response = self.llm.chat(
                self.history,
                model=self.model,
                **kwargs
            )
            response_content = response.content

            # 打印统计信息
            print(f"[{self.provider}] Token: {response.usage['total']}, Cost: ${response.cost:.4f}")

        # 添加助手消息
        self.history.append({
            "role": "assistant",
            "content": response_content
        })

        return response_content

    def switch_model(self, provider: str, model: str = None):
        """切换模型"""
        self.llm = LLMFactory.create(provider)
        self.provider = provider
        self.model = model or self._get_default_model(provider)

    def clear_history(self):
        """清除历史"""
        self.history = []

# 使用示例
chat = UnifiedChat("openai")
chat.chat("你好！")
chat.chat("请介绍一下Python")

# 切换模型
chat.switch_model("claude")
chat.chat("你好！")
```

## 五、批量处理

### 5.1 批量调用

```python
class BatchProcessor:
    """批量处理器"""

    def __init__(self, provider: str = "openai"):
        self.llm = LLMFactory.create(provider)

    def process_batch(
        self,
        prompts: List[str],
        model: str = None,
        max_concurrent: int = 5
    ) -> List[LLMResponse]:
        """批量处理"""
        import asyncio

        async def process_one(prompt):
            return self.llm.chat(
                [{"role": "user", "content": prompt}],
                model=model
            )

        async def process_all():
            semaphore = asyncio.Semaphore(max_concurrent)

            async def limited(prompt):
                async with semaphore:
                    return await asyncio.to_thread(
                        process_one, prompt
                    )

            tasks = [limited(p) for p in prompts]
            return await asyncio.gather(*tasks)

        return asyncio.run(process_all())

    def compare_models(
        self,
        prompt: str,
        providers: List[str] = None
    ) -> Dict[str, LLMResponse]:
        """比较不同模型"""
        providers = providers or ["openai", "claude"]

        results = {}
        for provider in providers:
            llm = LLMFactory.create(provider)
            results[provider] = llm.chat(
                [{"role": "user", "content": prompt}]
            )

        return results

# 使用示例
batch = BatchProcessor("openai")
results = batch.process_batch([
    "什么是Python？",
    "什么是JavaScript？",
    "什么是AI？"
])

for r in results:
    print(f"Response: {r.content[:50]}...")
```

## 六、成本追踪

### 6.1 成本监控器

```python
class CostTracker:
    """成本追踪"""

    def __init__(self):
        self.total_cost = 0
        self.call_count = 0
        self.records = []

    def track(self, response: LLMResponse):
        """记录成本"""
        self.total_cost += response.cost
        self.call_count += 1

        self.records.append({
            "model": response.model,
            "tokens": response.usage["total"],
            "cost": response.cost,
            "latency": response.latency
        })

    def get_report(self) -> Dict[str, Any]:
        """获取报告"""
        return {
            "total_cost": self.total_cost,
            "total_calls": self.call_count,
            "avg_cost": self.total_cost / self.call_count if self.call_count else 0,
            "details": self.records
        }

# 使用
tracker = CostTracker()

chat = UnifiedChat("openai")
response = chat.llm.chat([{"role": "user", "content": "你好"}])
tracker.track(response)

print(tracker.get_report())
```
