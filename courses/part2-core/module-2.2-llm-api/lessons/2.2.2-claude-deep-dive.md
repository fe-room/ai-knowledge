```python
def compare_models(prompt):

    results = {}

    for model in [
        "claude-3-opus-20240229",
        "claude-3-sonnet-20240229",
        "claude-3-haiku-20240307"
```

### 5.2 文档分块处理

```python
def chunked_document_processing(document, chunk_size=50000):
    """分块处理超大文档"""
    chunks = []
    for i in range(0, len(document), chunk_size):
        chunks.append(document[i:i+chunk_size])

    summaries = []
    for chunk in chunks:
        summary = client.messages.create(
            model="claude-3-haiku-20240307",
            max_tokens=500,
            messages=[
                {
                    "role": "user",
                    "content": f"总结这段文本：\n{chunk}"
                }
            ]
        )
        summaries.append(summary.content[0].text)

    # 合并总结
    combined = "\n\n".join(summaries)
    final_summary = client.messages.create(
        model="claude-3-opus-20240229",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"基于以下章节总结，生成完整总结：\n{combined}"
            }
        ]
    )

    return final_summary.content[0].text
```

## 六、多轮对话

### 6.1 对话历史管理

```python
class ClaudeChat:
    """Claude多轮对话"""

    def __init__(self, system_prompt=None):
        self.client = anthropic.Anthropic()
        self.messages = []
        self.system_prompt = system_prompt

    def chat(self, user_input):
        """发送消息"""
        # 添加用户消息
        self.messages.append({
            "role": "user",
            "content": user_input
        })

        # 调用API
        params = {
            "model": "claude-3-opus-20240229",
            "max_tokens": 1024,
            "messages": self.messages
        }

        if self.system_prompt:
            params["system"] = self.system_prompt

        response = self.client.messages.create(**params)

        # 添加助手消息
        self.messages.append({
            "role": "assistant",
            "content": response.content[0].text
        })

        return response.content[0].text

    def clear_history(self):
        """清除历史"""
        self.messages = []

# 使用
chat = ClaudeChat(
    system_prompt="你是一位友好的助手"
)

print(chat.chat("你好！"))
print(chat.chat("请介绍一下Python"))
print(chat.chat("能举个例子吗？"))
```

## 七、Token计算

### 7.1 Token统计

```python
def count_tokens(text):
    """计算Token数量"""
    # Claude使用自己的tokenizer
    # 粗略估算：英文约4字符=1token，中文约1.5字符=1token

    # 简单估算
    english_chars = sum(1 for c in text if c.isascii())
    chinese_chars = sum(1 for c in text if not c.isascii())

    estimated_tokens = (
        english_chars // 4 +
        chinese_chars // 2 +
        10  # 额外缓冲
    )

    return estimated_tokens

def track_usage(response):
    """跟踪使用情况"""
    return {
        "input_tokens": response.usage.input_tokens,
        "output_tokens": response.usage.output_tokens,
        "total_tokens": response.usage.input_tokens + response.usage.output_tokens
    }
```

## 八、流式响应

### 8.1 流式输出

```python
def stream_chat(user_message):
    """流式对话"""
    with client.messages.stream(
        model="claude-3-opus-20240229",
        max_tokens=1024,
        messages=[
            {"role": "user", "content": user_message}
        ]
    ) as stream:
        for text in stream.text_stream:
            print(text, end="", flush=True)

    print()  # 换行

# 使用
stream_chat("请写一个关于Python的简短介绍")
```

### 8.2 流式处理

```python
def process_stream(user_message, callback):
    """处理流式响应"""
    full_response = ""

    with client.messages.stream(
        model="claude-3-opus-20240229",
        max_tokens=1024,
        messages=[
            {"role": "user", "content": user_message}
        ]
    ) as stream:
        for text in stream.text_stream:
            full_response += text
            callback(text)  # 实时处理

    return full_response

# 使用示例：实时显示
def print_callback(text):
    print(text, end="", flush=True)

result = process_stream(
    "写一首短诗",
    print_callback
)
```
