## 一、API参数详解

### 1.1 核心参数

**model（模型选择）**

```python
# 可用模型
models = [
    "gpt-3.5-turbo",      # 快速、便宜
    "gpt-3.5-turbo-16k",  # 更长上下文
    "gpt-4",              # 最强性能
    "gpt-4-turbo",        # GPT-4 + 长上下文
    "gpt-4o",             # 多模态
]

# 选择建议
# 简单任务 → gpt-3.5-turbo
# 复杂任务 → gpt-4
# 长文本 → gpt-3.5-turbo-16k 或 gpt-4-turbo
```

**temperature（温度）**

```
范围：0-2

0.0-0.3：确定性输出，适合编程、事实查询
0.4-0.7：平衡，适合一般对话
0.8-1.0：创造性，适合写作、头脑风暴
1.0+：高度随机，实验性用途

示例：
temperature=0  → 同样输入总是得到相同输出
temperature=0.7 → 每次输出可能略有不同
temperature=1.5 → 输出变化很大，可能出人意料
```

**max_tokens（最大Token数）**

```python
# 限制输出长度
response = client.chat.completions.create(
    model="gpt-3.5-turbo",
    messages=[{"role": "user", "content": "写一首诗"}],
    max_tokens=100  # 输出不超过100个token
)

# 注意：
# - 设置过小可能导致输出截断
# - 影响API成本
# - 与模型上下文窗口相关
```

**top_p（核采样）**

```
范围：0-1

控制从概率最高的多少token中采样

top_p=0.1：只从概率最高的10%token中选择
top_p=0.9：从概率最高的90%token中选择
top_p=1.0：考虑所有token

建议：
- 通常只需调整temperature或top_p之一
- top_p=0.9是一个好的默认值
```

**frequency_penalty & presence_penalty**

```python
# frequency_penalty：惩罚频繁出现的token
# 范围：-2.0到2.0
# 正值减少重复，负值允许更多重复

# presence_penalty：惩罚出现过的token
# 范围：-2.0到2.0
# 正值鼓励谈论新话题

response = client.chat.completions.create(
    model="gpt-3.5-turbo",
    messages=messages,
    frequency_penalty=0.5,  # 减少重复
    presence_penalty=0.3    # 鼓励新话题
)
```

## 二、流式响应

### 2.1 基本流式调用

```python
from openai import OpenAI

client = OpenAI()

# 流式调用
stream = client.chat.completions.create(
    model="gpt-3.5-turbo",
    messages=[{"role": "user", "content": "写一篇短文"}],
    stream=True
)

# 处理流式响应
for chunk in stream:
    if chunk.choices[0].delta.content:
        print(chunk.choices[0].delta.content, end="", flush=True)
```

### 2.2 异步流式

```python
import asyncio
from openai import AsyncOpenAI

client = AsyncOpenAI()

async def stream_chat(prompt):
    stream = await client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    )

    async for chunk in stream:
        if chunk.choices[0].delta.content:
            print(chunk.choices[0].delta.content, end="", flush=True)

# 运行
asyncio.run(stream_chat("你好"))
```

### 2.3 流式响应结构

```python
# chunk结构
{
    "id": "chatcmpl-123",
    "object": "chat.completion.chunk",
    "created": 1234567890,
    "model": "gpt-3.5-turbo",
    "choices": [{
        "index": 0,
        "delta": {
            "content": "文本片段"  # 增量内容
        },
        "finish_reason": null  # 或 "stop"
    }]
}

# 处理完整响应
full_response = ""
for chunk in stream:
    content = chunk.choices[0].delta.content
    if content:
        full_response += content
        print(content, end="")

    # 检查是否完成
    if chunk.choices[0].finish_reason:
        print(f"\n完成原因: {chunk.choices[0].finish_reason}")
```

### 3.2 调用函数

```python
def chat_with_tools(user_input):
    """使用函数调用的对话"""

    # 第一次调用
    response = client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "user", "content": user_input}
        ],
        tools=tools,
        tool_choice="auto"
    )

    message = response.choices[0].message

    # 检查是否需要调用函数
    if message.tool_calls:
        # 执行函数调用
        tool_call = message.tool_calls[0]
        function_name = tool_call.function.name
        function_args = json.loads(tool_call.function.arguments)

        # 调用实际函数
        if function_name == "get_weather":
            result = get_weather(**function_args)
        elif function_name == "search_web":
            result = search_web(**function_args)

        # 第二次调用，传入函数结果
        second_response = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "user", "content": user_input},
                message,  # 第一次的响应
                {
                    "role": "tool",
                    "tool_call_id": tool_call.id,
                    "content": str(result)
                }
            ]
        )

        return second_response.choices[0].message.content

    return message.content

# 使用
result = chat_with_tools("北京今天天气怎么样？")
```

### 3.3 多函数调用

```python
def handle_multiple_tool_calls(user_input):
    """处理多个函数调用"""

    response = client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[{"role": "user", "content": user_input}],
        tools=tools,
        tool_choice="auto"
    )

    message = response.choices[0].message

    if message.tool_calls:
        # 准备工具响应消息
        tool_messages = []

        for tool_call in message.tool_calls:
            function_name = tool_call.function.name
            function_args = json.loads(tool_call.function.arguments)

            # 执行函数
            result = execute_function(function_name, function_args)

            # 添加工具响应
            tool_messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": str(result)
            })

        # 第二次调用
        final_response = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "user", "content": user_input},
                message,
                *tool_messages
            ]
        )

        return final_response.choices[0].message.content

    return message.content
```

## 五、Embedding API

### 5.1 生成嵌入向量

```python
def get_embedding(text, model="text-embedding-3-small"):
    """获取文本的嵌入向量"""
    response = client.embeddings.create(
        input=text,
        model=model
    )
    return response.data[0].embedding

# 使用
embedding = get_embedding("Hello, World!")
print(f"向量维度: {len(embedding)}")  # 1536
```

### 5.2 批量嵌入

```python
def get_batch_embeddings(texts):
    """批量获取嵌入向量"""
    response = client.embeddings.create(
        input=texts,
        model="text-embedding-3-small"
    )
    return [item.embedding for item in response.data]

# 使用
texts = ["文本1", "文本2", "文本3"]
embeddings = get_batch_embeddings(texts)
```

### 5.3 相似度计算

```python
import numpy as np

def cosine_similarity(a, b):
    """计算余弦相似度"""
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

# 使用
text1 = "人工智能"
text2 = "机器学习"
text3 = "天气预报"

emb1 = get_embedding(text1)
emb2 = get_embedding(text2)
emb3 = get_embedding(text3)

print(f"'{text1}' vs '{text2}': {cosine_similarity(emb1, emb2):.4f}")
print(f"'{text1}' vs '{text3}': {cosine_similarity(emb1, emb3):.4f}")
```

```python
import tiktoken

def count_tokens(text, model="gpt-3.5-turbo"):
    """计算Token数量"""
    encoding = tiktoken.encoding_for_model(model)
    return len(encoding.encode(text))

# 使用
```

### 6.2 成本估算

```python
def estimate_cost(input_tokens, output_tokens, model="gpt-3.5-turbo"):
    """估算API调用成本"""

    # 价格（美元/1K tokens）
    prices = {
        "gpt-3.5-turbo": {"input": 0.0005, "output": 0.0015},
        "gpt-4": {"input": 0.03, "output": 0.06},
        "gpt-4-turbo": {"input": 0.01, "output": 0.03}
    }

    price = prices[model]
    cost = (
        input_tokens * price["input"] / 1000 +
        output_tokens * price["output"] / 1000
    )

    return cost

# 使用
cost = estimate_cost(1000, 500)
print(f"预估成本: ${cost:.6f}")
```

## 七、错误处理

```python
from openai import (
    APIError,
    APIConnectionError,
    RateLimitError,
    AuthenticationError,
    BadRequestError
)

def safe_chat(messages, max_retries=3):
    """安全的API调用"""

    for attempt in range(max_retries):
        try:
            response = client.chat.completions.create(
                model="gpt-3.5-turbo",
                messages=messages
            )
            return response.choices[0].message.content

        except AuthenticationError:
            raise Exception("API密钥无效")

        except RateLimitError:
            if attempt < max_retries - 1:
                wait_time = 2 ** attempt
                print(f"速率限制，等待{wait_time}秒...")
                time.sleep(wait_time)
            else:
                raise Exception("达到速率限制")

        except APIConnectionError:
            if attempt < max_retries - 1:
                time.sleep(1)
            else:
                raise Exception("网络连接失败")

        except BadRequestError as e:
            raise Exception(f"请求错误: {e.message}")

        except APIError as e:
            if attempt < max_retries - 1:
                time.sleep(2)
            else:
                raise Exception(f"API错误: {e.message}")
```

## 总结

### 核心要点

1. **参数调优**：temperature控制随机性，max_tokens控制长度
2. **流式响应**：提供更好的用户体验

4. **Vision API**：图像理解能力
5. **成本优化**：精简提示词、使用缓存
