# 2.3.3 Plan-Execute模式

> 掌握先规划后执行的Agent设计模式

## 一、Plan-Execute模式介绍

### 1.1 什么是Plan-Execute？

**Plan-Execute = 先规划，后执行**

一种将任务分解为明确计划，然后按计划执行的Agent模式。

```
ReAct：边思考边执行（动态调整）
Plan-Execute：先规划再执行（静态计划）

Plan-Execute流程：
用户问题 → 制定计划 → 执行步骤 → 返回结果
```

### 1.2 与ReAct对比

```
ReAct模式：
问题 → 思考 → 动作 → 观察 → 思考 → 动作 → ...
         ↑__________↓ 动态循环

Plan-Execute模式：
问题 → 制定完整计划 → 步骤1 → 步骤2 → 步骤3 → 结果
       ↑________↓ 一次性规划，顺序执行
```

## 二、Plan-Execute实现

### 2.1 基础架构

```python
"""
Plan-Execute Agent基础实现
"""

from openai import OpenAI
import json

class PlanExecuteAgent:
    """Plan-Execute Agent"""

    def __init__(self):
        self.client = OpenAI()
        self.tools = {}

    def register_tool(self, name, function, description):
        """注册工具"""
        self.tools[name] = {
            "function": function,
            "description": description
        }

    def run(self, question):
        """运行"""
        # Step 1: 制定计划
        plan = self._create_plan(question)
        print(f"[计划]\n{plan}")

        # Step 2: 执行计划
        results = []
        for step in plan["steps"]:
            print(f"\n[执行步骤] {step['description']}")
            result = self._execute_step(step)
            results.append(result)
            print(f"[结果] {result}")

        # Step 3: 总结答案
        answer = self._summarize(question, plan, results)
        return answer

    def _create_plan(self, question):
        """制定计划"""
        tools_desc = "\n".join([
            f"- {name}: {desc}"
            for name, desc in [(k, v["description"]) for k, v in self.tools.items()]
        ])

        prompt = f"""
你需要制定一个执行计划来回答问题。

可用工具：
{tools_desc}

请制定一个清晰的执行计划，包含具体的步骤。

问题：{question}

输出JSON格式的计划：
```json
{{"steps": [
  {{ "step": 1, "description": "步骤描述", "tool": "工具名", "input": "输入" }},
  ...
]}}
```
"""

        response = self.client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}]
        )

        output = response.choices[0].message.content

        # 解析JSON
        try:
            # 提取JSON部分
            json_str = self._extract_json(output)
            return json.loads(json_str)
        except:
            return {"steps": []}

    def _execute_step(self, step):
        """执行步骤"""
        tool_name = step.get("tool")
        tool_input = step.get("input")

        if tool_name in self.tools:
            return self.tools[tool_name]["function"](tool_input)
        return "工具不存在"

    def _summarize(self, question, plan, results):
        """总结答案"""
        context = "\n".join([
            f"步骤{step['step']}: {step['description']}\n结果: {result}"
            for step, result in zip(plan["steps"], results)
        ])

        prompt = f"""
问题：{question}

执行结果：
{context}

请基于以上执行结果，回答问题。
"""

        response = self.client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}]
        )

        return response.choices[0].message.content

    def _extract_json(self, text):
        """提取JSON"""
        import re
        match = re.search(r'```json\s*(.*?)\s*```', text)
        if match:
            return match.group(1)
        return text

# 使用示例
agent = PlanExecuteAgent()
agent.register_tool("search", lambda q: f"搜索:{q}", "搜索信息")
agent.register_tool("calculate", lambda e: str(eval(e)), "计算")

result = agent.run("北京的人口密度是多少？")
print(f"\n[最终答案]\n{result}")
```

## 三、动态计划调整

### 3.1 自适应计划

```python
class AdaptivePlanAgent(PlanExecuteAgent):
    """自适应计划Agent"""

    def run(self, question):
        """运行，可调整计划"""
        # 制定初始计划
        plan = self._create_plan(question)
        print(f"[初始计划]\n{json.dumps(plan, indent=2)}")

        results = []
        for i, step in enumerate(plan["steps"]):
            print(f"\n[执行步骤{i+1}] {step['description']}")

            # 执行步骤
            result = self._execute_step(step)
            results.append(result)
            print(f"[结果] {result}")

            # 检查是否需要调整计划
            if self._needs_replan(step, result):
                print("[需要调整计划]")
                plan = self._replan(question, plan, results)
                print(f"[新计划]\n{json.dumps(plan, indent=2)}")

        return self._summarize(question, plan, results)

    def _needs_replan(self, step, result):
        """判断是否需要重新规划"""
        # 简化判断：如果结果包含错误信息
        error_indicators = ["错误", "失败", "不存在", "无法"]
        return any(ind in result for ind in error_indicators)

    def _replan(self, question, old_plan, results):
        """重新规划"""
        context = "\n".join([
            f"步骤{step['step']}: {step['description']}\n结果: {result}"
            for step, result in zip(old_plan["steps"], results)
        ])

        prompt = f"""
原始问题：{question}

已执行步骤和结果：
{context}

当前计划无法继续执行，请重新制定计划。

输出新的执行计划JSON。
"""

        response = self.client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}]
        )

        output = response.choices[0].message.content
        json_str = self._extract_json(output)
        return json.loads(json_str)
```

## 四、计划优化

### 4.1 并行执行

```python
class ParallelPlanAgent(PlanExecuteAgent):
    """支持并行执行的计划Agent"""

    def run(self, question):
        """运行，识别可并行步骤"""
        plan = self._create_plan(question)

        # 分析依赖关系
        groups = self._analyze_dependencies(plan)

        all_results = {}
        for group in groups:
            print(f"\n[并行执行组] {group}")

            # 并行执行组内步骤
            import concurrent.futures

            with concurrent.futures.ThreadPoolExecutor() as executor:
                futures = {}
                for step in group:
                    future = executor.submit(
                        self._execute_step, step
                    )
                    futures[future] = step

                for future in concurrent.futures.as_completed(futures):
                    step = futures[future]
                    result = future.result()
                    all_results[step["step"]] = result
                    print(f"[步骤{step['step']}] {result}")

        return self._summarize_with_results(question, plan, all_results)

    def _analyze_dependencies(self, plan):
        """分析依赖，返回可并行执行的组"""
        # 简化：假设前N个步骤可以并行
        parallel = []
        sequential = []

        for step in plan["steps"]:
            if step.get("parallel", False):
                parallel.append(step)
            else:
                sequential.append(step)

        if parallel:
            return [parallel] + [[s] for s in sequential]
        return [[s] for s in plan["steps"]]
```

## 六、Plan-Execute vs ReAct选择

### 6.1 选择决策

```python
def choose_pattern(task):
    """选择Agent模式"""

    # 判断标准
    criteria = {
        "plan_execute": [
            "任务步骤可预先确定",
            "需要优化执行顺序",
            "任务复杂但结构清晰",
            "可以并行执行部分步骤"
        ],
        "react": [
            "任务不确定性高",
            "需要根据反馈调整",
            "无法预先规划步骤",
            "探索性任务"
        ]
    }

    # 简化判断
    if "分析" in task or "处理" in task:
        return "plan_execute"
    elif "查询" in task or "搜索" in task:
        return "react"
    else:
        return "react"  # 默认
```

## 总结

### 核心要点

1. **Plan-Execute**：先规划，后执行
2. **流程**：制定计划 → 执行步骤 → 总结答案
3. **动态调整**：可根据执行结果重新规划
4. **并行优化**：识别可并行执行的步骤
5. **适用场景**：结构清晰的多步骤任务
