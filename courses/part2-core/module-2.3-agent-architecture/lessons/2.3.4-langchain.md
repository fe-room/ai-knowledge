## 一、LangChain概述

### 1.1 什么是LangChain？

**LangChain** 是一个用于构建LLM应用的开发框架。

**核心价值**：
- 组件化设计
- 链式组合
- 统一接口
- 易于扩展

## 二、核心组件

### 2.1 Models（模型）

```python
"""
LangChain模型组件
"""

from langchain_openai import ChatOpenAI, OpenAI

# Chat模型
chat_model = ChatOpenAI(
    model="gpt-3.5-turbo",
    temperature=0.7
)

# 调用
response = chat_model.invoke("你好")
print(response.content)

# 文本模型
llm = OpenAI(model="gpt-3.5-turbo-instruct")
response = llm.invoke("写一首诗")
print(response)
```

### 2.2 Prompts（提示词）

```python
"""
提示词模板
"""

from langchain.prompts import (
    PromptTemplate,
    ChatPromptTemplate,
    SystemMessagePromptTemplate,
    HumanMessagePromptTemplate
)

# 简单模板
template = PromptTemplate.from_template(
    "写一篇关于{topic}的短文"
)

prompt = template.format(topic="人工智能")
print(prompt)

# Chat模板
chat_template = ChatPromptTemplate.from_messages([
    SystemMessagePromptTemplate.from_template(
        "你是一位{role}"
    ),
    HumanMessagePromptTemplate.from_template(
        "{question}"
    )
])

messages = chat_template.format_messages(
    role="编程导师",
    question="什么是装饰器？"
)

print(messages)
```

### 2.3 Chains（链）

```python
"""
链式组合
"""

from langchain.chains import LLMChain, SimpleSequentialChain

# 单链
chain = LLMChain(
    llm=ChatOpenAI(),
    prompt=PromptTemplate.from_template(
        "翻译成中文: {text}"
    )
)

result = chain.run("Hello World")
print(result)

# 顺序链
chain1 = LLMChain(
    llm=ChatOpenAI(),
    prompt=PromptTemplate.from_template(
        "生成一个关于{topic}的标题"
    )
)

chain2 = LLMChain(
    llm=ChatOpenAI(),
    prompt=PromptTemplate.from_template(
        "基于标题写文章: {title}"
    )
)

sequential_chain = SimpleSequentialChain(
    chains=[chain1, chain2]
)

result = sequential_chain.run("AI")
print(result)
```

## 三、Output Parsers

### 3.1 输出解析器

```python
"""
解析LLM输出
"""

from langchain.output_parsers import (
    StrOutputParser,
    JsonOutputParser,
    PydanticOutputParser
)
from langchain.prompts import PromptTemplate
from pydantic import BaseModel, Field

# 字符串解析器
parser = StrOutputParser()

# JSON解析器
json_parser = JsonOutputParser()
template = PromptTemplate.from_template(
    "回答问题并输出JSON格式:\n{question}\n{format_instructions}"
).partial(format_instructions=json_parser.get_format_instructions())

# Pydantic解析器
class Answer(BaseModel):
    answer: str = Field(description="答案")
    confidence: float = Field(description="置信度")

pydantic_parser = PydanticOutputParser(pydantic_object=Answer)

template = PromptTemplate.from_template(
    "回答问题:\n{question}\n{format_instructions}"
).partial(format_instructions=pydantic_parser.get_format_instructions())

chain = template | ChatOpenAI() | pydantic_parser
result = chain.invoke({"question": "Python是什么？"})
print(result)
```

## 四、Agents

### 4.1 Agent基础

```python
"""
LangChain Agent
"""

from langchain.agents import AgentExecutor, create_tool_calling_agent
from langchain.tools import Tool

# 定义工具
def search(query):
    return f"搜索结果: {query}"

def calculate(expression):
    return str(eval(expression))

tools = [
    Tool(
        name="search",
        func=search,
        description="搜索信息"
    ),
    Tool(
        name="calculate",
        func=calculate,
        description="计算数学表达式"
    )
]

# 创建Agent
llm = ChatOpenAI(model="gpt-3.5-turbo")
agent = create_tool_calling_agent(llm, tools)
agent_executor = AgentExecutor(agent=agent, tools=tools)

# 运行
result = agent_executor.invoke({
    "input": "北京的人口是多少？"
})
print(result["output"])
```

### 4.2 自定义Agent

```python
"""
自定义Agent行为
"""

from langchain.agents import ConversationalAgent

class CustomAgent:
    """自定义Agent"""

    def __init__(self, llm, tools):
        self.llm = llm
        self.tools = tools

        # 创建Agent
        self.agent = create_tool_calling_agent(llm, tools)
        self.executor = AgentExecutor(
            agent=self.agent,
            tools=tools,
            max_iterations=10,
            handle_parsing_errors=True
        )

    def run(self, question):
        """运行"""
        return self.executor.invoke({"input": question})

    def run_with_memory(self, question, chat_history):
        """带记忆运行"""
        return self.executor.invoke({
            "input": question,
            "chat_history": chat_history
        })
```

## 五、Memory

### 5.1 内存组件

```python
"""
记忆管理
"""

from langchain.memory import (
    ConversationBufferMemory,
    ConversationSummaryMemory,
    ConversationBufferWindowMemory
)

# 缓冲记忆
buffer_memory = ConversationBufferMemory(
    return_messages=True
)

# 窗口记忆（限制数量）
window_memory = ConversationBufferWindowMemory(
    k=5,  # 保留最近5轮
    return_messages=True
)

# 概要记忆
summary_memory = ConversationSummaryMemory(
    llm=ChatOpenAI(),
    return_messages=True
)

# 使用记忆的Chain
from langchain.chains import ConversationChain

chain = ConversationChain(
    llm=ChatOpenAI(),
    memory=buffer_memory
)

# 多轮对话
chain.predict(input="你好")
chain.predict(input="我叫小明")
chain.predict(input="我叫什么名字？")
```

## 六、RAG实现

### 6.1 文档加载

```python
"""
文档处理
"""

from langchain.document_loaders import (
    TextLoader,
    PDFLoader,
    WebBaseLoader
)

# 加载文本
text_loader = TextLoader("document.txt")
docs = text_loader.load()

# 加载PDF
pdf_loader = PDFLoader("document.pdf")
pdf_docs = pdf_loader.load()

# 加载网页
web_loader = WebBaseLoader("https://example.com")
web_docs = web_loader.load()
```

### 6.2 分块与向量存储

```python
"""
文档分块和向量存储
"""

from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain.vectorstores import Chroma

# 分块
splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,
    chunk_overlap=50
)

chunks = splitter.split_documents(docs)

# 向量化
embeddings = OpenAIEmbeddings()

# 存储到向量数据库
vectorstore = Chroma.from_documents(
    documents=chunks,
    embedding=embeddings
)

# 检索
results = vectorstore.similarity_search("查询内容", k=3)
```

### 6.3 RAG Chain

```python
"""
RAG问答链
"""

from langchain.chains import RetrievalQA

qa_chain = RetrievalQA.from_chain_type(
    llm=ChatOpenAI(),
    chain_type="stuff",
    retriever=vectorstore.as_retriever()
)

# 问答
answer = qa_chain.run("文档的主要内容是什么？")
print(answer)
```

## 七、LCEL（LangChain Expression Language）

### 7.1 链式表达式

```python
"""
LCEL - LangChain表达式语言
"""

from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate
from langchain.output_parsers import StrOutputParser

# 定义组件
model = ChatOpenAI()
prompt = ChatPromptTemplate.from_template("告诉我关于{topic}的信息")
parser = StrOutputParser()

# 使用LCEL组合
chain = prompt | model | parser

# 运行
result = chain.invoke({"topic": "Python"})
print(result)

# 流式输出
for chunk in chain.stream({"topic": "Python"}):
    print(chunk, end="", flush=True)

# 批量处理
results = chain.batch([
    {"topic": "Python"},
    {"topic": "JavaScript"},
    {"topic": "AI"}
])
```
