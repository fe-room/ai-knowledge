# 4.3 AI能力集成

## 一、对话能力

```python
"""
AI对话服务
"""

class AIService:
    """AI服务"""

    def __init__(self, llm_client, conversation_manager):
        self.llm = llm_client
        self.conversation = conversation_manager

    async def chat(
        self,
        user_id: str,
        message: str,
        context: dict = None
    ) -> ChatResponse:
        """对话"""
        # 获取对话历史
        history = await self.conversation.get_history(user_id)

        # 构建上下文
        messages = self._build_messages(history, message, context)

        # 调用LLM
        response = await self.llm.generate(messages)

        # 保存对话
        await self.conversation.save_turn(user_id, message, response.content)

        return ChatResponse(
            content=response.content,
            tokens_used=response.usage.total_tokens
        )

    def _build_messages(self, history, message, context):
        """构建消息列表"""
        messages = []

        # 系统提示
        messages.append({
            "role": "system",
            "content": self._get_system_prompt(context)
        })

        # 历史
        messages.extend(history[-10:])  # 最近5轮

        # 当前消息
        messages.append({"role": "user", "content": message})

        return messages
```

## 二、文档处理

```python
class DocumentProcessor:
    """文档处理服务"""

    async def process_document(self, file_path: str, user_id: str) -> ProcessResult:
        """处理文档"""
        # 1. 解析文档
        content = await self._parse(file_path)

        # 2. 提取信息
        extracted = await self._extract(content)

        # 3. 生成摘要
        summary = await self._summarize(content)

        # 4. 索引
        await self._index(user_id, content)

        return ProcessResult(
            document_id=generate_id(),
            summary=summary,
            extracted_info=extracted
        )

    async def search_documents(
        self,
        user_id: str,
        query: str
    ) -> List[SearchResult]:
        """搜索文档"""
        results = await self.vector_store.search(user_id, query)
        return results
```

class DataAnalyzer:
    """数据分析服务"""

    async def analyze(self, data: List[dict], analysis_type: str) -> AnalysisResult:
        """分析数据"""
        if analysis_type == "trend":
            return await self._trend_analysis(data)
        elif analysis_type == "summary":
            return await self._summary_analysis(data)
        elif analysis_type == "anomaly":
            return await self._anomaly_detection(data)
