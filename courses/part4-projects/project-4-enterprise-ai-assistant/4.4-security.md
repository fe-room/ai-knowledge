# 4.4 安全体系

## 一、数据加密

```python
"""
安全服务
"""

from cryptography.fernet import Fernet
import os

class SecurityService:
    """安全服务"""

    def __init__(self):
        self.key = os.environ.get("ENCRYPTION_KEY")
        self.cipher = Fernet(self.key)

    def encrypt(self, data: str) -> str:
        """加密数据"""
        return self.cipher.encrypt(data.encode()).decode()

    def decrypt(self, encrypted_data: str) -> str:
        """解密数据"""
        return self.cipher.decrypt(encrypted_data.encode()).decode()

    def hash_data(self, data: str) -> str:
        """哈希数据"""
        import hashlib
        return hashlib.sha256(data.encode()).hexdigest()
```

## 二、访问控制

```python
class AccessControl:
    """访问控制"""

    def __init__(self, rbac_manager):
        self.rbac = rbac_manager

    async def check_access(
        self,
        user_id: str,
        resource: str,
        action: str
    ) -> bool:
        """检查访问权限"""
        permission = f"{resource}:{action}"
        return self.rbac.check_permission(user_id, permission)

    async def filter_data(
        self,
        user_id: str,
        data: List[dict],
        resource: str
    ) -> List[dict]:
        """过滤数据"""
        permissions = self.rbac.get_user_permissions(user_id)

        # 根据权限过滤敏感字段
        filtered = []
        for item in data:
            filtered_item = {}
            for key, value in item.items():
                if f"{resource}:{key}" in permissions or "*" in permissions:
                    filtered_item[key] = value
            filtered.append(filtered_item)

        return filtered
```

## 三、审计日志

```python
class AuditLogger:
    """审计日志器"""

    def __init__(self, db):
        self.db = db

    async def log(
        self,
        user_id: str,
        action: str,
        resource: str,
        details: dict = None
    ):
        """记录审计日志"""
        log_entry = AuditLog(
            id=generate_id(),
            user_id=user_id,
            action=action,
            resource=resource,
            details=details,
            ip_address=get_request_ip(),
            user_agent=get_user_agent(),
            timestamp=datetime.now()
        )

        await self.db.insert(log_entry)

    async def get_logs(
        self,
        user_id: str = None,
        action: str = None,
        start_time: datetime = None,
        end_time: datetime = None
    ) -> List[AuditLog]:
        """查询审计日志"""
        query = {}

        if user_id:
            query["user_id"] = user_id
        if action:
            query["action"] = action
        if start_time or end_time:
            query["timestamp"] = {}
            if start_time:
                query["timestamp"]["$gte"] = start_time
            if end_time:
                query["timestamp"]["$lte"] = end_time

        return await self.db.find(query)
```
