## 一、用户注册登录

```python
"""
认证服务
"""

from datetime import datetime, timedelta
import jwt
from passlib.context import CryptContext

class AuthService:
    """认证服务"""

    def __init__(self, secret_key: str):
        self.secret_key = secret_key
        self.pwd_context = CryptContext(schemes=["bcrypt"])

    async def register(self, user_data: UserCreate) -> User:
        """用户注册"""
        # 检查用户名是否存在
        if await self.user_exists(user_data.username):
            raise ValueError("用户名已存在")

        # 密码加密
        hashed_password = self.pwd_context.hash(user_data.password)

        # 创建用户
        user = User(
            id=generate_id(),
            username=user_data.username,
            email=user_data.email,
            hashed_password=hashed_password,
            created_at=datetime.now()
        )

        await self.save_user(user)
        return user

    async def login(self, username: str, password: str) -> Token:
        """用户登录"""
        # 验证用户
        user = await self.get_user(username)
        if not user:
            raise ValueError("用户不存在")

        # 验证密码
        if not self.pwd_context.verify(password, user.hashed_password):
            raise ValueError("密码错误")

        # 生成token
        access_token = self._create_access_token(user)

        return Token(
            access_token=access_token,
            token_type="bearer",
            expires_in=7200
        )

    def _create_access_token(self, user: User) -> str:
        """创建访问令牌"""
        payload = {
            "sub": user.id,
            "username": user.username,
            "roles": user.roles,
            "exp": datetime.utcnow() + timedelta(hours=2)
        }

        return jwt.encode(payload, self.secret_key, algorithm="HS256")

    def verify_token(self, token: str) -> TokenPayload:
        """验证令牌"""
        try:
            payload = jwt.decode(
                token,
                self.secret_key,
                algorithms=["HS256"]
            )
            return TokenPayload(**payload)
        except jwt.ExpiredSignatureError:
            raise ValueError("令牌已过期")
        except jwt.InvalidTokenError:
            raise ValueError("无效令牌")
```

## 二、权限管理

```python
"""
权限管理
"""

from typing import List, Set

class RBACManager:
    """RBAC权限管理"""

    def __init__(self):
        self.roles = {}
        self.permissions = {}

    def create_role(self, role_name: str, permissions: List[str]):
        """创建角色"""
        self.roles[role_name] = permissions

    def assign_role(self, user_id: str, role_name: str):
        """分配角色"""
        if user_id not in self.permissions:
            self.permissions[user_id] = set()

        if role_name in self.roles:
            self.permissions[user_id].add(role_name)

    def check_permission(self, user_id: str, permission: str) -> bool:
        """检查权限"""
        user_roles = self.permissions.get(user_id, set())

        for role in user_roles:
            role_permissions = self.roles.get(role, [])
            if permission in role_permissions:
                return True

        return False

    def get_user_permissions(self, user_id: str) -> Set[str]:
        """获取用户所有权限"""
        user_roles = self.permissions.get(user_id, set())
        all_permissions = set()

        for role in user_roles:
            all_permissions.update(self.roles.get(role, []))

        return all_permissions
```

## 三、会话管理

```python
"""
会话管理
"""

class SessionManager:
    """会话管理器"""

    def __init__(self, redis_client):
        self.redis = redis_client
        self.session_timeout = 7200  # 2小时

    async def create_session(self, user_id: str, token: str):
        """创建会话"""
        session_key = f"session:{user_id}"

        await self.redis.setex(
            session_key,
            self.session_timeout,
            json.dumps({
                "user_id": user_id,
                "token": token,
                "created_at": datetime.now().isoformat()
            })
        )

    async def get_session(self, user_id: str) -> Optional[dict]:
        """获取会话"""
        session_key = f"session:{user_id}"
        data = await self.redis.get(session_key)

        if data:
            return json.loads(data)

        return None

    async def refresh_session(self, user_id: str):
        """刷新会话"""
        session_key = f"session:{user_id}"
        await self.redis.expire(session_key, self.session_timeout)

    async def invalidate_session(self, user_id: str):
        """使会话失效"""
        session_key = f"session:{user_id}"
        await self.redis.delete(session_key)
```
