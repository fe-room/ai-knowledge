"""
对话管理模块设计
"""

class ConversationManager:
    """对话管理器"""

    def __init__(self):
        self.session_store = SessionStore()  # 会话存储
        self.context_manager = ContextManager()  # 上下文管理
        self.message_router = MessageRouter()  # 消息路由

    def create_session(self, user_id: str) -> Session:
        """创建会话"""
        session = Session(
            session_id=generate_session_id(),
            user_id=user_id,
            status="active",
            created_at=datetime.now()
        )
        self.session_store.save(session)
        return session

    def process_message(self, session_id: str, message: str) -> Response:
        """处理消息"""
        # 1. 获取会话
        session = self.session_store.get(session_id)
        if not session:
            raise SessionNotFound()

        # 2. 检查会话状态
        if session.status != "active":
            raise SessionClosed()

        # 3. 更新上下文
        self.context_manager.add_message(
            session_id,
            {"role": "user", "content": message}
        )

        # 4. 路由处理
        response = self.message_router.route(session, message)

        # 5. 更新上下文
        self.context_manager.add_message(
            session_id,
            {"role": "assistant", "content": response.content}
        )

        # 6. 更新会话
        self.session_store.update(session)

        return response

    def close_session(self, session_id: str):
        """关闭会话"""
        session = self.session_store.get(session_id)
        session.status = "closed"
        session.closed_at = datetime.now()
        self.session_store.update(session)


class ContextManager:
    """上下文管理器"""

    def __init__(self, max_length: int = 10):
        self.max_length = max_length
        self.contexts = {}  # session_id -> message list

    def add_message(self, session_id: str, message: dict):
        """添加消息"""
        if session_id not in self.contexts:
            self.contexts[session_id] = []

        self.contexts[session_id].append(message)

        # 长度控制
        if len(self.contexts[session_id]) > self.max_length:
            self.contexts[session_id] = self._compress(
                self.contexts[session_id]
            )

    def get_context(self, session_id: str) -> List[dict]:
        """获取上下文"""
        return self.contexts.get(session_id, [])

"""
意图识别模块设计
"""

class IntentClassifier:
    """意图分类器"""

    # 意图类别
    INTENTS = {
        "faq_query": "FAQ查询",
        "product_query": "产品咨询",
        "order_query": "订单查询",
        "complaint": "投诉反馈",
        "service_request": "服务请求",
        "greeting": "问候",
        "thanks": "感谢",
        "unknown": "未知"
    }

    def __init__(self):
        self.llm_classifier = LLMClassifier()
        self.rule_classifier = RuleClassifier()
        self.entity_extractor = EntityExtractor()

    def classify(self, message: str) -> IntentResult:
        """分类意图"""
        # 1. 规则快速匹配
        rule_result = self.rule_classifier.match(message)
        if rule_result:
            return rule_result

        # 2. LLM深度分类
        llm_result = self.llm_classifier.classify(message)

        # 3. 实体提取
        entities = self.entity_extractor.extract(message)

        # 4. 组合结果
        return IntentResult(
            intent=llm_result.intent,
            confidence=llm_result.confidence,
            entities=entities
        )


class EntityExtractor:
    """实体提取器"""

    # 实体类型
    ENTITY_TYPES = {
        "product_name": "产品名称",
        "order_number": "订单号",
        "phone_number": "手机号",
        "date": "日期",
        "price": "价格",
        "quantity": "数量"
    }

    def extract(self, message: str) -> List[Entity]:
        """提取实体"""
        entities = []

        for entity_type, pattern in self._get_patterns():
            matches = pattern.findall(message)
            for match in matches:
                entities.append(Entity(
                    type=entity_type,
                    value=match,
                    confidence=0.9
                ))

        return entities

    def _get_patterns(self) -> List[Tuple[str, Pattern]]:
        """获取正则模式"""
        return [
            ("order_number", r"订单号[:\s]*(\d+)"),
            ("phone_number", r"手机号[:\s]*(\d{11})"),
            ("date", r"\d{4}[-\年]\d{1,2}[-\月]\d{1,2}[\日]?"),
            ("quantity", r"[买购]\s*(\d+)\s*[个件台]")
        ]

## 四、架构决策记录

### ADR-001: 选择FastAPI作为Web框架

**决策**：使用FastAPI构建API服务

**理由**：
- 高性能异步支持
- 自动API文档生成
- 强类型验证
- WebSocket支持

**替代方案**：
- Flask：同步，性能较低
- Django：重量级，不适合

---

### ADR-002: 使用LangChain作为Agent框架

**决策**：使用LangChain构建Agent

**理由**：
- 完整的Agent工具链
- 记忆管理内置
- 大量现成组件
- 活跃的社区支持

**替代方案**：
- 直接API调用：需要自己构建全部
- AutoGen：更适合多Agent

---

### ADR-003: Redis作为缓存层

**决策**：使用Redis作为缓存和消息队列

**理由**：
- 高性能内存存储
- 支持多种数据结构
- 消息队列功能
- 会话存储支持

**替代方案**：
- Memcached：功能单一
- 纯内存：无持久化
