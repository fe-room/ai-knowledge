```python
import pytest

        assert session is not None
        assert session.session_id is not None
        assert session.user_id == "test_user"
```

```python
"""
性能测试
"""

import asyncio
import time
from app.services.conversation import ConversationService

async def test_concurrent_sessions():
    """测试并发会话"""
    service = ConversationService()

    async def create_and_chat(i):
        session = await service.create_session(f"user_{i}")
        result = await service.process_message(session, "你好")
        return result

    # 并发创建100个会话
    start = time.time()
    tasks = [create_and_chat(i) for i in range(100)]
    results = await asyncio.gather(*tasks)
    duration = time.time() - start

    print(f"100个并发请求完成时间: {duration:.2f}秒")
    print(f"平均响应时间: {duration/100*1000:.2f}ms")

    assert len(results) == 100
    assert all(r is not None for r in results)

async def test_session_throughput():
    """测试会话吞吐量"""
    service = ConversationService()

    count = 0
    start = time.time()

    while time.time() - start < 10:  # 10秒测试
        session = await service.create_session("test_user")
        await service.process_message(session, "你好")
        await service.close_session(session.session_id)
        count += 1

    print(f"10秒内处理会话数: {count}")
    print(f"吞吐量: {count/10:.2f} 会话/秒")

if __name__ == "__main__":
    asyncio.run(test_concurrent_sessions())
    asyncio.run(test_session_throughput())
```

### 2.2 响应时间测试

```python
"""
响应时间测试
"""

import time
import statistics

class ResponseTimeTest:
    """响应时间测试"""

    def __init__(self, service):
        self.service = service
        self.results = []

    async def test_intent_recognition(self, messages: list, iterations: int = 100):
        """测试意图识别响应时间"""
        times = []

        for _ in range(iterations):
            session = await self.service.create_session("test_user")

            for message in messages:
                start = time.time()
                await self.service.process_message(session, message)
                times.append(time.time() - start)

            await self.service.close_session(session.session_id)

        self.results = {
            "avg": statistics.mean(times),
            "min": min(times),
            "max": max(times),
            "p50": statistics.median(times),
            "p95": statistics.quantiles(times, n=20)[18] if len(times) >= 20 else max(times),
            "p99": statistics.quantiles(times, n=100)[98] if len(times) >= 100 else max(times)
        }

        return self.results

    def print_results(self):
        """打印结果"""
        print("响应时间统计:")
        print(f"  平均: {self.results['avg']*1000:.2f}ms")
        print(f"  最小: {self.results['min']*1000:.2f}ms")
        print(f"  最大: {self.results['max']*1000:.2f}ms")
        print(f"  P50: {self.results['p50']*1000:.2f}ms")
        print(f"  P95: {self.results['p95']*1000:.2f}ms")
        print(f"  P99: {self.results['p99']*1000:.2f}ms")
```

### 3.3 部署脚本

```bash
#!/bin/bash
# deploy.sh

set -e

echo "开始部署智能客服系统..."

# 1. 拉取代码
git pull origin main

# 2. 构建镜像
docker-compose build

# 3. 停止旧容器
docker-compose down

# 4. 启动新容器
docker-compose up -d

# 5. 等待健康检查
echo "等待服务启动..."
sleep 10

# 6. 健康检查
for i in {1..30}; do
    if curl -f http://localhost:8000/health > /dev/null 2>&1; then
        echo "服务健康检查通过"
        break
    fi
    echo "等待服务就绪... ($i/30)"
    sleep 1
done

# 7. 运行测试
docker-compose exec api pytest tests/

echo "部署完成!"
```

```yaml
    branches: [main]

          python-version: '3.11'

          pip install -r requirements.txt
          pip install pytest pytest-asyncio

      - name: Run tests
        run: pytest tests/ -v

      - name: Upload coverage
        uses: codecov/codecov-action@v3

  build:
    needs: test
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Build Docker image
        run: docker build -t customer-service-bot:${{ github.sha }} .

      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker push customer-service-bot:${{ github.sha }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'

    steps:
      - name: Deploy to production
        run: |
          # 部署到生产环境
```
