```python
"""
对话槽位管理
"""

from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from enum import Enum

class SlotStatus(Enum):
    """槽位状态"""
    EMPTY = "empty"  # 空
    COLLECTING = "collecting"  # 收集中
    FILLED = "filled"  # 已填充
    CONFIRMED = "confirmed"  # 已确认

@dataclass
class Slot:
    """对话槽位"""
    name: str
    description: str
    required: bool = True
    value: Any = None
    status: SlotStatus = SlotStatus.EMPTY
    prompt: str = ""
    validator: callable = None

    def is_valid(self) -> bool:
        """验证值"""
        if self.validator and self.value is not None:
            return self.validator(self.value)
        return self.value is not None

@dataclass
class DialogTemplate:
    """对话模板"""
    intent_id: str
    name: str
    description: str
    slots: List[Slot]
    confirmation_required: bool = True
    confirmation_template: str = "请确认：{summary}，是否正确？"
    success_message: str = "操作成功完成"
    failure_message: str = "操作失败，请稍后重试"

    def get_next_empty_slot(self) -> Optional[Slot]:
        """获取下一个空槽位"""
        for slot in self.slots:
            if slot.required and slot.status == SlotStatus.EMPTY:
                return slot
        return None

    def get_summary(self) -> str:
        """获取摘要"""
        filled = [
            f"{slot.description}: {slot.value}"
            for slot in self.slots
            if slot.status in [SlotStatus.FILLED, SlotStatus.CONFIRMED]
        ]
        return "，".join(filled)

    def is_complete(self) -> bool:
        """是否完成"""
        return all(
            slot.status in [SlotStatus.FILLED, SlotStatus.CONFIRMED]
            for slot in self.slots if slot.required
        )


# 预定义对话模板

class SlotFillingManager:
    """槽位填充管理器"""

    def __init__(self):
        self.templates = DIALOG_TEMPLATES
        self.sessions = {}  # session_id -> current dialog state

    def start_dialog(self, session_id: str, intent_id: str) -> dict:
        """开始对话"""
        template = self.templates.get(intent_id)

        if not template:
            return {
                "status": "error",
                "message": "未知的意图类型"
            }

        # 初始化对话状态
        self.sessions[session_id] = {
            "template": template,
            "slots": {
                slot.name: Slot(
                    name=slot.name,
                    description=slot.description,
                    required=slot.required,
                    prompt=slot.prompt,
                    validator=slot.validator
                )
                for slot in template.slots
            },
            "status": "collecting"
        }

        # 返回第一个问题
        next_slot = template.get_next_empty_slot()

        if next_slot:
            return {
                "status": "collecting",
                "message": next_slot.prompt,
                "current_slot": next_slot.name
            }
        else:
            # 无需收集信息
            return self._prepare_confirmation(session_id)

    def fill_slot(
        self,
        session_id: str,
        slot_name: str,
        value: Any
    ) -> dict:
        """填充槽位"""
        state = self.sessions.get(session_id)

        if not state:
            return {"status": "error", "message": "会话不存在"}

        slot = state["slots"].get(slot_name)

        if not slot:
            return {"status": "error", "message": "槽位不存在"}

        # 验证
        if slot.validator and not slot.validator(value):
            return {
                "status": "invalid",
                "message": f"无效的{slot.description}，请重新输入",
                "current_slot": slot_name
            }

        # 填充
        slot.value = value
        slot.status = SlotStatus.FILLED

        # 检查下一个槽位
        template = state["template"]
        next_slot = template.get_next_empty_slot()

        if next_slot:
            return {
                "status": "collecting",
                "message": next_slot.prompt,
                "current_slot": next_slot.name
            }
        elif template.confirmation_required:
            return self._prepare_confirmation(session_id)
        else:
            return self._execute(session_id)

    def confirm(self, session_id: str, confirmed: bool) -> dict:
        """确认"""
        state = self.sessions.get(session_id)

        if not state:
            return {"status": "error", "message": "会话不存在"}

        if confirmed:
            return self._execute(session_id)
        else:
            # 重置
            return self._reset(session_id)

    def _prepare_confirmation(self, session_id: str) -> dict:
        """准备确认"""
        state = self.sessions[session_id]
        template = state["template"]

        summary = template.get_summary()
        message = template.confirmation_template.format(summary=summary)

        state["status"] = "confirming"

        return {
            "status": "confirming",
            "message": message,
            "summary": summary
        }

    def _execute(self, session_id: str) -> dict:
        """执行操作"""
        state = self.sessions[session_id]
        template = state["template"]

        # 标记槽位为已确认
        for slot in state["slots"].values():
            if slot.status == SlotStatus.FILLED:
                slot.status = SlotStatus.CONFIRMED

        state["status"] = "completed"

        return {
            "status": "completed",
            "message": template.success_message,
            "data": {
                slot.name: slot.value
                for slot in state["slots"].values()
            }
        }

    def _reset(self, session_id: str) -> dict:
        """重置"""
        state = self.sessions.get(session_id)

        if state:
            for slot in state["slots"].values():
                slot.value = None
                slot.status = SlotStatus.EMPTY

            return self.start_dialog(
                session_id,
                state["template"].intent_id
            )

        return {"status": "error", "message": "重置失败"}
```

### 1.2 对话上下文传递

```python
"""
对话上下文传递
"""

from typing import Dict, List, Any, Optional

class DialogContext:
    """对话上下文"""

    def __init__(self, max_history: int = 10):
        self.max_history = max_history

        # 历史信息
        self.history: List[Dict] = []

        # 当前状态
        self.current_intent: Optional[str] = None
        self.current_entities: Dict[str, Any] = {}
        self.slot_values: Dict[str, Any] = {}

        # 话题追踪
        self.topics: List[str] = []
        self.last_topic: Optional[str] = None

        # 用户画像
        self.user_preferences: Dict = {}

        # 会话级变量
        self.variables: Dict[str, Any] = {}

    def add_turn(
        self,
        role: str,
        content: str,
        intent: str = None,
        entities: Dict = None
    ):
        """添加一轮对话"""
        turn = {
            "role": role,
            "content": content,
            "intent": intent or self.current_intent,
            "entities": entities or {},
            "timestamp": datetime.now().isoformat()
        }

        self.history.append(turn)

        # 维护历史长度
        if len(self.history) > self.max_history * 2:
            self._compress_history()

    def update_entities(self, entities: Dict):
        """更新实体"""
        self.current_entities.update(entities)

    def set_variable(self, key: str, value: Any):
        """设置变量"""
        self.variables[key] = value

    def get_variable(self, key: str, default: Any = None) -> Any:
        """获取变量"""
        return self.variables.get(key, default)

    def add_topic(self, topic: str):
        """添加话题"""
        if topic != self.last_topic:
            self.topics.append(topic)
            self.last_topic = topic

            # 限制话题数量
            if len(self.topics) > 5:
                self.topics = self.topics[-5:]

    def get_recent_context(self, turns: int = 3) -> List[Dict]:
        """获取最近上下文"""
        return self.history[-turns * 2:] if len(self.history) >= turns * 2 else self.history

    def get_context_summary(self) -> str:
        """获取上下文摘要"""
        parts = []

        # 当前意图
        if self.current_intent:
            parts.append(f"当前意图: {self.current_intent}")

        # 已提取实体
        if self.current_entities:
            entities_str = ", ".join(
                f"{k}: {v}" for k, v in self.current_entities.items()
            )
            parts.append(f"相关信息: {entities_str}")

        # 最近话题
        if self.topics:
            parts.append(f"讨论话题: {' → '.join(self.topics[-3:])}")

        return " | ".join(parts)

    def _compress_history(self):
        """压缩历史"""
        # 保留最近的，对早期的生成摘要
        recent = self.history[-self.max_history:]
        older = self.history[:-self.max_history]

        # 生成摘要（实际使用LLM）
        summary = self._summarize_older(older)

        # 重建历史
        self.history = [
            {"role": "system", "content": f"[历史摘要] {summary}"}
        ] + recent

    def _summarize_older(self, turns: List[Dict]) -> str:
        """总结早期对话"""
        key_points = []

        for turn in turns:
            if turn["role"] == "user":
                key_points.append(f"用户: {turn['content'][:30]}")
            elif turn["role"] == "assistant":
                key_points.append(f"助手: {turn['content'][:30]}")

        return " → ".join(key_points[-5:])

    def to_dict(self) -> dict:
        """序列化"""
        return {
            "history": self.history,
            "current_intent": self.current_intent,
            "current_entities": self.current_entities,
            "slot_values": self.slot_values,
            "topics": self.topics,
            "user_preferences": self.user_preferences,
            "variables": self.variables
        }

    @classmethod
    def from_dict(cls, data: dict) -> "DialogContext":
        """反序列化"""
        ctx = cls()
        ctx.history = data.get("history", [])
        ctx.current_intent = data.get("current_intent")
        ctx.current_entities = data.get("current_entities", {})
        ctx.slot_values = data.get("slot_values", {})
        ctx.topics = data.get("topics", [])
        ctx.user_preferences = data.get("user_preferences", {})
        ctx.variables = data.get("variables", {})
        return ctx
```

```python
"""
任务规划器
"""

from typing import List, Dict, Optional
from dataclasses import dataclass
from enum import Enum

class TaskStatus(Enum):
    """任务状态"""
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"

@dataclass
class Task:
    """任务"""
    task_id: str
    name: str
    description: str
    dependencies: List[str] = None
    status: TaskStatus = TaskStatus.PENDING
    result: Any = None
    error: str = None

class TaskPlanner:
    """任务规划器"""

    def __init__(self):
        self.task_templates = self._load_templates()

    def plan(self, intent_result: IntentResult) -> List[Task]:
        """规划任务"""
        intent_id = intent_result.intent_id

        # 获取任务模板
        template = self.task_templates.get(intent_id)

        if not template:
            return []

        # 创建任务实例
        tasks = []
        for i, task_def in enumerate(template):
            task = Task(
                task_id=f"{intent_id}_task_{i}",
                name=task_def["name"],
                description=task_def["description"],
                dependencies=task_def.get("dependencies", [])
            )
            tasks.append(task)

        return tasks

    def _load_templates(self) -> Dict[str, List[Dict]]:
        """加载任务模板"""
        return {
            "order_query": [
                {
                    "name": "验证订单号",
                    "description": "验证订单号有效性"
                },
                {
                    "name": "查询订单状态",
                    "description": "调用订单系统查询",
                    "dependencies": ["order_query_task_0"]
                },
                {
                    "name": "格式化结果",
                    "description": "格式化查询结果",
                    "dependencies": ["order_query_task_1"]
                }
            ],

            "return_request": [
                {
                    "name": "验证订单状态",
                    "description": "检查订单是否可退货"
                },
                {
                    "name": "创建退货单",
                    "description": "在系统中创建退货申请",
                    "dependencies": ["return_request_task_0"]
                },
                {
                    "name": "通知用户",
                    "description": "发送退货确认通知",
                    "dependencies": ["return_request_task_1"]
                }
            ]
        }


class TaskExecutor:
    """任务执行器"""

    def __init__(self):
        self.handlers = {}
        self._register_handlers()

    def execute(self, tasks: List[Task], context: DialogContext) -> Dict:
        """执行任务"""
        results = {}

        # 拓扑排序
        ordered = self._topological_sort(tasks)

        for task in ordered:
            # 检查依赖
            if not self._check_dependencies(task, results):
                task.status = TaskStatus.FAILED
                task.error = "依赖任务未完成"
                continue

            # 执行
            task.status = TaskStatus.RUNNING

            try:
                handler = self.handlers.get(task.name)
                if handler:
                    result = handler(context)
                    task.result = result
                    task.status = TaskStatus.COMPLETED
                    results[task.task_id] = result
                else:
                    raise ValueError(f"未找到处理器: {task.name}")

            except Exception as e:
                task.status = TaskStatus.FAILED
                task.error = str(e)

        return {
            "success": all(
                t.status == TaskStatus.COMPLETED
                for t in tasks
            ),
            "results": results,
            "tasks": [
                {
                    "task_id": t.task_id,
                    "name": t.name,
                    "status": t.status.value,
                    "result": t.result,
                    "error": t.error
                }
                for t in tasks
            ]
        }

    def _topological_sort(self, tasks: List[Task]) -> List[Task]:
        """拓扑排序"""
        # 简化实现
        return tasks

    def _check_dependencies(
        self,
        task: Task,
        completed: Dict
    ) -> bool:
        """检查依赖"""
        if not task.dependencies:
            return True

        return all(
            dep_id in completed
            for dep_id in task.dependencies
        )

    def _register_handlers(self):
        """注册处理器"""
        self.handlers = {
            "验证订单号": self._validate_order,
            "查询订单状态": self._query_order,
            "格式化结果": self._format_result
        }

    def _validate_order(self, context: DialogContext) -> dict:
        """验证订单号"""
        order_number = context.current_entities.get("order_number")

        if not order_number:
            return {"valid": False, "error": "缺少订单号"}

        # 模拟验证
        return {"valid": True, "order_number": order_number}

    def _query_order(self, context: DialogContext) -> dict:
        """查询订单"""
        # 模拟查询
        return {
            "order_number": context.current_entities.get("order_number"),
            "status": "已发货",
            "tracking_number": "SF1234567890"
        }

    def _format_result(self, context: DialogContext) -> str:
        """格式化结果"""
        # 从变量中获取查询结果
        query_result = context.get_variable("query_result", {})

        return f"您的订单{query_result.get('order_number')}目前状态：{query_result.get('status')}，快递单号：{query_result.get('tracking_number')}"
```

```python
"""
转人工判断

from typing import Dict, List
from dataclasses import dataclass

@dataclass
class TransferRule:
    """转人工规则"""
    rule_id: str
    name: str
    condition: callable
    priority: int
    message: str

class TransferDecider:
    """转人工决策器"""

    def __init__(self):
        self.rules = self._load_rules()
        self.transfer_history = {}

    def should_transfer(
        self,
        session: Session,
        intent_result: IntentResult,
        sentiment: Dict
    ) -> Dict:
        """判断是否转人工"""
        # 获取历史转人工次数
        transfer_count = self.transfer_history.get(session.session_id, 0)

        # 按优先级检查规则
        triggered = []

        for rule in sorted(self.rules, key=lambda r: r.priority, reverse=True):
            if rule.condition(session, intent_result, sentiment):
                triggered.append(rule)

        if triggered:
            # 记录
            self.transfer_history[session.session_id] = transfer_count + 1

            # 返回最高优先级规则
            top_rule = triggered[0]

            return {
                "should_transfer": True,
                "reason": top_rule.name,
                "message": top_rule.message,
                "priority": top_rule.priority
            }

        return {
            "should_transfer": False
        }

    def _load_rules(self) -> List[TransferRule]:
        """加载规则"""
        return [
            # 高优先级：用户直接要求
            TransferRule(
                rule_id="user_request",
                name="用户要求转人工",
                condition=lambda s, i, sen: any(
                    kw in s.messages[-1].get("content", "")
                    for kw in ["转人工", "人工客服", "人工服务"]
                ),
                priority=100,
                message="正在为您转接人工客服，请稍候..."
            ),

            # 强烈负面情感
            TransferRule(
                rule_id="strong_negative",
                name="强烈负面情绪",
                condition=lambda s, i, sen: (
                    sen.get("sentiment") == "negative" and
                    sen.get("intensity", 0) > 0.7
                ),
                priority=90,
                message="检测到您情绪激动，正在为您转接人工客服..."
            ),

            # 多次无法解决
            TransferRule(
                rule_id="multiple_failures",
                name="多次未解决问题",
                condition=lambda s, i, sen: (
                    len([m for m in s.messages if m.get("role") == "user"]) >= 5 and
                    not any("成功" in m.get("content", "") for m in s.messages[-4:])
                ),
                priority=80,
                message="看起来问题比较复杂，为您转接人工客服..."
            ),

            # 复杂问题
            TransferRule(
                rule_id="complex_issue",
                name="复杂问题",
                condition=lambda s, i, sen: (
                    i.intent_id in ["complaint", "service_request"] and
                    sen.get("negative_count", 0) >= 2
                ),
                priority=70,
                message="您的问题需要人工处理，正在转接..."
            ),

            # 未知意图
            TransferRule(
                rule_id="unknown_intent",
                name="意图识别失败",
                condition=lambda s, i, sen: i.intent_id == "unknown",
                priority=60,
                message="抱歉没能理解您的问题，为您转接人工客服..."
            )
        ]


class HumanHandoff:
    """人工转接"""

    def __init__(self, agent_service):
        self.agent_service = agent_service

    async def transfer(
        self,
        session: Session,
        reason: str
    ) -> Dict:
        """执行转接"""
        # 1. 查找可用客服
        agent = await self._find_available_agent()

        if not agent:
            return {
                "success": False,
                "message": "当前客服忙碌，请稍后再试"
            }

        # 2. 创建转接会话
        handoff_session = await self._create_handoff_session(
            session,
            agent
        )

        # 3. 传递上下文
        await self._transfer_context(session, agent)

        # 4. 更新状态
        session.status = SessionStatus.WAITING
        session.metadata["handoff_to"] = agent["agent_id"]
        session.metadata["handoff_reason"] = reason

        return {
            "success": True,
            "message": f"已转接至{agent['name']}，请稍候...",
            "agent": agent,
            "estimated_wait": agent.get("estimated_wait", 60)
        }

    async def _find_available_agent(self) -> Optional[Dict]:
        """查找可用客服"""
        # 调用客服系统API
        return await self.agent_service.get_available_agent()

    async def _create_handoff_session(
        self,
        session: Session,
        agent: Dict
    ) -> Dict:
        """创建转接会话"""
        return await self.agent_service.create_session(
            user_id=session.user_id,
            agent_id=agent["agent_id"],
            original_session=session.session_id
        )

    async def _transfer_context(
        self,
        session: Session,
        agent: Dict
    ):
        """传递上下文"""
        summary = self._summarize_conversation(session)

        await self.agent_service.send_context(
            agent_id=agent["agent_id"],
            session_id=session.session_id,
            summary=summary,
            key_info={
                "intent": session.current_intent,
                "entities": session.entities,
                "message_count": len(session.messages)
            }
        )

    def _summarize_conversation(self, session: Session) -> str:
        """总结对话"""
        parts = []

        for msg in session.messages[-6:]:  # 最近3轮
            role = "用户" if msg["role"] == "user" else "客服"
            parts.append(f"{role}: {msg['content'][:50]}")

        return "\n".join(parts)
```

## 四、对话质量评估

```python
"""
对话质量评估
"""

from typing import Dict, List
from dataclasses import dataclass

@dataclass
class DialogMetrics:
    """对话指标"""
    turn_count: int
    resolution_time: float
    intent_accuracy: float
    slot_filling_rate: float
    transfer_rate: float
    user_satisfaction: float = None

class DialogEvaluator:
    """对话评估器"""

    def evaluate(self, session: Session) -> DialogMetrics:
        """评估对话"""
        # 轮次
        turn_count = session.turn_count

        # 持续时间
        if session.created_at and session.last_activity:
            resolution_time = (
                session.last_activity - session.created_at
            ).total_seconds()
        else:
            resolution_time = 0

        # 槽位填充率
        slot_filling_rate = self._calculate_slot_filling_rate(session)

        # 意图准确率（基于置信度）
        intent_accuracy = self._calculate_intent_accuracy(session)

        # 转人工率
        transfer_rate = 1.0 if session.status == SessionStatus.WAITING else 0.0

        return DialogMetrics(
            turn_count=turn_count,
            resolution_time=resolution_time,
            intent_accuracy=intent_accuracy,
            slot_filling_rate=slot_filling_rate,
            transfer_rate=transfer_rate
        )

    def _calculate_slot_filling_rate(self, session: Session) -> float:
        """计算槽位填充率"""
        # 简化：基于实体提取
        entities = session.entities
        if not entities:
            return 0.0

        expected_entities = self._get_expected_entities(session.current_intent)

        if not expected_entities:
            return 1.0

        filled = sum(1 for e in expected_entities if e in entities)

        return filled / len(expected_entities)

    def _calculate_intent_accuracy(self, session: Session) -> float:
        """计算意图准确率"""
        # 基于消息中的置信度
        confidences = []

        for msg in session.messages:
            if "intent_confidence" in msg.get("metadata", {}):
                confidences.append(msg["metadata"]["intent_confidence"])

        if not confidences:
            return 0.8  # 默认值

        return sum(confidences) / len(confidences)

    def _get_expected_entities(self, intent_id: str) -> List[str]:
        """获取期望实体"""
        entity_map = {
            "order_query": ["order_number"],
            "return_request": ["order_number", "reason", "contact_phone"]
        }

        return entity_map.get(intent_id, [])
```
