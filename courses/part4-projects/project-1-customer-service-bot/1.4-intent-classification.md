```python
class IntentHierarchy:
    """意图层次结构"""

    def __init__(self):
        self.categories = {}
        self.intents = {}
        self._build_hierarchy()

    def _build_hierarchy(self):
        """构建层次结构"""
        for intent_id, definition in INTENT_DEFINITIONS.items():
            # 按类别分组
            category = definition.category
            if category not in self.categories:
                self.categories[category] = []
            self.categories[category].append(intent_id)

            # 存储意图定义
            self.intents[intent_id] = definition

    def get_category_intents(
        self,
        category: IntentCategory
    ) -> List[IntentDefinition]:
        """获取类别下所有意图"""
        intent_ids = self.categories.get(category, [])
        return [self.intents[iid] for iid in intent_ids]

    def get_intent(self, intent_id: str) -> Optional[IntentDefinition]:
        """获取意图定义"""
        return self.intents.get(intent_id)

    def find_by_keyword(self, keyword: str) -> List[IntentDefinition]:
        """按关键词查找"""
        matches = []

        for intent in self.intents.values():
            if keyword in intent.keywords:
                matches.append(intent)

        return matches
```

```python
"""
意图识别结果

from dataclasses import dataclass, field
from typing import List, Dict, Optional

@dataclass
class Entity:
    """实体"""
    entity_type: str
    value: str
    normalized_value: Optional[str] = None
    confidence: float = 1.0
    position: tuple = (0, 0)  # (start, end)

@dataclass
class IntentResult:
    """意图识别结果"""
    intent_id: str
    intent_name: str
    confidence: float
    category: IntentCategory

    # 提取的实体
    entities: List[Entity] = field(default_factory=list)

    # 相关信息
    keywords_matched: List[str] = field(default_factory=list)
    similar_intents: List[str] = field(default_factory=list)

    # 元数据
    processing_time: float = 0.0
    method: str = "hybrid"  # rule, llm, hybrid

    def has_entity(self, entity_type: str) -> bool:
        """检查是否有某类型实体"""
        return any(e.entity_type == entity_type for e in self.entities)

    def get_entity(self, entity_type: str) -> Optional[Entity]:
        """获取某类型实体"""
        for entity in self.entities:
            if entity.entity_type == entity_type:
                return entity
        return None

    def get_all_entities(self, entity_type: str = None) -> List[Entity]:
        """获取所有实体"""
        if entity_type:
            return [e for e in self.entities if e.entity_type == entity_type]
        return self.entities
```

## 二、意图识别方法

### 2.1 规则匹配

```python
"""
规则匹配
"""

import re
from typing import List, Tuple

class RuleMatcher:
    """规则匹配器"""

    def __init__(self, intent_hierarchy: IntentHierarchy):
        self.hierarchy = intent_hierarchy
        self.patterns = self._build_patterns()

    def _build_patterns(self) -> Dict[str, List]:
        """构建模式"""
        patterns = {}

        for intent_id, definition in self.hierarchy.intents.items():
            intent_patterns = []

            # 关键词模式
            for keyword in definition.keywords:
                pattern = re.compile(
                    r'.*' + re.escape(keyword) + r'.*',
                    re.IGNORECASE
                )
                intent_patterns.append(('keyword', keyword, pattern))

            # 示例模式
            for example in definition.examples:
                # 提取核心词
                core_words = self._extract_core_words(example)
                pattern_str = r'.*'.join(
                    re.escape(w) for w in core_words
                )
                pattern = re.compile(
                    r'.*' + pattern_str + r'.*',
                    re.IGNORECASE
                )
                intent_patterns.append(('example', example, pattern))

            patterns[intent_id] = intent_patterns

        return patterns

    def match(
        self,
        message: str,
        threshold: float = 0.5
    ) -> List[Tuple[IntentResult, float]]:
        """匹配意图"""
        results = []

        for intent_id, patterns in self.patterns.items():
            score = 0.0
            matched = []

            for pattern_type, source, pattern in patterns:
                if pattern.match(message):
                    if pattern_type == 'keyword':
                        score += 0.3
                    else:
                        score += 0.5
                    matched.append(source)

            if score > 0:
                # 归一化
                score = min(score, 1.0)

                if score >= threshold:
                    definition = self.hierarchy.get_intent(intent_id)

                    result = IntentResult(
                        intent_id=intent_id,
                        intent_name=definition.name,
                        confidence=score,
                        category=definition.category,
                        keywords_matched=matched,
                        method="rule"
                    )

                    results.append((result, score))

        # 排序
        results.sort(key=lambda x: x[1], reverse=True)

        return results

    def _extract_core_words(self, text: str) -> List[str]:
        """提取核心词"""
        # 去除停用词和标点
        stopwords = {"的", "了", "是", "在", "我", "有", "和", "就", "不", "吗", "呢"}
        words = [w for w in text if w not in stopwords and not re.match(r'[^\w]', w)]
        return words[:5]  # 最多5个核心词
```

### 2.2 LLM意图分类

```python
"""
LLM意图分类
"""

from typing import List, Optional
import json

class LLMIntentClassifier:
    """LLM意图分类器"""

    def __init__(self, llm_client, intent_hierarchy: IntentHierarchy):
        self.llm = llm_client
        self.hierarchy = intent_hierarchy

    def classify(
        self,
        message: str,
        context: List[Dict] = None
    ) -> IntentResult:
        """分类意图"""
        # 构建prompt
        prompt = self._build_prompt(message, context)

        # 调用LLM
        response = self.llm.generate(
            prompt,
            temperature=0.1
        )

        # 解析结果
        result = self._parse_response(response.content)

        return result

    def _build_prompt(
        self,
        message: str,
        context: List[Dict] = None
    ) -> str:
        """构建prompt"""
        # 意图列表
        intent_list = "\n".join([
            f"- {intent_id}: {defn.name} ({defn.description})"
            for intent_id, defn in self.hierarchy.intents.items()
        ])

        # 上下文
        context_str = ""
        if context:
            recent = context[-3:]  # 最近3轮
            context_str = "\n".join([
                f"{m['role']}: {m['content']}"
                for m in recent
            ])

        prompt = f"""请分析以下用户消息的意图。

可能的意图类型：
{intent_list}

{'对话历史：' + context_str if context_str else ''}
用户消息：{message}

请以JSON格式返回分析结果：
{{
    "intent_id": "意图ID",
    "confidence": 0.95,
    "entities": [
        {{"type": "实体类型", "value": "实体值"}}
    ],
    "reasoning": "判断理由"
}}

只返回JSON，不要其他内容。"""

        return prompt

    def _parse_response(self, response: str) -> IntentResult:
        """解析响应"""
        try:
            # 提取JSON
            json_start = response.find('{')
            json_end = response.rfind('}') + 1
            json_str = response[json_start:json_end]

            data = json.loads(json_str)

            intent_id = data.get("intent_id", "unknown")
            definition = self.hierarchy.get_intent(intent_id)

            if not definition:
                # 未识别意图
                return IntentResult(
                    intent_id="unknown",
                    intent_name="未识别",
                    confidence=0.0,
                    category=IntentCategory.QUERY,
                    method="llm"
                )

            # 构建结果
            entities = [
                Entity(
                    entity_type=e["type"],
                    value=e["value"]
                )
                for e in data.get("entities", [])
            ]

            return IntentResult(
                intent_id=intent_id,
                intent_name=definition.name,
                confidence=data.get("confidence", 0.5),
                category=definition.category,
                entities=entities,
                method="llm"
            )

        except Exception as e:
            # 解析失败，返回未知
            return IntentResult(
                intent_id="unknown",
                intent_name="未识别",
                confidence=0.0,
                category=IntentCategory.QUERY,
                method="llm"
            )
```

### 2.3 混合分类器

```python
"""
混合意图分类器
"""

from typing import List, Optional

class HybridIntentClassifier:
    """混合意图分类器"""

    def __init__(
        self,
        rule_matcher: RuleMatcher,
        llm_classifier: LLMIntentClassifier
    ):
        self.rule_matcher = rule_matcher
        self.llm_classifier = llm_classifier

    def classify(
        self,
        message: str,
        context: List[Dict] = None
    ) -> IntentResult:
        """混合分类"""
        # 1. 先尝试规则匹配
        rule_results = self.rule_matcher.match(message, threshold=0.7)

        if rule_results:
            top_result, score = rule_results[0]

            # 高置信度直接返回
            if score >= 0.8:
                # 补充实体提取
                entities = self._extract_entities(message, top_result.intent_id)
                top_result.entities = entities

                return top_result

        # 2. 使用LLM分类
        llm_result = self.llm_classifier.classify(message, context)

        # 3. 结果融合
        if rule_results:
            rule_result, rule_score = rule_results[0]

            # 如果规则和LLM结果一致，提高置信度
            if rule_result.intent_id == llm_result.intent_id:
                llm_result.confidence = min(
                    (llm_result.confidence + rule_score) / 2 + 0.1,
                    1.0
                )
            else:
                # 不一致，选择置信度高的
                if rule_score > llm_result.confidence + 0.2:
                    entities = self._extract_entities(message, rule_result.intent_id)
                    rule_result.entities = entities
                    return rule_result

        llm_result.method = "hybrid"
        return llm_result

    def _extract_entities(
        self,
        message: str,
        intent_id: str
    ) -> List[Entity]:
        """提取实体"""
        entities = []

        # 订单号
        order_match = re.search(r'订单[号:]?\s*(\d+)', message)
        if order_match:
            entities.append(Entity(
                entity_type="order_number",
                value=order_match.group(1),
                position=order_match.span()
            ))

        # 手机号
        phone_match = re.search(r'1[3-9]\d{9}', message)
        if phone_match:
            entities.append(Entity(
                entity_type="phone_number",
                value=phone_match.group(),
                position=phone_match.span()
            ))

        # 产品名称（简单匹配）
        product_keywords = ["手机", "电脑", "耳机", "平板"]
        for kw in product_keywords:
            if kw in message:
                entities.append(Entity(
                    entity_type="product_name",
                    value=kw,
                    confidence=0.7
                ))

        return entities


class IntentRouter:
    """意图路由器"""

    def __init__(self, classifier: HybridIntentClassifier):
        self.classifier = classifier
        self.handlers = {}

    def register_handler(
        self,
        intent_id: str,
        handler: callable
    ):
        """注册处理器"""
        self.handlers[intent_id] = handler

    def route(
        self,
        message: str,
        session: Session
    ) -> Response:
        """路由处理"""
        # 分类意图
        intent_result = self.classifier.classify(
            message,
            session.messages
        )

        # 更新会话
        session.current_intent = intent_result.intent_id
        session.entities.update({
            e.entity_type: e.value
            for e in intent_result.entities
        })

        # 调用处理器
        handler = self.handlers.get(intent_result.intent_id)

        if handler:
            return handler(message, session, intent_result)
        else:
            # 默认处理
            return self._default_handler(message, session, intent_result)

    def _default_handler(
        self,
        message: str,
        session: Session,
        intent_result: IntentResult
    ) -> Response:
        """默认处理器"""
        return Response(
            content="我理解您的问题了，让我为您查询一下。",
            intent=intent_result
        )
```

```python
"""
实体提取
"""

from typing import List, Dict, Callable
import re

@dataclass
class EntityType:
    """实体类型定义"""
    name: str
    description: str
    patterns: List[str]
    validator: Callable[[str], bool]
    normalizer: Callable[[str], str]

# 预定义实体类型
ENTITY_TYPES = {
    "order_number": EntityType(
        name="order_number",
        description="订单号",
        patterns=[
            r'订单[号:]?\s*(\d{10,20})',
            r'订单编号[是为]?\s*[：:]?\s*(\d{10,20})',
            r'(?:^|\s)(\d{15,20})(?:\s|$)'
        ],
        validator=lambda x: len(x) >= 10 and x.isdigit(),
        normalizer=lambda x: x
    ),

    "phone_number": EntityType(
        name="phone_number",
        description="手机号",
        patterns=[
            r'(1[3-9]\d{9})',
            r'手机[号:]?\s*(1[3-9]\d{9})'
        ],
        validator=lambda x: len(x) == 11 and x.startswith('1'),
        normalizer=lambda x: x
    ),

    "date": EntityType(
        name="date",
        description="日期",
        patterns=[
            r'(\d{4}[-/年]\d{1,2}[-/月]\d{1,2}[日]?)',
            r'(\d{1,2}[-/月]\d{1,2}[日]?)'
        ],
        validator=lambda x: True,  # 简化验证
        normalizer=lambda x: x.replace('年', '-').replace('月', '-').replace('日', '')
    ),

    "money": EntityType(
        name="money",
        description="金额",
        patterns=[
            r'(\d+(?:\.\d{1,2})?)\s*[元块]',
            r'[￥$]\s*(\d+(?:\.\d{1,2})?)'
        ],
        validator=lambda x: True,
        normalizer=lambda x: float(x)
    ),

    "product_name": EntityType(
        name="product_name",
        description="产品名称",
        patterns=[],  # 通过词典匹配
        validator=lambda x: True,
        normalizer=lambda x: x
    )
}


class EntityExtractor:
    """实体提取器"""

    def __init__(self):
        self.entity_types = ENTITY_TYPES
        self.product_dict = self._load_product_dict()

    def extract(self, text: str) -> List[Entity]:
        """提取实体"""
        entities = []

        for entity_type_name, entity_type in self.entity_types.items():
            # 模式匹配
            for pattern in entity_type.patterns:
                matches = re.finditer(pattern, text)

                for match in matches:
                    value = match.group(1) if match.lastindex else match.group()

                    # 验证
                    if entity_type.validator(value):
                        # 标准化
                        normalized = entity_type.normalizer(value)

                        entities.append(Entity(
                            entity_type=entity_type_name,
                            value=value,
                            normalized_value=str(normalized),
                            confidence=0.9,
                            position=match.span()
                        ))

        # 产品名称匹配
        products = self._match_products(text)
        entities.extend(products)

        # 去重
        entities = self._deduplicate(entities)

        return entities

    def _load_product_dict(self) -> Dict[str, str]:
        """加载产品词典"""
        # 实际从数据库或配置加载
        return {
            "iPhone": "iPhone",
            "iPhone15": "iPhone 15",
            "华为": "华为手机",
            "小米": "小米手机"
        }

    def _match_products(self, text: str) -> List[Entity]:
        """匹配产品名称"""
        entities = []

        for keyword, product_name in self.product_dict.items():
            if keyword in text:
                # 查找位置
                start = text.find(keyword)
                end = start + len(keyword)

                entities.append(Entity(
                    entity_type="product_name",
                    value=keyword,
                    normalized_value=product_name,
                    confidence=0.8,
                    position=(start, end)
                ))

        return entities

    def _deduplicate(self, entities: List[Entity]) -> List[Entity]:
        """去重"""
        seen = set()
        unique = []

        for entity in entities:
            key = (entity.entity_type, entity.value)
            if key not in seen:
                seen.add(key)
                unique.append(entity)

        return unique
```

```python
"""

from typing import Dict, List

class SentimentAnalyzer:
    """情感分析器"""

    def __init__(self):
        self.positive_words = self._load_words("positive")
        self.negative_words = self._load_words("negative")
        self.intensifiers = {"很", "非常", "太", "特别", "极其"}

    def analyze(self, text: str) -> Dict:
        """分析情感"""
        # 分词（简化）
        words = list(text)

        # 计算情感分数
        score = 0
        positive_count = 0
        negative_count = 0

        for i, word in enumerate(words):
            # 检查是否增强词
            if word in self.intensifiers and i + 1 < len(words):
                next_word = words[i + 1]
                if next_word in self.positive_words:
                    score += 2
                    positive_count += 1
                elif next_word in self.negative_words:
                    score -= 2
                    negative_count += 1
            else:
                if word in self.positive_words:
                    score += 1
                    positive_count += 1
                elif word in self.negative_words:
                    score -= 1
                    negative_count += 1

        # 判断情感
        if score > 0:
            sentiment = "positive"
        elif score < 0:
            sentiment = "negative"
        else:
            sentiment = "neutral"

        # 计算强度
        intensity = min(abs(score) / 5, 1.0)

        return {
            "sentiment": sentiment,
            "score": score,
            "intensity": intensity,
            "positive_count": positive_count,
            "negative_count": negative_count
        }

    def _load_words(self, sentiment: str) -> set:
        """加载情感词典"""
        if sentiment == "positive":
            return {"好", "棒", "满意", "感谢", "优秀", "喜欢", "开心", "高兴"}
        else:
            return {"差", "烂", "不满", "投诉", "失望", "讨厌", "生气", "愤怒"}

    def needs_human_attention(self, analysis: Dict) -> bool:
        """是否需要人工关注"""
        # 强烈负面情感
        if analysis["sentiment"] == "negative" and analysis["intensity"] > 0.6:
            return True

        # 多个负面词
        if analysis["negative_count"] >= 3:
            return True

        return False
```
