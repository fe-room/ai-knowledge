```typescript
// src/app.ts

import express, { Express, Request, Response } from 'express';
import cors from 'cors';
import helmet from 'helmet';

import chatRoutes from './routes/chat';
import adminRoutes from './routes/admin';
import { errorHandler } from './middleware/errorHandler';
import { rateLimiter } from './middleware/rateLimit';
import { ConversationService } from './services/ConversationService';
import { KnowledgeService } from './services/KnowledgeService';

declare global {
    namespace Express {
        interface Request {
            conversationService: ConversationService;
            knowledgeService: KnowledgeService;
        }
    }
}

export async function createApp(): Promise<Express> {
    const app = express();

    // 中间件
    app.use(helmet());
    app.use(cors());
    app.use(express.json());
    app.use(rateLimiter);

    // 初始化服务
    const conversationService = new ConversationService();
    const knowledgeService = new KnowledgeService();

    // 加载知识库
    await knowledgeService.loadKnowledge('data/knowledge.json');

    // 服务注入
    app.use((req: Request, res: Response, next) => {
        req.conversationService = conversationService;
        req.knowledgeService = knowledgeService;
        next();
    });

    // 路由
    app.use('/api/v1', chatRoutes);
    app.use('/api/v1/admin', adminRoutes);

    // 健康检查
    app.get('/health', (req: Request, res: Response) => {
        res.json({ status: 'healthy' });
    });

    // 错误处理
    app.use(errorHandler);

    return app;
}
```

### 2.3 会话模型

```typescript
// src/models/Session.ts

export enum SessionStatus {
    ACTIVE = 'active',
    WAITING = 'waiting',
    CLOSED = 'closed',
    TIMEOUT = 'timeout'
}

export interface Message {
    role: 'user' | 'assistant' | 'system';
    content: string;
    timestamp: Date;
    intent?: string;
    entities?: Record<string, any>;
}

export interface SessionData {
    sessionId: string;
    userId: string;
    status: SessionStatus;
    createdAt: Date;
    lastActivity: Date;
    messages: Message[];
    currentIntent?: string;
    entities: Record<string, any>;
    metadata: Record<string, any>;
}

export class Session {
    sessionId: string;
    userId: string;
    status: SessionStatus;
    createdAt: Date;
    lastActivity: Date;
    messages: Message[];
    currentIntent?: string;
    entities: Record<string, any>;
    metadata: Record<string, any>;
    messageCount: number = 0;
    turnCount: number = 0;

    constructor(data: Partial<SessionData>) {
        this.sessionId = data.sessionId || this.generateId();
        this.userId = data.userId || 'anonymous';
        this.status = data.status || SessionStatus.ACTIVE;
        this.createdAt = data.createdAt || new Date();
        this.lastActivity = data.lastActivity || new Date();
        this.messages = data.messages || [];
        this.currentIntent = data.currentIntent;
        this.entities = data.entities || {};
        this.metadata = data.metadata || {};
    }

    addMessage(role: 'user' | 'assistant', content: string): void {
        this.messages.push({
            role,
            content,
            timestamp: new Date()
        });

        this.messageCount++;
        if (role === 'user') {
            this.turnCount++;
        }
        this.lastActivity = new Date();
    }

    isExpired(timeoutMinutes: number = 30): boolean {
        const elapsed = (Date.now() - this.lastActivity.getTime()) / 1000;
        return elapsed > timeoutMinutes * 60;
    }

    private generateId(): string {
        return Math.random().toString(36).substring(2, 10);
    }

    toJSON(): SessionData {
        return {
            sessionId: this.sessionId,
            userId: this.userId,
            status: this.status,
            createdAt: this.createdAt,
            lastActivity: this.lastActivity,
            messages: this.messages,
            currentIntent: this.currentIntent,
            entities: this.entities,
            metadata: this.metadata
        };
    }

    static fromJSON(data: SessionData): Session {
        return new Session(data);
    }
}
```

### 2.4 对话服务

```typescript
// src/services/ConversationService.ts

import { Session, SessionStatus } from '../models/Session';
import { IntentService } from './IntentService';
import { KnowledgeService } from './KnowledgeService';
import { LLMService } from './LLMService';
import { RedisCache } from '../utils/cache';

interface ProcessResult {
    response: string;
    intent?: string;
    entities?: Record<string, any>;
    suggestions?: string[];
    transfer?: boolean;
}

export class ConversationService {
    private cache: RedisCache;
    private intentService: IntentService;
    private knowledgeService: KnowledgeService;
    private llmService: LLMService;
    private activeSessions: Map<string, Session>;

    constructor() {
        this.cache = new RedisCache();
        this.intentService = new IntentService();
        this.knowledgeService = new KnowledgeService();
        this.llmService = new LLMService();
        this.activeSessions = new Map();
    }

    async createSession(userId: string, metadata: Record<string, any> = {}): Promise<Session> {
        const session = new Session({ userId, metadata });

        await this.cache.set(`session:${session.sessionId}`, session.toJSON(), 1800);
        this.activeSessions.set(session.sessionId, session);

        return session;
    }

    async getSession(sessionId: string): Promise<Session | null> {
        // 先查活跃缓存
        if (this.activeSessions.has(sessionId)) {
            return this.activeSessions.get(sessionId)!;
        }

        // 查Redis
        const data = await this.cache.get(`session:${sessionId}`);

        if (data) {
            const session = Session.fromJSON(data as any);

            if (session.isExpired()) {
                return null;
            }

            this.activeSessions.set(sessionId, session);
            return session;
        }

        return null;
    }

    async processMessage(session: Session, message: string): Promise<ProcessResult> {
        // 添加用户消息
        session.addMessage('user', message);

        // 意图识别
        const intentResult = await this.intentService.classify(message, session.messages);
        session.currentIntent = intentResult.intentId;
        Object.assign(session.entities, intentResult.entities);

        // 路由处理
        const result = await this.routeByIntent(session, message, intentResult);

        // 添加助手消息
        session.addMessage('assistant', result.response);

        // 更新会话
        await this.updateSession(session);

        return result;
    }

    private async routeByIntent(
        session: Session,
        message: string,
        intentResult: any
    ): Promise<ProcessResult> {
        const intentId = intentResult.intentId;

        // 社交类
        if (['greeting', 'thanks', 'goodbye'].includes(intentId)) {
            return this.handleSocial(intentId);
        }

        // FAQ
        if (intentId === 'faq_query') {
            return this.handleFAQ(message);
        }

        // 订单查询
        if (intentId === 'order_query') {
            return this.handleOrderQuery(session);
        }

        // 退货
        if (intentId === 'return_request') {
            return this.handleReturnRequest(session, message);
        }

        // 默认
        return this.handleDefault(session, message, intentResult);
    }

    private handleSocial(intentId: string): ProcessResult {
        const responses: Record<string, string> = {
            greeting: '您好！我是智能客服助手，很高兴为您服务。请问有什么可以帮助您？',
            thanks: '不客气，很高兴能帮到您！',
            goodbye: '感谢您的咨询，再见！'
        };

        const suggestions: Record<string, string[]> = {
            greeting: ['查询订单', '产品咨询', '售后服务'],
            thanks: ['还有其他问题吗？'],
            goodbye: []
        };

        return {
            response: responses[intentId],
            suggestions: suggestions[intentId],
            intent: intentId
        };
    }

    private async handleFAQ(message: string): Promise<ProcessResult> {
        const results = await this.knowledgeService.search(message, 3);

        if (results.length > 0) {
            return {
                response: results[0].answer,
                intent: 'faq_query',
                suggestions: ['还有其他问题吗？', '转人工']
            };
        }

        const response = await this.llmService.generateResponse(message, []);
        return { response, intent: 'faq_query' };
    }

    private async handleOrderQuery(session: Session): Promise<ProcessResult> {
        const orderNumber = session.entities.order_number;

        if (!orderNumber) {
            return {
                response: '请提供您的订单号，我来帮您查询订单状态。',
                intent: 'order_query'
            };
        }

        const orderInfo = await this.queryOrder(orderNumber);

        return {
            response: `您的订单${orderNumber}目前状态：${orderInfo.status}。物流信息：${orderInfo.tracking}`,
            intent: 'order_query',
            entities: { order_number: orderNumber }
        };
    }

    private async handleReturnRequest(session: Session, message: string): Promise<ProcessResult> {
        const required = ['order_number', 'reason'];
        const missing = required.filter(field => !session.entities[field]);

        if (missing.length > 0) {
            const prompts: Record<string, string> = {
                order_number: '请提供您的订单号',
                reason: '请说明退货原因'
            };

            return {
                response: prompts[missing[0]],
                intent: 'return_request'
            };
        }

        const result = await this.createReturnRequest(session);

        return {
            response: `退货申请已提交，退货单号：${result.returnId}。我们会在3个工作日内处理。`,
            intent: 'return_request'
        };
    }

    private async handleDefault(
        session: Session,
        message: string,
        intentResult: any
    ): Promise<ProcessResult> {
        const knowledgeResults = await this.knowledgeService.search(message);

        const context = session.messages.slice(-6);
        if (knowledgeResults.length > 0) {
            context.push({
                role: 'system',
                content: `相关知识：${knowledgeResults[0].answer}`,
                timestamp: new Date()
            });
        }

        const response = await this.llmService.generateResponse(message, context);

        return {
            response,
            intent: intentResult.intentId
        };
    }

    async closeSession(sessionId: string): Promise<void> {
        const session = await this.getSession(sessionId);

        if (session) {
            session.status = SessionStatus.CLOSED;
            await this.updateSession(session);
            this.activeSessions.delete(sessionId);
        }
    }

    private async updateSession(session: Session): Promise<void> {
        session.lastActivity = new Date();
        await this.cache.set(`session:${session.sessionId}`, session.toJSON(), 1800);
    }

    private async queryOrder(orderNumber: string): Promise<any> {
        return { status: '已发货', tracking: 'SF1234567890' };
    }

    private async createReturnRequest(session: Session): Promise<any> {
        return { returnId: `RT${Date.now()}` };
    }
}
```

### 2.5 意图服务

```typescript
// src/services/IntentService.ts

import { LLMService } from './LLMService';

export interface IntentResult {
    intentId: string;
    intentName: string;
    confidence: number;
    entities: Record<string, any>;
}

export interface Entity {
    type: string;
    value: string;
}

export class IntentService {
    private llm: LLMService;
    private patterns: Map<string, { pattern: RegExp; name: string; confidence: number }[]>;

    constructor() {
        this.llm = new LLMService();
        this.patterns = this.loadPatterns();
    }

    async classify(message: string, context: any[]): Promise<IntentResult> {
        // 规则匹配
        const ruleResult = this.ruleMatch(message);

        if (ruleResult && ruleResult.confidence > 0.8) {
            const entities = this.extractEntities(message);
            return {
                intentId: ruleResult.intentId,
                intentName: ruleResult.name,
                confidence: ruleResult.confidence,
                entities
            };
        }

        // LLM分类
        return this.llmClassify(message, context);
    }

    private ruleMatch(message: string): { intentId: string; name: string; confidence: number } | null {
        for (const [intentId, patterns] of this.patterns) {
            for (const { pattern, name, confidence } of patterns) {
                if (pattern.test(message)) {
                    return { intentId, name, confidence };
                }
            }
        }
        return null;
    }

    private extractEntities(message: string): Record<string, string> {
        const entities: Record<string, string> = {};

        // 订单号
        const orderMatch = message.match(/订单[号:]?\s*(\d+)/);
        if (orderMatch) {
            entities.order_number = orderMatch[1];
        }

        // 手机号
        const phoneMatch = message.match(/(1[3-9]\d{9})/);
        if (phoneMatch) {
            entities.phone_number = phoneMatch[1];
        }

        return entities;
    }

    private async llmClassify(message: string, context: any[]): Promise<IntentResult> {
        const prompt = `
        分析用户消息的意图，返回JSON：
        {
            "intent_id": "意图ID",
            "intent_name": "意图名称",
            "confidence": 0.95,
            "entities": {}
        }

        可能的意图：greeting, thanks, goodbye, order_query, product_query, faq_query, return_request, complaint, unknown

        用户消息：${message}

        只返回JSON。
        `;

        const response = await this.llm.generate(prompt, 0.1);

        try {
            return JSON.parse(response);
        } catch {
            return {
                intentId: 'unknown',
                intentName: '未知',
                confidence: 0,
                entities: {}
            };
        }
    }

    async analyzeSentiment(message: string): Promise<{ sentiment: string; intensity: number }> {
        const positiveWords = new Set(['好', '棒', '满意', '感谢', '喜欢']);
        const negativeWords = new Set(['差', '烂', '不满', '投诉', '失望', '讨厌']);

        let score = 0;
        for (const char of message) {
            if (positiveWords.has(char)) score++;
            else if (negativeWords.has(char)) score--;
        }

        if (score > 0) {
            return { sentiment: 'positive', intensity: Math.min(score / 3, 1) };
        } else if (score < 0) {
            return { sentiment: 'negative', intensity: Math.min(Math.abs(score) / 3, 1) };
        }
        return { sentiment: 'neutral', intensity: 0 };
    }

    private loadPatterns(): Map<string, { pattern: RegExp; name: string; confidence: number }[]> {
        return new Map([
            ['greeting', [
                { pattern: /(你好|您好|在吗)/i, name: '问候', confidence: 0.9 }
            ]],
            ['thanks', [
                { pattern: /(谢谢|感谢|多谢)/i, name: '感谢', confidence: 0.9 }
            ]],
            ['goodbye', [
                { pattern: /(再见|拜拜)/i, name: '告别', confidence: 0.9 }
            ]],
            ['order_query', [
                { pattern: /(订单|快递|物流|到哪了)/, name: '订单查询', confidence: 0.85 }
            ]],
            ['return_request', [
                { pattern: /(退货|退款)/, name: '退货申请', confidence: 0.85 }
            ]],
            ['complaint', [
                { pattern: /(投诉|不满|太差)/, name: '投诉', confidence: 0.85 }
            ]]
        ]);
    }
}
```

### 2.6 路由

```typescript
// src/routes/chat.ts

import { Router, Request, Response } from 'express';

const router = Router();

interface ChatRequest {
    message: string;
    session_id?: string;
    user_id?: string;
    metadata?: Record<string, any>;
}

router.post('/chat', async (req: Request, res: Response) => {
    const { message, session_id, user_id, metadata } = req.body as ChatRequest;

    try {
        const service = req.conversationService;

        // 获取或创建会话
        let session = session_id
            ? await service.getSession(session_id)
            : null;

        if (!session) {
            session = await service.createSession(user_id || 'anonymous', metadata);
        }

        // 处理消息
        const result = await service.processMessage(session, message);

        res.json({
            response: result.response,
            session_id: session.sessionId,
            intent: result.intent,
            entities: result.entities,
            suggestions: result.suggestions,
            transfer: result.transfer
        });
    } catch (error) {
        res.status(500).json({ error: (error as Error).message });
    }
});

router.post('/session/create', async (req: Request, res: Response) => {
    const { user_id } = req.body;

    const session = await req.conversationService.createSession(user_id || 'anonymous');

    res.json({ session_id: session.sessionId });
});

router.get('/session/:sessionId', async (req: Request, res: Response) => {
    const session = await req.conversationService.getSession(req.params.sessionId);

    if (!session) {
        return res.status(404).json({ error: '会话不存在' });
    }

    res.json({
        session_id: session.sessionId,
        status: session.status,
        message_count: session.messageCount,
        created_at: session.createdAt
    });
});

router.post('/session/:sessionId/close', async (req: Request, res: Response) => {
    await req.conversationService.closeSession(req.params.sessionId);
    res.json({ message: '会话已关闭' });
});

export default router;
```
