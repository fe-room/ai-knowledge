```python
"""

from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import datetime

@dataclass
class KnowledgeItem:
    """知识条目"""
    id: str
    question: str  # 问题/关键词
    answer: str  # 答案
    category: str  # 分类
    tags: List[str] = field(default_factory=list)

    # 元数据
    source: Optional[str] = None
    confidence: float = 1.0
    view_count: int = 0
    helpful_count: int = 0

    # 时间信息
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    version: int = 1

    # 向量
    embedding: Optional[List[float]] = None

    def to_dict(self) -> dict:
        """转换为字典"""
        return {
            "id": self.id,
            "question": self.question,
            "answer": self.answer,
            "category": self.category,
            "tags": self.tags,
            "source": self.source,
            "confidence": self.confidence,
            "view_count": self.view_count,
            "helpful_count": self.helpful_count,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
            "version": self.version
        }


@dataclass
class KnowledgeCategory:
    """知识分类"""
    id: str
    name: str
    description: str
    parent_id: Optional[str] = None
    children: List[str] = field(default_factory=list)


class KnowledgeBase:
    """知识库"""

    def __init__(self):
        self.items: Dict[str, KnowledgeItem] = {}
        self.categories: Dict[str, KnowledgeCategory] = {}
        self.index: Dict[str, List[str]] = {}  # tag -> item_ids

    def add_item(self, item: KnowledgeItem):
        """添加知识"""
        self.items[item.id] = item

        # 更新标签索引
        for tag in item.tags:
            if tag not in self.index:
                self.index[tag] = []
            self.index[tag].append(item.id)

    def get_item(self, item_id: str) -> Optional[KnowledgeItem]:
        """获取知识"""
        return self.items.get(item_id)

    def update_item(self, item: KnowledgeItem):
        """更新知识"""
        if item.id in self.items:
            old_item = self.items[item.id]

            # 更新标签索引
            old_tags = set(old_item.tags)
            new_tags = set(item.tags)

            # 移除旧标签
            for tag in old_tags - new_tags:
                self.index[tag].remove(item.id)

            # 添加新标签
            for tag in new_tags - old_tags:
                if tag not in self.index:
                    self.index[tag] = []
                self.index[tag].append(item.id)

            # 更新版本
            item.version = old_item.version + 1
            item.updated_at = datetime.now()

            self.items[item.id] = item

    def delete_item(self, item_id: str):
        """删除知识"""
        if item_id in self.items:
            item = self.items[item_id]

            # 移除标签索引
            for tag in item.tags:
                self.index[tag].remove(item_id)

            del self.items[item_id]

    def search_by_tag(self, tag: str) -> List[KnowledgeItem]:
        """按标签搜索"""
        item_ids = self.index.get(tag, [])
        return [self.items[iid] for iid in item_ids if iid in self.items]
```

```python
"""
FAQ知识库
"""

import json
from typing import List, Tuple

class FAQManager:
    """FAQ管理器"""

    def __init__(self):
        self.faqs: List[Dict] = []
        self.patterns: Dict[str, List[str]] = {}  # 标准问题 -> 变体列表

    def load_from_file(self, file_path: str):
        """从文件加载"""
        with open(file_path, 'r', encoding='utf-8') as f:
            data = json.load(f)

        self.faqs = data.get("faqs", [])
        self.patterns = data.get("patterns", {})

        # 构建倒排索引
        self._build_index()

    def add_faq(
        self,
        question: str,
        answer: str,
        category: str = "general",
        variants: List[str] = None
    ):
        """添加FAQ"""
        faq_id = f"faq_{len(self.faqs)}"

        faq = {
            "id": faq_id,
            "question": question,
            "answer": answer,
            "category": category,
            "variants": variants or []
        }

        self.faqs.append(faq)

        # 添加变体
        if variants:
            self.patterns[question] = variants

        return faq_id

    def match(
        self,
        query: str,
        threshold: float = 0.85
    ) -> List[Tuple[Dict, float]]:
        """匹配FAQ"""
        results = []

        for faq in self.faqs:
            # 检查标准问题
            score = self._calculate_similarity(query, faq["question"])

            if score >= threshold:
                results.append((faq, score))
                continue

            # 检查变体
            for variant in faq.get("variants", []):
                score = self._calculate_similarity(query, variant)

                if score >= threshold:
                    results.append((faq, score * 0.95))  # 变体稍降权
                    break

        # 排序
        results.sort(key=lambda x: x[1], reverse=True)

        return results

    def _calculate_similarity(self, text1: str, text2: str) -> float:
        """计算相似度"""
        # 使用多种方法组合
        scores = []

        # 1. 编辑距离
        edit_score = self._edit_distance_score(text1, text2)
        scores.append(edit_score)

        # 2. Jaccard相似度
        jaccard_score = self._jaccard_score(text1, text2)
        scores.append(jaccard_score)

        # 3. 关键词匹配
        keyword_score = self._keyword_match_score(text1, text2)
        scores.append(keyword_score)

        # 加权平均
        weights = [0.3, 0.3, 0.4]
        final_score = sum(s * w for s, w in zip(scores, weights))

        return final_score

    def _edit_distance_score(self, text1: str, text2: str) -> float:
        """编辑距离相似度"""
        # Levenshtein距离
        m, n = len(text1), len(text2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]

        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j

        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if text1[i-1] == text2[j-1]:
                    dp[i][j] = dp[i-1][j-1]
                else:
                    dp[i][j] = min(
                        dp[i-1][j] + 1,
                        dp[i][j-1] + 1,
                        dp[i-1][j-1] + 1
                    )

        distance = dp[m][n]
        max_len = max(m, n)

        return 1 - distance / max_len if max_len > 0 else 1.0

    def _jaccard_score(self, text1: str, text2: str) -> float:
        """Jaccard相似度"""
        # 分词
        words1 = set(text1)
        words2 = set(text2)

        intersection = words1 & words2
        union = words1 | words2

        return len(intersection) / len(union) if union else 0.0

    def _keyword_match_score(self, text1: str, text2: str) -> float:
        """关键词匹配"""
        # 提取关键词
        keywords1 = self._extract_keywords(text1)
        keywords2 = self._extract_keywords(text2)

        if not keywords1 or not keywords2:
            return 0.0

        # 计算匹配
        matches = sum(1 for kw in keywords1 if kw in keywords2)

        return matches / max(len(keywords1), len(keywords2))

    def _extract_keywords(self, text: str) -> List[str]:
        """提取关键词"""
        # 简单实现：去除停用词
        stopwords = {"的", "了", "是", "在", "我", "有", "和", "就", "不", "人", "都", "一", "一个", "上", "也", "很", "到", "说", "要", "去", "你", "会", "着", "没有", "看", "好", "自己", "这"}

        words = [w for w in text if w not in stopwords and len(w) > 1]

        return list(set(words))
```

```python
"""
向量化处理
"""

from typing import List
import numpy as np

class EmbeddingService:
    """向量嵌入服务"""

    def __init__(self, model_name: str = "text-embedding-3-small"):
        self.model_name = model_name
        self.client = self._init_client()
        self.cache = {}

    def _init_client(self):
        """初始化客户端"""
        from openai import OpenAI
        return OpenAI()

    def embed(self, text: str) -> List[float]:
        """生成向量"""
        # 检查缓存
        if text in self.cache:
            return self.cache[text]

        # 调用API
        response = self.client.embeddings.create(
            model=self.model_name,
            input=text
        )

        embedding = response.data[0].embedding

        # 缓存
        self.cache[text] = embedding

        return embedding

    def embed_batch(self, texts: List[str]) -> List[List[float]]:
        """批量生成向量"""
        # 过滤已缓存的
        uncached = []
        uncached_indices = []

        embeddings = [None] * len(texts)

        for i, text in enumerate(texts):
            if text in self.cache:
                embeddings[i] = self.cache[text]
            else:
                uncached.append(text)
                uncached_indices.append(i)

        # 批量请求
        if uncached:
            response = self.client.embeddings.create(
                model=self.model_name,
                input=uncached
            )

            for j, data in enumerate(response.data):
                embedding = data.embedding
                embeddings[uncached_indices[j]] = embedding
                self.cache[uncached[j]] = embedding

        return embeddings

    def cosine_similarity(
        self,
        vec1: List[float],
        vec2: List[float]
    ) -> float:
        """计算余弦相似度"""
        a = np.array(vec1)
        b = np.array(vec2)

        return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))


class VectorIndex:
    """向量索引"""

    def __init__(self, dimension: int = 1536):
        self.dimension = dimension
        self.vectors = {}  # id -> vector
        self.metadata = {}  # id -> metadata
        self.index = None  # FAISS索引

    def add(self, item_id: str, vector: List[float], metadata: dict):
        """添加向量"""
        self.vectors[item_id] = vector
        self.metadata[item_id] = metadata

    def build_index(self):
        """构建索引"""
        import faiss

        # 创建索引
        self.index = faiss.IndexFlatIP(self.dimension)

        # 添加向量
        ids = list(self.vectors.keys())
        vectors = np.array([self.vectors[i] for i in ids])

        # 归一化
        faiss.normalize_L2(vectors)

        self.index.add(vectors)
        self.id_map = {i: ids[i] for i in range(len(ids))}

    def search(
        self,
        query_vector: List[float],
        top_k: int = 5
    ) -> List[Tuple[str, float, dict]]:
        """搜索相似向量"""
        # 归一化查询向量
        query = np.array([query_vector])
        faiss.normalize_L2(query)

        # 搜索
        scores, indices = self.index.search(query, top_k)

        results = []
        for i, score in zip(indices[0], scores[0]):
            if i >= 0:  # 有效索引
                item_id = self.id_map[i]
                results.append((
                    item_id,
                    float(score),
                    self.metadata[item_id]
                ))

        return results
```

```python
class HybridRetriever:
    """混合检索器"""

    def __init__(
        self,
        faq_manager: FAQManager,
        vector_index: VectorIndex,
        embedding_service: EmbeddingService
    ):
        self.faq = faq_manager
        self.vector = vector_index
        self.embedding = embedding_service

    def retrieve(
        self,
        query: str,
        top_k: int = 5
    ) -> List[SearchResult]:
        """混合检索"""
        results = []

        # 1. FAQ精确匹配
        faq_results = self._retrieve_faq(query)
        results.extend(faq_results)

        # 2. 向量语义检索
        vector_results = self._retrieve_vector(query, top_k)
        results.extend(vector_results)

        # 3. 关键词检索
        keyword_results = self._retrieve_keyword(query, top_k)
        results.extend(keyword_results)

        # 4. 去重合并
        merged = self._merge_results(results)

        # 5. 重排序
        reranked = self._rerank(merged, query)

        return reranked[:top_k]
```

```python
"""
知识管理
"""

import os
import json
from typing import List, Dict

class KnowledgeManager:
    """知识管理器"""

    def __init__(
        self,
        knowledge_base: KnowledgeBase,
        vector_index: VectorIndex,
        embedding_service: EmbeddingService
    ):
        self.kb = knowledge_base
        self.index = vector_index
        self.embedding = embedding_service

    def import_from_file(self, file_path: str):
        """从文件导入"""
        ext = os.path.splitext(file_path)[1]

        if ext == ".json":
            self._import_json(file_path)
        elif ext == ".csv":
            self._import_csv(file_path)
        elif ext == ".md":
            self._import_markdown(file_path)

        # 重建索引
        self._rebuild_index()

    def _import_json(self, file_path: str):
        """导入JSON"""
        with open(file_path, 'r', encoding='utf-8') as f:
            data = json.load(f)

        for item_data in data.get("items", []):
            item = KnowledgeItem(
                id=item_data["id"],
                question=item_data["question"],
                answer=item_data["answer"],
                category=item_data.get("category", "general"),
                tags=item_data.get("tags", [])
            )

            self.kb.add_item(item)

    def add_knowledge(
        self,
        question: str,
        answer: str,
        category: str = "general",
        tags: List[str] = None
    ) -> str:
        """添加知识"""
        import uuid

        # 创建条目
        item_id = str(uuid.uuid4())[:8]

        item = KnowledgeItem(
            id=item_id,
            question=question,
            answer=answer,
            category=category,
            tags=tags or []
        )

        # 生成向量
        text = f"{question}\n{answer}"
        item.embedding = self.embedding.embed(text)

        # 添加到知识库
        self.kb.add_item(item)

        # 更新向量索引
        self.index.add(
            item_id,
            item.embedding,
            {
                "question": question,
                "content": answer,
                "category": category
            }
        )

        return item_id

    def update_knowledge(
        self,
        item_id: str,
        question: str = None,
        answer: str = None
    ):
        """更新知识"""
        item = self.kb.get_item(item_id)

        if not item:
            raise ValueError(f"Knowledge {item_id} not found")

        # 更新字段
        if question:
            item.question = question
        if answer:
            item.answer = answer

        # 重新生成向量
        text = f"{item.question}\n{item.answer}"
        item.embedding = self.embedding.embed(text)

        # 更新
        self.kb.update_item(item)

        # 更新向量索引
        self.index.add(
            item_id,
            item.embedding,
            {
                "question": item.question,
                "content": item.answer,
                "category": item.category
            }
        )

    def _rebuild_index(self):
        """重建索引"""
        # 收集所有向量
        for item_id, item in self.kb.items.items():
            if not item.embedding:
                text = f"{item.question}\n{item.answer}"
                item.embedding = self.embedding.embed(text)

            self.index.add(
                item_id,
                item.embedding,
                {
                    "question": item.question,
                    "content": item.answer,
                    "category": item.category
                }
            )

        # 构建索引
        self.index.build_index()

    def export_knowledge(self, file_path: str):
        """导出知识"""
        data = {
            "items": [item.to_dict() for item in self.kb.items.values()],
            "categories": {
                cid: {
                    "name": cat.name,
                    "description": cat.description
                }
                for cid, cat in self.kb.categories.items()
            }
        }

        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
```

```python
"""
检索缓存
"""

from typing import List, Optional
import hashlib

class RetrievalCache:
    """检索缓存"""

    def __init__(self, redis_client, ttl: int = 3600):
        self.redis = redis_client
        self.ttl = ttl
        self.prefix = "retrieval:"

    def get(self, query: str) -> Optional[List[dict]]:
        """获取缓存"""
        key = self._make_key(query)
        value = self.redis.get(key)

        if value:
            import json
            return json.loads(value)

        return None

    def set(self, query: str, results: List[dict]):
        """设置缓存"""
        key = self._make_key(query)
        import json
        value = json.dumps(results, ensure_ascii=False)

        self.redis.setex(key, self.ttl, value)

    def invalidate(self, item_id: str = None):
        """失效缓存"""
        if item_id:
            # 删除包含该item的缓存
            pattern = f"{self.prefix}*"
            keys = self.redis.keys(pattern)

            for key in keys:
                value = self.redis.get(key)
                if value and item_id in value:
                    self.redis.delete(key)
        else:
            # 清除所有缓存
            pattern = f"{self.prefix}*"
            keys = self.redis.keys(pattern)
            if keys:
                self.redis.delete(*keys)

    def _make_key(self, query: str) -> str:
        """生成缓存key"""
        # 规范化查询
        normalized = query.strip().lower()

        # 生成hash
        hash_value = hashlib.md5(normalized.encode()).hexdigest()[:16]

        return f"{self.prefix}{hash_value}"


class QueryCache:
    """查询缓存"""

    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self.cache = {}
        self.access_order = []

    def get(self, query: str) -> Optional[List[dict]]:
        """获取缓存"""
        normalized = self._normalize(query)

        if normalized in self.cache:
            # 更新访问顺序
            self.access_order.remove(normalized)
            self.access_order.append(normalized)

            return self.cache[normalized]

        return None

    def set(self, query: str, results: List[dict]):
        """设置缓存"""
        normalized = self._normalize(query)

        # 检查容量
        if len(self.cache) >= self.max_size:
            # LRU淘汰
            oldest = self.access_order.pop(0)
            del self.cache[oldest]

        self.cache[normalized] = results
        self.access_order.append(normalized)

    def _normalize(self, query: str) -> str:
        """规范化查询"""
        return query.strip().lower()
```

- FAQ精确匹配优先于向量检索
- 使用缓存减少重复计算
- 定期更新和优化知识库
- 监控检索质量指标
