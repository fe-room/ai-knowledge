```python
# app/main.py

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from contextlib import asynccontextmanager

from app.config import settings
from app.api import chat, admin
from app.services import ConversationService, KnowledgeService
from app.utils import setup_logging

@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期"""
    # 启动
    setup_logging()

    # 初始化服务
    app.state.conversation_service = ConversationService()
    app.state.knowledge_service = KnowledgeService()

    # 加载知识库
    await app.state.knowledge_service.load_knowledge("data/knowledge.json")

    yield

    # 关闭
    await app.state.conversation_service.close()

# 创建应用
app = FastAPI(
    title="智能客服系统",
    description="AI客服机器人API",
    version="1.0.0",
    lifespan=lifespan
)

# CORS
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 注册路由
app.include_router(chat.router, prefix="/api/v1", tags=["chat"])
app.include_router(admin.router, prefix="/api/v1/admin", tags=["admin"])


@app.get("/health")
async def health_check():
    """健康检查"""
    return {"status": "healthy"}


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

### 2.3 对话API

```python
# app/api/chat.py

from fastapi import APIRouter, Depends, HTTPException
from fastapi import Request
from pydantic import BaseModel
from typing import Optional, List, Dict

from app.services.conversation import ConversationService

router = APIRouter()

# 请求模型
class ChatRequest(BaseModel):
    """聊天请求"""
    message: str
    session_id: Optional[str] = None
    user_id: Optional[str] = None
    metadata: Optional[Dict] = None

class ChatResponse(BaseModel):
    """聊天响应"""
    response: str
    session_id: str
    intent: Optional[str] = None
    entities: Optional[Dict] = None
    suggestions: Optional[List[str]] = None
    transfer: bool = False

# 获取服务
def get_conversation_service(request: Request) -> ConversationService:
    return request.app.state.conversation_service

@router.post("/chat", response_model=ChatResponse)
async def chat(
    request: ChatRequest,
    service: ConversationService = Depends(get_conversation_service)
):
    """处理聊天消息"""
    try:
        # 创建或获取会话
        if request.session_id:
            session = await service.get_session(request.session_id)
        else:
            session = await service.create_session(
                user_id=request.user_id or "anonymous",
                metadata=request.metadata
            )

        # 处理消息
        result = await service.process_message(
            session,
            request.message
        )

        return ChatResponse(
            response=result["response"],
            session_id=session.session_id,
            intent=result.get("intent"),
            entities=result.get("entities"),
            suggestions=result.get("suggestions"),
            transfer=result.get("transfer", False)
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.post("/session/create")
async def create_session(
    user_id: str,
    service: ConversationService = Depends(get_conversation_service)
):
    """创建新会话"""
    session = await service.create_session(user_id=user_id)
    return {"session_id": session.session_id}

@router.get("/session/{session_id}")
async def get_session(
    session_id: str,
    service: ConversationService = Depends(get_conversation_service)
):
    """获取会话信息"""
    session = await service.get_session(session_id)

    if not session:
        raise HTTPException(status_code=404, detail="会话不存在")

    return {
        "session_id": session.session_id,
        "status": session.status.value,
        "message_count": session.message_count,
        "created_at": session.created_at.isoformat()
    }

@router.post("/session/{session_id}/close")
async def close_session(
    session_id: str,
    service: ConversationService = Depends(get_conversation_service)
):
    """关闭会话"""
    await service.close_session(session_id)
    return {"message": "会话已关闭"}
```

### 2.4 对话服务

```python
# app/services/conversation.py

from typing import Optional, Dict, List
from datetime import datetime
import uuid

from app.models.session import Session, SessionStatus
from app.services.intent import IntentService
from app.services.knowledge import KnowledgeService
from app.services.llm import LLMService
from app.utils.cache import RedisCache

class ConversationService:
    """对话服务"""

    def __init__(self):
        self.cache = RedisCache()
        self.intent_service = IntentService()
        self.knowledge_service = KnowledgeService()
        self.llm_service = LLMService()

        self.active_sessions: Dict[str, Session] = {}

    async def create_session(
        self,
        user_id: str,
        metadata: Dict = None
    ) -> Session:
        """创建会话"""
        session_id = str(uuid.uuid4())[:8]

        session = Session(
            session_id=session_id,
            user_id=user_id,
            status=SessionStatus.ACTIVE,
            created_at=datetime.now(),
            last_activity=datetime.now(),
            metadata=metadata or {}
        )

        # 存储
        await self.cache.set(
            f"session:{session_id}",
            session.to_dict(),
            ttl=1800
        )

        self.active_sessions[session_id] = session

        return session

    async def get_session(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        # 先查活跃缓存
        if session_id in self.active_sessions:
            return self.active_sessions[session_id]

        # 查Redis
        data = await self.cache.get(f"session:{session_id}")

        if data:
            session = Session.from_dict(data)

            # 检查超时
            if session.is_expired():
                return None

            self.active_sessions[session_id] = session
            return session

        return None

    async def process_message(
        self,
        session: Session,
        message: str
    ) -> Dict:
        """处理消息"""
        # 添加用户消息
        session.add_message("user", message)

        # 1. 意图识别
        intent_result = await self.intent_service.classify(
            message,
            session.messages
        )

        session.current_intent = intent_result.intent_id
        session.entities.update(intent_result.entities_dict)

        # 2. 根据意图处理
        response = await self._route_by_intent(
            session,
            message,
            intent_result
        )

        # 3. 添加助手消息
        session.add_message("assistant", response["response"])

        # 4. 更新会话
        await self._update_session(session)

        return response

    async def _route_by_intent(
        self,
        session: Session,
        message: str,
        intent_result
    ) -> Dict:
        """意图路由"""
        intent_id = intent_result.intent_id

        # 社交类意图
        if intent_id in ["greeting", "thanks", "goodbye"]:
            return await self._handle_social(intent_id)

        # FAQ查询
        if intent_id == "faq_query":
            return await self._handle_faq(message)

        # 订单查询
        if intent_id == "order_query":
            return await self._handle_order_query(session)

        # 退货申请
        if intent_id == "return_request":
            return await self._handle_return_request(session, message)

        # 投诉
        if intent_id == "complaint":
            return await self._handle_complaint(session, message)

        # 默认：知识库检索 + LLM生成
        return await self._handle_default(session, message, intent_result)

    async def _handle_social(self, intent_id: str) -> Dict:
        """处理社交意图"""
        responses = {
            "greeting": "您好！我是智能客服助手，很高兴为您服务。请问有什么可以帮助您？",
            "thanks": "不客气，很高兴能帮到您！",
            "goodbye": "感谢您的咨询，再见！"
        }

        suggestions = {
            "greeting": ["查询订单", "产品咨询", "售后服务"],
            "thanks": ["还有其他问题吗？"],
            "goodbye": []
        }

        return {
            "response": responses[intent_id],
            "suggestions": suggestions.get(intent_id),
            "intent": intent_id
        }

    async def _handle_faq(self, message: str) -> Dict:
        """处理FAQ查询"""
        # 检索知识库
        results = await self.knowledge_service.search(message, top_k=3)

        if results:
            top_result = results[0]

            return {
                "response": top_result["answer"],
                "intent": "faq_query",
                "suggestions": ["还有其他问题吗？", "转人工"]
            }

        # 无结果，使用LLM
        response = await self.llm_service.generate_response(
            message,
            context=[]
        )

        return {
            "response": response,
            "intent": "faq_query"
        }

    async def _handle_order_query(self, session: Session) -> Dict:
        """处理订单查询"""
        # 检查是否有订单号
        order_number = session.entities.get("order_number")

        if not order_number:
            return {
                "response": "请提供您的订单号，我来帮您查询订单状态。",
                "intent": "order_query",
                "entities_needed": ["order_number"]
            }

        # 查询订单（模拟）
        order_info = await self._query_order(order_number)

        return {
            "response": f"您的订单{order_number}目前状态：{order_info['status']}。物流信息：{order_info['tracking']}",
            "intent": "order_query",
            "entities": {"order_number": order_number}
        }

    async def _handle_return_request(
        self,
        session: Session,
        message: str
    ) -> Dict:
        """处理退货申请"""
        # 槽位检查
        required = ["order_number", "reason"]

        missing = [
            field for field in required
            if not session.entities.get(field)
        ]

        if missing:
            prompts = {
                "order_number": "请提供您的订单号",
                "reason": "请说明退货原因"
            }

            return {
                "response": prompts.get(missing[0], "请提供更多信息"),
                "intent": "return_request",
                "entities_needed": missing
            }

        # 创建退货申请
        result = await self._create_return_request(session)

        return {
            "response": f"退货申请已提交，退货单号：{result['return_id']}。我们会在3个工作日内处理。",
            "intent": "return_request"
        }

    async def _handle_complaint(
        self,
        session: Session,
        message: str
    ) -> Dict:
        """处理投诉"""
        # 情感分析
        sentiment = await self.intent_service.analyze_sentiment(message)

        # 强烈负面情绪，转人工
        if sentiment["sentiment"] == "negative" and sentiment["intensity"] > 0.7:
            return {
                "response": "非常抱歉给您带来不好的体验，正在为您转接人工客服...",
                "intent": "complaint",
                "transfer": True
            }

        # 记录投诉
        complaint_id = await self._record_complaint(session, message)

        return {
            "response": f"非常抱歉给您带来困扰，您的投诉已记录（投诉编号：{complaint_id}），我们会尽快处理并给您反馈。",
            "intent": "complaint"
        }

    async def _handle_default(
        self,
        session: Session,
        message: str,
        intent_result
    ) -> Dict:
        """默认处理"""
        # 检索知识库
        knowledge_results = await self.knowledge_service.search(message)

        # 构建上下文
        context = session.messages[-6:]  # 最近3轮
        context.append({
            "role": "system",
            "content": f"相关知识：{knowledge_results[0]['answer'] if knowledge_results else '无'}"
        })

        # LLM生成
        response = await self.llm_service.generate_response(
            message,
            context=context
        )

        return {
            "response": response,
            "intent": intent_result.intent_id
        }

    async def close_session(self, session_id: str):
        """关闭会话"""
        session = await self.get_session(session_id)

        if session:
            session.status = SessionStatus.CLOSED
            await self._update_session(session)
            self.active_sessions.pop(session_id, None)

    async def _update_session(self, session: Session):
        """更新会话"""
        session.last_activity = datetime.now()
        await self.cache.set(
            f"session:{session.session_id}",
            session.to_dict(),
            ttl=1800
        )

    async def _query_order(self, order_number: str) -> Dict:
        """查询订单（模拟）"""
        return {
            "status": "已发货",
            "tracking": "SF1234567890"
        }

    async def _create_return_request(self, session: Session) -> Dict:
        """创建退货申请（模拟）"""
        return {
            "return_id": f"RT{datetime.now().strftime('%Y%m%d%H%M%S')}"
        }

    async def _record_complaint(self, session: Session, message: str) -> str:
        """记录投诉（模拟）"""
        return f"CP{datetime.now().strftime('%Y%m%d%H%M%S')}"

    async def close(self):
        """关闭服务"""
        await self.cache.close()
```

### 2.5 意图服务

```python
# app/services/intent.py

from typing import Dict, List, Optional
import re

from app.models.intent import IntentResult, Entity
from app.services.llm import LLMService

class IntentService:
    """意图识别服务"""

    def __init__(self):
        self.llm = LLMService()
        self.intent_patterns = self._load_patterns()

    async def classify(
        self,
        message: str,
        context: List[Dict] = None
    ) -> IntentResult:
        """分类意图"""
        # 1. 规则匹配
        rule_result = self._rule_match(message)

        if rule_result and rule_result["confidence"] > 0.8:
            # 提取实体
            entities = self._extract_entities(message)

            return IntentResult(
                intent_id=rule_result["intent_id"],
                intent_name=rule_result["name"],
                confidence=rule_result["confidence"],
                entities=entities
            )

        # 2. LLM分类
        llm_result = await self._llm_classify(message, context)

        return llm_result

    def _rule_match(self, message: str) -> Optional[Dict]:
        """规则匹配"""
        for intent_id, patterns in self.intent_patterns.items():
            for pattern_info in patterns:
                if pattern_info["pattern"].search(message):
                    return {
                        "intent_id": intent_id,
                        "name": pattern_info["name"],
                        "confidence": pattern_info["confidence"]
                    }

        return None

    def _extract_entities(self, message: str) -> List[Entity]:
        """提取实体"""
        entities = []

        # 订单号
        order_match = re.search(r'订单[号:]?\s*(\d+)', message)
        if order_match:
            entities.append(Entity(
                entity_type="order_number",
                value=order_match.group(1)
            ))

        # 手机号
        phone_match = re.search(r'(1[3-9]\d{9})', message)
        if phone_match:
            entities.append(Entity(
                entity_type="phone_number",
                value=phone_match.group(1)
            ))

        return entities

    async def _llm_classify(
        self,
        message: str,
        context: List[Dict] = None
    ) -> IntentResult:
        """LLM分类"""
        prompt = f"""
        分析以下用户消息的意图，返回JSON格式：
        {{
            "intent_id": "意图ID",
            "intent_name": "意图名称",
            "confidence": 0.95,
            "entities": []
        }}

        可能的意图：
        - greeting: 问候
        - thanks: 感谢
        - goodbye: 告别
        - order_query: 订单查询
        - product_query: 产品咨询
        - faq_query: 常见问题
        - return_request: 退货申请
        - complaint: 投诉建议
        - unknown: 未知

        用户消息：{message}

        只返回JSON。
        """

        response = await self.llm.generate(prompt, temperature=0.1)

        # 解析结果
        import json
        try:
            data = json.loads(response)
            return IntentResult(
                intent_id=data.get("intent_id", "unknown"),
                intent_name=data.get("intent_name", "未知"),
                confidence=data.get("confidence", 0.5),
                entities=[]
            )
        except:
            return IntentResult(
                intent_id="unknown",
                intent_name="未知",
                confidence=0.0,
                entities=[]
            )

    async def analyze_sentiment(self, message: str) -> Dict:
        """情感分析"""
        positive_words = {"好", "棒", "满意", "感谢", "喜欢"}
        negative_words = {"差", "烂", "不满", "投诉", "失望", "讨厌"}

        score = 0
        for word in message:
            if word in positive_words:
                score += 1
            elif word in negative_words:
                score -= 1

        if score > 0:
            return {"sentiment": "positive", "intensity": min(score / 3, 1.0)}
        elif score < 0:
            return {"sentiment": "negative", "intensity": min(abs(score) / 3, 1.0)}
        else:
            return {"sentiment": "neutral", "intensity": 0.0}

    def _load_patterns(self) -> Dict:
        """加载模式"""
        return {
            "greeting": [
                {"pattern": re.compile(r'(你好|您好|在吗)', re.I), "name": "问候", "confidence": 0.9}
            ],
            "thanks": [
                {"pattern": re.compile(r'(谢谢|感谢|多谢)', re.I), "name": "感谢", "confidence": 0.9}
            ],
            "goodbye": [
                {"pattern": re.compile(r'(再见|拜拜)', re.I), "name": "告别", "confidence": 0.9}
            ],
            "order_query": [
                {"pattern": re.compile(r'(订单|快递|物流|到哪了)'), "name": "订单查询", "confidence": 0.85}
            ],
            "return_request": [
                {"pattern": re.compile(r'(退货|退款)'), "name": "退货申请", "confidence": 0.85}
            ],
            "complaint": [
                {"pattern": re.compile(r'(投诉|不满|太差)'), "name": "投诉", "confidence": 0.85}
            ]
        }
```
