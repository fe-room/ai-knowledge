## 一、对话状态设计

### 1.1 会话状态模型

```python
"""
会话状态模型
"""

from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Dict, Optional
from enum import Enum

class SessionStatus(Enum):
    """会话状态"""
    ACTIVE = "active"  # 活跃
    WAITING = "waiting"  # 等待人工
    CLOSED = "closed"  # 已关闭
    TIMEOUT = "timeout"  # 超时

@dataclass
class Session:
    """会话"""
    session_id: str
    user_id: str
    status: SessionStatus = SessionStatus.ACTIVE
    created_at: datetime = field(default_factory=datetime.now)
    last_activity: datetime = field(default_factory=datetime.now)

    # 对话信息
    messages: List[Dict] = field(default_factory=list)
    current_intent: Optional[str] = None
    entities: Dict = field(default_factory=dict)

    # 统计信息
    message_count: int = 0
    turn_count: int = 0

    # 元数据
    metadata: Dict = field(default_factory=dict)

    def is_expired(self, timeout_minutes: int = 30) -> bool:
        """检查是否超时"""
        elapsed = (datetime.now() - self.last_activity).total_seconds()
        return elapsed > timeout_minutes * 60

    def add_message(self, role: str, content: str):
        """添加消息"""
        self.messages.append({
            "role": role,
            "content": content,
            "timestamp": datetime.now().isoformat()
        })
        self.message_count += 1
        if role == "user":
            self.turn_count += 1
        self.last_activity = datetime.now()


class SessionManager:
    """会话管理器"""

    def __init__(self, storage: SessionStorage, timeout: int = 30):
        self.storage = storage
        self.timeout = timeout
        self.active_sessions = {}

    def create_session(self, user_id: str, metadata: Dict = None) -> Session:
        """创建会话"""
        import uuid
        session_id = str(uuid.uuid4())[:8]

        session = Session(
            session_id=session_id,
            user_id=user_id,
            metadata=metadata or {}
        )

        # 存储
        self.storage.save(session)
        self.active_sessions[session_id] = session

        return session

    def get_session(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        # 先查活跃缓存
        if session_id in self.active_sessions:
            session = self.active_sessions[session_id]

            # 检查超时
            if session.is_expired(self.timeout):
                self._handle_timeout(session)
                return None

            return session

        # 查持久化存储
        session = self.storage.get(session_id)
        if session and session.status == SessionStatus.ACTIVE:
            if not session.is_expired(self.timeout):
                self.active_sessions[session_id] = session
                return session

        return None

    def update_session(self, session: Session):
        """更新会话"""
        session.last_activity = datetime.now()
        self.storage.update(session)
        if session.session_id in self.active_sessions:
            self.active_sessions[session.session_id] = session

    def close_session(self, session_id: str, reason: str = "user_closed"):
        """关闭会话"""
        session = self.get_session(session_id)
        if session:
            session.status = SessionStatus.CLOSED
            session.metadata["close_reason"] = reason
            self.storage.update(session)
            self.active_sessions.pop(session_id, None)

    def _handle_timeout(self, session: Session):
        """处理超时"""
        session.status = SessionStatus.TIMEOUT
        session.metadata["timeout_at"] = datetime.now().isoformat()
        self.storage.update(session)
        self.active_sessions.pop(session.session_id, None)

    def cleanup_expired(self):
        """清理过期会话"""
        expired = [
            sid for sid, session in self.active_sessions.items()
            if session.is_expired(self.timeout)
        ]

        for sid in expired:
            session = self.active_sessions[sid]
            self._handle_timeout(session)
```

### 1.2 会话持久化

```python
"""
会话存储实现
"""

import json
from typing import Optional
from datetime import datetime

class SessionStorage:
    """会话存储"""

    def __init__(self, db_connection):
        self.db = db_connection

    def save(self, session: Session):
        """保存会话"""
        query = """
            INSERT INTO sessions (
                session_id, user_id, status, created_at,
                last_activity, messages, metadata
            ) VALUES (?, ?, ?, ?, ?, ?, ?)
        """

        self.db.execute(query, (
            session.session_id,
            session.user_id,
            session.status.value,
            session.created_at.isoformat(),
            session.last_activity.isoformat(),
            json.dumps(session.messages),
            json.dumps(session.metadata)
        ))

    def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        query = """
            SELECT * FROM sessions WHERE session_id = ?
        """

        result = self.db.fetch_one(query, (session_id,))

        if result:
            return self._deserialize(result)

        return None

    def update(self, session: Session):
        """更新会话"""
        query = """
            UPDATE sessions SET
                status = ?,
                last_activity = ?,
                messages = ?,
                current_intent = ?,
                entities = ?,
                metadata = ?
            WHERE session_id = ?
        """

        self.db.execute(query, (
            session.status.value,
            session.last_activity.isoformat(),
            json.dumps(session.messages),
            session.current_intent,
            json.dumps(session.entities),
            json.dumps(session.metadata),
            session.session_id
        ))

    def _deserialize(self, data: dict) -> Session:
        """反序列化"""
        return Session(
            session_id=data["session_id"],
            user_id=data["user_id"],
            status=SessionStatus(data["status"]),
            created_at=datetime.fromisoformat(data["created_at"]),
            last_activity=datetime.fromisoformat(data["last_activity"]),
            messages=json.loads(data["messages"]),
            metadata=json.loads(data["metadata"])
        )


class RedisSessionCache:
    """Redis会话缓存"""

    def __init__(self, redis_client):
        self.redis = redis_client
        self.prefix = "session:"
        self.ttl = 3600  # 1小时

    def set(self, session: Session):
        """设置缓存"""
        key = f"{self.prefix}{session.session_id}"
        value = json.dumps({
            "session_id": session.session_id,
            "user_id": session.user_id,
            "status": session.status.value,
            "messages": session.messages,
            "last_activity": session.last_activity.isoformat()
        })

        self.redis.setex(key, self.ttl, value)

    def get(self, session_id: str) -> Optional[dict]:
        """获取缓存"""
        key = f"{self.prefix}{session_id}"
        value = self.redis.get(key)

        if value:
            return json.loads(value)

        return None

    def delete(self, session_id: str):
        """删除缓存"""
        key = f"{self.prefix}{session_id}"
        self.redis.delete(key)
```

### 2.1 消息历史管理

```python
"""
消息历史管理
"""

from typing import List, Dict
from dataclasses import dataclass

@dataclass
class Message:
    """消息"""
    role: str  # user, assistant, system
    content: str
    timestamp: datetime
    intent: Optional[str] = None
    entities: Optional[Dict] = None
    metadata: Optional[Dict] = None

class MessageHistory:
    """消息历史"""

    def __init__(self, max_length: int = 20):
        self.max_length = max_length
        self.messages: List[Message] = []

    def add(self, message: Message):
        """添加消息"""
        self.messages.append(message)

        # 超长处理
        if len(self.messages) > self.max_length:
            self._trim()

    def get_for_llm(self) -> List[Dict]:
        """获取LLM格式消息"""
        return [
            {"role": m.role, "content": m.content}
            for m in self.messages
        ]

    def get_last_n(self, n: int) -> List[Message]:
        """获取最近N条"""
        return self.messages[-n:]

    def _trim(self):
        """裁剪历史"""
        # 保留系统消息 + 最近消息
        system_messages = [
            m for m in self.messages if m.role == "system"
        ]
        recent_messages = self.messages[-(self.max_length - len(system_messages)):]

        # 如果裁剪太多，生成摘要
        removed_count = len(self.messages) - len(system_messages) - len(recent_messages)

        if removed_count > 5:
            summary = self._generate_summary(
                self.messages[len(system_messages):-len(recent_messages)]
            )
            system_messages.append(Message(
                role="system",
                content=f"历史摘要：{summary}",
                timestamp=datetime.now()
            ))

        self.messages = system_messages + recent_messages

    def _generate_summary(self, messages: List[Message]) -> str:
        """生成历史摘要"""
        # 提取关键信息
        key_points = []

        for m in messages:
            if m.role == "user":
                key_points.append(f"用户问：{m.content[:50]}")
            elif m.role == "assistant":
                key_points.append(f"助手答：{m.content[:50]}")

        return " | ".join(key_points[-10:])


class ContextWindow:
    """上下文窗口"""

    def __init__(self, max_tokens: int = 4000):
        self.max_tokens = max_tokens
        self.token_counter = TokenCounter()

    def build_context(
        self,
        history: MessageHistory,
        current_query: str,
        system_prompt: str
    ) -> List[Dict]:
        """构建上下文"""
        messages = [{"role": "system", "content": system_prompt}]

        # 添加历史（动态裁剪）
        history_messages = history.get_for_llm()

        # 计算token
        current_tokens = self.token_counter.count(
            [messages[0]] +
            [{"role": "user", "content": current_query}]
        )

        # 按重要性添加历史
        remaining_tokens = self.max_tokens - current_tokens

        for msg in reversed(history_messages):
            msg_tokens = self.token_counter.count([msg])

            if remaining_tokens >= msg_tokens:
                messages.insert(1, msg)
                remaining_tokens -= msg_tokens
            else:
                break

        # 添加当前查询
        messages.append({"role": "user", "content": current_query})

        return messages


class TokenCounter:
    """Token计数器"""

    def __init__(self):
        # 使用tiktoken
        import tiktoken
        self.encoder = tiktoken.get_encoding("cl100k_base")

    def count(self, messages: List[Dict]) -> int:
        """计算token数"""
        total = 0

        for msg in messages:
            # 每条消息的格式开销
            total += 4  # role + content 格式

            # 内容token
            total += len(self.encoder.encode(msg["content"]))

        return total + 2  # 结束标记
```

### 2.2 上下文优化策略

```python
"""
上下文优化策略
"""

class ContextOptimizer:
    """上下文优化器"""

    def __init__(self):
        self.strategies = [
            TrimmingStrategy(),
            SummarizationStrategy(),
            ImportanceStrategy()
        ]

    def optimize(
        self,
        messages: List[Dict],
        max_tokens: int,
        current_query: str
    ) -> List[Dict]:
        """优化上下文"""
        # 计算当前token
        token_count = self._count_tokens(messages +
            [{"role": "user", "content": current_query}]
        )

        if token_count <= max_tokens:
            return messages

        # 应用优化策略
        optimized = messages

        for strategy in self.strategies:
            optimized = strategy.apply(optimized, max_tokens, current_query)

            token_count = self._count_tokens(
                optimized + [{"role": "user", "content": current_query}]
            )

            if token_count <= max_tokens:
                break

        return optimized


class ImportanceStrategy:
    """重要性策略"""

    def apply(
        self,
        messages: List[Dict],
        max_tokens: int,
        current_query: str
    ) -> List[Dict]:
        """按重要性保留"""
        # 评估每条消息重要性
        scored_messages = []

        for msg in messages:
            score = self._calculate_importance(msg, current_query)
            scored_messages.append((score, msg))

        # 按重要性排序
        scored_messages.sort(key=lambda x: x[0], reverse=True)

        # 保留高重要性消息
        selected = []
        current_tokens = self._count_tokens(
            [{"role": "user", "content": current_query}]
        )

        for score, msg in scored_messages:
            msg_tokens = self._count_tokens([msg])
            if current_tokens + msg_tokens <= max_tokens:
                selected.append(msg)
                current_tokens += msg_tokens

        # 恢复时间顺序
        selected.sort(key=lambda m: messages.index(m))

        return selected

    def _calculate_importance(self, msg: Dict, query: str) -> float:
        """计算重要性"""
        score = 0.0

        # 系统消息最重要
        if msg["role"] == "system":
            return 100.0

        # 内容与查询相关性
        similarity = self._semantic_similarity(msg["content"], query)
        score += similarity * 30

        # 最近的消息更重要
        score += 10

        # 包含关键信息
        if any(kw in msg["content"] for kw in ["订单", "产品", "问题"]):
            score += 15

        return score


class SummarizationStrategy:
    """摘要策略"""

    def __init__(self, llm_client):
        self.llm = llm_client

    def apply(
        self,
        messages: List[Dict],
        max_tokens: int,
        current_query: str
    ) -> List[Dict]:
        """生成摘要替代"""
        # 分离早期和近期消息
        split_point = len(messages) * 2 // 3

        early_messages = messages[:split_point]
        recent_messages = messages[split_point:]

        # 生成早期摘要
        summary = self._generate_summary(early_messages)

        # 组合
        optimized = [{"role": "system", "content": summary}]
        optimized.extend(recent_messages)

        return optimized

    def _generate_summary(self, messages: List[Dict]) -> str:
        """生成摘要"""
        # 构建摘要请求
        prompt = f"""
        请将以下对话历史总结为简洁的摘要，保留关键信息：

        {self._format_messages(messages)}

        摘要要求：
        1. 保留用户的主要意图和问题
        2. 保留已讨论的关键信息（产品、订单等）
        3. 保持在100字以内
        """

        response = self.llm.generate(prompt)
        return f"对话历史摘要：{response.content}"
```

## 三、对话流程控制

### 3.1 流程状态机

```python
"""
对话流程状态机
"""

from enum import Enum
from typing import Dict, Callable, Optional

class DialogState(Enum):
    """对话状态"""
    GREETING = "greeting"  # 问候
    COLLECTING_INFO = "collecting_info"  # 收集信息
    PROCESSING = "processing"  # 处理请求
    WAITING_CONFIRMATION = "waiting_confirmation"  # 等待确认
    COMPLETED = "completed"  # 完成
    TRANSFER = "transfer"  # 转人工

class DialogStateMachine:
    """对话状态机"""

    def __init__(self):
        self.state = DialogState.GREETING
        self.context = {}
        self.handlers = self._setup_handlers()

    def _setup_handlers(self) -> Dict[DialogState, Callable]:
        """设置状态处理器"""
        return {
            DialogState.GREETING: self._handle_greeting,
            DialogState.COLLECTING_INFO: self._handle_collecting,
            DialogState.PROCESSING: self._handle_processing,
            DialogState.WAITING_CONFIRMATION: self._handle_confirmation,
            DialogState.COMPLETED: self._handle_completed,
            DialogState.TRANSFER: self._handle_transfer
        }

    def process(self, message: str, session: Session) -> DialogResponse:
        """处理消息"""
        handler = self.handlers[self.state]
        response = handler(message, session)

        return response

    def _handle_greeting(
        self,
        message: str,
        session: Session
    ) -> DialogResponse:
        """处理问候"""
        # 检测意图
        if self._is_greeting(message):
            return DialogResponse(
                content="您好！我是智能客服助手，很高兴为您服务。请问有什么可以帮助您？",
                state=DialogState.GREETING,
                suggestions=["查询订单", "产品咨询", "售后服务", "投诉建议"]
            )

        # 有具体问题，进入信息收集
        self.state = DialogState.COLLECTING_INFO
        return self._handle_collecting(message, session)

    def _handle_collecting(
        self,
        message: str,
        session: Session
    ) -> DialogResponse:
        """收集信息"""
        intent = self._detect_intent(message)
        entities = self._extract_entities(message)

        session.current_intent = intent
        session.entities.update(entities)

        # 检查是否需要更多信息
        missing_info = self._check_missing_info(intent, session.entities)

        if missing_info:
            return DialogResponse(
                content=f"请问您的{missing_info}是什么？",
                state=DialogState.COLLECTING_INFO
            )

        # 信息完整，进入处理
        self.state = DialogState.PROCESSING
        return self._handle_processing(message, session)

    def _handle_processing(
        self,
        message: str,
        session: Session
    ) -> DialogResponse:
        """处理请求"""
        # 根据意图执行相应处理
        result = self._execute_intent(session)

        if result.needs_confirmation:
            self.state = DialogState.WAITING_CONFIRMATION
            return DialogResponse(
                content=result.content,
                state=DialogState.WAITING_CONFIRMATION,
                options=["确认", "取消", "修改"]
            )

        self.state = DialogState.COMPLETED
        return self._handle_completed(message, session)

    def _handle_confirmation(
        self,
        message: str,
        session: Session
    ) -> DialogResponse:
        """处理确认"""
        if "确认" in message or "是的" in message:
            # 执行最终操作
            self._execute_action(session)
            self.state = DialogState.COMPLETED
            return DialogResponse(
                content="操作已完成，还有其他问题吗？",
                state=DialogState.COMPLETED
            )

        elif "取消" in message or "不是" in message:
            self.state = DialogState.COLLECTING_INFO
            return DialogResponse(
                content="好的，请问您需要什么帮助？",
                state=DialogState.COLLECTING_INFO
            )

        else:
            return DialogResponse(
                content="请回复：确认、取消或修改",
                state=DialogState.WAITING_CONFIRMATION
            )

    def reset(self):
        """重置状态"""
        self.state = DialogState.GREETING
        self.context = {}


@dataclass
class DialogResponse:
    """对话响应"""
    content: str
    state: DialogState
    suggestions: List[str] = None
    options: List[str] = None
    transfer: bool = False
    metadata: Dict = None
```

## 四、会话恢复与迁移

### 4.1 会话恢复

```python
"""
会话恢复
"""

class SessionRestorer:
    """会话恢复器"""

    def __init__(self, storage: SessionStorage):
        self.storage = storage

    def restore(self, user_id: str) -> Optional[Session]:
        """恢复最近会话"""
        # 查找用户最近活跃会话
        recent_session = self.storage.find_recent(user_id)

        if recent_session and recent_session.status == SessionStatus.ACTIVE:
            return recent_session

        # 尝试恢复超时会话
        if recent_session and recent_session.status == SessionStatus.TIMEOUT:
            # 检查是否可以恢复
            elapsed = (datetime.now() - recent_session.last_activity).total_seconds()

            if elapsed < 3600:  # 1小时内
                return self._reactivate(recent_session)

        return None

    def _reactivate(self, session: Session) -> Session:
        """重新激活会话"""
        session.status = SessionStatus.ACTIVE
        session.last_activity = datetime.now()
        session.metadata["restored"] = True

        self.storage.update(session)

        return session

    def resume_context(self, session: Session) -> str:
        """恢复上下文提示"""
        # 提取最后话题
        last_user_msg = None
        last_intent = None

        for msg in reversed(session.messages):
            if msg["role"] == "user":
                last_user_msg = msg["content"]
                break

        if last_user_msg:
            return f"我们上次在讨论关于'{last_user_msg[:30]}...'的问题，是否继续？"

        return "欢迎回来！请问有什么可以帮助您？"
```

## 总结

### 核心要点

1. **会话管理**：创建、维护、关闭、超时处理
2. **上下文管理**：消息历史、token控制、动态裁剪
3. **流程控制**：状态机、意图驱动、多步骤处理
4. **持久化**：数据库存储、缓存加速、会话恢复

### 最佳实践

- 合理设置会话超时时间
- 动态裁剪上下文而非硬性截断
- 使用摘要保留关键信息
- 支持会话恢复提升体验
