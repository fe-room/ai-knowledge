```python
# main.py

from fastapi import FastAPI, UploadFile, File
from pydantic import BaseModel

from document_processor import DocumentPipeline
from index_manager import IndexManager
from retriever import HybridRetriever
from generator import AnswerGenerator

app = FastAPI(title="知识库问答系统")

# 初始化服务
pipeline = DocumentPipeline()
index_manager = IndexManager()
retriever = HybridRetriever()
generator = AnswerGenerator()

class QueryRequest(BaseModel):
    query: str
    top_k: int = 5

class QueryResponse(BaseModel):
    answer: str
    sources: List[dict]
    confidence: float

@app.post("/documents/upload")
async def upload_document(file: UploadFile = File(...)):
    """上传文档"""
    # 保存文件
    file_path = f"/tmp/{file.filename}"
    with open(file_path, "wb") as f:
        f.write(await file.read())

    # 处理并索引
    chunks = pipeline.process(file_path)
    count = index_manager.index_chunks(chunks)

    return {"indexed_chunks": count}

@app.post("/query", response_model=QueryResponse)
async def query(request: QueryRequest):
    """问答查询"""
    # 1. 检索
    results = retriever.retrieve(request.query, request.top_k)

    # 2. 构建上下文
    context = build_context(results)

    # 3. 生成答案
    answer = generator.generate(request.query, context)

    return QueryResponse(
        answer=answer.content,
        sources=results,
        confidence=answer.confidence
    )
```

### 1.2 JavaScript版本

```typescript
// app.ts

import express from 'express';
import multer from 'multer';

import { DocumentProcessor } from './services/DocumentProcessor';
import { IndexManager } from './services/IndexManager';
import { Retriever } from './services/Retriever';
import { Generator } from './services/Generator';

const app = express();
const upload = multer({ dest: '/tmp/' });

const processor = new DocumentProcessor();
const indexManager = new IndexManager();
const retriever = new Retriever();
const generator = new Generator();

app.post('/documents/upload', upload.single('file'), async (req, res) => {
    const file = req.file;

    // 处理并索引
    const chunks = await processor.process(file.path);
    const count = await indexManager.indexChunks(chunks);

    res.json({ indexed_chunks: count });
});

app.post('/query', async (req, res) => {
    const { query, top_k = 5 } = req.body;

    // 检索
    const results = await retriever.retrieve(query, top_k);

    // 生成答案
    const answer = await generator.generate(query, results);

    res.json({
        answer: answer.content,
        sources: results,
        confidence: answer.confidence
    });
});

app.listen(3000, () => {
    console.log('Server running on port 3000');
});
```

## 二、性能测试

```python
"""
性能测试
"""

import pytest
import time

class TestPerformance:
    """性能测试"""

    def test_index_speed(self):
        """测试索引速度"""
        start = time.time()

        # 索引100个文档
        for i in range(100):
            index_manager.index_document(f"docs/doc_{i}.txt")

        duration = time.time() - start
        speed = 100 / duration

        print(f"索引速度: {speed:.2f} docs/sec")
        assert speed >= 1  # 至少1 doc/sec

    def test_query_latency(self):
        """测试查询延迟"""
        latencies = []

        for _ in range(100):
            start = time.time()
            retriever.retrieve("测试查询")
            latencies.append(time.time() - start)

        avg = sum(latencies) / len(latencies)

        print(f"平均查询延迟: {avg*1000:.2f}ms")
        assert avg < 0.5  # 小于500ms
```
