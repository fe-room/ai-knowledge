## 一、混合检索策略

```python
"""
混合检索
"""

from typing import List, Dict

class HybridRetriever:
    """混合检索器"""

    def __init__(
        self,
        vector_store: VectorStore,
        embedding_service: EmbeddingService
    ):
        self.vector_store = vector_store
        self.embedding = embedding_service

    def retrieve(
        self,
        query: str,
        top_k: int = 5
    ) -> List[Dict]:
        """混合检索"""
        # 1. 向量检索
        query_vector = self.embedding.embed(query)
        vector_results = self.vector_store.search(query_vector, top_k * 2)

        # 2. 关键词检索（BM25）
        keyword_results = self._keyword_search(query, top_k * 2)

        # 3. 融合排序
        merged = self._merge_results(vector_results, keyword_results)

        return merged[:top_k]

    def _keyword_search(self, query: str, top_k: int) -> List[Dict]:
        """关键词检索"""
        # 简化实现：使用倒排索引
        # 实际使用 Elasticsearch 或 Whoosh
        return []

    def _merge_results(
        self,
        vector_results: List[Dict],
        keyword_results: List[Dict]
    ) -> List[Dict]:
        """融合结果"""
        # RRF (Reciprocal Rank Fusion)
        scores = {}

        for rank, result in enumerate(vector_results):
            doc_id = result["id"]
            scores[doc_id] = scores.get(doc_id, 0) + 1 / (rank + 60)
            # 保存内容
            if doc_id not in scores:
                scores[doc_id] = {"result": result, "score": 0}
            scores[doc_id]["result"] = result

        for rank, result in enumerate(keyword_results):
            doc_id = result["id"]
            if doc_id not in scores:
                scores[doc_id] = {"result": result, "score": 0}
            scores[doc_id]["score"] += 1 / (rank + 60)

        # 排序
        sorted_results = sorted(
            scores.items(),
            key=lambda x: x[1]["score"],
            reverse=True
        )

        return [item[1]["result"] for item in sorted_results]
```

        self.llm = llm_client

        self,
        query: str,
        results: List[Dict],
        top_k: int = 5

        response = self.llm.generate(prompt)

        # 解析排序结果
        ranked_ids = self._parse_response(response.content)

        # 重排序
        id_to_result = {r["id"]: r for r in results}
        ranked_results = [id_to_result[i] for i in ranked_ids if i in id_to_result]

        return ranked_results[:top_k]

    def _build_prompt(self, query: str, results: List[Dict]) -> str:
        return f"""
        对以下检索结果按与查询的相关性排序：

        查询：{query}

        结果：
        {self._format_results(results)}

        返回排序后的ID列表，JSON格式：["id1", "id2", ...]
        """

    def _format_results(self, results: List[Dict]) -> str:
        lines = []
        for i, r in enumerate(results[:10]):
            lines.append(f"ID: {r['id']}, 内容: {r['content'][:100]}...")
        return "\n".join(lines)

## 三、查询优化

```python
"""
查询优化
"""

class QueryOptimizer:
    """查询优化器"""

    def optimize(self, query: str) -> str:
        """优化查询"""
        # 1. 查询扩展
        expanded = self._expand_query(query)

        # 2. 查询改写
        rewritten = self._rewrite_query(expanded)

        return rewritten

    def _expand_query(self, query: str) -> str:
        """查询扩展"""
        # 添加同义词
        synonyms = {
            "查询": ["查找", "搜索", "检索"],
            "问题": ["疑问", "困惑"]
        }

        words = list(query)
        for i, word in enumerate(words):
            if word in synonyms:
                # 添加OR同义词
                words[i] = f"({word} OR {' OR '.join(synonyms[word])})"

        return "".join(words)

    def _rewrite_query(self, query: str) -> str:
        """查询改写"""
        # 移除停用词
        stopwords = {"的", "了", "是", "在", "我", "有"}
        words = [w for w in query if w not in stopwords]

        return "".join(words)
```
