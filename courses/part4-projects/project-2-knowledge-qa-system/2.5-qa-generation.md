# 2.5 问答生成模块

## 一、上下文构建

```python
"""
上下文构建
"""

from typing import List, Dict

class ContextBuilder:
    """上下文构建器"""

    def __init__(self, max_tokens: int = 3000):
        self.max_tokens = max_tokens

    def build(
        self,
        query: str,
        retrieved_docs: List[Dict]
    ) -> str:
        """构建上下文"""
        context_parts = []
        current_tokens = 0

        for i, doc in enumerate(retrieved_docs):
            content = doc["content"]
            tokens = self._count_tokens(content)

            if current_tokens + tokens > self.max_tokens:
                # 截断
                remaining = self.max_tokens - current_tokens
                if remaining > 100:
                    content = self._truncate(content, remaining)
                    context_parts.append(f"[文档{i+1}]\n{content}")
                break

            context_parts.append(f"[文档{i+1}]\n{content}")
            current_tokens += tokens

        return "\n\n".join(context_parts)

    def _count_tokens(self, text: str) -> int:
        """计算token数"""
        return len(text) // 2  # 简化估算

    def _truncate(self, text: str, max_tokens: int) -> str:
        """截断文本"""
        return text[:max_tokens * 2]
```

## 二、答案生成

```python
"""
答案生成
"""

class AnswerGenerator:
    """答案生成器"""

    def __init__(self, llm_client):
        self.llm = llm_client

    def generate(
        self,
        query: str,
        context: str
    ) -> Answer:
        """生成答案"""
        # 构建提示词
        prompt = self._build_prompt(query, context)

        # 调用LLM
        response = self.llm.generate(prompt, temperature=0.3)

        # 解析答案
        answer = self._parse_response(response.content)

        return answer

    def _build_prompt(self, query: str, context: str) -> str:
        return f"""基于以下参考文档回答问题。

参考文档：
{context}

问题：{query}

要求：
1. 仅基于参考文档回答
2. 如果文档中没有答案，请说明
3. 标注引用来源，如[文档1]
4. 回答准确、简洁

回答："""

    def _parse_response(self, response: str) -> Answer:
        """解析响应"""
        # 提取引用
        import re
        citations = re.findall(r'\[文档(\d+)\]', response)

        return Answer(
            content=response,
            citations=[int(c) for c in citations]
        )


@dataclass
class Answer:
    """答案"""
    content: str
    citations: List[int]
    confidence: float = 1.0
```

## 三、引用标注

```python
"""
引用标注
"""

class CitationManager:
    """引用管理器"""

    def add_citations(
        self,
        answer: str,
        retrieved_docs: List[Dict]
    ) -> AnnotatedAnswer:
        """添加引用"""
        # 1. 提取答案中的关键句
        sentences = self._split_sentences(answer)

        # 2. 为每个句子找来源
        annotated_sentences = []

        for sentence in sentences:
            source = self._find_source(sentence, retrieved_docs)

            if source:
                annotated_sentences.append({
                    "text": sentence,
                    "source": source
                })
            else:
                annotated_sentences.append({
                    "text": sentence,
                    "source": None
                })

        return AnnotatedAnswer(
            sentences=annotated_sentences,
            sources=retrieved_docs
        )

    def _split_sentences(self, text: str) -> List[str]:
        """分句"""
        import re
        return re.split(r'(?<=[。！？])', text)

    def _find_source(self, sentence: str, docs: List[Dict]) -> Dict:
        """找来源"""
        for doc in docs:
            if self._is_similar(sentence, doc["content"]):
                return {
                    "id": doc["id"],
                    "content": doc["content"][:100] + "...",
                    "score": doc.get("score", 0)
                }

        return None

    def _is_similar(self, text1: str, text2: str) -> bool:
        """判断相似"""
        # 简化：关键词匹配
        words1 = set(text1)
        words2 = set(text2)

        overlap = len(words1 & words2)
        return overlap > len(words1) * 0.3
```

## 四、质量评估

```python
"""
答案质量评估
"""

class AnswerEvaluator:
    """答案评估器"""

    def evaluate(
        self,
        query: str,
        answer: Answer,
        ground_truth: str = None
    ) -> EvaluationResult:
        """评估答案"""
        metrics = {}

        # 1. 相关性
        metrics["relevance"] = self._evaluate_relevance(query, answer.content)

        # 2. 完整性
        metrics["completeness"] = self._evaluate_completeness(query, answer.content)

        # 3. 准确性（如有ground truth）
        if ground_truth:
            metrics["accuracy"] = self._evaluate_accuracy(
                answer.content,
                ground_truth
            )

        # 4. 引用质量
        metrics["citation_quality"] = self._evaluate_citations(answer)

        return EvaluationResult(
            metrics=metrics,
            overall=sum(metrics.values()) / len(metrics)
        )

    def _evaluate_relevance(self, query: str, answer: str) -> float:
        """评估相关性"""
        # 使用LLM评估
        return 0.8  # 模拟分数

    def _evaluate_completeness(self, query: str, answer: str) -> float:
        """评估完整性"""
        return 0.75

    def _evaluate_accuracy(self, answer: str, ground_truth: str) -> float:
        """评估准确性"""
        return 0.85

    def _evaluate_citations(self, answer: Answer) -> float:
        """评估引用质量"""
        if not answer.citations:
            return 0.5

        # 检查引用是否合理
        return 0.9
```
