## 一、Embedding选择

### 1.1 模型对比

```
Embedding模型对比：

| 模型 | 维度 | 性能 | 成本 |
|------|------|------|------|
| text-embedding-3-small | 1536 | 高 | 低 |
| text-embedding-3-large | 3072 | 很高 | 中 |

推荐：
- 英文为主：text-embedding-3-small
- 中文为主：bge-large-zh
- 高精度需求：text-embedding-3-large
```

### 1.2 Embedding服务

```python
"""
Embedding服务
"""

from typing import List
import numpy as np

class EmbeddingService:
    """Embedding服务"""

    def __init__(self, model_name: str = "text-embedding-3-small"):
        self.model_name = model_name
        self.client = self._init_client()
        self.cache = {}

    def _init_client(self):
        from openai import OpenAI
        return OpenAI()

    def embed(self, text: str) -> List[float]:
        """生成向量"""
        # 检查缓存
        cache_key = self._make_cache_key(text)
        if cache_key in self.cache:
            return self.cache[cache_key]

        # 调用API
        response = self.client.embeddings.create(
            model=self.model_name,
            input=text
        )

        embedding = response.data[0].embedding

        # 缓存
        self.cache[cache_key] = embedding

        return embedding

    def embed_batch(self, texts: List[str]) -> List[List[float]]:
        """批量生成"""
        # 过滤已缓存
        uncached = []
        uncached_indices = []

        embeddings = [None] * len(texts)

        for i, text in enumerate(texts):
            cache_key = self._make_cache_key(text)
            if cache_key in self.cache:
                embeddings[i] = self.cache[cache_key]
            else:
                uncached.append(text)
                uncached_indices.append(i)

        # 批量请求
        if uncached:
            response = self.client.embeddings.create(
                model=self.model_name,
                input=uncached
            )

            for j, data in enumerate(response.data):
                embedding = data.embedding
                embeddings[uncached_indices[j]] = embedding
                self.cache[self._make_cache_key(uncached[j])] = embedding

        return embeddings

    def _make_cache_key(self, text: str) -> str:
        import hashlib
        return hashlib.md5(text.encode()).hexdigest()
```

## 二、向量数据库

### 2.1 Qdrant配置

```python
"""
Qdrant向量数据库
"""

from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams, PointStruct

class VectorStore:
    """向量存储"""

    def __init__(self, url: str = "http://localhost:6333"):
        self.client = QdrantClient(url=url)
        self.collection_name = "knowledge_base"

    def create_collection(self, vector_size: int = 1536):
        """创建集合"""
        self.client.create_collection(
            collection_name=self.collection_name,
            vectors_config=VectorParams(
                size=vector_size,
                distance=Distance.COSINE
            )
        )

    def add_vectors(
        self,
        ids: List[str],
        vectors: List[List[float]],
        payloads: List[dict]
    ):
        """添加向量"""
        points = [
            PointStruct(
                id=idx,
                vector=vector,
                payload=payload
            )
            for idx, (vector, payload) in enumerate(zip(vectors, payloads))
        ]

        self.client.upsert(
            collection_name=self.collection_name,
            points=points
        )

    def search(
        self,
        query_vector: List[float],
        top_k: int = 5
    ) -> List[dict]:
        """搜索向量"""
        results = self.client.search(
            collection_name=self.collection_name,
            query_vector=query_vector,
            limit=top_k
        )

        return [
            {
                "id": r.id,
                "score": r.score,
                "content": r.payload.get("content"),
                "metadata": r.payload.get("metadata", {})
            }
            for r in results
        ]

    def delete(self, ids: List[str]):
        """删除向量"""
        self.client.delete(
            collection_name=self.collection_name,
            points_selector=ids
        )
```

## 三、索引管理

```python
"""
索引管理器
"""

class IndexManager:
    """索引管理器"""

    def __init__(
        self,
        embedding_service: EmbeddingService,
        vector_store: VectorStore
    ):
        self.embedding = embedding_service
        self.store = vector_store

    def index_chunks(self, chunks: List[Chunk]) -> int:
        """索引文档块"""
        # 批量生成向量
        texts = [chunk.content for chunk in chunks]
        vectors = self.embedding.embed_batch(texts)

        # 准备payload
        payloads = [
            {
                "content": chunk.content,
                "metadata": chunk.metadata
            }
            for chunk in chunks
        ]

        # 生成ID
        ids = [self._generate_id(chunk) for chunk in chunks]

        # 存储
        self.store.add_vectors(ids, vectors, payloads)

        return len(chunks)

    def index_document(self, file_path: str) -> int:
        """索引单个文档"""
        # 处理文档
        pipeline = DocumentPipeline()
        chunks = pipeline.process(file_path)

        # 索引

        import hashlib
        content = chunk.content + str(chunk.metadata)
```
