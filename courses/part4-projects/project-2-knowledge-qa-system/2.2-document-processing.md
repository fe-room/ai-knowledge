# 2.2 文档处理Pipeline

## 一、文档加载

### 1.1 多格式支持

```python
"""
文档加载器
"""

from typing import List, Dict
from pathlib import Path

class DocumentLoader:
    """文档加载器"""

    def __init__(self):
        self.loaders = {
            '.pdf': self._load_pdf,
            '.docx': self._load_docx,
            '.md': self._load_markdown,
            '.txt': self._load_text,
            '.html': self._load_html
        }

    def load(self, file_path: str) -> Document:
        """加载文档"""
        ext = Path(file_path).suffix.lower()

        if ext not in self.loaders:
            raise ValueError(f"不支持的格式: {ext}")

        return self.loaders[ext](file_path)

    def _load_pdf(self, file_path: str) -> Document:
        """加载PDF"""
        from PyPDF2 import PdfReader

        reader = PdfReader(file_path)
        text = ""

        for page in reader.pages:
            text += page.extract_text() + "\n"

        return Document(
            content=text,
            metadata={
                "source": file_path,
                "type": "pdf",
                "pages": len(reader.pages)
            }
        )

    def _load_docx(self, file_path: str) -> Document:
        """加载Word"""
        from docx import Document as DocxDocument

        doc = DocxDocument(file_path)
        text = "\n".join([p.text for p in doc.paragraphs])

        return Document(
            content=text,
            metadata={
                "source": file_path,
                "type": "docx"
            }
        )

    def _load_markdown(self, file_path: str) -> Document:
        """加载Markdown"""
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        return Document(
            content=content,
            metadata={
                "source": file_path,
                "type": "markdown"
            }
        )

    def _load_text(self, file_path: str) -> Document:
        """加载文本"""
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        return Document(
            content=content,
            metadata={
                "source": file_path,
                "type": "text"
            }
        )

    def _load_html(self, file_path: str) -> Document:
        """加载HTML"""
        from bs4 import BeautifulSoup

        with open(file_path, 'r', encoding='utf-8') as f:
            soup = BeautifulSoup(f.read(), 'html.parser')

        # 提取文本
        text = soup.get_text(separator='\n')

        return Document(
            content=text,
            metadata={
                "source": file_path,
                "type": "html"
            }
        )
```

## 二、智能分块

### 2.1 分块策略

```python
"""
智能分块
"""

from typing import List

class Chunker:
    """分块器"""

    def __init__(
        self,
        chunk_size: int = 500,
        chunk_overlap: int = 50
    ):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap

    def chunk(self, document: Document) -> List[Chunk]:
        """分块文档"""
        # 根据文档类型选择策略
        if document.metadata.get("type") == "markdown":
            return self._chunk_by_headers(document)
        else:
            return self._chunk_by_size(document)

    def _chunk_by_size(self, document: Document) -> List[Chunk]:
        """按大小分块"""
        chunks = []
        text = document.content

        # 分句
        sentences = self._split_sentences(text)

        current_chunk = []
        current_size = 0

        for sentence in sentences:
            sentence_size = len(sentence)

            if current_size + sentence_size > self.chunk_size:
                if current_chunk:
                    chunks.append(Chunk(
                        content="".join(current_chunk),
                        metadata=document.metadata.copy()
                    ))

                # 保留overlap
                overlap_text = self._get_overlap(current_chunk)
                current_chunk = [overlap_text, sentence]
                current_size = len(overlap_text) + sentence_size
            else:
                current_chunk.append(sentence)
                current_size += sentence_size

        # 最后一个块
        if current_chunk:
            chunks.append(Chunk(
                content="".join(current_chunk),
                metadata=document.metadata.copy()
            ))

        return chunks

    def _chunk_by_headers(self, document: Document) -> List[Chunk]:
        """按标题分块（Markdown）"""
        import re

        chunks = []
        lines = document.content.split('\n')

        current_header = ""
        current_content = []

        for line in lines:
            # 检测标题
            header_match = re.match(r'^(#{1,6})\s+(.+)$', line)

            if header_match:
                # 保存前一块
                if current_content:
                    chunks.append(Chunk(
                        content="\n".join(current_content),
                        metadata={
                            **document.metadata,
                            "header": current_header
                        }
                    ))

                current_header = header_match.group(2)
                current_content = [line]
            else:
                current_content.append(line)

        # 最后一个块
        if current_content:
            chunks.append(Chunk(
                content="\n".join(current_content),
                metadata={
                    **document.metadata,
                    "header": current_header
                }
            ))

        return chunks

    def _split_sentences(self, text: str) -> List[str]:
        """分句"""
        import re
        return re.split(r'(?<=[。！？\n])', text)

    def _get_overlap(self, chunk_parts: List[str]) -> str:
        """获取重叠文本"""
        if not chunk_parts:
            return ""

        text = "".join(chunk_parts)
        return text[-self.chunk_overlap:] if len(text) > self.chunk_overlap else text
```

## 三、处理Pipeline

```python
"""
文档处理Pipeline
"""

class DocumentPipeline:
    """文档处理Pipeline"""

    def __init__(self):
        self.loader = DocumentLoader()
        self.chunker = Chunker()
        self.cleaner = TextCleaner()

    def process(self, file_path: str) -> List[Chunk]:
        """处理文档"""
        # 1. 加载
        document = self.loader.load(file_path)

        # 2. 清洗
        document = self.cleaner.clean(document)

        # 3. 分块
        chunks = self.chunker.chunk(document)

        return chunks

class TextCleaner:
    """文本清洗器"""

    def clean(self, document: Document) -> Document:
        """清洗文本"""
        content = document.content

        # 移除多余空白
        content = self._remove_extra_whitespace(content)

        # 移除特殊字符
        content = self._remove_special_chars(content)

        # 标准化换行
        content = self._normalize_newlines(content)

        document.content = content
        return document

    def _remove_extra_whitespace(self, text: str) -> str:
        import re
        return re.sub(r'\s+', ' ', text)

    def _remove_special_chars(self, text: str) -> str:
        import re
        # 保留中文、英文、数字、基本标点
        return re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9\s，。！？、；：""''（）\-\.\,\!\?\;\:\"\'\(\)]', '', text)

    def _normalize_newlines(self, text: str) -> str:
        import re
        return re.sub(r'\n{3,}', '\n\n', text)
```
