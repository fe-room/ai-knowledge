# 3.5 Reviewer Agent

## 一、质量审核

```python
"""
Reviewer Agent
"""

class ReviewerAgent:
    """审核Agent"""

    def __init__(self, llm_client):
        self.llm = llm_client

    def review(self, document: Document, criteria: List[str] = None) -> ReviewResult:
        """审核文档"""
        # 默认审核标准
        if not criteria:
            criteria = ["准确性", "完整性", "清晰度", "格式规范"]

        # 逐项审核
        results = {}

        for criterion in criteria:
            score, feedback = self._evaluate(document, criterion)
            results[criterion] = {
                "score": score,
                "feedback": feedback
            }

        # 综合评估
        overall = self._calculate_overall(results)

        return ReviewResult(
            document_id=document.id,
            criteria_results=results,
            overall_score=overall,
            needs_revision=overall < 0.8
        )

    def _evaluate(self, document: Document, criterion: str) -> Tuple[float, str]:
        """评估"""
        prompt = f"""
        评估文档的{criterion}：

        文档内容：
        {document.content[:2000]}

        返回JSON：
        {{
            "score": 0.0-1.0,
            "feedback": "评估意见"
        }}
        """

        response = self.llm.generate(prompt, temperature=0.2)
        result = self._parse_response(response.content)

        return result["score"], result["feedback"]

    def suggest_revisions(self, review_result: ReviewResult) -> List[str]:
        """建议修改"""
        suggestions = []

        for criterion, result in review_result.criteria_results.items():
            if result["score"] < 0.7:
                suggestions.append(f"【{criterion}】{result['feedback']}")

        return suggestions
```

## 二、反馈生成

```python
class FeedbackGenerator:
    """反馈生成器"""

    def generate(self, review_result: ReviewResult) -> Feedback:
        """生成反馈"""
        # 汇总问题
        issues = self._identify_issues(review_result)

        # 生成建议
        suggestions = self._generate_suggestions(issues)

        return Feedback(
            overall_assessment=self._overall_assessment(review_result),
            issues=issues,
            suggestions=suggestions,
            revision_required=review_result.needs_revision
        )
```
