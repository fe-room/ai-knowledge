"""
Agent角色定义
"""

from dataclasses import dataclass
from typing import List, Dict

@dataclass
class AgentRole:
    """Agent角色"""
    name: str
    description: str
    capabilities: List[str]
    tools: List[str]
    llm_config: Dict

# 定义角色
ROLES = {
    "planner": AgentRole(
        name="Planner",
        description="任务规划师，负责分析任务、制定计划、分配工作",
        capabilities=["task_decomposition", "planning", "scheduling"],
        tools=["calendar", "task_manager"],
        llm_config={"model": "gpt-4", "temperature": 0.3}

    "researcher": AgentRole(

## 三、协作模式

```python
"""
协作模式设计
"""

from enum import Enum

class CollaborationMode(Enum):
    """协作模式"""
    SEQUENTIAL = "sequential"  # 顺序执行
    PARALLEL = "parallel"  # 并行执行
    HIERARCHICAL = "hierarchical"  # 层级协作

class Workflow:
    """工作流"""

    def __init__(self, mode: CollaborationMode):
        self.mode = mode
        self.agents = []
        self.transitions = []

    def add_agent(self, agent: AgentRole):
        """添加Agent"""
        self.agents.append(agent)

    def add_transition(self, from_agent: str, to_agent: str, condition: str = None):
        """添加转换"""
        self.transitions.append({
            "from": from_agent,
            "to": to_agent,
            "condition": condition
        })

# 示例：研究报告工作流
research_workflow = Workflow(CollaborationMode.SEQUENTIAL)
research_workflow.add_agent(ROLES["planner"])
research_workflow.add_agent(ROLES["researcher"])
research_workflow.add_agent(ROLES["writer"])
research_workflow.add_agent(ROLES["reviewer"])

research_workflow.add_transition("planner", "researcher")
research_workflow.add_transition("researcher", "writer")
research_workflow.add_transition("writer", "reviewer")
research_workflow.add_transition("reviewer", "writer", condition="needs_revision")
```

## 四、通信机制

```python
"""
Agent通信
"""

@dataclass
class Message:
    """消息"""
    sender: str
    receiver: str
    content: str
    message_type: str  # task, result, feedback
    metadata: Dict = None

class MessageBus:
    """消息总线"""

    def __init__(self):
        self.messages = []
        self.handlers = {}

    def register(self, agent_name: str, handler: callable):
        """注册处理器"""
        self.handlers[agent_name] = handler

    def send(self, message: Message):
        """发送消息"""
        self.messages.append(message)

        # 通知接收者
        if message.receiver in self.handlers:
            self.handlers[message.receiver](message)

    def broadcast(self, sender: str, content: str):
        """广播消息"""
        for agent_name in self.handlers:
            if agent_name != sender:
                self.send(Message(
                    sender=sender,
                    receiver=agent_name,
                    content=content,
                    message_type="broadcast"
                ))
```
