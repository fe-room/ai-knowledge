## 一、Python实现

```python
# main.py

from fastapi import FastAPI
from pydantic import BaseModel

from orchestrator import Orchestrator
from agents import PlannerAgent, ResearcherAgent, WriterAgent, ReviewerAgent

app = FastAPI(title="多Agent协作系统")

# 初始化
orchestrator = Orchestrator()
orchestrator.register_agent("planner", PlannerAgent())
orchestrator.register_agent("researcher", ResearcherAgent())
orchestrator.register_agent("writer", WriterAgent())
orchestrator.register_agent("reviewer", ReviewerAgent())

class TaskRequest(BaseModel):
    task: str
    options: Dict = None

@app.post("/execute")
async def execute_task(request: TaskRequest):
    """执行任务"""
    result = orchestrator.execute_workflow(request.task)

    return {
        "document": result.document.content,
        "review": result.review.to_dict()
    }
```

## 二、JavaScript实现

```typescript
// app.ts

import express from 'express';
import { Orchestrator } from './orchestrator';
import { PlannerAgent } from './agents/PlannerAgent';
import { ResearcherAgent } from './agents/ResearcherAgent';
import { WriterAgent } from './agents/WriterAgent';
import { ReviewerAgent } from './agents/ReviewerAgent';

const app = express();
app.use(express.json());

const orchestrator = new Orchestrator();
orchestrator.registerAgent('planner', new PlannerAgent());
orchestrator.registerAgent('researcher', new ResearcherAgent());
orchestrator.registerAgent('writer', new WriterAgent());
orchestrator.registerAgent('reviewer', new ReviewerAgent());

app.post('/execute', async (req, res) => {
    const { task, options } = req.body;

    const result = await orchestrator.executeWorkflow(task);

    res.json({
        document: result.document.content,
        review: result.review
    });
});

app.listen(3000, () => {
    console.log('Multi-Agent system running on port 3000');
});
```
