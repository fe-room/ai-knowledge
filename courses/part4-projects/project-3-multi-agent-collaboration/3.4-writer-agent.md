# 3.4 Writer Agent

## 一、内容生成

```python
"""
Writer Agent
"""

class WriterAgent:
    """写作者Agent"""

    def __init__(self, llm_client):
        self.llm = llm_client

    def write(self, task: str, research_data: OrganizedData) -> Document:
        """生成内容"""
        # 1. 生成大纲
        outline = self._create_outline(task, research_data)

        # 2. 逐节写作
        sections = []
        for section in outline.sections:
            content = self._write_section(section, research_data)
            sections.append(content)

        # 3. 整合文档
        document = self._assemble_document(sections)

        return document

    def _create_outline(self, task: str, data: OrganizedData) -> Outline:
        """创建大纲"""
        prompt = f"""
        为以下任务创建文档大纲：

        任务：{task}
        可用资料：{data.summary}

        返回JSON格式大纲：
        {{
            "title": "标题",
            "sections": [
                {{"title": "章节标题", "points": ["要点1", "要点2"]}}
            ]
        }}
        """

        response = self.llm.generate(prompt)
        return Outline(**self._parse_response(response.content))

    def _write_section(self, section, data: OrganizedData) -> str:
        """写作章节"""
        prompt = f"""
        撰写以下章节：

        章节标题：{section.title}
        要点：{section.points}
        参考资料：{data.get_relevant_sources(section.title)}

        要求：
        1. 内容充实
        2. 逻辑清晰
        3. 标注引用
        """

        return self.llm.generate(prompt, temperature=0.7).content
```

## 二、格式处理

```python
class Formatter:
    """格式化器"""

    def format(self, document: Document, style: str = "markdown") -> str:
        """格式化文档"""
        if style == "markdown":
            return self._to_markdown(document)
        elif style == "html":
            return self._to_html(document)
        else:
            return document.raw_content

    def _to_markdown(self, document: Document) -> str:
        """转Markdown"""
        lines = [f"# {document.title}", ""]

        for section in document.sections:
            lines.append(f"## {section.title}")
            lines.append(section.content)
            lines.append("")

        return "\n".join(lines)
```
