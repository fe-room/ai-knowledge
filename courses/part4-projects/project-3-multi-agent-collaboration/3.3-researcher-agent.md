# 3.3 Researcher Agent

## 一、信息检索

```python
"""
Researcher Agent
"""

from typing import List, Dict

class ResearcherAgent:
    """研究员Agent"""

    def __init__(self, llm_client, search_tool):
        self.llm = llm_client
        self.search = search_tool

    def research(self, topic: str, depth: str = "medium") -> ResearchResult:
        """研究主题"""
        # 1. 生成搜索查询
        queries = self._generate_queries(topic)

        # 2. 执行搜索
        all_results = []
        for query in queries:
            results = self.search.search(query)
            all_results.extend(results)

        # 3. 去重和筛选
        filtered = self._filter_results(all_results)

        # 4. 整理和汇总
        summary = self._summarize_findings(filtered)

        return ResearchResult(
            topic=topic,
            sources=filtered,
            summary=summary
        )

    def _generate_queries(self, topic: str) -> List[str]:
        """生成搜索查询"""
        prompt = f"""
        为以下研究主题生成3-5个搜索查询：

        主题：{topic}

        返回JSON数组：["查询1", "查询2", ...]
        """

        response = self.llm.generate(prompt)
        return self._parse_response(response.content)

    def _filter_results(self, results: List[Dict]) -> List[Dict]:
        """筛选结果"""
        # 去重
        seen = set()
        filtered = []

        for result in results:
            key = result.get("url") or result.get("title")
            if key not in seen:
                seen.add(key)
                filtered.append(result)

        return filtered[:20]  # 最多20个

    def _summarize_findings(self, results: List[Dict]) -> str:
        """总结发现"""
        contents = [r.get("content", "")[:500] for r in results[:5]]

        prompt = f"""
        总结以下研究发现：

        资料：
        {chr(10).join(contents)}

        要求：
        1. 提取关键信息
        2. 保留重要数据
        3. 标注来源
        """

        return self.llm.generate(prompt).content


@dataclass
class ResearchResult:
    """研究结果"""
    topic: str
    sources: List[Dict]
    summary: str
```

## 二、资料整理

```python
"""
资料整理
"""

class DataOrganizer:
    """数据整理器"""

    def organize(self, research_result: ResearchResult) -> OrganizedData:
        """整理资料"""
        # 按主题分类
        categories = self._categorize(research_result.sources)

        # 提取关键信息
        key_points = self._extract_key_points(research_result.summary)

        # 生成引用
        citations = self._generate_citations(research_result.sources)

        return OrganizedData(
            categories=categories,
            key_points=key_points,
            citations=citations
        )

    def _categorize(self, sources: List[Dict]) -> Dict[str, List]:
        """分类"""
        categories = {}

        for source in sources:
            category = source.get("category", "general")

            if category not in categories:
                categories[category] = []

            categories[category].append(source)

        return categories
```
