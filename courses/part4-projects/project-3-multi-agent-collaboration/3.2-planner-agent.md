## 一、任务分解

```python
"""
Planner Agent
"""

from typing import List, Dict

class PlannerAgent:
    """规划Agent"""

    def __init__(self, llm_client):
        self.llm = llm_client

    def analyze_task(self, task: str) -> TaskAnalysis:
        """分析任务"""
        prompt = f"""
        分析以下任务，提取关键信息：

        任务：{task}

        返回JSON格式：
        {{
            "task_type": "任务类型",
            "objectives": ["目标1", "目标2"],
            "constraints": ["约束条件"],
            "required_resources": ["资源"],
            "estimated_complexity": "low/medium/high"
        }}
        """

        response = self.llm.generate(prompt, temperature=0.2)
        analysis = self._parse_response(response.content)

        return TaskAnalysis(**analysis)

    def decompose_task(self, task: str, analysis: TaskAnalysis) -> List[SubTask]:
        """分解任务"""
        prompt = f"""
        将以下任务分解为子任务：

        任务：{task}
        分析：{analysis.to_dict()}

        返回JSON格式子任务列表：
        [
            {{
                "id": "subtask_1",
                "name": "子任务名称",
                "description": "描述",
                "assigned_to": "researcher/writer/reviewer",
                "dependencies": [],
                "estimated_time": "10min"
            }}
        ]
        """

        response = self.llm.generate(prompt, temperature=0.3)
        subtasks = self._parse_response(response.content)

        return [SubTask(**st) for st in subtasks]

    def create_plan(self, subtasks: List[SubTask]) -> Plan:
        """创建执行计划"""
        # 拓扑排序
        ordered = self._topological_sort(subtasks)

        # 创建计划
        plan = Plan(
            tasks=ordered,
            status="pending",
            created_at=datetime.now()
        )

        return plan

    def _topological_sort(self, subtasks: List[SubTask]) -> List[SubTask]:
        """拓扑排序"""
        # 实现依赖排序
        return subtasks  # 简化


@dataclass
class TaskAnalysis:
    """任务分析"""
    task_type: str
    objectives: List[str]
    constraints: List[str]
    required_resources: List[str]
    estimated_complexity: str

@dataclass
class SubTask:
    """子任务"""
    id: str
    name: str
    description: str
    assigned_to: str
    dependencies: List[str]
    estimated_time: str
    status: str = "pending"

@dataclass
class Plan:
    """执行计划"""
    tasks: List[SubTask]
    status: str
    created_at: datetime
```

## 二、进度跟踪

```python
"""
进度跟踪
"""

class ProgressTracker:
    """进度跟踪器"""

    def __init__(self):
        self.task_status = {}
        self.completion_times = {}

    def update(self, task_id: str, status: str):
        """更新状态"""
        self.task_status[task_id] = status

        if status == "completed":
            self.completion_times[task_id] = datetime.now()

    def get_progress(self) -> Dict:
        """获取进度"""
        total = len(self.task_status)
        completed = sum(1 for s in self.task_status.values() if s == "completed")

        return {
            "total": total,
            "completed": completed,
            "percentage": completed / total * 100 if total > 0 else 0
        }

    def get_next_task(self) -> Optional[str]:
        """获取下一个任务"""
        for task_id, status in self.task_status.items():
            if status == "pending":
                return task_id
        return None
```
